# P9.1 — Fully Typed React App

**Phase:** 9 — TypeScript, Accessibility & Production

**Type:** Project

## Goal

Rebuild or retrofit a React app with TypeScript strict mode enabled throughout. Zero `any` types, all API responses validated with Zod, routes are type-safe, and every custom hook has an explicit, precise return type.

---

## What to Build

A typed **Task Management** app (CRUD for tasks with categories and priorities):

- Task list with filter by status and category
- Add / edit / delete tasks via a modal form
- API responses validated with Zod (mock API)
- Type-safe route parameters with React Router
- Generic `useMutation` hook with full type safety
- All `useRef` and `useContext` properly typed
- Zero TypeScript errors with strict + `noUncheckedIndexedAccess`

### Zod schemas — single source of truth

```ts
// src/schemas/task.ts
import { z } from 'zod';

export const PrioritySchema = z.enum(['low', 'medium', 'high', 'critical']);
export const StatusSchema = z.enum(['todo', 'in-progress', 'done', 'cancelled']);

export const TaskSchema = z.object({
  id: z.string().uuid(),
  title: z.string().min(1).max(200),
  description: z.string().optional(),
  priority: PrioritySchema,
  status: StatusSchema,
  category: z.string(),
  dueDate: z.string().datetime().optional(),
  createdAt: z.string().datetime(),
  updatedAt: z.string().datetime(),
  assignee: z.object({
    id: z.string(),
    name: z.string(),
    avatarUrl: z.string().url().optional(),
  }).optional(),
});

export const CreateTaskSchema = TaskSchema.omit({
  id: true,
  createdAt: true,
  updatedAt: true,
});

export const UpdateTaskSchema = CreateTaskSchema.partial();

export const TaskListResponseSchema = z.object({
  data: z.array(TaskSchema),
  total: z.number(),
  page: z.number(),
  pageSize: z.number(),
});

// TypeScript types derived from schemas
export type Priority = z.infer<typeof PrioritySchema>;
export type Status = z.infer<typeof StatusSchema>;
export type Task = z.infer<typeof TaskSchema>;
export type CreateTaskInput = z.infer<typeof CreateTaskSchema>;
export type UpdateTaskInput = z.infer<typeof UpdateTaskSchema>;
export type TaskListResponse = z.infer<typeof TaskListResponseSchema>;
```

### Type-safe API client

```ts
// src/api/client.ts
import { z } from 'zod';

export class ApiError extends Error {
  constructor(
    public readonly status: number,
    public readonly body: unknown,
    message: string
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

export async function apiFetch<T>(
  url: string,
  schema: z.ZodType<T>,
  options?: RequestInit
): Promise<T> {
  const res = await fetch(url, {
    headers: { 'Content-Type': 'application/json', ...options?.headers },
    ...options,
  });

  if (!res.ok) {
    const body = await res.json().catch(() => null);
    throw new ApiError(res.status, body, `API error ${res.status}`);
  }

  const raw = await res.json();

  // Validate and parse — throws ZodError if shape is wrong
  return schema.parse(raw);
}
```

```ts
// src/api/tasks.ts
import { apiFetch } from './client';
import {
  TaskSchema,
  TaskListResponseSchema,
  type Task,
  type CreateTaskInput,
  type UpdateTaskInput,
  type TaskListResponse,
} from '../schemas/task';
import { z } from 'zod';

const BASE = '/api/tasks';

export const tasksApi = {
  list: (params?: { status?: string; page?: number }): Promise<TaskListResponse> => {
    const qs = new URLSearchParams(
      Object.entries(params ?? {})
        .filter(([, v]) => v !== undefined)
        .map(([k, v]) => [k, String(v)])
    ).toString();
    return apiFetch(`${BASE}${qs ? `?${qs}` : ''}`, TaskListResponseSchema);
  },

  get: (id: string): Promise<Task> =>
    apiFetch(`${BASE}/${id}`, TaskSchema),

  create: (input: CreateTaskInput): Promise<Task> =>
    apiFetch(BASE, TaskSchema, {
      method: 'POST',
      body: JSON.stringify(input),
    }),

  update: (id: string, input: UpdateTaskInput): Promise<Task> =>
    apiFetch(`${BASE}/${id}`, TaskSchema, {
      method: 'PATCH',
      body: JSON.stringify(input),
    }),

  delete: (id: string): Promise<void> =>
    apiFetch(`${BASE}/${id}`, z.void(), { method: 'DELETE' }),
};
```

### Generic mutation hook

```ts
// src/hooks/useTaskMutation.ts
import { useState, useCallback } from 'react';
import { ZodError } from 'zod';
import { ApiError } from '../api/client';

type MutationState<T> =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: ApiError | ZodError | Error };

export function useTaskMutation<TInput, TOutput>(
  mutationFn: (input: TInput) => Promise<TOutput>
) {
  const [state, setState] = useState<MutationState<TOutput>>({ status: 'idle' });

  const mutate = useCallback(
    async (input: TInput): Promise<TOutput | null> => {
      setState({ status: 'loading' });
      try {
        const data = await mutationFn(input);
        setState({ status: 'success', data });
        return data;
      } catch (error) {
        const err = error instanceof Error ? error : new Error(String(error));
        setState({ status: 'error', error: err });
        return null;
      }
    },
    [mutationFn]
  );

  const reset = useCallback(() => setState({ status: 'idle' }), []);

  return {
    mutate,
    reset,
    isIdle: state.status === 'idle',
    isLoading: state.status === 'loading',
    isSuccess: state.status === 'success',
    isError: state.status === 'error',
    data: state.status === 'success' ? state.data : null,
    error: state.status === 'error' ? state.error : null,
  };
}
```

### Type-safe search params hook

```ts
// src/hooks/useTypedSearchParams.ts
import { useSearchParams } from 'react-router-dom';
import { z } from 'zod';
import { useCallback } from 'react';

/**
 * useTypedSearchParams — parse and set URL search params with Zod validation.
 * Invalid params fall back to defaults without throwing.
 */
export function useTypedSearchParams<T extends z.ZodRawShape>(
  schema: z.ZodObject<T>,
  defaults: z.infer<z.ZodObject<T>>
): [z.infer<z.ZodObject<T>>, (updates: Partial<z.infer<z.ZodObject<T>>>) => void] {
  const [searchParams, setSearchParams] = useSearchParams();

  const raw: Record<string, string> = {};
  searchParams.forEach((value, key) => {
    raw[key] = value;
  });

  const parsed = schema.safeParse(raw);
  const params = parsed.success ? parsed.data : defaults;

  const setParams = useCallback(
    (updates: Partial<z.infer<z.ZodObject<T>>>) => {
      setSearchParams(prev => {
        const next = new URLSearchParams(prev);
        for (const [key, value] of Object.entries(updates)) {
          if (value === undefined || value === null || value === '') {
            next.delete(key);
          } else {
            next.set(key, String(value));
          }
        }
        return next;
      });
    },
    [setSearchParams]
  );

  return [params, setParams];
}
```

### Usage in TaskFilters

```tsx
// src/components/TaskFilters.tsx
import { useTypedSearchParams } from '../hooks/useTypedSearchParams';
import { z } from 'zod';
import { StatusSchema, PrioritySchema } from '../schemas/task';

const filtersSchema = z.object({
  status: StatusSchema.optional(),
  priority: PrioritySchema.optional(),
  page: z.coerce.number().min(1).optional().default(1),
  search: z.string().optional(),
});

const defaults = { page: 1 } as const;

export function TaskFilters() {
  const [filters, setFilters] = useTypedSearchParams(filtersSchema, defaults);

  return (
    <div className="flex gap-3">
      <select
        value={filters.status ?? ''}
        onChange={e => setFilters({ status: e.target.value as z.infer<typeof StatusSchema> || undefined })}
        className="border rounded-lg px-3 py-2 text-sm"
      >
        <option value="">All statuses</option>
        <option value="todo">To Do</option>
        <option value="in-progress">In Progress</option>
        <option value="done">Done</option>
      </select>

      <select
        value={filters.priority ?? ''}
        onChange={e => setFilters({ priority: e.target.value as z.infer<typeof PrioritySchema> || undefined })}
        className="border rounded-lg px-3 py-2 text-sm"
      >
        <option value="">All priorities</option>
        <option value="low">Low</option>
        <option value="medium">Medium</option>
        <option value="high">High</option>
        <option value="critical">Critical</option>
      </select>
    </div>
  );
}
```

### PriorityBadge — discriminated union

```tsx
// src/components/PriorityBadge.tsx
import type { Priority } from '../schemas/task';
import { cn } from '../lib/utils';

const config: Record<Priority, { label: string; className: string }> = {
  low: { label: 'Low', className: 'bg-gray-100 text-gray-600' },
  medium: { label: 'Medium', className: 'bg-yellow-100 text-yellow-700' },
  high: { label: 'High', className: 'bg-orange-100 text-orange-700' },
  critical: { label: 'Critical', className: 'bg-red-100 text-red-700 font-bold' },
};

export function PriorityBadge({ priority }: { priority: Priority }) {
  // config[priority] is guaranteed to exist — Priority is the key type
  const { label, className } = config[priority];
  return (
    <span className={cn('px-2 py-0.5 rounded-full text-xs', className)}>
      {label}
    </span>
  );
}
```
