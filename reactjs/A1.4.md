## Goal
Understand when and how to run side effects in React components using useEffect, and why proper cleanup prevents bugs and memory leaks.

## Background

### What is a Side Effect?
A side effect is anything a component does beyond returning JSX — fetching data, setting up a timer, subscribing to an event, updating the document title, or writing to localStorage. React separates rendering (pure, no side effects) from effects (run after render).

`useEffect(callback, deps)` runs the callback after the browser paints. The `deps` array controls when it re-runs: an empty array `[]` runs only on mount, a list of values runs whenever any of those values change, and no array runs after every render (rarely what you want).

### The Cleanup Function
Your effect callback can return a cleanup function. React calls it before the component unmounts and before the effect runs again (if deps changed). Without cleanup: event listeners pile up, intervals run forever, and cancelled fetch requests try to update unmounted component state.

### Avoiding the Async useEffect Mistake
You cannot pass an `async` function directly to `useEffect` because async functions return a Promise, and React expects either `undefined` or a cleanup function. The pattern is to define an async function inside the effect and call it immediately.

## Tasks

### Task 1 — Document Title Effect

```tsx
// src/components/PageTitle.tsx
import { useEffect } from 'react'

interface PageTitleProps {
  title: string
  suffix?: string
}

export function useDocumentTitle(title: string, suffix = 'My App') {
  useEffect(() => {
    const previous = document.title
    document.title = `${title} | ${suffix}`

    // Cleanup: restore previous title when component unmounts
    return () => {
      document.title = previous
    }
  }, [title, suffix])  // re-run when title or suffix changes
}

export function PageWithTitle({ title }: PageTitleProps) {
  useDocumentTitle(title)

  return <h1>{title}</h1>
}
```

### Task 2 — Fetch on Mount with Loading and Error States

```tsx
// src/components/PostList.tsx
import { useState, useEffect } from 'react'

interface Post {
  id: number
  title: string
  body: string
  userId: number
}

type FetchState<T> =
  | { status: 'idle' }
  | { status: 'loading' }
  | { status: 'success'; data: T }
  | { status: 'error'; error: string }

export function PostList() {
  const [state, setState] = useState<FetchState<Post[]>>({ status: 'idle' })

  useEffect(() => {
    // Cannot write: useEffect(async () => { ... })
    // Must define async function inside and call it
    let cancelled = false  // abort flag for cleanup

    setState({ status: 'loading' })

    async function fetchPosts() {
      try {
        const res = await fetch('https://jsonplaceholder.typicode.com/posts?_limit=10')

        if (!res.ok) throw new Error(`HTTP error ${res.status}`)

        const data: Post[] = await res.json()

        if (!cancelled) {  // only update state if still mounted
          setState({ status: 'success', data })
        }
      } catch (err) {
        if (!cancelled) {
          setState({ status: 'error', error: err instanceof Error ? err.message : 'Unknown error' })
        }
      }
    }

    fetchPosts()

    // Cleanup: set flag so in-flight fetch doesn't update state after unmount
    return () => {
      cancelled = true
    }
  }, [])  // empty array: run only on mount

  if (state.status === 'idle' || state.status === 'loading') {
    return <div>Loading posts...</div>
  }

  if (state.status === 'error') {
    return <div style={{ color: 'red' }}>Error: {state.error}</div>
  }

  return (
    <ul>
      {state.data.map(post => (
        <li key={post.id} style={{ marginBottom: '1rem' }}>
          <strong>{post.title}</strong>
          <p style={{ color: '#666', margin: '4px 0 0', fontSize: 14 }}>{post.body.slice(0, 80)}...</p>
        </li>
      ))}
    </ul>
  )
}
```

### Task 3 — Effect with Changing Dependencies

```tsx
// src/components/UserSearch.tsx
import { useState, useEffect } from 'react'

interface User {
  id: number
  name: string
  email: string
  company: { name: string }
}

export function UserSearch() {
  const [userId, setUserId] = useState(1)
  const [user, setUser] = useState<User | null>(null)
  const [loading, setLoading] = useState(false)

  useEffect(() => {
    let cancelled = false
    setLoading(true)
    setUser(null)

    async function loadUser() {
      try {
        const res = await fetch(`https://jsonplaceholder.typicode.com/users/${userId}`)
        const data: User = await res.json()
        if (!cancelled) {
          setUser(data)
          setLoading(false)
        }
      } catch {
        if (!cancelled) setLoading(false)
      }
    }

    loadUser()

    return () => { cancelled = true }
  }, [userId])  // re-run whenever userId changes

  return (
    <div style={{ maxWidth: 400 }}>
      <div style={{ display: 'flex', gap: '0.5rem', marginBottom: '1rem' }}>
        {[1, 2, 3, 4, 5].map(id => (
          <button
            key={id}
            onClick={() => setUserId(id)}
            style={{
              background: userId === id ? '#3b82f6' : '#f1f5f9',
              color: userId === id ? 'white' : 'inherit',
              border: 'none',
              borderRadius: 4,
              padding: '6px 12px',
              cursor: 'pointer',
            }}
          >
            User {id}
          </button>
        ))}
      </div>

      {loading && <p>Loading user {userId}...</p>}

      {user && !loading && (
        <div style={{ padding: '1rem', border: '1px solid #e5e7eb', borderRadius: 6 }}>
          <h3 style={{ margin: '0 0 4px' }}>{user.name}</h3>
          <p style={{ margin: '0 0 4px', color: '#6b7280', fontSize: 14 }}>{user.email}</p>
          <p style={{ margin: 0, fontSize: 13 }}>Company: {user.company.name}</p>
        </div>
      )}
    </div>
  )
}
```

### Task 4 — Timer with Cleanup

```tsx
// src/components/Stopwatch.tsx
import { useState, useEffect, useRef } from 'react'

export function Stopwatch() {
  const [running, setRunning] = useState(false)
  const [elapsed, setElapsed] = useState(0)  // ms

  useEffect(() => {
    if (!running) return  // early return: no setup needed when paused

    const start = Date.now() - elapsed  // account for already-elapsed time

    const intervalId = setInterval(() => {
      setElapsed(Date.now() - start)
    }, 10)  // update every 10ms

    // Cleanup: always clear interval to prevent memory leaks
    return () => clearInterval(intervalId)
  }, [running])  // re-run when running changes
  // Note: 'elapsed' is intentionally omitted — we capture it at start

  function formatTime(ms: number) {
    const centiseconds = Math.floor((ms % 1000) / 10)
    const seconds = Math.floor((ms / 1000) % 60)
    const minutes = Math.floor(ms / 60000)
    return `${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}.${String(centiseconds).padStart(2, '0')}`
  }

  return (
    <div style={{ textAlign: 'center' }}>
      <div style={{ fontSize: 48, fontFamily: 'monospace', margin: '1rem 0' }}>
        {formatTime(elapsed)}
      </div>
      <div style={{ display: 'flex', gap: '0.5rem', justifyContent: 'center' }}>
        <button onClick={() => setRunning(r => !r)}>
          {running ? 'Pause' : 'Start'}
        </button>
        <button
          onClick={() => { setRunning(false); setElapsed(0) }}
          disabled={elapsed === 0}
        >
          Reset
        </button>
      </div>
    </div>
  )
}
```

### Task 5 — Event Listener with Cleanup

```tsx
// src/components/MouseTracker.tsx
import { useState, useEffect } from 'react'

export function MouseTracker() {
  const [position, setPosition] = useState({ x: 0, y: 0 })
  const [visible, setVisible] = useState(false)

  useEffect(() => {
    function handleMouseMove(e: MouseEvent) {
      setPosition({ x: e.clientX, y: e.clientY })
    }

    // Add listener
    window.addEventListener('mousemove', handleMouseMove)

    // Cleanup: remove listener when component unmounts
    // Without this, the listener keeps running even after component is gone
    return () => {
      window.removeEventListener('mousemove', handleMouseMove)
    }
  }, [])  // empty array: set up once on mount, clean up on unmount

  return (
    <div>
      <button onClick={() => setVisible(v => !v)}>
        {visible ? 'Hide' : 'Show'} tracker
      </button>

      {visible && (
        <p>
          Mouse: {position.x}px, {position.y}px
        </p>
      )}
    </div>
  )
}
```

### Task 6 — LocalStorage Sync

```tsx
// src/components/ThemeToggle.tsx
import { useState, useEffect } from 'react'

type Theme = 'light' | 'dark'

export function ThemeToggle() {
  // Lazy initializer: read from localStorage once on mount
  const [theme, setTheme] = useState<Theme>(() => {
    const saved = localStorage.getItem('theme')
    return (saved === 'dark' ? 'dark' : 'light') as Theme
  })

  useEffect(() => {
    // Sync to localStorage whenever theme changes
    localStorage.setItem('theme', theme)

    // Also apply to document for CSS variables
    document.documentElement.setAttribute('data-theme', theme)
  }, [theme])

  return (
    <button
      onClick={() => setTheme(t => (t === 'light' ? 'dark' : 'light'))}
      style={{
        padding: '8px 16px',
        borderRadius: 6,
        border: '1px solid #d1d5db',
        cursor: 'pointer',
        background: theme === 'dark' ? '#1e293b' : 'white',
        color: theme === 'dark' ? 'white' : '#1e293b',
      }}
    >
      {theme === 'light' ? '🌙 Dark' : '☀️ Light'} mode
    </button>
  )
}
```
