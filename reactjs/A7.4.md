## Goal
Render lists of tens of thousands of items smoothly by rendering only what's visible in the viewport using virtualization.

## Background

### The DOM Node Problem
Rendering 10,000 `<div>` elements creates 10,000 DOM nodes. Each DOM node costs memory and layout time. Scrolling through 10,000 items requires the browser to keep all of them in layout, causing severe performance degradation on low-end devices. Virtualization (also called "windowing") is the solution: only render the items currently visible in the scrollable area, plus a small buffer above and below. As the user scrolls, items outside the viewport are removed and new items are added.

### TanStack Virtual
`@tanstack/react-virtual` (react-virtual v3) is a headless virtualization library — it gives you the measurements and offsets needed to position items, but you manage the DOM yourself. This gives maximum flexibility. `react-window` is an older alternative that provides pre-built list components. Both work; TanStack Virtual is more flexible and actively maintained.

### Measuring the Gain
Before implementing virtualization, open Chrome DevTools Performance tab, record scrolling through a long list, and note the FPS. After virtualization, record again. The difference demonstrates the value of windowing. Measure DOM node count before and after with `document.querySelectorAll('li').length`.

### Task 2 — Virtualized List

```tsx
// src/components/VirtualList.tsx
import { useVirtualizer } from '@tanstack/react-virtual'
import { useRef, useMemo, useState } from 'react'

interface Row {
  id: number
  name: string
  email: string
  status: 'active' | 'inactive' | 'pending'
  score: number
}

// Generate 100,000 rows
const allRows: Row[] = Array.from({ length: 100_000 }, (_, i) => ({
  id: i + 1,
  name: `User ${i + 1}`,
  email: `user${i + 1}@example.com`,
  status: ['active', 'inactive', 'pending'][i % 3] as Row['status'],
  score: Math.floor(Math.random() * 1000),
}))

const statusColors = { active: '#dcfce7', inactive: '#fee2e2', pending: '#fef9c3' }

export function VirtualList() {
  const [search, setSearch] = useState('')
  const [statusFilter, setStatusFilter] = useState<Row['status'] | 'all'>('all')

  const filtered = useMemo(() => {
    const lowerSearch = search.toLowerCase()
    return allRows.filter(row => {
      if (statusFilter !== 'all' && row.status !== statusFilter) return false
      if (search && !row.name.toLowerCase().includes(lowerSearch) && !row.email.includes(lowerSearch)) return false
      return true
    })
  }, [search, statusFilter])

  const parentRef = useRef<HTMLDivElement>(null)

  const rowVirtualizer = useVirtualizer({
    count: filtered.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 52,  // estimated row height in px
    overscan: 10,            // render 10 extra rows above/below viewport
  })

  return (
    <div>
      <div style={{ display: 'flex', gap: 8, marginBottom: 12, flexWrap: 'wrap' }}>
        <input
          type="search"
          value={search}
          onChange={e => setSearch(e.target.value)}
          placeholder="Search users..."
          style={{ flex: 1, padding: '6px 10px' }}
        />
        <select value={statusFilter} onChange={e => setStatusFilter(e.target.value as typeof statusFilter)}>
          <option value="all">All statuses</option>
          <option value="active">Active</option>
          <option value="inactive">Inactive</option>
          <option value="pending">Pending</option>
        </select>
      </div>

      <p style={{ fontSize: 13, color: '#6b7280', marginBottom: 8 }}>
        Showing {filtered.length.toLocaleString()} of {allRows.length.toLocaleString()} rows
        · DOM nodes: ~{Math.min(rowVirtualizer.getVirtualItems().length, filtered.length)}
      </p>

      {/* Scrollable container */}
      <div
        ref={parentRef}
        style={{ height: 500, overflow: 'auto', border: '1px solid #e5e7eb', borderRadius: 8 }}
      >
        {/* Total height spacer — makes scrollbar correct */}
        <div style={{ height: rowVirtualizer.getTotalSize(), position: 'relative' }}>
          {rowVirtualizer.getVirtualItems().map(virtualItem => {
            const row = filtered[virtualItem.index]
            return (
              <div
                key={row.id}
                style={{
                  position: 'absolute',
                  top: 0,
                  left: 0,
                  width: '100%',
                  transform: `translateY(${virtualItem.start}px)`,
                  height: virtualItem.size,
                  display: 'grid',
                  gridTemplateColumns: '60px 1fr 1fr 80px 80px',
                  alignItems: 'center',
                  padding: '0 12px',
                  borderBottom: '1px solid #f0f0f0',
                  background: virtualItem.index % 2 === 0 ? 'white' : '#fafafa',
                  fontSize: 14,
                }}
              >
                <span style={{ color: '#9ca3af' }}>#{row.id}</span>
                <span style={{ fontWeight: 500 }}>{row.name}</span>
                <span style={{ color: '#6b7280', overflow: 'hidden', textOverflow: 'ellipsis', whiteSpace: 'nowrap' }}>{row.email}</span>
                <span style={{ padding: '2px 8px', borderRadius: 12, background: statusColors[row.status], fontSize: 12, textAlign: 'center' }}>
                  {row.status}
                </span>
                <span style={{ textAlign: 'right', fontVariantNumeric: 'tabular-nums' }}>{row.score}</span>
              </div>
            )
          })}
        </div>
      </div>
    </div>
  )
}
```

### Task 3 — Variable Height Rows

```tsx
// src/components/VariableHeightList.tsx
import { useVirtualizer } from '@tanstack/react-virtual'
import { useRef } from 'react'

interface Post {
  id: number
  title: string
  body: string  // variable length
}

export function VariableHeightList({ posts }: { posts: Post[] }) {
  const parentRef = useRef<HTMLDivElement>(null)

  const rowVirtualizer = useVirtualizer({
    count: posts.length,
    getScrollElement: () => parentRef.current,
    estimateSize: () => 100,  // initial estimate
    // measureElement is called after render to get actual height
  })

  return (
    <div ref={parentRef} style={{ height: 600, overflow: 'auto' }}>
      <div style={{ height: rowVirtualizer.getTotalSize(), position: 'relative' }}>
        {rowVirtualizer.getVirtualItems().map(virtualItem => {
          const post = posts[virtualItem.index]
          return (
            <div
              key={post.id}
              data-index={virtualItem.index}
              ref={rowVirtualizer.measureElement}  // measure actual height
              style={{
                position: 'absolute',
                top: 0,
                left: 0,
                width: '100%',
                transform: `translateY(${virtualItem.start}px)`,
                padding: '1rem',
                borderBottom: '1px solid #f0f0f0',
              }}
            >
              <h3 style={{ margin: '0 0 8px', fontSize: 16 }}>{post.title}</h3>
              <p style={{ margin: 0, fontSize: 14, color: '#6b7280', lineHeight: 1.6 }}>{post.body}</p>
            </div>
          )
        })}
      </div>
    </div>
  )
}
```
