## Goal
Define form validation schemas with Zod and connect them to React Hook Form, getting TypeScript types for free from the schema.

## Background

### What is Zod?
Zod is a TypeScript-first schema validation library. You define a schema that describes the shape and constraints of your data. Zod then validates data against that schema at runtime and infers TypeScript types from it at compile time. This means you write the schema once and get both validation logic and TypeScript types — no duplication.

### Schema-First Validation
Before Zod, you might write a TypeScript interface for your form, then separately write validation logic. Zod collapses these: you define a schema, call `z.infer<typeof schema>` to get the TypeScript type, and pass the schema to `zodResolver` to get React Hook Form validation. Change the schema and both validation and types update.

### Key Zod APIs
- `z.string()`, `z.number()`, `z.boolean()`, `z.date()` — primitive types
- `.min()`, `.max()`, `.email()`, `.url()`, `.regex()` — validators with custom messages
- `.optional()`, `.nullable()` — allow undefined/null
- `.transform()` — transform valid input to a different type
- `.refine()` — custom validation with full access to the value
- `.superRefine()` — custom validation with access to the entire object (for cross-field validation)
- `z.infer<typeof schema>` — extract TypeScript type from schema

### Task 2 — Registration Schema

```tsx
// src/schemas/registration.ts
import { z } from 'zod'

export const registrationSchema = z
  .object({
    username: z
      .string()
      .min(3, 'At least 3 characters')
      .max(30, 'At most 30 characters')
      .regex(/^[a-z0-9_]+$/, 'Only lowercase letters, numbers, and underscores'),

    email: z.string().email('Enter a valid email address'),

    password: z
      .string()
      .min(8, 'At least 8 characters')
      .regex(/[A-Z]/, 'Must contain an uppercase letter')
      .regex(/[0-9]/, 'Must contain a number'),

    confirmPassword: z.string(),

    birthYear: z
      .string()
      .transform(val => Number(val))  // input is a string, transform to number
      .pipe(z.number().min(1900).max(new Date().getFullYear())),

    agreeToTerms: z.literal(true, {
      errorMap: () => ({ message: 'You must agree to the terms' }),
    }),
  })
  .refine(data => data.password === data.confirmPassword, {
    message: 'Passwords do not match',
    path: ['confirmPassword'],  // attach error to this field
  })

// TypeScript type inferred from schema — no separate interface needed
export type RegistrationValues = z.infer<typeof registrationSchema>
```

```tsx
// src/components/RegistrationForm.tsx
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { registrationSchema, type RegistrationValues } from '../schemas/registration'

function FormField({ label, error, children }: { label: string; error?: string; children: React.ReactNode }) {
  return (
    <div>
      <label style={{ display: 'block', fontWeight: 500, marginBottom: 4 }}>{label}</label>
      {children}
      {error && <p style={{ color: '#ef4444', fontSize: 12, margin: '4px 0 0' }}>{error}</p>}
    </div>
  )
}

export function RegistrationForm() {
  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting, isValid },
  } = useForm<RegistrationValues>({
    resolver: zodResolver(registrationSchema),  // Zod validates on every change
    mode: 'onChange',  // validate on each change for real-time feedback
  })

  async function onSubmit(data: RegistrationValues) {
    await new Promise(r => setTimeout(r, 1000))
    console.log('Registered:', data)
    // data.birthYear is a number here, not a string — Zod's transform worked
  }

  const inputStyle = (hasError: boolean) => ({
    display: 'block',
    width: '100%',
    padding: '8px 10px',
    border: `1px solid ${hasError ? '#ef4444' : '#d1d5db'}`,
    borderRadius: 4,
  })

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ display: 'flex', flexDirection: 'column', gap: '1rem', maxWidth: 400 }}>
      <FormField label="Username" error={errors.username?.message}>
        <input type="text" {...register('username')} style={inputStyle(!!errors.username)} />
      </FormField>

      <FormField label="Email" error={errors.email?.message}>
        <input type="email" {...register('email')} style={inputStyle(!!errors.email)} />
      </FormField>

      <FormField label="Password" error={errors.password?.message}>
        <input type="password" {...register('password')} style={inputStyle(!!errors.password)} />
      </FormField>

      <FormField label="Confirm Password" error={errors.confirmPassword?.message}>
        <input type="password" {...register('confirmPassword')} style={inputStyle(!!errors.confirmPassword)} />
      </FormField>

      <FormField label="Birth Year" error={errors.birthYear?.message}>
        <input type="number" {...register('birthYear')} style={inputStyle(!!errors.birthYear)} placeholder="e.g. 1990" />
      </FormField>

      <label style={{ display: 'flex', gap: 8, alignItems: 'flex-start' }}>
        <input type="checkbox" {...register('agreeToTerms')} style={{ marginTop: 3 }} />
        <span>I agree to the <a href="/terms">Terms of Service</a></span>
      </label>
      {errors.agreeToTerms && <p style={{ color: '#ef4444', fontSize: 12 }}>{errors.agreeToTerms.message}</p>}

      <button type="submit" disabled={!isValid || isSubmitting} style={{ padding: '10px', borderRadius: 4, background: isValid ? '#3b82f6' : '#9ca3af', color: 'white', border: 'none', cursor: isValid ? 'pointer' : 'not-allowed' }}>
        {isSubmitting ? 'Creating account...' : 'Create Account'}
      </button>
    </form>
  )
}
```

### Task 3 — API Response Validation with Zod

```tsx
// src/schemas/api.ts — validate API responses at runtime
import { z } from 'zod'

export const postSchema = z.object({
  id: z.number(),
  userId: z.number(),
  title: z.string(),
  body: z.string(),
})

export const postListSchema = z.array(postSchema)

export type Post = z.infer<typeof postSchema>

// In your API function:
export async function fetchPost(id: number): Promise<Post> {
  const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`)
  const raw = await res.json()
  // This throws if the shape doesn't match — catches API changes early
  return postSchema.parse(raw)
}

// Safe parse (returns { success, data } | { success: false, error })
export async function safelyFetchPosts(): Promise<Post[]> {
  const res = await fetch('https://jsonplaceholder.typicode.com/posts?_limit=5')
  const raw = await res.json()
  const result = postListSchema.safeParse(raw)
  if (!result.success) {
    console.error('API schema mismatch:', result.error.format())
    return []
  }
  return result.data
}
```
