# P9.3 — Production SPA

**Phase:** 9 — TypeScript, Accessibility & Production
**Topic:** TypeScript + testing + i18n + a11y + Docker + CI/CD + performance budget
**Type:** Project

---

## Goal

Build and ship a complete, production-grade React SPA that combines everything from the curriculum: TypeScript strict mode, comprehensive testing, internationalization, accessibility, Docker deployment, and a full CI/CD pipeline with a performance budget gate.

---

## What to Build

A **Bookmark Manager** SPA — a real-world app complex enough to exercise all requirements:

- Save, organize, tag, and search bookmarks
- TypeScript strict mode, Zod API validation
- Unit + integration + E2E test suite (80%+ coverage)
- English + French i18n
- WCAG 2.1 AA accessibility
- Docker + nginx deployment
- GitHub Actions CI/CD
- Lighthouse CI performance budget (LCP < 2.5 s, TBT < 200 ms, CLS < 0.1)

### 2. Bookmark schema and API

```ts
// src/schemas/bookmark.ts
import { z } from 'zod';

export const BookmarkSchema = z.object({
  id: z.string().uuid(),
  url: z.string().url(),
  title: z.string().min(1).max(200),
  description: z.string().optional(),
  tags: z.array(z.string()),
  favicon: z.string().url().optional(),
  createdAt: z.string().datetime(),
  updatedAt: z.string().datetime(),
  folderId: z.string().uuid().nullable(),
  isRead: z.boolean(),
});

export const CreateBookmarkSchema = BookmarkSchema.omit({
  id: true, createdAt: true, updatedAt: true,
});

export type Bookmark = z.infer<typeof BookmarkSchema>;
export type CreateBookmark = z.infer<typeof CreateBookmarkSchema>;
```

### 4. Accessibility — key patterns

All from P9.2 apply. Additional requirements:

```tsx
// Page title updates on route change
import { useTranslation } from 'react-i18next';
import { useEffect } from 'react';

function usePageTitle(key: string) {
  const { t } = useTranslation();
  useEffect(() => {
    document.title = `${t(key)} — ${t('app.title')}`;
  }, [key, t]);
}

// Loading states announced to screen readers
function BookmarkList() {
  const { t } = useTranslation();
  const { data, isLoading } = useBookmarks();

  return (
    <>
      <div role="status" aria-live="polite" className="sr-only">
        {isLoading ? t('messages.loading') : t('bookmark.count', { count: data?.length ?? 0 })}
      </div>
      {/* ... */}
    </>
  );
}
```

### 6. Dockerfile

```dockerfile
FROM node:20-alpine AS builder
WORKDIR /app
RUN corepack enable && corepack prepare pnpm@latest --activate
COPY pnpm-lock.yaml package.json ./
RUN pnpm install --frozen-lockfile
COPY . .
RUN pnpm build

FROM nginx:1.27-alpine AS runner
COPY --from=builder /app/dist /usr/share/nginx/html
COPY nginx.conf /etc/nginx/conf.d/default.conf
EXPOSE 80
CMD ["nginx", "-g", "daemon off;"]
```

### 7. GitHub Actions — full pipeline

```yaml
# .github/workflows/ci.yml
name: CI / CD

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  quality:
    name: TypeScript + Lint + Tests
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v3
        with: { version: 9 }
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: pnpm }
      - run: pnpm install --frozen-lockfile
      - run: pnpm exec tsc --noEmit
      - run: pnpm lint
      - run: pnpm test:coverage
      - name: Assert 80% coverage
        run: |
          COVERAGE=$(cat coverage/coverage-summary.json | jq '.total.lines.pct')
          if (( $(echo "$COVERAGE < 80" | bc -l) )); then
            echo "Coverage $COVERAGE% is below 80%"
            exit 1
          fi

  build:
    name: Production Build
    needs: quality
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v3
        with: { version: 9 }
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: pnpm }
      - run: pnpm install --frozen-lockfile
      - run: pnpm build
        env:
          VITE_API_URL: ${{ secrets.VITE_API_URL }}
      - uses: actions/upload-artifact@v4
        with: { name: dist, path: dist/ }

  e2e:
    name: E2E Tests
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: pnpm/action-setup@v3
        with: { version: 9 }
      - uses: actions/setup-node@v4
        with: { node-version: 20, cache: pnpm }
      - run: pnpm install --frozen-lockfile
      - run: pnpm exec playwright install --with-deps chromium
      - uses: actions/download-artifact@v4
        with: { name: dist, path: dist/ }
      - run: pnpm test:e2e
        env: { CI: true }
      - uses: actions/upload-artifact@v4
        if: always()
        with: { name: playwright-report, path: playwright-report/ }

  lighthouse:
    name: Lighthouse Performance Budget
    needs: build
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/download-artifact@v4
        with: { name: dist, path: dist/ }
      - name: Serve and audit
        uses: treosh/lighthouse-ci-action@v11
        with:
          uploadArtifacts: true
          temporaryPublicStorage: true
          configPath: '.lighthouserc.json'

  docker:
    name: Push Docker Image
    needs: [e2e, lighthouse]
    if: github.ref == 'refs/heads/main'
    runs-on: ubuntu-latest
    permissions: { contents: read, packages: write }
    steps:
      - uses: actions/checkout@v4
      - uses: docker/login-action@v3
        with:
          registry: ghcr.io
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}
      - uses: docker/metadata-action@v5
        id: meta
        with:
          images: ghcr.io/${{ github.repository }}
          tags: |
            type=sha,prefix=sha-
            type=raw,value=latest
      - uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          cache-from: type=gha
          cache-to: type=gha,mode=max
```

### 9. E2E test covering all pillars

```ts
// e2e/bookmarks.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Bookmark Manager', () => {
  test('add and view a bookmark (English)', async ({ page }) => {
    await page.goto('/');
    await expect(page.getByRole('heading', { name: 'Bookmarks' })).toBeVisible();

    await page.getByRole('button', { name: /add bookmark/i }).click();
    const dialog = page.getByRole('dialog');
    await dialog.getByLabel(/url/i).fill('https://react.dev');
    await dialog.getByLabel(/title/i).fill('React Docs');
    await dialog.getByRole('button', { name: /save/i }).click();

    await expect(page.getByText('Bookmark saved.')).toBeVisible();
    await expect(page.getByRole('link', { name: 'React Docs' })).toBeVisible();
  });

  test('language switches to French', async ({ page }) => {
    await page.goto('/');
    await page.getByRole('combobox', { name: /language/i }).selectOption('fr');
    await expect(page.getByRole('heading', { name: 'Signets' })).toBeVisible();
    await expect(page.getByRole('button', { name: /ajouter un signet/i })).toBeVisible();
  });

  test('keyboard-only workflow works', async ({ page }) => {
    await page.goto('/');
    await page.keyboard.press('Tab'); // Skip link
    await page.keyboard.press('Tab'); // First nav item
    // Tab through to Add button
    await page.keyboard.press('Enter');
    await expect(page.getByRole('dialog')).toBeVisible();
    await page.keyboard.press('Escape');
    await expect(page.getByRole('dialog')).not.toBeVisible();
  });
});
```
