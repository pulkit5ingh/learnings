# A7.2 — useMemo & useCallback in Practice

**Phase:** 7 — Performance
**Topic:** Memoize expensive computations, stable callbacks, when NOT to memoize
**Type:** Assignment

---

## Goal
Apply useMemo and useCallback correctly — knowing when they help, when they hurt, and how to measure the difference.

## Background

### The Cost of Memoization
`useMemo` and `useCallback` are not free. Every call costs: storing the previous result, comparing dependencies on every render, and the cognitive overhead of understanding memoized code. The rule of thumb: memoize only when you have a measured performance problem or a specific referential equality requirement. Premature memoization adds complexity without benefit.

### When useMemo Is Worth It
1. Computationally expensive operations: sorting/filtering thousands of items, complex math, deep object transformations. If the function takes less than 1ms, memoization overhead is not worth it.
2. Stable reference requirements: when passing a derived object to a `React.memo` child or as a `useEffect` dependency.

### When useCallback Is Worth It
1. Passing callbacks to `React.memo` children — without `useCallback`, a new function reference every render defeats memo.
2. Callbacks used as `useEffect` dependencies — without `useCallback`, the effect runs on every render.
3. Callbacks passed to event listeners in custom hooks.

### Task 1 — Measuring Before Memoizing

```tsx
// src/components/ExpensiveSort.tsx
import { useState, useMemo } from 'react'

// Simulate expensive computation
function slowSort(items: number[], order: 'asc' | 'desc'): number[] {
  // Artificially slow for demo — in real code, this would be a complex sort
  const start = performance.now()
  const sorted = [...items].sort((a, b) => order === 'asc' ? a - b : b - a)
  while (performance.now() - start < 50) { /* spin for 50ms */ }
  return sorted
}

const items = Array.from({ length: 1000 }, () => Math.floor(Math.random() * 10000))

export function ExpensiveSort() {
  const [order, setOrder] = useState<'asc' | 'desc'>('asc')
  const [unrelated, setUnrelated] = useState(0)

  // Without useMemo: runs slowSort on EVERY render, including unrelated state changes
  // const sorted = slowSort(items, order)

  // With useMemo: runs slowSort only when `order` or `items` changes
  const sorted = useMemo(() => {
    console.time('sort')
    const result = slowSort(items, order)
    console.timeEnd('sort')
    return result
  }, [order])  // items is stable (module-level const), so omitting it is safe

  return (
    <div>
      <div style={{ display: 'flex', gap: 8, marginBottom: 16 }}>
        <button onClick={() => setOrder('asc')} style={{ background: order === 'asc' ? '#3b82f6' : '#f1f5f9', color: order === 'asc' ? 'white' : 'inherit', border: 'none', borderRadius: 4, padding: '6px 12px', cursor: 'pointer' }}>Ascending</button>
        <button onClick={() => setOrder('desc')} style={{ background: order === 'desc' ? '#3b82f6' : '#f1f5f9', color: order === 'desc' ? 'white' : 'inherit', border: 'none', borderRadius: 4, padding: '6px 12px', cursor: 'pointer' }}>Descending</button>
        <button onClick={() => setUnrelated(n => n + 1)}>Unrelated state change ({unrelated})</button>
      </div>
      <p style={{ fontSize: 13, color: '#6b7280' }}>
        Sorted {items.length} items. Check console for sort timing.
        Unrelated changes should NOT trigger re-sort.
      </p>
      <div style={{ display: 'flex', gap: 4, flexWrap: 'wrap', maxHeight: 100, overflow: 'hidden' }}>
        {sorted.slice(0, 20).map((n, i) => (
          <span key={i} style={{ padding: '2px 6px', background: '#f1f5f9', borderRadius: 4, fontSize: 13 }}>{n}</span>
        ))}
        <span style={{ fontSize: 13, color: '#9ca3af' }}>...{items.length - 20} more</span>
      </div>
    </div>
  )
}
```

### Task 2 — Stable Callbacks for Child Components

```tsx
// src/components/SearchWithCallback.tsx
import { useState, useCallback, memo } from 'react'

// This child component is expensive to render
const SearchResults = memo(function SearchResults({
  query,
  onResultClick,
}: {
  query: string
  onResultClick: (id: number) => void
}) {
  console.log('SearchResults rendered with query:', query)

  const mockResults = Array.from({ length: 5 }, (_, i) => ({
    id: i + 1,
    title: `Result ${i + 1} for "${query}"`,
  }))

  return (
    <ul style={{ listStyle: 'none', padding: 0 }}>
      {mockResults.map(r => (
        <li
          key={r.id}
          onClick={() => onResultClick(r.id)}
          style={{ padding: '0.5rem', cursor: 'pointer', borderBottom: '1px solid #f0f0f0' }}
        >
          {r.title}
        </li>
      ))}
    </ul>
  )
})

export function SearchWithCallback() {
  const [query, setQuery] = useState('')
  const [selectedId, setSelectedId] = useState<number | null>(null)

  // Stable callback — doesn't change between renders
  // SearchResults won't re-render when parent re-renders for other reasons
  const handleResultClick = useCallback((id: number) => {
    setSelectedId(id)
    console.log('Selected:', id)
  }, [])  // no dependencies — this function never needs to change

  return (
    <div>
      <input
        value={query}
        onChange={e => setQuery(e.target.value)}
        placeholder="Search..."
        style={{ width: '100%', padding: '8px 10px', marginBottom: 8 }}
      />
      {selectedId && <p>Selected ID: {selectedId}</p>}
      {query && <SearchResults query={query} onResultClick={handleResultClick} />}
    </div>
  )
}
```

### Task 3 — When NOT to Memoize

```tsx
// src/components/OverMemoized.tsx

// BEFORE: over-memoized — adds complexity with no benefit
function BadCounter() {
  const [count, setCount] = useState(0)

  // Pointless: adding 1 to count is not "expensive"
  const doubled = useMemo(() => count * 2, [count])

  // Pointless: this inline handler is used once, on a non-memo'd component
  const handleClick = useCallback(() => setCount(c => c + 1), [])

  return <div onClick={handleClick}>{count} × 2 = {doubled}</div>
}

// AFTER: simpler and equivalent
function GoodCounter() {
  const [count, setCount] = useState(0)
  const doubled = count * 2  // just compute it — it's instant
  return <div onClick={() => setCount(c => c + 1)}>{count} × 2 = {doubled}</div>
}

// Rules of thumb:
// - Never memoize primitives (strings, numbers, booleans)
// - Never memoize computations that take < 1ms
// - Never memoize callbacks unless the child is React.memo'd OR the callback is a useEffect dep
// - Profile first, optimize second
```
