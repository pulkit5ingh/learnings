**Type:** Assignment

## Goal
Load data at the route level, handle missing resources with 404s, and split your bundle with lazy-loaded routes.

## Background

### Route-Level Data Loading
React Router v6.4+ introduced loaders — functions you attach to routes that run before the route renders. This moves data fetching to the routing layer instead of inside components with `useEffect`. The route doesn't render until the data is ready. Use `useLoaderData()` inside the component to access what the loader returned. This pattern prevents the flash of empty content you get with useEffect-based fetching.

### Lazy Loading Routes
As your app grows, bundling every page into one JavaScript file makes the initial load slow. `React.lazy()` dynamically imports a component only when it's needed. Combined with `<Suspense>`, it shows a fallback while the chunk loads. Route-based code splitting is the most effective form — you only load code for the page the user visits.

### 404 Handling Two Ways
A catch-all `<Route path="*">` handles unknown routes at the router level. For routes that exist but whose data doesn't (e.g., `/users/99999`), you need to handle the 404 inside the component or loader — check if the fetched data is null/undefined and render a "not found" state.

## Tasks

### Task 1 — Route Loader Pattern

```tsx
// src/loaders/userLoader.ts
import type { LoaderFunctionArgs } from 'react-router-dom'

export interface UserLoaderData {
  user: {
    id: number
    name: string
    email: string
    company: { name: string }
  }
}

export async function userLoader({ params }: LoaderFunctionArgs): Promise<UserLoaderData> {
  const res = await fetch(`https://jsonplaceholder.typicode.com/users/${params.userId}`)

  if (res.status === 404) {
    throw new Response('User not found', { status: 404 })
  }

  if (!res.ok) {
    throw new Error(`Failed to load user: HTTP ${res.status}`)
  }

  const user = await res.json()
  return { user }
}
```

```tsx
// src/pages/UserDetailPage.tsx — using useLoaderData instead of useEffect fetch
import { useLoaderData, Link } from 'react-router-dom'
import type { UserLoaderData } from '../loaders/userLoader'

export function UserDetailPage() {
  const { user } = useLoaderData() as UserLoaderData

  return (
    <div>
      <Link to="/users">← All Users</Link>
      <h1>{user.name}</h1>
      <p>{user.email}</p>
      <p>{user.company.name}</p>
    </div>
  )
}
```

```tsx
// src/router.tsx — using createBrowserRouter with loaders
import { createBrowserRouter, RouterProvider } from 'react-router-dom'
import { userLoader } from './loaders/userLoader'
import { RootLayout } from './layouts/RootLayout'
import { HomePage } from './pages/HomePage'
import { UsersPage } from './pages/UsersPage'
import { UserDetailPage } from './pages/UserDetailPage'
import { NotFoundPage } from './pages/NotFoundPage'
import { ErrorPage } from './pages/ErrorPage'

const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    errorElement: <ErrorPage />,
    children: [
      { index: true, element: <HomePage /> },
      { path: 'users', element: <UsersPage /> },
      {
        path: 'users/:userId',
        element: <UserDetailPage />,
        loader: userLoader,  // runs before UserDetailPage renders
      },
      { path: '*', element: <NotFoundPage /> },
    ],
  },
])

export function App() {
  return <RouterProvider router={router} />
}
```

### Task 2 — Error Boundary for Routes

```tsx
// src/pages/ErrorPage.tsx
import { useRouteError, Link, isRouteErrorResponse } from 'react-router-dom'

export function ErrorPage() {
  const error = useRouteError()

  // React Router's own error responses (thrown with `new Response(...)`)
  if (isRouteErrorResponse(error)) {
    return (
      <div style={{ textAlign: 'center', padding: '4rem' }}>
        <h1 style={{ fontSize: 64 }}>{error.status}</h1>
        <p>{error.status === 404 ? 'Page not found' : error.statusText}</p>
        <Link to="/">← Go home</Link>
      </div>
    )
  }

  // Regular JavaScript errors
  const message = error instanceof Error ? error.message : 'Something went wrong'
  return (
    <div style={{ textAlign: 'center', padding: '4rem' }}>
      <h1>Oops!</h1>
      <p style={{ color: '#ef4444' }}>{message}</p>
      <Link to="/">← Go home</Link>
    </div>
  )
}
```

### Task 3 — Lazy-Loaded Routes

```tsx
// src/router.tsx — lazy loading large pages
import { lazy, Suspense } from 'react'
import { createBrowserRouter, RouterProvider } from 'react-router-dom'

// Vite will code-split each of these into a separate chunk
const DashboardPage = lazy(() => import('./pages/DashboardPage'))
const SettingsPage = lazy(() => import('./pages/SettingsPage'))
const ReportsPage = lazy(() => import('./pages/ReportsPage'))

function PageLoader() {
  return (
    <div style={{ display: 'flex', justifyContent: 'center', padding: '4rem' }}>
      <div style={{ width: 32, height: 32, border: '3px solid #e5e7eb', borderTopColor: '#3b82f6', borderRadius: '50%', animation: 'spin 0.8s linear infinite' }} />
    </div>
  )
}

// Wrap lazy component with Suspense
function LazyRoute({ Component }: { Component: React.ComponentType }) {
  return (
    <Suspense fallback={<PageLoader />}>
      <Component />
    </Suspense>
  )
}

const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    errorElement: <ErrorPage />,
    children: [
      { index: true, element: <HomePage /> },  // not lazy — loads immediately
      {
        path: 'dashboard',
        element: <LazyRoute Component={DashboardPage} />,
      },
      {
        path: 'settings',
        element: <LazyRoute Component={SettingsPage} />,
      },
      {
        path: 'reports',
        element: <LazyRoute Component={ReportsPage} />,
      },
    ],
  },
])
```

### Task 4 — Auth Guard (Loader Version)

```tsx
// src/loaders/requireAuth.ts
import type { LoaderFunctionArgs } from 'react-router-dom'
import { redirect } from 'react-router-dom'

// Simple token check — replace with your actual auth logic
function isAuthenticated() {
  return !!localStorage.getItem('auth-token')
}

export function requireAuth({ request }: LoaderFunctionArgs) {
  if (!isAuthenticated()) {
    const url = new URL(request.url)
    return redirect(`/login?from=${encodeURIComponent(url.pathname)}`)
  }
  return null  // allow access
}
```
