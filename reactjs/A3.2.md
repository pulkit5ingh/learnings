## Goal
Read dynamic URL segments and query string parameters to drive component behavior from the URL.

## Background

### URL as State
The URL is a powerful piece of application state. When filters, search queries, pagination, and selected items live in the URL, users can bookmark pages, share links, and use the browser back button to undo navigation. This is fundamentally better than storing the same state in `useState` — that state is lost on page refresh.

### Route Params vs Search Params
Route params (`:id` in a path like `/users/:id`) identify a specific resource. They are required — a route without the param won't match. Use them for IDs and slugs. Search params (`?sort=asc&page=2`) are optional key-value pairs in the query string. Use them for filters, sorting, pagination, and anything that can be absent or have a default. `useParams()` reads route params. `useSearchParams()` reads and writes search params, returning a `URLSearchParams` object and a setter.

### Task 1 — Route Params with useParams

```tsx
// src/pages/UserDetailPage.tsx
import { useParams, Link } from 'react-router-dom'
import { useFetch } from '../hooks/useFetch'

interface User {
  id: number
  name: string
  email: string
  phone: string
  website: string
  company: { name: string; catchPhrase: string }
  address: { city: string; street: string }
}

export function UserDetailPage() {
  const { userId } = useParams<{ userId: string }>()  // typed params

  const { data: user, loading, error } = useFetch<User>(
    userId ? `https://jsonplaceholder.typicode.com/users/${userId}` : null
  )

  if (loading) return <p>Loading user {userId}...</p>
  if (error) return (
    <div>
      <p style={{ color: '#ef4444' }}>Failed to load user: {error}</p>
      <Link to="/users">← Back to users</Link>
    </div>
  )
  if (!user) return null

  return (
    <div style={{ maxWidth: 500 }}>
      <Link to="/users" style={{ color: '#3b82f6', fontSize: 14 }}>← All users</Link>
      <h1 style={{ marginTop: '0.5rem' }}>{user.name}</h1>
      <p style={{ color: '#6b7280' }}>{user.company.name}</p>
      <dl style={{ display: 'grid', gridTemplateColumns: '120px 1fr', gap: '4px 8px', fontSize: 14 }}>
        <dt style={{ fontWeight: 600 }}>Email</dt><dd>{user.email}</dd>
        <dt style={{ fontWeight: 600 }}>Phone</dt><dd>{user.phone}</dd>
        <dt style={{ fontWeight: 600 }}>City</dt><dd>{user.address.city}</dd>
        <dt style={{ fontWeight: 600 }}>Catchphrase</dt><dd style={{ fontStyle: 'italic' }}>"{user.company.catchPhrase}"</dd>
      </dl>
    </div>
  )
}
```

### Task 2 — Users List with Link to Detail

```tsx
// src/pages/UsersPage.tsx
import { Link } from 'react-router-dom'
import { useFetch } from '../hooks/useFetch'

interface User { id: number; name: string; email: string; company: { name: string } }

export function UsersPage() {
  const { data: users, loading } = useFetch<User[]>('https://jsonplaceholder.typicode.com/users')

  if (loading) return <p>Loading...</p>

  return (
    <div>
      <h1>Users</h1>
      <ul style={{ listStyle: 'none', padding: 0 }}>

          <li key={user.id} style={{ borderBottom: '1px solid #f0f0f0' }}>
            {/* Link to dynamic route: /users/1, /users/2, etc. */}

              <span style={{ fontSize: 13, color: '#6b7280' }}>{user.email} · {user.company.name}</span>
            </Link>
          </li>
        ))}
      </ul>
    </div>
  )
}
```

### Task 3 — useSearchParams for Filters

```tsx
// src/pages/PostsPage.tsx
import { useSearchParams, Link } from 'react-router-dom'
import { useMemo } from 'react'
import { useFetch } from '../hooks/useFetch'

interface Post { id: number; userId: number; title: string; body: string }

const POSTS_PER_PAGE = 5

export function PostsPage() {
  const [searchParams, setSearchParams] = useSearchParams()

  // Read all filter state from the URL
  const search = searchParams.get('search') ?? ''
  const userId = searchParams.get('userId') ?? ''
  const sortBy = (searchParams.get('sort') ?? 'id') as 'id' | 'title'
  const page = Number(searchParams.get('page') ?? '1')

  const { data: allPosts, loading } = useFetch<Post[]>('https://jsonplaceholder.typicode.com/posts')

  // All filtering happens in useMemo — pure derived state from URL params
  const { posts, totalPages } = useMemo(() => {
    if (!allPosts) return { posts: [], totalPages: 1 }

    let filtered = allPosts
    if (search) filtered = filtered.filter(p => p.title.includes(search.toLowerCase()))
    if (userId) filtered = filtered.filter(p => p.userId === Number(userId))
    if (sortBy === 'title') filtered = [...filtered].sort((a, b) => a.title.localeCompare(b.title))

    const totalPages = Math.ceil(filtered.length / POSTS_PER_PAGE)
    const posts = filtered.slice((page - 1) * POSTS_PER_PAGE, page * POSTS_PER_PAGE)
    return { posts, totalPages }
  }, [allPosts, search, userId, sortBy, page])

  // Update URL params — merges with existing params
  function setParam(key: string, value: string) {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev)
      if (value) next.set(key, value)
      else next.delete(key)
      if (key !== 'page') next.delete('page')  // reset page on filter change
      return next
    })
  }

  if (loading) return <p>Loading posts...</p>

  return (
    <div style={{ maxWidth: 700 }}>
      <h1>Posts</h1>

      {/* Filters — all changes are reflected in the URL */}
      <div style={{ display: 'flex', gap: 8, marginBottom: 16, flexWrap: 'wrap' }}>
        <input
          type="search"
          value={search}
          onChange={e => setParam('search', e.target.value)}
          placeholder="Search titles..."
          style={{ flex: 1, padding: '6px 10px', minWidth: 150 }}
        />
        <select value={userId} onChange={e => setParam('userId', e.target.value)}>
          <option value="">All users</option>
          {Array.from({ length: 10 }, (_, i) => i + 1).map(id => (
            <option key={id} value={id}>User {id}</option>
          ))}
        </select>
        <select value={sortBy} onChange={e => setParam('sort', e.target.value)}>
          <option value="id">Default</option>
          <option value="title">Title A-Z</option>
        </select>
      </div>

      {/* Current URL state for debugging */}
      <p style={{ fontSize: 12, color: '#9ca3af', fontFamily: 'monospace' }}>
        ?{searchParams.toString()}
      </p>

      {posts.map(post => (
        <div key={post.id} style={{ borderBottom: '1px solid #f0f0f0', padding: '0.75rem 0' }}>
          <Link to={`/posts/${post.id}`} style={{ fontWeight: 600, color: '#1d4ed8', textDecoration: 'none', display: 'block' }}>
            {post.title}
          </Link>
          <p style={{ margin: '4px 0 0', fontSize: 13, color: '#6b7280' }}>{post.body.slice(0, 100)}…</p>
        </div>
      ))}

      {/* Pagination — also in URL */}
      <div style={{ display: 'flex', gap: 4, marginTop: 16 }}>
        {Array.from({ length: totalPages }, (_, i) => i + 1).map(p => (
          <button
            key={p}
            onClick={() => setParam('page', String(p))}
            style={{
              padding: '4px 10px',
              background: p === page ? '#3b82f6' : '#f1f5f9',
              color: p === page ? 'white' : 'inherit',
              border: 'none',
              borderRadius: 4,
              cursor: 'pointer',
            }}
          >
            {p}
          </button>
        ))}
      </div>
    </div>
  )
}
```
