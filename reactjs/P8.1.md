### Hook tests with MSW

```ts
// src/__tests__/hooks/useMovieSearch.test.ts
import { describe, it, expect } from 'vitest';
import { renderHook, waitFor } from '@testing-library/react';
import { useMovieSearch } from '../../hooks/useMovieSearch';
import { server } from '../../mocks/server';
import { http, HttpResponse } from 'msw';

describe('useMovieSearch', () => {
  it('returns empty array when query is empty', () => {
    const { result } = renderHook(() => useMovieSearch(''));
    expect(result.current.movies).toEqual([]);
    expect(result.current.loading).toBe(false);
  });

  it('fetches movies when query is provided', async () => {
    const { result } = renderHook(() => useMovieSearch('shawshank'));
    expect(result.current.loading).toBe(true);

    await waitFor(() => expect(result.current.loading).toBe(false));
    expect(result.current.movies).toHaveLength(2);
    expect(result.current.movies[0].Title).toBe('The Shawshank Redemption');
  });

  it('sets error when movie not found', async () => {
    const { result } = renderHook(() => useMovieSearch('notfound'));
    await waitFor(() => expect(result.current.loading).toBe(false));
    expect(result.current.error).toBe('Movie not found!');
    expect(result.current.movies).toHaveLength(0);
  });

  it('handles network error', async () => {
    server.use(
      http.get('https://www.omdbapi.com', () => HttpResponse.error())
    );
    const { result } = renderHook(() => useMovieSearch('test'));
    await waitFor(() => expect(result.current.loading).toBe(false));
    expect(result.current.error).toBeTruthy();
  });
});
```

### Favorites hook test

```ts
// src/__tests__/hooks/useFavorites.test.ts
import { describe, it, expect, beforeEach } from 'vitest';
import { renderHook, act } from '@testing-library/react';
import { useFavorites } from '../../hooks/useFavorites';

const mockMovie = {
  imdbID: 'tt0111161',
  Title: 'The Shawshank Redemption',
  Year: '1994',
  Poster: 'N/A',
  Type: 'movie' as const,
};

describe('useFavorites', () => {
  beforeEach(() => {
    localStorage.clear();
  });

  it('starts with empty favorites', () => {
    const { result } = renderHook(() => useFavorites());
    expect(result.current.favorites).toHaveLength(0);
  });

  it('adds a movie to favorites', () => {
    const { result } = renderHook(() => useFavorites());
    act(() => result.current.toggle(mockMovie));
    expect(result.current.favorites).toHaveLength(1);
    expect(result.current.isFavorite('tt0111161')).toBe(true);
  });

  it('removes a movie from favorites on second toggle', () => {
    const { result } = renderHook(() => useFavorites());
    act(() => result.current.toggle(mockMovie));
    act(() => result.current.toggle(mockMovie));
    expect(result.current.favorites).toHaveLength(0);
  });

  it('persists to localStorage', () => {
    const { result } = renderHook(() => useFavorites());
    act(() => result.current.toggle(mockMovie));

    const saved = JSON.parse(localStorage.getItem('movie-favorites') ?? '[]');
    expect(saved).toHaveLength(1);
    expect(saved[0].imdbID).toBe('tt0111161');
  });
});
```

### Playwright E2E

```ts
// e2e/search.spec.ts
import { test, expect } from '@playwright/test';

test.describe('Movie search', () => {
  test('user can search and see results', async ({ page }) => {
    await page.goto('/');
    await page.getByRole('searchbox').fill('Shawshank');
    await expect(page.getByTestId('movie-grid')).toBeVisible();
    await expect(page.getByText('The Shawshank Redemption')).toBeVisible();
  });

  test('user can open movie detail modal', async ({ page }) => {
    await page.goto('/');
    await page.getByRole('searchbox').fill('Shawshank');
    await page.getByText('The Shawshank Redemption').click();
    await expect(page.getByRole('dialog')).toBeVisible();
    await expect(page.getByText('Frank Darabont')).toBeVisible();
  });

  test('user can add to favorites and see them on favorites tab', async ({ page }) => {
    await page.goto('/');
    await page.getByRole('searchbox').fill('Shawshank');
    await page.getByRole('button', { name: /add.*favorites/i }).first().click();

    await page.getByRole('button', { name: /favorites/i }).click();
    await expect(page.getByText('The Shawshank Redemption')).toBeVisible();
  });

  test('favorites persist after page reload', async ({ page }) => {
    await page.goto('/');
    await page.getByRole('searchbox').fill('Shawshank');
    await page.getByRole('button', { name: /add.*favorites/i }).first().click();

    await page.reload();
    await page.getByRole('button', { name: /favorites/i }).click();
    await expect(page.getByText('The Shawshank Redemption')).toBeVisible();
  });
});
```
