# P5.2 — CMS-style Post Editor

**Phase:** 5 — Forms & Validation
**Topic:** TipTap rich text, tag input, image upload preview, autosave, React Hook Form
**Type:** Project

---

## Goal

Build a blog post editor similar to a headless CMS. It features a TipTap rich text editor, tag input with keyboard handling, image upload with preview, and autosave that fires 2 seconds after the user stops typing.

---

## What to Build

- Title input (plain text, required)
- Rich text body editor (TipTap) with toolbar: bold, italic, underline, headings, lists, blockquote, code, links
- Slug field (auto-generated from title, editable)
- Tag input with keyboard support (Enter / comma adds a tag, Backspace removes last)
- Cover image upload with drag-and-drop and preview
- Autosave indicator ("Saving..." / "Saved 2s ago" / "Unsaved changes")
- Publish vs. Draft toggle
- Manual save / publish button
- Character count for the title

```ts
export function slugify(text: string): string {
  return text
    .toLowerCase()
    .trim()
    .replace(/[^\w\s-]/g, '')
    .replace(/[\s_-]+/g, '-')
    .replace(/^-+|-+$/g, '');
}
```

### Autosave hook

```ts
// src/hooks/useAutosave.ts
import { useEffect, useRef, useState } from 'react';

type SaveStatus = 'idle' | 'unsaved' | 'saving' | 'saved';

export function useAutosave<T>(
  data: T,
  saveFn: (data: T) => Promise<void>,
  delay = 2000
) {
  const [status, setStatus] = useState<SaveStatus>('idle');
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);
  const lastSavedRef = useRef<number | null>(null);
  const isFirstRender = useRef(true);

  useEffect(() => {
    if (isFirstRender.current) {
      isFirstRender.current = false;
      return;
    }

    setStatus('unsaved');

    if (timerRef.current) clearTimeout(timerRef.current);

    timerRef.current = setTimeout(async () => {
      setStatus('saving');
      try {
        await saveFn(data);
        lastSavedRef.current = Date.now();
        setStatus('saved');
      } catch {
        setStatus('unsaved');
      }
    }, delay);

    return () => {
      if (timerRef.current) clearTimeout(timerRef.current);
    };
  }, [data, delay]); // saveFn excluded intentionally to avoid re-triggering

  return { status, lastSaved: lastSavedRef.current };
}
```

```tsx
// src/components/RichTextEditor.tsx
import { useEditor, EditorContent } from '@tiptap/react';
import StarterKit from '@tiptap/starter-kit';
import Underline from '@tiptap/extension-underline';
import Link from '@tiptap/extension-link';
import Placeholder from '@tiptap/extension-placeholder';
import { EditorToolbar } from './EditorToolbar';
import { useEffect } from 'react';

interface Props {
  content: string;
  onChange: (html: string) => void;
}

export function RichTextEditor({ content, onChange }: Props) {
  const editor = useEditor({
    extensions: [
      StarterKit,
      Underline,
      Link.configure({ openOnClick: false }),
      Placeholder.configure({ placeholder: 'Write your post...' }),
    ],
    content,
    onUpdate: ({ editor }) => {
      onChange(editor.getHTML());
    },
  });

  // Sync external content changes (e.g., load from draft)
  useEffect(() => {
    if (editor && content !== editor.getHTML()) {
      editor.commands.setContent(content, false);
    }
  }, [content, editor]);

  return (
    <div className="border rounded-xl overflow-hidden">
      <EditorToolbar editor={editor} />
      <EditorContent
        editor={editor}
        className="prose max-w-none p-4 min-h-64 focus:outline-none"
      />
    </div>
  );
}
```

### TagInput

```tsx
// src/components/TagInput.tsx
import { useState, useRef } from 'react';

interface Props {
  tags: string[];
  onChange: (tags: string[]) => void;
}

export function TagInput({ tags, onChange }: Props) {
  const [input, setInput] = useState('');
  const inputRef = useRef<HTMLInputElement>(null);

  const addTag = (tag: string) => {
    const trimmed = tag.trim().toLowerCase();
    if (trimmed && !tags.includes(trimmed)) {
      onChange([...tags, trimmed]);
    }
    setInput('');
  };

  const removeTag = (tag: string) => {
    onChange(tags.filter(t => t !== tag));
  };

  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter' || e.key === ',') {
      e.preventDefault();
      addTag(input);
    } else if (e.key === 'Backspace' && !input && tags.length > 0) {
      removeTag(tags[tags.length - 1]);
    }
  };

  return (
    <div
      className="flex flex-wrap gap-2 border rounded-lg p-2 cursor-text min-h-10"
      onClick={() => inputRef.current?.focus()}
    >
      {tags.map(tag => (
        <span
          key={tag}
          className="flex items-center gap-1 bg-blue-100 text-blue-700 px-2 py-0.5 rounded-full text-sm"
        >
          {tag}
          <button
            type="button"
            onClick={() => removeTag(tag)}
            aria-label={`Remove tag ${tag}`}
            className="text-blue-400 hover:text-blue-700 font-bold"
          >
            ×
          </button>
        </span>
      ))}
      <input
        ref={inputRef}
        value={input}
        onChange={e => setInput(e.target.value)}
        onKeyDown={handleKeyDown}
        onBlur={() => { if (input.trim()) addTag(input); }}
        placeholder={tags.length === 0 ? 'Add tags...' : ''}
        className="flex-1 min-w-20 outline-none text-sm"
        aria-label="Add tag"
      />
    </div>
  );
}
```

### ImageUpload

```tsx
// src/components/ImageUpload.tsx
import { useCallback, useRef } from 'react';

interface Props {
  imageUrl: string | null;
  onChange: (url: string) => void;
}

export function ImageUpload({ imageUrl, onChange }: Props) {
  const inputRef = useRef<HTMLInputElement>(null);

  const handleFile = useCallback((file: File) => {
    if (!file.type.startsWith('image/')) return;
    const reader = new FileReader();
    reader.onload = e => {
      if (e.target?.result) onChange(e.target.result as string);
    };
    reader.readAsDataURL(file);
  }, [onChange]);

  const handleDrop = (e: React.DragEvent) => {
    e.preventDefault();
    const file = e.dataTransfer.files[0];
    if (file) handleFile(file);
  };

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const file = e.target.files?.[0];
    if (file) handleFile(file);
  };

  return (
    <div
      onDrop={handleDrop}
      onDragOver={e => e.preventDefault()}
      className="border-2 border-dashed border-gray-300 rounded-xl overflow-hidden hover:border-blue-400 transition cursor-pointer"
      onClick={() => inputRef.current?.click()}
    >
      {imageUrl ? (
        <div className="relative">
          <img src={imageUrl} alt="Cover" className="w-full h-48 object-cover" />
          <button
            type="button"
            onClick={e => { e.stopPropagation(); onChange(''); }}
            className="absolute top-2 right-2 bg-black/50 text-white rounded-full w-7 h-7 flex items-center justify-center"
            aria-label="Remove cover image"
          >
            ×
          </button>
        </div>
      ) : (
        <div className="h-32 flex flex-col items-center justify-center text-gray-400">
          <span className="text-3xl">🖼</span>
          <p className="text-sm mt-1">Drop image here or click to upload</p>
        </div>
      )}
      <input
        ref={inputRef}
        type="file"
        accept="image/*"
        onChange={handleChange}
        className="hidden"
      />
    </div>
  );
}
```

### PostEditor (main form)

```tsx
// src/components/PostEditor.tsx
import { useEffect } from 'react';
import { useForm, Controller } from 'react-hook-form';
import { z } from 'zod';
import { zodResolver } from '@hookform/resolvers/zod';
import { RichTextEditor } from './RichTextEditor';
import { TagInput } from './TagInput';
import { ImageUpload } from './ImageUpload';
import { SaveStatus } from './SaveStatus';
import { useAutosave } from '../hooks/useAutosave';
import { slugify } from '../utils/slugify';

const postSchema = z.object({
  title: z.string().min(1, 'Title is required').max(100),
  slug: z.string().min(1, 'Slug is required').regex(/^[a-z0-9-]+$/, 'Slug must be lowercase with hyphens'),
  body: z.string().min(1, 'Body is required'),
  tags: z.array(z.string()),
  coverImage: z.string(),
  status: z.enum(['draft', 'published']),
});

type PostData = z.infer<typeof postSchema>;

const defaultValues: PostData = {
  title: '',
  slug: '',
  body: '',
  tags: [],
  coverImage: '',
  status: 'draft',
};

async function savePost(data: PostData) {
  await new Promise(r => setTimeout(r, 300));
  localStorage.setItem('post-draft', JSON.stringify(data));
}

export function PostEditor() {
  const { register, handleSubmit, watch, setValue, control, formState: { errors } } = useForm<PostData>({
    resolver: zodResolver(postSchema),
    defaultValues: () => {
      const saved = localStorage.getItem('post-draft');
      return saved ? JSON.parse(saved) : defaultValues;
    },
  });

  const watchedData = watch();
  const titleValue = watch('title');

  // Auto-generate slug from title
  useEffect(() => {
    setValue('slug', slugify(titleValue));
  }, [titleValue, setValue]);

  const { status } = useAutosave(watchedData, savePost, 2000);

  const onSubmit = async (data: PostData) => {
    console.log('Published:', data);
    alert(`Post "${data.title}" published!`);
  };

  return (
    <form onSubmit={handleSubmit(onSubmit)} className="space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-xl font-bold">New Post</h1>
        <SaveStatus status={status} />
      </div>

      <div>
        <div className="flex justify-between">
          <label className="text-sm font-medium">Title</label>
          <span className="text-xs text-gray-400">{titleValue.length}/100</span>
        </div>
        <input
          {...register('title')}
          className="w-full border rounded-lg px-3 py-2 mt-1 text-xl font-semibold"
          placeholder="Post title..."
        />
        {errors.title && <p className="text-red-500 text-sm mt-1">{errors.title.message}</p>}
      </div>

      <div>
        <label className="text-sm font-medium">Slug</label>
        <input
          {...register('slug')}
          className="w-full border rounded-lg px-3 py-2 mt-1 font-mono text-sm"
        />
        {errors.slug && <p className="text-red-500 text-sm mt-1">{errors.slug.message}</p>}
      </div>

      <div>
        <label className="text-sm font-medium">Cover Image</label>
        <div className="mt-1">
          <Controller
            name="coverImage"
            control={control}
            render={({ field }) => (
              <ImageUpload imageUrl={field.value || null} onChange={field.onChange} />
            )}
          />
        </div>
      </div>

      <div>
        <label className="text-sm font-medium">Body</label>
        <div className="mt-1">
          <Controller
            name="body"
            control={control}
            render={({ field }) => (
              <RichTextEditor content={field.value} onChange={field.onChange} />
            )}
          />
        </div>
        {errors.body && <p className="text-red-500 text-sm mt-1">{errors.body.message}</p>}
      </div>

      <div>
        <label className="text-sm font-medium">Tags</label>
        <div className="mt-1">
          <Controller
            name="tags"
            control={control}
            render={({ field }) => (
              <TagInput tags={field.value} onChange={field.onChange} />
            )}
          />
        </div>
      </div>

      <div className="flex items-center gap-4 pt-4 border-t">
        <Controller
          name="status"
          control={control}
          render={({ field }) => (
            <label className="flex items-center gap-2 cursor-pointer">
              <div
                onClick={() => field.onChange(field.value === 'published' ? 'draft' : 'published')}
                className={`relative w-12 h-6 rounded-full transition-colors ${
                  field.value === 'published' ? 'bg-green-500' : 'bg-gray-300'
                }`}
              >
                <div className={`absolute top-1 w-4 h-4 bg-white rounded-full shadow transition-transform ${
                  field.value === 'published' ? 'translate-x-7' : 'translate-x-1'
                }`} />
              </div>
              <span className="text-sm font-medium">
                {field.value === 'published' ? 'Published' : 'Draft'}
              </span>
            </label>
          )}
        />
        <button
          type="submit"
          className="ml-auto px-6 py-2 bg-blue-500 text-white rounded-lg font-semibold hover:bg-blue-600"
        >
          {watch('status') === 'published' ? 'Publish' : 'Save Draft'}
        </button>
      </div>
    </form>
  );
}
```
