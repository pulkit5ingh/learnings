## Background

Accessibility (a11y) is not a feature — it is a baseline. WCAG 2.1 AA is the legal and ethical standard for web apps. In React, accessibility problems often come from:
- Using `<div>` and `<span>` where semantic elements exist
- Missing `aria-label` on icon buttons
- Focus traps that do not close, or modals that do not trap focus
- Dynamic content that is invisible to screen readers

Tools:
- `axe-core` / `jest-axe` — automated audit
- NVDA (Windows) / VoiceOver (macOS) — manual screen reader testing
- `eslint-plugin-jsx-a11y` — lint-time warnings
- Browser DevTools accessibility tree

### 2. Semantic HTML in React

```tsx
// Bad — divs everywhere
function BadNav() {
  return (
    <div className="nav">
      <div onClick={() => navigate('/')}>Home</div>
      <div onClick={() => navigate('/about')}>About</div>
    </div>
  );
}

// Good — semantic elements
function GoodNav() {
  return (
    <nav aria-label="Main navigation">
      <ul role="list">
        <li><a href="/">Home</a></li>
        <li><a href="/about">About</a></li>
      </ul>
    </nav>
  );
}

// Semantic landmarks
function PageLayout({ children }: { children: React.ReactNode }) {
  return (
    <>
      <header>
        <nav aria-label="Main navigation">{/* ... */}</nav>
      </header>
      <main id="main-content">{children}</main>
      <footer>{/* ... */}</footer>
    </>
  );
}
```

### 3. ARIA attributes

```tsx
// aria-label — name an element with no visible text
function IconButton({ icon, label, onClick }: {
  icon: React.ReactNode;
  label: string;
  onClick: () => void;
}) {
  return (
    <button aria-label={label} onClick={onClick}>
      {icon}
    </button>
  );
}

// aria-describedby — link a field to helper text
function PasswordInput() {
  return (
    <div>
      <label htmlFor="password">Password</label>
      <input
        id="password"
        type="password"
        aria-describedby="password-hint"
        aria-required="true"
      />
      <p id="password-hint">Must be at least 8 characters with one number.</p>
    </div>
  );
}

// aria-expanded — toggleable regions
function Accordion({ title, children }: { title: string; children: React.ReactNode }) {
  const [open, setOpen] = useState(false);
  const contentId = useId();

  return (
    <div>
      <button
        aria-expanded={open}
        aria-controls={contentId}
        onClick={() => setOpen(o => !o)}
      >
        {title}
      </button>
      <div id={contentId} hidden={!open}>
        {children}
      </div>
    </div>
  );
}

// role — override implicit semantics
function StatusBadge({ status }: { status: 'online' | 'offline' }) {
  return (
    <span role="status" aria-live="polite">
      {status === 'online' ? 'Online' : 'Offline'}
    </span>
  );
}

// aria-live — announce dynamic updates
function Notification({ message }: { message: string }) {
  return (
    <div aria-live="assertive" aria-atomic="true" className="sr-only">
      {message}
    </div>
  );
}
```

### 4. Keyboard navigation

```tsx
// onKeyDown for non-button interactive elements
function Tag({ label, onRemove }: { label: string; onRemove: () => void }) {
  return (
    <span className="tag">
      {label}
      <button
        aria-label={`Remove ${label}`}
        onClick={onRemove}
        // Buttons already respond to Enter and Space — no onKeyDown needed
      >
        ×
      </button>
    </span>
  );
}

// Keyboard-navigable custom list
function SelectableList({ items }: { items: string[] }) {
  const [selectedIndex, setSelectedIndex] = useState(0);

  const handleKeyDown = (e: React.KeyboardEvent<HTMLUListElement>) => {
    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        setSelectedIndex(i => Math.min(i + 1, items.length - 1));
        break;
      case 'ArrowUp':
        e.preventDefault();
        setSelectedIndex(i => Math.max(i - 1, 0));
        break;
      case 'Home':
        setSelectedIndex(0);
        break;
      case 'End':
        setSelectedIndex(items.length - 1);
        break;
    }
  };

  return (
    <ul
      role="listbox"
      aria-label="Select an option"
      tabIndex={0}
      onKeyDown={handleKeyDown}
    >
      {items.map((item, index) => (
        <li
          key={item}
          role="option"
          aria-selected={index === selectedIndex}
          onClick={() => setSelectedIndex(index)}
        >
          {item}
        </li>
      ))}
    </ul>
  );
}
```

### 5. Focus management with `useRef`

```tsx
// Return focus to trigger element when modal closes
function Modal({ isOpen, onClose, children }: {
  isOpen: boolean;
  onClose: () => void;
  children: React.ReactNode;
}) {
  const dialogRef = useRef<HTMLDivElement>(null);
  const triggerRef = useRef<HTMLButtonElement>(null);

  // Move focus into modal when opened
  useEffect(() => {
    if (isOpen) {
      dialogRef.current?.focus();
    }
  }, [isOpen]);

  // Trap focus within the modal
  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Escape') {
      onClose();
      triggerRef.current?.focus(); // return focus to trigger
      return;
    }

    if (e.key !== 'Tab') return;

    const focusable = dialogRef.current?.querySelectorAll<HTMLElement>(
      'a[href], button:not([disabled]), input, select, textarea, [tabindex]:not([tabindex="-1"])'
    );
    if (!focusable || focusable.length === 0) return;

    const first = focusable[0];
    const last = focusable[focusable.length - 1];

    if (e.shiftKey && document.activeElement === first) {
      e.preventDefault();
      last.focus();
    } else if (!e.shiftKey && document.activeElement === last) {
      e.preventDefault();
      first.focus();
    }
  };

  if (!isOpen) return null;

  return (
    <div
      role="dialog"
      aria-modal="true"
      aria-labelledby="modal-title"
      ref={dialogRef}
      tabIndex={-1}
      onKeyDown={handleKeyDown}
      className="modal-overlay"
    >
      <div className="modal-content">
        <h2 id="modal-title">Modal Title</h2>
        {children}
        <button onClick={onClose}>Close</button>
      </div>
    </div>
  );
}
```

### 6. `tabIndex` guidelines

```tsx
// tabIndex={0}  — add to focusable order (only when element has interaction)
// tabIndex={-1} — reachable programmatically but not via Tab key
// tabIndex={1+} — AVOID: changes natural tab order, hard to maintain

// Skip navigation link (visible on focus)
function SkipLink() {
  return (
    <a
      href="#main-content"
      className="sr-only focus:not-sr-only focus:absolute focus:top-4 focus:left-4 z-50"
    >
      Skip to main content
    </a>
  );
}
```

### 7. Automated a11y testing with jest-axe

```tsx
// src/components/__tests__/Button.a11y.test.tsx
import { render } from '@testing-library/react';
import { axe, toHaveNoViolations } from 'jest-axe';
import { Button } from '../Button';

expect.extend(toHaveNoViolations);

describe('Button accessibility', () => {
  it('has no a11y violations', async () => {
    const { container } = render(
      <Button onClick={() => {}}>Click me</Button>
    );
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });

  it('icon button with aria-label has no violations', async () => {
    const { container } = render(
      <button aria-label="Close dialog">
        <svg aria-hidden="true">{/* ... */}</svg>
      </button>
    );
    const results = await axe(container);
    expect(results).toHaveNoViolations();
  });
});
```

```tsx
// Test keyboard navigation
import { render, screen } from '@testing-library/react';
import userEvent from '@testing-library/user-event';
import { Accordion } from '../Accordion';

test('accordion opens and closes with Enter key', async () => {
  const user = userEvent.setup();
  render(<Accordion title="FAQ">Answer here</Accordion>);

  const button = screen.getByRole('button', { name: 'FAQ' });
  expect(button).toHaveAttribute('aria-expanded', 'false');

  await user.keyboard('{Tab}'); // focus the button
  await user.keyboard('{Enter}');
  expect(button).toHaveAttribute('aria-expanded', 'true');

  await user.keyboard('{Enter}');
  expect(button).toHaveAttribute('aria-expanded', 'false');
});
```

### 8. Screen reader testing checklist (manual)

**VoiceOver (macOS):** `Cmd + F5` to start, `Ctrl + Option + Right` to move forward

**NVDA (Windows):** Free download from nvaccess.org, `Insert + F7` for landmark list

Check:
- [ ] All images have meaningful `alt` text (or `alt=""` if decorative)
- [ ] Form fields have visible labels connected via `for`/`id` or `aria-label`
- [ ] Error messages are announced (`aria-live="assertive"` or associated via `aria-describedby`)
- [ ] Modals trap focus and return focus to trigger on close
- [ ] Page title updates on route change
- [ ] Loading states are announced
