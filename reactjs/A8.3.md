## Goal
Replace manual `fetch` mocks with Mock Service Worker (MSW), which intercepts network requests at the service worker level for realistic API mocking in both tests and development.

## Background

### Why MSW Over fetch Mocks?
Mocking `global.fetch` directly is brittle. You mock the transport layer, not the actual API contract. MSW runs at the network level: your code calls `fetch('/api/posts')` and MSW intercepts it before it leaves the browser (or Node.js). This means your components, React Query hooks, and Axios calls all work exactly as they would in production — you just control what the server responds with. No code changes needed in what you're testing.

### Request Handlers
MSW uses "handlers" — functions that match a URL pattern and return a mocked response. `http.get('/api/posts', () => HttpResponse.json(data))` intercepts all GET requests to `/api/posts`. You can have handlers for every endpoint in your API. Organize them by domain: `src/mocks/handlers/posts.ts`, `src/mocks/handlers/users.ts`.

### Test Server vs Browser Service Worker
MSW runs in two modes: `setupServer` (for Node.js/tests) and `setupWorker` (for the browser). In tests, use `setupServer` from `msw/node`. In development, use `setupWorker` from `msw/browser` to mock APIs in the browser without a backend running.

```ts
// src/mocks/handlers/posts.ts
import { http, HttpResponse } from 'msw'

export interface Post {
  id: number
  title: string
  body: string
  userId: number
}

const posts: Post[] = [
  { id: 1, title: 'First Post', body: 'Content of first post', userId: 1 },
  { id: 2, title: 'Second Post', body: 'Content of second post', userId: 1 },
  { id: 3, title: 'Third Post', body: 'Content of third post', userId: 2 },
]

export const postHandlers = [
  http.get('/api/posts', ({ request }) => {
    const url = new URL(request.url)
    const userId = url.searchParams.get('userId')

    const filtered = userId
      ? posts.filter(p => p.userId === Number(userId))
      : posts

    return HttpResponse.json(filtered)
  }),

  http.get('/api/posts/:id', ({ params }) => {
    const post = posts.find(p => p.id === Number(params.id))
    if (!post) return new HttpResponse(null, { status: 404 })
    return HttpResponse.json(post)
  }),

  http.post('/api/posts', async ({ request }) => {
    const body = await request.json() as Omit<Post, 'id'>
    const newPost: Post = { ...body, id: Date.now() }
    posts.push(newPost)
    return HttpResponse.json(newPost, { status: 201 })
  }),

  http.delete('/api/posts/:id', ({ params }) => {
    const index = posts.findIndex(p => p.id === Number(params.id))
    if (index === -1) return new HttpResponse(null, { status: 404 })
    posts.splice(index, 1)
    return new HttpResponse(null, { status: 204 })
  }),
]
```

```ts
// src/mocks/handlers/users.ts
import { http, HttpResponse } from 'msw'

const users = [
  { id: 1, name: 'Alice', email: 'alice@example.com', role: 'admin' },
  { id: 2, name: 'Bob', email: 'bob@example.com', role: 'user' },
]

export const userHandlers = [
  http.get('/api/users', () => HttpResponse.json(users)),
  http.get('/api/users/:id', ({ params }) => {
    const user = users.find(u => u.id === Number(params.id))
    return user ? HttpResponse.json(user) : new HttpResponse(null, { status: 404 })
  }),
  http.post('/api/auth/login', async ({ request }) => {
    const { email, password } = await request.json() as { email: string; password: string }
    if (password !== 'correct-password') {
      return HttpResponse.json({ error: 'Invalid credentials' }, { status: 401 })
    }
    const user = users.find(u => u.email === email) ?? users[0]
    return HttpResponse.json({ user, token: 'mock-jwt-token' })
  }),
]
```

### Task 3 — Tests Using MSW

```tsx
// src/components/__tests__/PostListWithMSW.test.tsx
import { render, screen, waitFor } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { http, HttpResponse } from 'msw'
import { server } from '../../mocks/server'
import { PostList } from '../PostList'
import { QueryClientWrapper } from '../../test/QueryClientWrapper'

describe('PostList with MSW', () => {
  it('renders posts from API', async () => {
    render(<PostList />, { wrapper: QueryClientWrapper })

    // MSW intercepts the real fetch and returns mock data
    await waitFor(() => {
      expect(screen.getByText('First Post')).toBeInTheDocument()
      expect(screen.getByText('Second Post')).toBeInTheDocument()
    })
  })

  it('shows error state when API returns 500', async () => {
    // Override handler for this test only
    server.use(
      http.get('/api/posts', () => {
        return HttpResponse.json({ error: 'Server error' }, { status: 500 })
      })
    )

    render(<PostList />, { wrapper: QueryClientWrapper })

    await waitFor(() => {
      expect(screen.getByText(/failed to load/i)).toBeInTheDocument()
    })
  })

  it('shows empty state when API returns empty array', async () => {
    server.use(
      http.get('/api/posts', () => HttpResponse.json([]))
    )

    render(<PostList />, { wrapper: QueryClientWrapper })

    await waitFor(() => {
      expect(screen.getByText(/no posts/i)).toBeInTheDocument()
    })
  })
})
```

```tsx
// src/test/QueryClientWrapper.tsx
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import type { ReactNode } from 'react'

export function QueryClientWrapper({ children }: { children: ReactNode }) {
  const queryClient = new QueryClient({
    defaultOptions: { queries: { retry: false, staleTime: 0 } },
  })
  return <QueryClientProvider client={queryClient}>{children}</QueryClientProvider>
}
```

```ts
// src/main.tsx — start worker in development
async function enableMocking() {
  if (process.env.NODE_ENV !== 'development') return
  const { worker } = await import('./mocks/browser')
  return worker.start({ onUnhandledRequest: 'bypass' })
}

enableMocking().then(() => {
  createRoot(document.getElementById('root')!).render(<App />)
})
```
