## Goal
Handle file uploads with preview, drag-and-drop, and integrate a rich text editor into a React Hook Form-based form.

## Background

### File Inputs in React
File inputs are special — their value is write-protected by the browser (for security). You cannot make them controlled in the normal way. Instead, read the file from `e.target.files` in the `onChange` handler and store it in state separately from the input. React Hook Form's `register` works for file inputs but you typically manage the file object yourself and use `setValue` to update the form's file field.

### Creating Previews
For images, use `URL.createObjectURL(file)` to create a temporary URL pointing to the file in memory. This URL is valid for the lifetime of the page session. Call `URL.revokeObjectURL(url)` when the preview is no longer needed to free memory.

### TipTap
TipTap is a headless rich text editor built on ProseMirror. "Headless" means it provides all the editing logic but no UI — you build the toolbar yourself. This gives complete styling control. It integrates cleanly with React Hook Form via a controlled pattern: call `editor.getHTML()` to get the current content and use `setValue` to update the form field.

### Task 2 — File Upload with Preview

```tsx
// src/components/AvatarUpload.tsx
import { useState, useCallback } from 'react'
import { useDropzone } from 'react-dropzone'  // pnpm add react-dropzone

interface AvatarUploadProps {
  value?: File | null
  onChange: (file: File | null) => void
  currentUrl?: string
}

export function AvatarUpload({ value, onChange, currentUrl }: AvatarUploadProps) {
  const [preview, setPreview] = useState<string | null>(currentUrl ?? null)

  const onDrop = useCallback((acceptedFiles: File[]) => {
    const file = acceptedFiles[0]
    if (!file) return

    onChange(file)

    // Create preview URL
    const url = URL.createObjectURL(file)
    setPreview(url)

    // Clean up old preview URL
    return () => URL.revokeObjectURL(url)
  }, [onChange])

  const { getRootProps, getInputProps, isDragActive, fileRejections } = useDropzone({
    onDrop,
    accept: { 'image/*': ['.png', '.jpg', '.jpeg', '.webp'] },
    maxSize: 5 * 1024 * 1024,  // 5MB
    maxFiles: 1,
  })

  function handleRemove(e: React.MouseEvent) {
    e.stopPropagation()
    onChange(null)
    setPreview(null)
  }

  const error = fileRejections[0]?.errors[0]

  return (
    <div>
      <div
        {...getRootProps()}
        style={{
          border: `2px dashed ${isDragActive ? '#3b82f6' : '#d1d5db'}`,
          borderRadius: 12,
          padding: '1.5rem',
          textAlign: 'center',
          cursor: 'pointer',
          background: isDragActive ? '#eff6ff' : '#fafafa',
          transition: 'all 0.2s',
        }}
      >
        <input {...getInputProps()} />

        {preview ? (
          <div style={{ position: 'relative', display: 'inline-block' }}>
            <img
              src={preview}
              alt="Avatar preview"
              style={{ width: 120, height: 120, borderRadius: '50%', objectFit: 'cover' }}
            />
            <button
              onClick={handleRemove}
              style={{ position: 'absolute', top: 0, right: 0, width: 24, height: 24, borderRadius: '50%', background: '#ef4444', color: 'white', border: 'none', cursor: 'pointer', fontSize: 16, lineHeight: 1, display: 'flex', alignItems: 'center', justifyContent: 'center' }}
            >
              ×
            </button>
          </div>
        ) : (
          <div>
            <div style={{ fontSize: 40, marginBottom: 8 }}>📷</div>
            <p style={{ margin: 0, color: '#6b7280' }}>
              {isDragActive ? 'Drop the image here' : 'Drag & drop or click to upload'}
            </p>
            <p style={{ margin: '4px 0 0', fontSize: 12, color: '#9ca3af' }}>PNG, JPG, WEBP up to 5MB</p>
          </div>
        )}
      </div>

      {error && (
        <p style={{ color: '#ef4444', fontSize: 13, margin: '8px 0 0' }}>
          {error.code === 'file-too-large' ? 'File exceeds 5MB limit' : error.message}
        </p>
      )}
    </div>
  )
}
```

### Task 3 — TipTap Rich Text Editor

```tsx
// src/components/RichTextEditor.tsx
import { useEditor, EditorContent } from '@tiptap/react'
import StarterKit from '@tiptap/starter-kit'
import Placeholder from '@tiptap/extension-placeholder'
import Link from '@tiptap/extension-link'

interface RichTextEditorProps {
  value: string
  onChange: (html: string) => void
  placeholder?: string
}

function ToolbarButton({ onClick, active, title, children }: {
  onClick: () => void
  active?: boolean
  title: string
  children: React.ReactNode
}) {
  return (
    <button
      type="button"
      onClick={onClick}
      title={title}
      style={{
        padding: '4px 8px',
        border: 'none',
        borderRadius: 4,
        cursor: 'pointer',
        background: active ? '#dbeafe' : 'transparent',
        color: active ? '#1d4ed8' : '#374151',
        fontWeight: active ? 600 : 400,
      }}
    >
      {children}
    </button>
  )
}

export function RichTextEditor({ value, onChange, placeholder = 'Start writing...' }: RichTextEditorProps) {
  const editor = useEditor({
    extensions: [
      StarterKit,
      Placeholder.configure({ placeholder }),
      Link.configure({ openOnClick: false }),
    ],
    content: value,
    onUpdate: ({ editor }) => {
      onChange(editor.getHTML())
    },
  })

  if (!editor) return null

  return (
    <div style={{ border: '1px solid #d1d5db', borderRadius: 8, overflow: 'hidden' }}>
      {/* Toolbar */}
      <div style={{ borderBottom: '1px solid #e5e7eb', padding: '8px', display: 'flex', flexWrap: 'wrap', gap: 2, background: '#f9fafb' }}>
        <ToolbarButton onClick={() => editor.chain().focus().toggleBold().run()} active={editor.isActive('bold')} title="Bold (Ctrl+B)"><strong>B</strong></ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleItalic().run()} active={editor.isActive('italic')} title="Italic (Ctrl+I)"><em>I</em></ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleStrike().run()} active={editor.isActive('strike')} title="Strikethrough"><s>S</s></ToolbarButton>
        <div style={{ width: 1, background: '#e5e7eb', margin: '0 4px' }} />
        <ToolbarButton onClick={() => editor.chain().focus().toggleHeading({ level: 2 }).run()} active={editor.isActive('heading', { level: 2 })} title="Heading 2">H2</ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleHeading({ level: 3 }).run()} active={editor.isActive('heading', { level: 3 })} title="Heading 3">H3</ToolbarButton>
        <div style={{ width: 1, background: '#e5e7eb', margin: '0 4px' }} />
        <ToolbarButton onClick={() => editor.chain().focus().toggleBulletList().run()} active={editor.isActive('bulletList')} title="Bullet list">• List</ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleOrderedList().run()} active={editor.isActive('orderedList')} title="Numbered list">1. List</ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleBlockquote().run()} active={editor.isActive('blockquote')} title="Blockquote">"</ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().toggleCodeBlock().run()} active={editor.isActive('codeBlock')} title="Code block">{`<>`}</ToolbarButton>
        <div style={{ width: 1, background: '#e5e7eb', margin: '0 4px' }} />
        <ToolbarButton onClick={() => editor.chain().focus().undo().run()} title="Undo">↩</ToolbarButton>
        <ToolbarButton onClick={() => editor.chain().focus().redo().run()} title="Redo">↪</ToolbarButton>
      </div>

      {/* Editor area */}
      <EditorContent
        editor={editor}
        style={{ minHeight: 200, padding: '0.75rem 1rem', fontSize: 15, lineHeight: 1.6 }}
      />
    </div>
  )
}
```

### Task 4 — Integrated Post Editor Form

```tsx
// src/components/PostEditorForm.tsx
import { useForm, Controller } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'
import { AvatarUpload } from './AvatarUpload'
import { RichTextEditor } from './RichTextEditor'

const postSchema = z.object({
  title: z.string().min(5, 'Title must be at least 5 characters'),
  body: z.string().min(20, 'Write at least 20 characters').refine(
    val => val !== '<p></p>' && val !== '',
    'Body is required'
  ),
  coverImage: z.instanceof(File, { message: 'Cover image required' }).nullable(),
  tags: z.string().transform(s => s.split(',').map(t => t.trim()).filter(Boolean)),
})

type PostValues = z.infer<typeof postSchema>

export function PostEditorForm() {
  const { register, control, handleSubmit, formState: { errors, isSubmitting } } = useForm<PostValues>({
    resolver: zodResolver(postSchema),
    defaultValues: { title: '', body: '', coverImage: null, tags: '' },
  })

  function onSubmit(data: PostValues) {
    console.log('Post data:', { ...data, body: data.body.slice(0, 50) + '...' })
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ maxWidth: 700, display: 'flex', flexDirection: 'column', gap: '1.5rem' }}>
      <div>
        <label style={{ fontWeight: 600 }}>Cover Image</label>
        <div style={{ marginTop: 8 }}>
          <Controller
            control={control}
            name="coverImage"
            render={({ field }) => (
              <AvatarUpload value={field.value} onChange={field.onChange} />
            )}
          />
        </div>
        {errors.coverImage && <p style={{ color: '#ef4444', fontSize: 13 }}>{errors.coverImage.message}</p>}
      </div>

      <div>
        <label style={{ fontWeight: 600 }}>Title *</label>
        <input {...register('title')} style={{ display: 'block', width: '100%', marginTop: 6, padding: '10px 12px', fontSize: 18, fontWeight: 500, border: `1px solid ${errors.title ? '#ef4444' : '#d1d5db'}`, borderRadius: 6 }} placeholder="Your post title" />
        {errors.title && <p style={{ color: '#ef4444', fontSize: 13 }}>{errors.title.message}</p>}
      </div>

      <div>
        <label style={{ fontWeight: 600 }}>Body *</label>
        <div style={{ marginTop: 6 }}>
          <Controller
            control={control}
            name="body"
            render={({ field }) => (
              <RichTextEditor value={field.value} onChange={field.onChange} placeholder="Write your post..." />
            )}
          />
        </div>
        {errors.body && <p style={{ color: '#ef4444', fontSize: 13 }}>{errors.body.message}</p>}
      </div>

      <div>
        <label style={{ fontWeight: 600 }}>Tags (comma separated)</label>
        <input {...register('tags')} style={{ display: 'block', width: '100%', marginTop: 6, padding: '8px 12px' }} placeholder="react, typescript, webdev" />
      </div>

      <button type="submit" disabled={isSubmitting} style={{ padding: '10px 20px', background: '#3b82f6', color: 'white', border: 'none', borderRadius: 6, cursor: 'pointer', fontSize: 15 }}>
        {isSubmitting ? 'Publishing...' : 'Publish Post'}
      </button>
    </form>
  )
}
```
