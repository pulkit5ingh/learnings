# P6.2 — Animated Portfolio

**Phase:** 6 — Styling & UI
**Topic:** Framer Motion page transitions, scroll animations, dark mode, Tailwind CSS

## Goal

Build an animated personal portfolio site with Framer Motion page transitions, scroll-triggered animations, a sticky animated header, and a dark mode toggle. Practice `motion.div`, `AnimatePresence`, `variants`, `useInView`, and `useScroll`.

---

## What to Build

- **Home** — hero section with staggered text animation, animated CTA button
- **About** — scroll-triggered skill bars, reveal-on-scroll paragraphs
- **Projects** — grid of project cards with hover lift effect and staggered entrance
- **Contact** — form with field-focus animations
- Sticky header that changes opacity/blur on scroll
- Page transition animation (fade + slide) between routes
- Dark mode toggle with smooth colour transition
- Mobile-responsive layout

### Dark mode hook

```ts
// src/hooks/useDarkMode.ts
import { useEffect, useState } from 'react';

export function useDarkMode() {
  const [dark, setDark] = useState(() => {
    if (typeof window === 'undefined') return false;
    return localStorage.getItem('theme') === 'dark' ||
      (!localStorage.getItem('theme') && window.matchMedia('(prefers-color-scheme: dark)').matches);
  });

  useEffect(() => {
    document.documentElement.classList.toggle('dark', dark);
    localStorage.setItem('theme', dark ? 'dark' : 'light');
  }, [dark]);

  return { dark, toggle: () => setDark(d => !d) };
}
```

### Page transition wrapper

```tsx
// src/components/PageTransition.tsx
import { motion, AnimatePresence } from 'framer-motion';
import { useLocation } from 'react-router-dom';

const pageVariants = {
  initial: { opacity: 0, y: 20 },
  animate: { opacity: 1, y: 0, transition: { duration: 0.4, ease: 'easeOut' } },
  exit: { opacity: 0, y: -20, transition: { duration: 0.25, ease: 'easeIn' } },
};

export function PageTransition({ children }: { children: React.ReactNode }) {
  const location = useLocation();

  return (
    <AnimatePresence mode="wait">
      <motion.div
        key={location.pathname}
        variants={pageVariants}
        initial="initial"
        animate="animate"
        exit="exit"
      >
        {children}
      </motion.div>
    </AnimatePresence>
  );
}
```

### ScrollReveal

```tsx
// src/components/ScrollReveal.tsx
import { motion, useInView } from 'framer-motion';
import { useRef } from 'react';

interface Props {
  children: React.ReactNode;
  delay?: number;
  direction?: 'up' | 'left' | 'right';
  className?: string;
}

export function ScrollReveal({ children, delay = 0, direction = 'up', className }: Props) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once: true, margin: '0px 0px -80px 0px' });

  const variants = {
    hidden: {
      opacity: 0,
      y: direction === 'up' ? 40 : 0,
      x: direction === 'left' ? -40 : direction === 'right' ? 40 : 0,
    },
    visible: {
      opacity: 1,
      y: 0,
      x: 0,
      transition: { duration: 0.6, delay, ease: [0.25, 0.1, 0.25, 1] },
    },
  };

  return (
    <motion.div
      ref={ref}
      variants={variants}
      initial="hidden"
      animate={isInView ? 'visible' : 'hidden'}
      className={className}
    >
      {children}
    </motion.div>
  );
}
```

### AnimatedText (staggered characters)

```tsx
// src/components/AnimatedText.tsx
import { motion } from 'framer-motion';

interface Props {
  text: string;
  className?: string;
  as?: keyof JSX.IntrinsicElements;
}

const containerVariants = {
  hidden: {},
  visible: { transition: { staggerChildren: 0.04 } },
};

const charVariants = {
  hidden: { opacity: 0, y: 20 },
  visible: { opacity: 1, y: 0, transition: { duration: 0.4, ease: 'easeOut' } },
};

export function AnimatedText({ text, className, as: Tag = 'span' }: Props) {
  return (
    <motion.span
      variants={containerVariants}
      initial="hidden"
      animate="visible"
      className={className}
      aria-label={text}
    >
      {text.split('').map((char, i) => (
        <motion.span key={i} variants={charVariants} aria-hidden="true">
          {char === ' ' ? ' ' : char}
        </motion.span>
      ))}
    </motion.span>
  );
}
```

### Sticky Header with scroll effect

```tsx
// src/components/Header.tsx
import { motion, useScroll, useTransform } from 'framer-motion';
import { NavLink } from 'react-router-dom';
import { DarkModeToggle } from './DarkModeToggle';
import { useDarkMode } from '../hooks/useDarkMode';

export function Header() {
  const { scrollY } = useScroll();
  const { dark, toggle } = useDarkMode();

  // Increase backdrop blur and add shadow as page scrolls
  const backdropBlur = useTransform(scrollY, [0, 100], ['blur(0px)', 'blur(12px)']);
  const bgOpacity = useTransform(scrollY, [0, 80], [0, 0.9]);

  const links = [
    { to: '/', label: 'Home' },
    { to: '/about', label: 'About' },
    { to: '/projects', label: 'Projects' },
    { to: '/contact', label: 'Contact' },
  ];

  return (
    <motion.header
      className="fixed top-0 left-0 right-0 z-50 px-6 py-4"
      style={{ backdropFilter: backdropBlur }}
    >
      <motion.div
        className="absolute inset-0 bg-white dark:bg-gray-900"
        style={{ opacity: bgOpacity }}
      />
      <nav className="relative max-w-5xl mx-auto flex items-center justify-between">
        <NavLink to="/" className="font-bold text-xl text-gray-900 dark:text-white">
          Portfolio
        </NavLink>
        <div className="flex items-center gap-6">
          {links.map(({ to, label }) => (
            <NavLink
              key={to}
              to={to}
              end={to === '/'}
              className={({ isActive }) =>
                `text-sm font-medium transition-colors ${
                  isActive
                    ? 'text-blue-600 dark:text-blue-400'
                    : 'text-gray-600 dark:text-gray-400 hover:text-gray-900 dark:hover:text-white'
                }`
              }
            >
              {label}
            </NavLink>
          ))}
          <DarkModeToggle dark={dark} onToggle={toggle} />
        </div>
      </nav>
    </motion.header>
  );
}
```

import { motion } from 'framer-motion';

interface Project {
  id: number;
  title: string;
  description: string;
  tags: string[];
  url: string;
  image?: string;
}

export function ProjectCard({ project }: { project: Project }) {

      href={project.url}
      target="_blank"

        <img src={project.image} alt={project.title} className="w-full h-48 object-cover" />

            </span>
          ))}

### SkillBar with scroll-triggered animation

```tsx
// src/components/SkillBar.tsx
import { motion, useInView } from 'framer-motion';
import { useRef } from 'react';

interface Props {
  skill: string;
  level: number; // 0–100
  delay?: number;
}

export function SkillBar({ skill, level, delay = 0 }: Props) {
  const ref = useRef<HTMLDivElement>(null);
  const inView = useInView(ref, { once: true });

  return (
    <div ref={ref}>
      <div className="flex justify-between text-sm mb-1">
        <span className="font-medium text-gray-700 dark:text-gray-300">{skill}</span>
        <span className="text-gray-500">{level}%</span>
      </div>
      <div className="h-2 bg-gray-200 dark:bg-gray-700 rounded-full overflow-hidden">
        <motion.div
          className="h-full bg-blue-500 rounded-full"
          initial={{ width: 0 }}
          animate={inView ? { width: `${level}%` } : { width: 0 }}
          transition={{ duration: 1.2, delay, ease: 'easeOut' }}
        />
      </div>
    </div>
  );
}
```
