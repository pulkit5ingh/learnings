## Goal
Learn SWR as an alternative data-fetching library and understand the trade-offs between SWR and React Query.

## Background

### What is SWR?
SWR (stale-while-revalidate) is a data-fetching library from Vercel. Like React Query, it caches data, refetches in the background, and handles loading/error states. SWR is lighter and simpler. React Query is more feature-rich (mutations, infinite queries, devtools, offline support). For many projects, either works. Knowing both makes you adaptable.

### SWR's Revalidation Model
SWR always serves cached (stale) data immediately, then revalidates (refetches) in the background. This is the "stale-while-revalidate" HTTP cache strategy. By default, SWR revalidates on focus, on reconnect, and when a tab refocuses. You can configure deduplication intervals (how long to wait before a new request for the same key) and revalidation behavior.

### Key API Differences
- React Query: `useQuery({ queryKey, queryFn })` — key and fetcher separate
- SWR: `useSWR(key, fetcher)` — key is a string or array, fetcher is a function taking the key
- React Query mutations: `useMutation` hook
- SWR mutations: `mutate` function from `useSWRMutation` or the global `mutate` function

### Task 1 — Setup and Basic Usage

```bash
pnpm add swr
```

```tsx
// src/main.tsx — add SWRConfig for global defaults
import { SWRConfig } from 'swr'

// Global fetcher function
const fetcher = (url: string) =>
  fetch(url).then(res => {
    if (!res.ok) throw new Error(`HTTP ${res.status}`)
    return res.json()
  })

// Wrap app (or just the parts that use SWR)
<SWRConfig
  value={{
    fetcher,
    revalidateOnFocus: true,
    revalidateOnReconnect: true,
    dedupingInterval: 2000,   // don't re-request same key within 2s
    errorRetryCount: 3,
  }}
>
  <App />
</SWRConfig>
```

```tsx
// src/components/SWRPostList.tsx
import useSWR from 'swr'

interface Post { id: number; title: string; body: string }

export function SWRPostList() {
  const { data, error, isLoading, isValidating } = useSWR<Post[]>(
    'https://jsonplaceholder.typicode.com/posts?_limit=10'
    // fetcher is provided globally via SWRConfig
  )

  if (isLoading) return <p>Loading...</p>
  if (error) return <p style={{ color: '#ef4444' }}>Error: {error.message}</p>

  return (
    <div>
      <div style={{ display: 'flex', gap: 8, alignItems: 'center' }}>
        <h2>Posts (SWR)</h2>
        {isValidating && <span style={{ fontSize: 12, color: '#6b7280' }}>Refreshing...</span>}
      </div>
      <ul style={{ listStyle: 'none', padding: 0 }}>
        {data?.map(post => (
          <li key={post.id} style={{ borderBottom: '1px solid #f0f0f0', padding: '0.75rem 0' }}>
            <strong>{post.title}</strong>
          </li>
        ))}
      </ul>
    </div>
  )
}
```

### Task 2 — SWR with Dynamic Keys

```tsx
// src/components/SWRUserDetail.tsx
import useSWR from 'swr'
import { useState } from 'react'

interface User { id: number; name: string; email: string }

export function SWRUserDetail() {
  const [userId, setUserId] = useState(1)

  // null key skips the request
  const { data: user, isLoading } = useSWR<User>(
    userId ? `https://jsonplaceholder.typicode.com/users/${userId}` : null
  )

  return (
    <div>
      <div style={{ display: 'flex', gap: 8, marginBottom: 12 }}>
        {[1, 2, 3, 4, 5].map(id => (
          <button
            key={id}
            onClick={() => setUserId(id)}
            style={{ background: userId === id ? '#3b82f6' : '#f1f5f9', color: userId === id ? 'white' : 'inherit', border: 'none', borderRadius: 4, padding: '6px 12px', cursor: 'pointer' }}
          >
            {id}
          </button>
        ))}
      </div>
      {isLoading && <p>Loading user {userId}...</p>}
      {user && (
        <div>
          <strong>{user.name}</strong>
          <p style={{ color: '#6b7280' }}>{user.email}</p>
        </div>
      )}
    </div>
  )
}
```

### Task 3 — useSWRMutation

```tsx
// src/components/SWRCreatePost.tsx
import useSWRMutation from 'swr/mutation'
import { useState } from 'react'

async function createPost(url: string, { arg }: { arg: { title: string; body: string } }) {
  const res = await fetch(url, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ ...arg, userId: 1 }),
  })
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
  return res.json()
}

export function SWRCreatePost() {
  const [title, setTitle] = useState('')
  const [body, setBody] = useState('')

  const { trigger, isMutating, error, data } = useSWRMutation(
    'https://jsonplaceholder.typicode.com/posts',
    createPost
  )

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    try {
      await trigger({ title, body })
      setTitle('')
      setBody('')
    } catch { /* error is in the error state */ }
  }

  return (
    <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem', maxWidth: 400 }}>
      {error && <p style={{ color: '#ef4444' }}>{error.message}</p>}
      {data && <p style={{ color: '#15803d' }}>Created: {data.title}</p>}
      <input value={title} onChange={e => setTitle(e.target.value)} placeholder="Title" required style={{ padding: '8px 10px' }} />
      <textarea value={body} onChange={e => setBody(e.target.value)} rows={3} placeholder="Body" required style={{ padding: '8px 10px' }} />
      <button type="submit" disabled={isMutating}>{isMutating ? 'Creating...' : 'Create'}</button>
    </form>
  )
}
```

### Task 4 — SWR vs React Query Comparison

```tsx
// Reference comparison table — keep in your notes

// | Feature                  | SWR                         | React Query                    |
// |--------------------------|-----------------------------|---------------------------------|
// | Bundle size              | ~4kB                        | ~13kB                          |
// | Setup                    | SWRConfig                   | QueryClient + QueryClientProvider |
// | Query hook               | useSWR(key, fetcher)        | useQuery({ queryKey, queryFn }) |
// | Mutation hook            | useSWRMutation              | useMutation                    |
// | Infinite queries         | useSWRInfinite              | useInfiniteQuery                |
// | Devtools                 | No official devtools        | @tanstack/react-query-devtools  |
// | Optimistic updates       | Manual                      | onMutate/onError/onSettled      |
// | Offline support          | Limited                     | Full (persistQueryClient)      |
// | Cache manipulation       | mutate(key, data)           | setQueryData, invalidateQueries |
// | When to choose           | Simple fetch needs, Next.js | Complex apps, full feature set |
```
