## Background

### Utility-First CSS
Tailwind CSS provides small, single-purpose utility classes. Instead of writing `.card { padding: 16px; border-radius: 8px; }` in a separate file, you write `className="p-4 rounded-lg"` directly in JSX. This co-locates styles with markup, eliminates specificity battles, and makes unused styles easy to identify. The trade-off is that className strings can get long.

### Responsive and Dark Mode
Tailwind's responsive prefixes (`sm:`, `md:`, `lg:`, `xl:`) apply styles at specific breakpoints. The `dark:` prefix applies styles when the user prefers dark mode. You can configure Tailwind to use a CSS class (`dark` on the `<html>` element) instead of the OS preference, giving your app manual dark mode control.

### Conditional Classes
In React, `className` is just a string. To conditionally apply classes, you concatenate strings — but this gets messy fast. The `clsx` library cleanly handles conditional class lists. The `cn` utility (used by shadcn/ui) combines `clsx` with `tailwind-merge`, which de-duplicates conflicting Tailwind classes: `cn('p-4', isPadded && 'p-8')` correctly outputs `p-8`, not `p-4 p-8`.

### Task 2 — The cn Utility

```ts
// src/lib/cn.ts
import { clsx, type ClassValue } from 'clsx'
import { twMerge } from 'tailwind-merge'

// Use this everywhere instead of bare className strings
export function cn(...inputs: ClassValue[]) {
  return twMerge(clsx(inputs))
}

// Examples:
// cn('px-4 py-2', isActive && 'bg-blue-600 text-white')
// cn('p-4', 'p-8')  → 'p-8'  (tailwind-merge resolves conflict)
// cn({ 'ring-2 ring-blue-500': isFocused, 'border-red-500': hasError })
```

### Task 4 — Dark Mode Toggle

```tsx
// src/components/DarkModeToggle.tsx
import { useEffect, useState } from 'react'
import { cn } from '../lib/cn'

export function DarkModeToggle() {
  const [dark, setDark] = useState(() =>
    document.documentElement.classList.contains('dark')
  )

  useEffect(() => {
    document.documentElement.classList.toggle('dark', dark)
    localStorage.setItem('theme', dark ? 'dark' : 'light')
  }, [dark])

  return (
    <button
      onClick={() => setDark(d => !d)}
      className={cn(
        'relative inline-flex h-6 w-11 items-center rounded-full transition-colors duration-200',
        dark ? 'bg-blue-600' : 'bg-slate-200'
      )}
      aria-label="Toggle dark mode"
    >
      <span
        className={cn(
          'inline-block h-4 w-4 rounded-full bg-white shadow-sm transition-transform duration-200',
          dark ? 'translate-x-6' : 'translate-x-1'
        )}
      />
    </button>
  )
}
```

### Task 5 — Dynamic Button with cn

```tsx
// src/components/Button.tsx
import { cn } from '../lib/cn'
import type { ButtonHTMLAttributes } from 'react'

interface ButtonProps extends ButtonHTMLAttributes<HTMLButtonElement> {
  variant?: 'primary' | 'secondary' | 'danger' | 'ghost'
  size?: 'sm' | 'md' | 'lg'
  loading?: boolean
}

export function Button({ variant = 'primary', size = 'md', loading, className, disabled, children, ...props }: ButtonProps) {
  return (
    <button
      {...props}
      disabled={disabled || loading}
      className={cn(
        // Base styles
        'inline-flex items-center justify-center rounded-md font-medium transition-colors focus-visible:outline-none focus-visible:ring-2 focus-visible:ring-offset-2 disabled:pointer-events-none disabled:opacity-50',
        // Variant
        {
          'bg-blue-600 text-white hover:bg-blue-700 focus-visible:ring-blue-500': variant === 'primary',
          'border border-slate-200 bg-white hover:bg-slate-50 text-slate-700': variant === 'secondary',
          'bg-red-600 text-white hover:bg-red-700 focus-visible:ring-red-500': variant === 'danger',
          'hover:bg-slate-100 text-slate-600': variant === 'ghost',
        },
        // Size
        {
          'h-8 px-3 text-sm': size === 'sm',
          'h-10 px-4 text-sm': size === 'md',
          'h-11 px-6 text-base': size === 'lg',
        },
        className
      )}
    >
      {loading && <span className="mr-2 h-4 w-4 animate-spin rounded-full border-2 border-current border-t-transparent" />}
      {children}
    </button>
  )
}
```
