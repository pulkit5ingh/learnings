# A9.1 — TypeScript with React

**Phase:** 9 — TypeScript, Accessibility & Production
**Topic:** Typing React components, hooks, events, and utility types
**Type:** Assignment

---

## Goal

Apply TypeScript to React code confidently — typing props, state, refs, event handlers, and API responses. Use utility types and discriminated unions to eliminate runtime bugs at compile time.

---

## Background

TypeScript's value in React is catching mistakes before the browser does: wrong prop types, missing null checks, incorrect event handler signatures. With `strict: true` in `tsconfig.json` you get the strongest guarantees.

Key areas:
- `FC<Props>` vs plain function component syntax
- `useState<Type>()` generic
- `useRef<HTMLElement>(null)`
- Event types: `React.ChangeEvent<HTMLInputElement>`, `React.FormEvent<HTMLFormElement>`, `React.MouseEvent<HTMLButtonElement>`
- Utility types: `Partial<T>`, `Required<T>`, `Pick<T, K>`, `Omit<T, K>`, `ReturnType<typeof fn>`
- Discriminated unions for component variants

### 2. Typing props — `FC<Props>` vs function syntax

```tsx
// Prefer explicit function declaration over React.FC<Props>
// React.FC adds an implicit children prop (pre-React 18) and hides return type issues

// Recommended approach
interface ButtonProps {
  label: string;
  variant?: 'primary' | 'secondary' | 'danger';
  disabled?: boolean;
  onClick?: (event: React.MouseEvent<HTMLButtonElement>) => void;
}

function Button({ label, variant = 'primary', disabled = false, onClick }: ButtonProps) {
  return (
    <button
      className={`btn btn-${variant}`}
      disabled={disabled}
      onClick={onClick}
    >
      {label}
    </button>
  );
}

// FC<Props> approach (also valid, slightly less preferred)
const Button2: React.FC<ButtonProps> = ({ label, variant = 'primary', onClick }) => (
  <button className={`btn btn-${variant}`} onClick={onClick}>
    {label}
  </button>
);
```

### 3. Typing `useState`

```tsx
interface User {
  id: number;
  name: string;
  email: string;
}

function UserProfile() {
  // Explicit generic when initial value is null/undefined
  const [user, setUser] = useState<User | null>(null);
  const [count, setCount] = useState(0); // inferred as number
  const [items, setItems] = useState<string[]>([]); // explicit for empty array

  const loadUser = async () => {
    const data = await fetchUser(1);
    setUser(data); // TypeScript checks data matches User
  };

  // Functional update — prev is typed correctly
  const increment = () => setCount(prev => prev + 1);

  return <div>{user?.name ?? 'Loading...'}</div>;
}
```

### 4. Typing `useRef`

```tsx
function SearchInput() {
  // Generic is the DOM element type; initial value must be null
  const inputRef = useRef<HTMLInputElement>(null);
  const timerRef = useRef<ReturnType<typeof setTimeout> | null>(null);

  const focusInput = () => {
    // TypeScript knows inputRef.current could be null (before mount)
    inputRef.current?.focus();
  };

  const startTimer = () => {
    timerRef.current = setTimeout(() => {
      console.log('done');
    }, 1000);
  };

  return <input ref={inputRef} type="text" />;
}
```

### 5. Event types

```tsx
function LoginForm() {
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');

  // ChangeEvent is generic over the element type
  const handleEmailChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setEmail(e.target.value);
  };

  // FormEvent for form submissions
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault();
    console.log({ email, password });
  };

  // KeyboardEvent for keyboard interactions
  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter') {
      // handle enter
    }
  };

  // MouseEvent with currentTarget
  const handleButtonClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    e.currentTarget.blur();
  };

  return (
    <form onSubmit={handleSubmit}>
      <input
        type="email"
        value={email}
        onChange={handleEmailChange}
        onKeyDown={handleKeyDown}
      />
      <input
        type="password"
        value={password}
        onChange={e => setPassword(e.target.value)} // inline — inferred
      />
      <button type="submit" onClick={handleButtonClick}>Login</button>
    </form>
  );
}
```

### 6. Utility types in practice

```tsx
interface Product {
  id: number;
  name: string;
  price: number;
  description: string;
  imageUrl: string;
  category: string;
  inStock: boolean;
}

// Pick — only what you need in a card
type ProductCardProps = Pick<Product, 'id' | 'name' | 'price' | 'imageUrl'>;

// Omit — everything except internal fields
type CreateProductInput = Omit<Product, 'id'>;

// Partial — all fields optional for update
type UpdateProductInput = Partial<Omit<Product, 'id'>>;

// Required — ensure optional fields are present
type CompleteProduct = Required<Product>;

// ReturnType — infer return type of a function
async function fetchProduct(id: number): Promise<Product> {
  const res = await fetch(`/api/products/${id}`);
  return res.json();
}
type FetchProductReturn = Awaited<ReturnType<typeof fetchProduct>>; // Product

// Record — typed dictionary
type CategoryMap = Record<string, Product[]>;
```

### 7. Discriminated unions for component variants

```tsx
// Instead of boolean props that conflict, use a union
type AlertVariant =
  | { variant: 'info'; message: string }
  | { variant: 'error'; message: string; code?: number }
  | { variant: 'success'; message: string; onDismiss: () => void };

function Alert(props: AlertVariant) {
  // TypeScript narrows the type inside each branch
  switch (props.variant) {
    case 'info':
      return <div className="alert-info">{props.message}</div>;

    case 'error':
      return (
        <div className="alert-error">
          {props.message}
          {props.code && <span> (code: {props.code})</span>}
        </div>
      );

    case 'success':
      return (
        <div className="alert-success">
          {props.message}
          <button onClick={props.onDismiss}>Dismiss</button>
        </div>
      );
  }
}

// Usage — TypeScript enforces correct props per variant
<Alert variant="success" message="Saved!" onDismiss={() => {}} />
<Alert variant="error" message="Failed" code={500} />
// <Alert variant="success" message="Ok" /> // Error: missing onDismiss
```

### 8. Typing custom hooks

```tsx
interface UseFetchResult<T> {
  data: T | null;
  loading: boolean;
  error: Error | null;
  refetch: () => void;
}

function useFetch<T>(url: string): UseFetchResult<T> {
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<Error | null>(null);

  const fetchData = useCallback(async () => {
    setLoading(true);
    setError(null);
    try {
      const res = await fetch(url);
      if (!res.ok) throw new Error(`HTTP ${res.status}`);
      const json: T = await res.json();
      setData(json);
    } catch (err) {
      setError(err instanceof Error ? err : new Error('Unknown error'));
    } finally {
      setLoading(false);
    }
  }, [url]);

  useEffect(() => { fetchData(); }, [fetchData]);

  return { data, loading, error, refetch: fetchData };
}

// Usage — T is inferred from the explicit type parameter
const { data: user } = useFetch<User>('/api/me');
// user is User | null — TypeScript knows this
```

### 9. Typing API responses with Zod

```tsx
import { z } from 'zod';

const UserSchema = z.object({
  id: z.number(),
  name: z.string(),
  email: z.string().email(),
  role: z.enum(['admin', 'user', 'guest']),
});

// Derive TypeScript type from Zod schema — single source of truth
type User = z.infer<typeof UserSchema>;

async function getUser(id: number): Promise<User> {
  const res = await fetch(`/api/users/${id}`);
  const raw = await res.json();
  return UserSchema.parse(raw); // throws ZodError if shape is wrong
}
```
