# P4.2 — Real-time Dashboard

**Phase:** 4 — Data Fetching & Server State
**Topic:** React Query refetchInterval, optimistic updates, error boundary integration
**Type:** Project

---

## Goal

Build a live metrics dashboard that auto-refreshes data at set intervals, supports optimistic updates for user actions, and gracefully recovers from errors with an error boundary. Practice React Query's polling and mutation patterns.

---

## What to Build

- Live metrics panel (CPU %, memory %, active users, requests/sec) — auto-refreshes every 5 s
- Alerts list — auto-refreshes every 10 s
- Acknowledge alert action with optimistic update (badge disappears instantly)
- Services health table — auto-refreshes every 15 s
- Toggle service on/off with optimistic update
- Error boundary for the entire dashboard with a retry button
- "Last updated" timestamp per panel
- Pause / resume auto-refresh toggle

```ts
// src/hooks/useAlerts.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { mockApi, type Alert } from '../lib/mockApi';

export function useAlerts(pollingEnabled: boolean) {
  const queryClient = useQueryClient();

  const alertsQuery = useQuery({
    queryKey: ['alerts'],
    queryFn: mockApi.getAlerts,
    refetchInterval: pollingEnabled ? 10000 : false,
  });

  const acknowledgeMutation = useMutation({
    mutationFn: mockApi.acknowledgeAlert,

    // Optimistic update
    onMutate: async (alertId: string) => {
      await queryClient.cancelQueries({ queryKey: ['alerts'] });

      const previous = queryClient.getQueryData<Alert[]>(['alerts']);

      queryClient.setQueryData<Alert[]>(['alerts'], old =>
        old?.map(a => a.id === alertId ? { ...a, acknowledged: true } : a)
      );

      return { previous };
    },

    // Rollback on error
    onError: (_err, _alertId, context) => {
      if (context?.previous) {
        queryClient.setQueryData(['alerts'], context.previous);
      }
    },

    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['alerts'] });
    },
  });

  return { alertsQuery, acknowledgeMutation };
}
```

```ts
// src/hooks/useServices.ts
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { mockApi, type Service } from '../lib/mockApi';

export function useServices(pollingEnabled: boolean) {
  const queryClient = useQueryClient();

  const servicesQuery = useQuery({
    queryKey: ['services'],
    queryFn: mockApi.getServices,
    refetchInterval: pollingEnabled ? 15000 : false,
  });

  const toggleMutation = useMutation({
    mutationFn: ({ id, enabled }: { id: string; enabled: boolean }) =>
      mockApi.toggleService(id, enabled),

    onMutate: async ({ id, enabled }) => {
      await queryClient.cancelQueries({ queryKey: ['services'] });
      const previous = queryClient.getQueryData<Service[]>(['services']);

      queryClient.setQueryData<Service[]>(['services'], old =>
        old?.map(s => s.id === id ? { ...s, enabled } : s)
      );

      return { previous };
    },

    onError: (_err, _vars, context) => {
      if (context?.previous) {
        queryClient.setQueryData(['services'], context.previous);
      }
    },

    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['services'] });
    },
  });

  return { servicesQuery, toggleMutation };
}
```

### Error Boundary

```tsx
// src/components/DashboardErrorBoundary.tsx
import { Component, type ReactNode } from 'react';

interface Props { children: ReactNode }
interface State { hasError: boolean; error: Error | null }

export class DashboardErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false, error: null };

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  componentDidCatch(error: Error, info: React.ErrorInfo) {
    console.error('Dashboard error:', error, info);
  }

  render() {
    if (this.state.hasError) {
      return (
        <div className="flex flex-col items-center justify-center h-screen gap-4">
          <div className="text-red-500 text-5xl">⚠️</div>
          <h2 className="text-xl font-bold text-gray-800">Dashboard unavailable</h2>
          <p className="text-gray-500">{this.state.error?.message}</p>
          <button
            onClick={() => this.setState({ hasError: false, error: null })}
            className="px-6 py-2 bg-blue-500 text-white rounded-lg hover:bg-blue-600"
          >
            Retry
          </button>
        </div>
      );
    }
    return this.props.children;
  }
}
```

```tsx
// src/components/MetricsPanel.tsx
import { useMetrics } from '../hooks/useMetrics';

const metricConfig = [
  { key: 'cpu' as const, label: 'CPU Usage', suffix: '%', warn: 70, crit: 90 },
  { key: 'memory' as const, label: 'Memory', suffix: '%', warn: 70, crit: 85 },
  { key: 'activeUsers' as const, label: 'Active Users', suffix: '', warn: Infinity, crit: Infinity },
  { key: 'requestsPerSec' as const, label: 'Req/sec', suffix: '', warn: Infinity, crit: Infinity },
];

export function MetricsPanel({ pollingEnabled }: { pollingEnabled: boolean }) {
  const { data, isFetching, dataUpdatedAt } = useMetrics(pollingEnabled);

  return (
    <section>
      <div className="flex items-center justify-between mb-3">
        <h2 className="font-bold text-gray-700">Live Metrics</h2>
        <span className="text-xs text-gray-400 flex items-center gap-1">
          {isFetching && <span className="animate-spin">⟳</span>}
          {dataUpdatedAt ? `Updated ${new Date(dataUpdatedAt).toLocaleTimeString()}` : ''}
        </span>
      </div>
      <div className="grid grid-cols-4 gap-4">
        {metricConfig.map(({ key, label, suffix, warn, crit }) => {
          const value = data?.[key] ?? 0;
          const color = value >= crit ? 'text-red-600' : value >= warn ? 'text-yellow-600' : 'text-green-600';
          return (
            <div key={key} className="bg-white rounded-xl p-4 shadow">
              <p className="text-xs text-gray-500">{label}</p>
              <p className={`text-3xl font-bold ${color}`}>
                {value}{suffix}
              </p>
            </div>
          );
        })}
      </div>
    </section>
  );
}
```

### App

```tsx
// src/App.tsx
import { useState } from 'react';
import { QueryClientProvider } from '@tanstack/react-query';
import { ReactQueryDevtools } from '@tanstack/react-query-devtools';
import { queryClient } from './lib/queryClient';
import { DashboardErrorBoundary } from './components/DashboardErrorBoundary';
import { MetricsPanel } from './components/MetricsPanel';
import { AlertsList } from './components/AlertsList';
import { ServicesTable } from './components/ServicesTable';

function Dashboard() {
  const [pollingEnabled, setPollingEnabled] = useState(true);

  return (
    <div className="min-h-screen bg-gray-100 p-6 space-y-6">
      <div className="flex items-center justify-between">
        <h1 className="text-2xl font-bold text-gray-800">Operations Dashboard</h1>
        <button
          onClick={() => setPollingEnabled(e => !e)}
          className={`px-4 py-2 rounded-lg text-sm font-semibold transition ${
            pollingEnabled ? 'bg-green-100 text-green-700' : 'bg-gray-200 text-gray-600'
          }`}
        >
          {pollingEnabled ? '⏸ Pause refresh' : '▶ Resume refresh'}
        </button>
      </div>

      <MetricsPanel pollingEnabled={pollingEnabled} />
      <AlertsList pollingEnabled={pollingEnabled} />
      <ServicesTable pollingEnabled={pollingEnabled} />
    </div>
  );
}

export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <DashboardErrorBoundary>
        <Dashboard />
      </DashboardErrorBoundary>
      <ReactQueryDevtools initialIsOpen={false} />
    </QueryClientProvider>
  );
}
```
