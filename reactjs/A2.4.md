## Goal
Learn to extract reusable stateful logic from components into custom hooks, and understand the rules that govern where hooks can be used.

## Background

### What Makes a Custom Hook?
A custom hook is any function whose name starts with `use` and that calls other hooks inside it. The `use` prefix is a convention that tells React (and linting tools) that this function follows the rules of hooks. You can call `useState`, `useEffect`, `useContext`, and other hooks inside custom hooks, just as you can inside components.

### Why Extract into Custom Hooks?
Custom hooks let you share stateful logic between components without rendering anything. Unlike component composition (which shares UI), custom hooks share behavior. The same `useFetch` hook can be used in a dozen components. The same `useLocalStorage` hook can be used wherever persistence is needed. You aren't duplicating the logic — you're centralizing it.

### Rules of Hooks
1. Only call hooks at the top level — never inside conditions, loops, or nested functions.
2. Only call hooks from React function components or custom hooks — never from plain JavaScript functions.
These rules exist because React identifies hooks by call order. Violating them causes bugs that are very hard to debug. The `eslint-plugin-react-hooks` package enforces these rules automatically.

## Tasks

### Task 1 — useToggle

```tsx
// src/hooks/useToggle.ts
import { useState, useCallback } from 'react'

export function useToggle(initialValue = false) {
  const [value, setValue] = useState(initialValue)

  const toggle = useCallback(() => setValue(v => !v), [])
  const setTrue = useCallback(() => setValue(true), [])
  const setFalse = useCallback(() => setValue(false), [])

  return { value, toggle, setTrue, setFalse } as const
}

// Usage:
// const { value: isOpen, toggle, setFalse: close } = useToggle()
// <button onClick={toggle}>Toggle</button>
// <Modal isOpen={isOpen} onClose={close} />
```

### Task 2 — useLocalStorage

```tsx
// src/hooks/useLocalStorage.ts
import { useState, useCallback } from 'react'

export function useLocalStorage<T>(key: string, initialValue: T) {
  const [storedValue, setStoredValue] = useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key)
      return item ? (JSON.parse(item) as T) : initialValue
    } catch {
      console.warn(`Error reading localStorage key "${key}"`)
      return initialValue
    }
  })

  const setValue = useCallback(
    (value: T | ((prev: T) => T)) => {
      try {
        const valueToStore = value instanceof Function ? value(storedValue) : value
        setStoredValue(valueToStore)
        window.localStorage.setItem(key, JSON.stringify(valueToStore))
      } catch {
        console.warn(`Error setting localStorage key "${key}"`)
      }
    },
    [key, storedValue]
  )

  const removeValue = useCallback(() => {
    try {
      window.localStorage.removeItem(key)
      setStoredValue(initialValue)
    } catch {
      console.warn(`Error removing localStorage key "${key}"`)
    }
  }, [key, initialValue])

  return [storedValue, setValue, removeValue] as const
}
```

### Task 3 — useDebounce

```tsx
// src/hooks/useDebounce.ts
import { useState, useEffect } from 'react'

export function useDebounce<T>(value: T, delay = 300): T {
  const [debouncedValue, setDebouncedValue] = useState<T>(value)

  useEffect(() => {
    const timer = setTimeout(() => {
      setDebouncedValue(value)
    }, delay)

    // Cleanup: cancel the timer if value changes before delay completes
    return () => clearTimeout(timer)
  }, [value, delay])

  return debouncedValue
}

// Usage in a search component:
// const [query, setQuery] = useState('')
// const debouncedQuery = useDebounce(query, 400)
// useEffect(() => { fetchResults(debouncedQuery) }, [debouncedQuery])
// → API is called only after user stops typing for 400ms
```

### Task 4 — useFetch

```tsx
// src/hooks/useFetch.ts
import { useState, useEffect, useCallback } from 'react'

interface FetchState<T> {
  data: T | null
  loading: boolean
  error: string | null
}

interface UseFetchReturn<T> extends FetchState<T> {
  refetch: () => void
}

export function useFetch<T>(url: string | null, options?: RequestInit): UseFetchReturn<T> {
  const [state, setState] = useState<FetchState<T>>({
    data: null,
    loading: false,
    error: null,
  })
  const [refreshCount, setRefreshCount] = useState(0)

  const refetch = useCallback(() => setRefreshCount(c => c + 1), [])

  useEffect(() => {
    if (!url) return  // null URL = skip fetch

    let cancelled = false
    const controller = new AbortController()  // more modern than cancel flag

    setState(prev => ({ ...prev, loading: true, error: null }))

    async function load() {
      try {
        const res = await fetch(url, { ...options, signal: controller.signal })
        if (!res.ok) throw new Error(`HTTP ${res.status}: ${res.statusText}`)
        const data: T = await res.json()
        if (!cancelled) setState({ data, loading: false, error: null })
      } catch (err) {
        if (err instanceof Error && err.name === 'AbortError') return  // cancelled
        if (!cancelled)
          setState({ data: null, loading: false, error: err instanceof Error ? err.message : 'Fetch failed' })
      }
    }

    load()

    return () => {
      cancelled = true
      controller.abort()
    }
  }, [url, refreshCount])  // re-fetch when URL changes or refetch() is called

  return { ...state, refetch }
}
```

### Task 5 — useWindowSize

```tsx
// src/hooks/useWindowSize.ts
import { useState, useEffect } from 'react'

interface WindowSize {
  width: number
  height: number
}

export function useWindowSize(): WindowSize {
  const [size, setSize] = useState<WindowSize>({
    width: window.innerWidth,
    height: window.innerHeight,
  })

  useEffect(() => {
    function handleResize() {
      setSize({ width: window.innerWidth, height: window.innerHeight })
    }

    window.addEventListener('resize', handleResize)
    return () => window.removeEventListener('resize', handleResize)
  }, [])

  return size
}
```

### Task 6 — Compose Custom Hooks in a Component

```tsx
// src/components/SearchPage.tsx
import { useState } from 'react'
import { useDebounce } from '../hooks/useDebounce'
import { useFetch } from '../hooks/useFetch'
import { useLocalStorage } from '../hooks/useLocalStorage'
import { useToggle } from '../hooks/useToggle'

interface SearchResult {
  id: number
  title: string
  body: string
}

export function SearchPage() {
  const [query, setQuery] = useState('')
  const debouncedQuery = useDebounce(query, 400)

  // Fetch only when debouncedQuery has content
  const url = debouncedQuery
    ? `https://jsonplaceholder.typicode.com/posts?q=${encodeURIComponent(debouncedQuery)}&_limit=5`
    : null

  const { data: results, loading, error, refetch } = useFetch<SearchResult[]>(url)

  const [recentSearches, setRecentSearches] = useLocalStorage<string[]>('recent-searches', [])
  const { value: showRecent, setTrue: openRecent, setFalse: closeRecent } = useToggle()

  function handleSearch(q: string) {
    setQuery(q)
    closeRecent()
    if (q && !recentSearches.includes(q)) {
      setRecentSearches(prev => [q, ...prev].slice(0, 5))
    }
  }

  return (
    <div style={{ maxWidth: 500 }}>
      <div style={{ position: 'relative' }}>
        <input
          type="search"
          value={query}
          onChange={e => setQuery(e.target.value)}
          onFocus={openRecent}
          placeholder="Search posts..."
          style={{ width: '100%', padding: '8px 12px', fontSize: 16 }}
        />

        {showRecent && recentSearches.length > 0 && (
          <div style={{ position: 'absolute', top: '100%', left: 0, right: 0, background: 'white', border: '1px solid #e5e7eb', borderRadius: '0 0 6px 6px', zIndex: 10 }}>
            {recentSearches.map(s => (
              <button
                key={s}
                onClick={() => handleSearch(s)}
                style={{ display: 'block', width: '100%', textAlign: 'left', padding: '8px 12px', background: 'none', border: 'none', cursor: 'pointer' }}
              >
                🕐 {s}
              </button>
            ))}
          </div>
        )}
      </div>

      {loading && <p style={{ color: '#6b7280' }}>Searching for "{debouncedQuery}"...</p>}
      {error && <p style={{ color: '#ef4444' }}>{error} <button onClick={refetch}>Retry</button></p>}
      {results && (
        <ul style={{ listStyle: 'none', padding: 0 }}>
          {results.map(r => (
            <li key={r.id} style={{ padding: '0.75rem 0', borderBottom: '1px solid #f0f0f0' }}>
              <strong>{r.title}</strong>
              <p style={{ margin: '4px 0 0', fontSize: 13, color: '#6b7280' }}>{r.body.slice(0, 80)}…</p>
            </li>
          ))}
        </ul>
      )}
      {results?.length === 0 && !loading && (
        <p style={{ color: '#9ca3af', textAlign: 'center' }}>No results for "{debouncedQuery}"</p>
      )}
    </div>
  )
}
```
