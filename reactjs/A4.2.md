## Goal
Perform create, update, and delete operations with useMutation, then keep the cache in sync using invalidation and optimistic updates.

## Background

### useMutation
While `useQuery` is for reading data, `useMutation` is for writing data (POST, PUT, PATCH, DELETE). It returns a `mutate` function you call when the user takes an action. Callbacks (`onSuccess`, `onError`, `onSettled`) let you react to the result — update the cache, show a notification, or navigate.

### Two Strategies for Cache Updates
**Invalidation**: After a mutation succeeds, call `queryClient.invalidateQueries(queryKey)`. This marks the cached data as stale and triggers a refetch. Simple and safe, but the UI momentarily shows old data before refetching. Best for list mutations where you want the server to reorder or recalculate.

**Direct update with setQueryData**: Manually update the cache without a network request. Instant UI update, no extra fetch. Best when you know exactly what the new data should look like (e.g., after adding a known item). Use `setQueryData(key, updater)` where `updater` is a function from old data to new data.

### Optimistic Updates
An optimistic update assumes the mutation will succeed and updates the UI immediately, before the server responds. If the mutation fails, you roll back. This makes the UI feel instant. React Query provides `onMutate` (run before the request), `onError` (run if it fails — restore previous cache), and `onSettled` (run regardless — refetch to ensure consistency).

// src/api/posts.ts — add mutation functions
export async function createPost(data: Omit<Post, 'id'>): Promise<Post> {
  const res = await fetch('https://jsonplaceholder.typicode.com/posts', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data),
  })
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
  return res.json()
}

export async function updatePost(id: number, data: Partial<Post>): Promise<Post> {
  const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(data),
  })
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
  return res.json()
}

export async function deletePost(id: number): Promise<void> {
  const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`, { method: 'DELETE' })
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
}

### Task 2 — Optimistic Update (Toggle Like)

```tsx
// src/components/LikeButton.tsx
import { useMutation, useQueryClient } from '@tanstack/react-query'

interface LikeButtonProps {
  postId: number
  liked: boolean
  likeCount: number
}

async function toggleLike(postId: number, liked: boolean): Promise<{ liked: boolean; count: number }> {
  // Simulate API call
  await new Promise(r => setTimeout(r, 500))
  if (Math.random() < 0.2) throw new Error('Server error')  // 20% failure for demo
  return { liked: !liked, count: liked ? likeCount - 1 : likeCount + 1 }
}

export function LikeButton({ postId, liked, likeCount }: LikeButtonProps) {
  const queryClient = useQueryClient()
  const queryKey = ['posts', postId, 'like']

  const { mutate, isPending } = useMutation({
    mutationFn: () => toggleLike(postId, liked),

    onMutate: async () => {
      // 1. Cancel any in-flight queries for this data
      await queryClient.cancelQueries({ queryKey })

      // 2. Snapshot the current value for rollback
      const previous = queryClient.getQueryData(queryKey)

      // 3. Optimistically update the cache
      queryClient.setQueryData(queryKey, { liked: !liked, count: liked ? likeCount - 1 : likeCount + 1 })

      // 4. Return context for rollback
      return { previous }
    },

    onError: (_err, _vars, context) => {
      // Rollback to snapshot on error
      if (context?.previous !== undefined) {
        queryClient.setQueryData(queryKey, context.previous)
      }
    },

    onSettled: () => {
      // Always refetch to ensure server consistency
      queryClient.invalidateQueries({ queryKey })
    },
  })

  return (
    <button
      onClick={() => mutate()}
      disabled={isPending}
      style={{
        background: liked ? '#fce7f3' : '#f9fafb',
        border: `1px solid ${liked ? '#f9a8d4' : '#e5e7eb'}`,
        borderRadius: 20,
        padding: '4px 12px',
        cursor: isPending ? 'not-allowed' : 'pointer',
        color: liked ? '#be185d' : '#374151',
        display: 'flex',
        alignItems: 'center',
        gap: 4,
        fontSize: 14,
      }}
    >
      {liked ? '♥' : '♡'} {likeCount}
    </button>
  )
}
```

### Task 3 — Delete with Cache Update

```tsx
// src/components/PostActions.tsx
import { useMutation, useQueryClient } from '@tanstack/react-query'
import { useNavigate } from 'react-router-dom'
import { deletePost, type Post } from '../api/posts'
import { queryKeys } from '../api/queryKeys'

export function DeletePostButton({ postId }: { postId: number }) {
  const queryClient = useQueryClient()
  const navigate = useNavigate()

  const { mutate, isPending } = useMutation({
    mutationFn: () => deletePost(postId),

    onSuccess: () => {
      // Remove this specific post from the cache
      queryClient.removeQueries({ queryKey: queryKeys.post(postId) })

      // Update the posts list without refetching
      queryClient.setQueryData(queryKeys.posts(), (old: Post[] | undefined) =>
        old ? old.filter(p => p.id !== postId) : []
      )

      navigate('/posts', { replace: true })
    },
  })

  function handleDelete() {
    if (window.confirm('Delete this post?')) {
      mutate()
    }
  }

  return (
    <button
      onClick={handleDelete}
      disabled={isPending}
      style={{ color: '#ef4444', background: 'none', border: '1px solid #ef4444', borderRadius: 4, padding: '6px 12px', cursor: 'pointer' }}
    >
      {isPending ? 'Deleting...' : 'Delete'}
    </button>
  )
}
```
