## Goal

## Background

### What is a Component?
A React component is a JavaScript function that returns JSX — a syntax extension that looks like HTML but compiles to `React.createElement()` calls. Components are the building blocks of every React UI. They can be as small as a single button or as large as an entire page.

In React 18+, all components are written as plain functions. Class components still work but are considered legacy. Every component name must start with a capital letter — lowercase names are treated as HTML elements.

### JSX Rules
JSX has stricter rules than HTML. Every element must be closed (including `<br />` and `<img />`). Adjacent elements must be wrapped in a single parent — either a real element like `<div>` or a React fragment (`<>...</>`). Fragments avoid adding extra DOM nodes. Attributes use camelCase (`className` instead of `class`, `onClick` instead of `onclick`). JavaScript expressions go inside curly braces `{}`.

### Conditional Rendering & Lists
React has no template syntax like `v-if` or `*ngIf`. Instead, you use JavaScript directly. The `&&` operator renders the right side only when the left side is truthy. The ternary operator renders one of two things. An early return exits the component before rendering anything. For lists, you use `.map()` and every rendered element needs a stable, unique `key` prop — this is how React tracks which items changed, added, or removed.

// Rule 1: Single root — use fragment to avoid extra div
function FragmentDemo() {
  return (
    <>
      <h2>Title</h2>

    </>
  )

// Rule 2: className instead of class
function StyledBox() {
  const isActive = true
  return (
    <div
      className={isActive ? 'box active' : 'box'}
      style={{ padding: '1rem', background: isActive ? 'lightblue' : 'white' }}
    >
      I am {isActive ? 'active' : 'inactive'}
    </div>
  )

// Rule 3: self-closing tags must close
function SelfClosing() {
  return (
    <div>

    </div>
  )
}

  return (
    <>
      <FragmentDemo />
      <StyledBox />
      <SelfClosing />
    </>

### Task 3 — Conditional Rendering

```tsx
// src/components/ConditionalDemo.tsx

interface UserCardProps {
  name: string
  isOnline: boolean
  role?: string
}

// Pattern 1: && operator
function OnlineIndicator({ isOnline }: { isOnline: boolean }) {
  return (
    <span>
      {isOnline && <span style={{ color: 'green' }}>● Online</span>}
      {!isOnline && <span style={{ color: 'gray' }}>● Offline</span>}
    </span>
  )
}

// Pattern 2: ternary
function Badge({ role }: { role?: string }) {
  return role ? (
    <span style={{ background: '#e0e0ff', padding: '2px 8px', borderRadius: 4 }}>
      {role}
    </span>
  ) : null
}

// Pattern 3: early return
function UserCard({ name, isOnline, role }: UserCardProps) {
  if (!name) return null  // early return

  return (
    <div style={{ border: '1px solid #ccc', padding: '1rem', margin: '0.5rem 0' }}>
      <strong>{name}</strong>
      <OnlineIndicator isOnline={isOnline} />
      <Badge role={role} />
    </div>
  )
}

export default function ConditionalDemo() {
  return (
    <div>
      <UserCard name="Alice" isOnline={true} role="Admin" />
      <UserCard name="Bob" isOnline={false} />
      <UserCard name="" isOnline={true} />  {/* renders nothing */}
    </div>
  )
}
```

### Task 4 — Lists and Keys

```tsx
// src/components/ListDemo.tsx

interface Task {
  id: number
  title: string
  done: boolean
}

const tasks: Task[] = [
  { id: 1, title: 'Learn React components', done: true },
  { id: 2, title: 'Understand JSX', done: true },
  { id: 3, title: 'Practice conditional rendering', done: false },
  { id: 4, title: 'Build a list with keys', done: false },
]

function TaskItem({ task }: { task: Task }) {
  return (
    <li
      style={{
        textDecoration: task.done ? 'line-through' : 'none',
        color: task.done ? '#999' : 'inherit',
        padding: '0.25rem 0',
      }}
    >
      {task.done ? '✓' : '○'} {task.title}
    </li>
  )
}

function TaskList({ items }: { items: Task[] }) {
  if (items.length === 0) {
    return <p>No tasks found.</p>  // empty state
  }

  return (
    <ul style={{ listStyle: 'none', padding: 0 }}>
      {items.map(task => (
        <TaskItem key={task.id} task={task} />
        // key must be on the outermost element returned by .map()
        // key must be a stable, unique string or number — not the array index
      ))}
    </ul>
  )
}

export default function ListDemo() {
  const pending = tasks.filter(t => !t.done)
  const completed = tasks.filter(t => t.done)

  return (
    <div>
      <h3>Pending ({pending.length})</h3>
      <TaskList items={pending} />

      <h3>Completed ({completed.length})</h3>
      <TaskList items={completed} />
    </div>
  )
}
```

### Task 5 — Nested Components

```tsx
// src/components/ProfileCard.tsx

interface Profile {
  id: number
  name: string
  title: string
  skills: string[]
  avatarUrl: string
}

function Avatar({ src, name }: { src: string; name: string }) {
  return (
    <img
      src={src}
      alt={`${name}'s avatar`}
      style={{ width: 64, height: 64, borderRadius: '50%', objectFit: 'cover' }}
    />
  )
}

function SkillTag({ skill }: { skill: string }) {
  return (
    <span
      style={{
        display: 'inline-block',
        background: '#f0f0f0',
        padding: '2px 10px',
        borderRadius: 12,
        fontSize: 13,
        margin: '2px',
      }}
    >
      {skill}
    </span>
  )
}

function ProfileCard({ profile }: { profile: Profile }) {
  return (
    <div style={{ border: '1px solid #ddd', borderRadius: 8, padding: '1rem', maxWidth: 300 }}>
      <Avatar src={profile.avatarUrl} name={profile.name} />
      <h3 style={{ margin: '0.5rem 0 0.25rem' }}>{profile.name}</h3>
      <p style={{ color: '#666', margin: '0 0 0.75rem', fontSize: 14 }}>{profile.title}</p>
      <div>
        {profile.skills.map(skill => (
          <SkillTag key={skill} skill={skill} />
        ))}
      </div>
    </div>
  )
}

const profiles: Profile[] = [
  {
    id: 1,
    name: 'Sarah Chen',
    title: 'Frontend Engineer',
    skills: ['React', 'TypeScript', 'CSS'],
    avatarUrl: 'https://i.pravatar.cc/64?img=1',
  },
  {
    id: 2,
    name: 'Marcus Davis',
    title: 'Full Stack Developer',
    skills: ['Node.js', 'React', 'PostgreSQL'],
    avatarUrl: 'https://i.pravatar.cc/64?img=2',
  },
]

export default function ProfileCardList() {
  return (
    <div style={{ display: 'flex', gap: '1rem', flexWrap: 'wrap' }}>
      {profiles.map(profile => (
        <ProfileCard key={profile.id} profile={profile} />
      ))}
    </div>
  )
}
```
