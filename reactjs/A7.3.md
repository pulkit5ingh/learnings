## Goal
Reduce initial bundle size by splitting code into chunks that load on demand, and analyze the bundle to find optimization opportunities.

## Background

### Why Code Splitting Matters
A typical React app with all pages bundled into one file might produce a 500kB+ JavaScript bundle. The user downloads ALL of this on the first visit, even pages they never navigate to. Code splitting divides the bundle into smaller chunks. The initial chunk only contains what's needed for the first render. Other chunks load on demand, when the user navigates to that page or triggers that feature.

### Route-Based Splitting
The highest-impact form of code splitting is at the route level. Each page becomes a separate chunk. When the user visits a route, that chunk downloads — but it's typically much smaller than the full app bundle, and subsequent visits use the cached chunk.

### Vite Bundle Analysis
`rollup-plugin-visualizer` generates an interactive treemap showing exactly what's in each chunk and how much it contributes to bundle size. Large dependencies (PDF libraries, charting libraries, rich text editors) that you use on one page are good candidates for dynamic imports.

### Task 1 — React.lazy and Suspense

```tsx
// src/router.tsx
import { lazy, Suspense, type ReactNode } from 'react'
import { createBrowserRouter } from 'react-router-dom'

// Eagerly loaded — critical for first render
import { RootLayout } from './layouts/RootLayout'
import { HomePage } from './pages/HomePage'
import { ErrorPage } from './pages/ErrorPage'

// Lazily loaded — loaded on first visit to that route
// Vite creates separate chunks for each of these
const DashboardPage = lazy(() => import('./pages/DashboardPage'))
const SettingsPage = lazy(() => import('./pages/SettingsPage'))
const ReportsPage = lazy(() => import('./pages/ReportsPage'))
const AdminPage = lazy(() => import('./pages/AdminPage'))

function PageSuspense({ children }: { children: ReactNode }) {
  return (
    <Suspense
      fallback={
        <div style={{ display: 'flex', justifyContent: 'center', alignItems: 'center', minHeight: 200 }}>
          <div style={{ display: 'flex', flexDirection: 'column', alignItems: 'center', gap: 12 }}>
            <div
              style={{
                width: 36,
                height: 36,
                border: '3px solid #e5e7eb',
                borderTopColor: '#3b82f6',
                borderRadius: '50%',
                animation: 'spin 0.8s linear infinite',
              }}
            />
            <p style={{ color: '#6b7280', fontSize: 14 }}>Loading...</p>
          </div>
        </div>
      }
    >
      {children}
    </Suspense>
  )
}

export const router = createBrowserRouter([
  {
    path: '/',
    element: <RootLayout />,
    errorElement: <ErrorPage />,
    children: [
      { index: true, element: <HomePage /> },
      { path: 'dashboard', element: <PageSuspense><DashboardPage /></PageSuspense> },
      { path: 'settings', element: <PageSuspense><SettingsPage /></PageSuspense> },
      { path: 'reports', element: <PageSuspense><ReportsPage /></PageSuspense> },
      { path: 'admin', element: <PageSuspense><AdminPage /></PageSuspense> },
    ],
  },
])
```

### Task 2 — Dynamic Import for Heavy Dependencies

```tsx
// src/components/ChartContainer.tsx
import { lazy, Suspense, useState } from 'react'

// Chart library is heavy — only load when user needs it
const LazyChart = lazy(() =>
  // Named export: import { Chart } from './Chart'
  import('./HeavyChart').then(module => ({ default: module.HeavyChart }))
)

export function ChartContainer() {
  const [showChart, setShowChart] = useState(false)

  return (
    <div>
      <button onClick={() => setShowChart(true)}>Load Chart</button>

      {showChart && (
        <Suspense fallback={<div style={{ height: 300, background: '#f9fafb', display: 'flex', alignItems: 'center', justifyContent: 'center' }}>Loading chart...</div>}>
          <LazyChart />
        </Suspense>
      )}
    </div>
  )
}

// Heavy component lives in a separate file — only loaded when needed
// src/HeavyChart.tsx
// import { LineChart, ... } from 'recharts'  ← large dependency
// export function HeavyChart() { ... }
```

### Task 3 — Bundle Analysis

```bash

pnpm add -D rollup-plugin-visualizer
```

```ts
// vite.config.ts
import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import { visualizer } from 'rollup-plugin-visualizer'

export default defineConfig({
  plugins: [
    react(),
    visualizer({
      filename: 'dist/stats.html',
      open: true,         // auto-open browser after build
      gzipSize: true,     // show gzip sizes
      brotliSize: true,   // show brotli sizes
    }),
  ],
  build: {
    // Manually control chunk splitting
    rollupOptions: {
      output: {
        manualChunks: {
          // Put all React core into one chunk
          'vendor-react': ['react', 'react-dom', 'react-router-dom'],
          // Put React Query into its own chunk
          'vendor-query': ['@tanstack/react-query'],
          // Heavy UI libraries
          'vendor-ui': ['framer-motion', '@radix-ui/react-dialog'],
        },
      },
    },
  },
})
```

### Task 4 — Preloading Critical Chunks

```tsx
// src/components/PreloadLink.tsx
import { Link } from 'react-router-dom'

// Preload a lazy chunk when the user hovers the nav link
// The chunk is fetched before they click, so navigation feels instant
export function PreloadNavLink({ to, preload, children }: {
  to: string
  preload: () => Promise<unknown>
  children: React.ReactNode
}) {
  return (
    <Link
      to={to}
      onMouseEnter={() => {
        // Trigger the dynamic import to start loading the chunk
        preload().catch(() => {})  // ignore preload errors
      }}
    >
      {children}
    </Link>
  )
}

// Usage in nav:
// <PreloadNavLink
//   to="/dashboard"
//   preload={() => import('./pages/DashboardPage')}
// >
//   Dashboard
// </PreloadNavLink>
```
