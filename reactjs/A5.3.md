**Type:** Assignment

## Goal
Handle advanced form patterns: repeating fields, nested field groups, multi-step wizards, and fields that appear based on other field values.

## Background

### useFieldArray
When users can add/remove items in a form (e.g., adding multiple phone numbers, work experiences, or cart items), `useFieldArray` manages a dynamic list of fields. It provides `fields` (the current array), `append`, `prepend`, `remove`, `move`, and `swap`. Each field has a stable `id` property for use as a key — don't use the array index as a key in dynamic lists.

### Multi-Step Forms
A multi-step form breaks a long form into pages. Each step validates only its own fields before advancing. The most common approach is to store `currentStep` in state and conditionally render step components. Share the same `useForm` instance across steps — all steps use the same form state and submitting on the last step submits everything. Alternatively, manage each step's data in `useState` and merge at the end.

### Nested Objects in Forms
React Hook Form supports nested field names using dot notation: `register('address.city')`. Use `z.object()` with nested schemas in Zod to validate nested objects.

### Task 1 — Dynamic Fields with useFieldArray

```tsx
// src/components/WorkExperienceForm.tsx
import { useForm, useFieldArray } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'

const experienceSchema = z.object({
  experiences: z
    .array(
      z.object({
        company: z.string().min(1, 'Company name required'),
        role: z.string().min(1, 'Role required'),
        startYear: z.string().regex(/^\d{4}$/, 'Enter a 4-digit year'),
        endYear: z.string().optional(),
        current: z.boolean(),
        description: z.string().optional(),
      })
    )
    .min(1, 'Add at least one experience'),
})

type ExperienceValues = z.infer<typeof experienceSchema>

export function WorkExperienceForm() {
  const { register, control, handleSubmit, watch, formState: { errors } } = useForm<ExperienceValues>({
    resolver: zodResolver(experienceSchema),
    defaultValues: {
      experiences: [{ company: '', role: '', startYear: '', current: false }],
    },
  })

  const { fields, append, remove } = useFieldArray({
    control,
    name: 'experiences',
  })

  function onSubmit(data: ExperienceValues) {
    console.log('Experience:', JSON.stringify(data, null, 2))
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ maxWidth: 600 }}>
      <h3>Work Experience</h3>

      {fields.map((field, index) => {
        const isCurrent = watch(`experiences.${index}.current`)
        const hasError = (key: string) => !!(errors.experiences?.[index] as Record<string, unknown>)?.[key]

        return (
          <div key={field.id} style={{ border: '1px solid #e5e7eb', borderRadius: 8, padding: '1rem', marginBottom: '1rem', position: 'relative' }}>
            <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: '0.75rem' }}>
              <div>
                <label>Company *</label>
                <input
                  {...register(`experiences.${index}.company`)}
                  style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px', border: `1px solid ${hasError('company') ? '#ef4444' : '#d1d5db'}` }}
                />
              </div>
              <div>
                <label>Role *</label>
                <input
                  {...register(`experiences.${index}.role`)}
                  style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px', border: `1px solid ${hasError('role') ? '#ef4444' : '#d1d5db'}` }}
                />
              </div>
              <div>
                <label>Start Year *</label>
                <input {...register(`experiences.${index}.startYear`)} placeholder="2020" style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px' }} />
              </div>
              {!isCurrent && (
                <div>
                  <label>End Year</label>
                  <input {...register(`experiences.${index}.endYear`)} placeholder="2023" style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px' }} />
                </div>
              )}
            </div>

            <label style={{ display: 'flex', gap: 8, alignItems: 'center', marginTop: '0.75rem' }}>
              <input type="checkbox" {...register(`experiences.${index}.current`)} />
              Currently working here
            </label>

            {fields.length > 1 && (
              <button
                type="button"
                onClick={() => remove(index)}
                style={{ position: 'absolute', top: 8, right: 8, background: 'none', border: 'none', cursor: 'pointer', color: '#ef4444', fontSize: 18 }}
              >
                ×
              </button>
            )}
          </div>
        )
      })}

      {errors.experiences?.root && (
        <p style={{ color: '#ef4444', fontSize: 13 }}>{errors.experiences.root.message}</p>
      )}

      <div style={{ display: 'flex', gap: 8 }}>
        <button
          type="button"
          onClick={() => append({ company: '', role: '', startYear: '', current: false })}
          style={{ padding: '8px 16px', background: '#f0f9ff', border: '1px solid #bae6fd', borderRadius: 4, cursor: 'pointer' }}
        >
          + Add Experience
        </button>
        <button type="submit" style={{ padding: '8px 16px', background: '#3b82f6', color: 'white', border: 'none', borderRadius: 4, cursor: 'pointer' }}>
          Save
        </button>
      </div>
    </form>
  )
}
```

### Task 2 — Multi-Step Form

```tsx
// src/components/MultiStepForm.tsx
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'
import { useState } from 'react'

const step1Schema = z.object({
  firstName: z.string().min(1, 'Required'),
  lastName: z.string().min(1, 'Required'),
  email: z.string().email('Valid email required'),
})

const step2Schema = z.object({
  plan: z.enum(['free', 'pro', 'enterprise']),
  billingCycle: z.enum(['monthly', 'yearly']),
})

const step3Schema = z.object({
  cardNumber: z.string().regex(/^\d{16}$/, '16-digit card number'),
  expiry: z.string().regex(/^\d{2}\/\d{2}$/, 'MM/YY format'),
  cvv: z.string().regex(/^\d{3,4}$/, '3-4 digit CVV'),
})

const fullSchema = step1Schema.merge(step2Schema).merge(step3Schema)
type FormValues = z.infer<typeof fullSchema>

const STEP_SCHEMAS = [step1Schema, step2Schema, step3Schema]
const STEP_LABELS = ['Account', 'Plan', 'Payment']

export function MultiStepForm() {
  const [step, setStep] = useState(0)
  const [submitted, setSubmitted] = useState(false)

  const {
    register,
    handleSubmit,
    trigger,
    formState: { errors, isSubmitting },
    getValues,
  } = useForm<FormValues>({
    resolver: zodResolver(fullSchema),
    mode: 'onBlur',
  })

  async function nextStep() {
    // Validate only current step's fields before advancing
    const stepSchema = STEP_SCHEMAS[step]
    const fields = Object.keys(stepSchema.shape) as (keyof FormValues)[]
    const valid = await trigger(fields)
    if (valid) setStep(s => s + 1)
  }

  async function onSubmit(data: FormValues) {
    await new Promise(r => setTimeout(r, 1000))
    console.log('Final form data:', data)
    setSubmitted(true)
  }

  if (submitted) return <div style={{ textAlign: 'center', padding: '2rem' }}><h2>✓ Account created!</h2></div>

  return (
    <div style={{ maxWidth: 480 }}>
      {/* Progress indicator */}
      <div style={{ display: 'flex', gap: 8, marginBottom: '2rem' }}>
        {STEP_LABELS.map((label, i) => (
          <div key={i} style={{ flex: 1, textAlign: 'center' }}>
            <div style={{ width: 32, height: 32, borderRadius: '50%', border: '2px solid', borderColor: i <= step ? '#3b82f6' : '#d1d5db', background: i < step ? '#3b82f6' : 'white', color: i < step ? 'white' : i === step ? '#3b82f6' : '#9ca3af', display: 'flex', alignItems: 'center', justifyContent: 'center', margin: '0 auto 4px', fontWeight: 600, fontSize: 14 }}>
              {i < step ? '✓' : i + 1}
            </div>
            <span style={{ fontSize: 12, color: i === step ? '#3b82f6' : '#6b7280' }}>{label}</span>
          </div>
        ))}
      </div>

      <form onSubmit={handleSubmit(onSubmit)}>
        {step === 0 && (
          <div style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem' }}>
            <h3>Account Info</h3>
            <input {...register('firstName')} placeholder="First name" style={{ padding: '8px 10px', border: `1px solid ${errors.firstName ? '#ef4444' : '#d1d5db'}` }} />
            {errors.firstName && <p style={{ color: '#ef4444', fontSize: 12, margin: 0 }}>{errors.firstName.message}</p>}
            <input {...register('lastName')} placeholder="Last name" style={{ padding: '8px 10px', border: `1px solid ${errors.lastName ? '#ef4444' : '#d1d5db'}` }} />
            <input {...register('email')} type="email" placeholder="Email" style={{ padding: '8px 10px', border: `1px solid ${errors.email ? '#ef4444' : '#d1d5db'}` }} />
            {errors.email && <p style={{ color: '#ef4444', fontSize: 12, margin: 0 }}>{errors.email.message}</p>}
          </div>
        )}

        {step === 1 && (
          <div style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem' }}>
            <h3>Choose Plan</h3>
            {['free', 'pro', 'enterprise'].map(plan => (
              <label key={plan} style={{ display: 'flex', gap: 8, alignItems: 'center', padding: '0.75rem', border: '1px solid #e5e7eb', borderRadius: 6 }}>
                <input type="radio" value={plan} {...register('plan')} />
                <span style={{ textTransform: 'capitalize', fontWeight: 500 }}>{plan}</span>
              </label>
            ))}
            <select {...register('billingCycle')} style={{ padding: '8px 10px' }}>
              <option value="monthly">Monthly</option>
              <option value="yearly">Yearly (save 20%)</option>
            </select>
          </div>
        )}

        {step === 2 && (
          <div style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem' }}>
            <h3>Payment</h3>
            <input {...register('cardNumber')} placeholder="Card number (16 digits)" style={{ padding: '8px 10px', border: `1px solid ${errors.cardNumber ? '#ef4444' : '#d1d5db'}` }} maxLength={16} />
            {errors.cardNumber && <p style={{ color: '#ef4444', fontSize: 12, margin: 0 }}>{errors.cardNumber.message}</p>}
            <div style={{ display: 'grid', gridTemplateColumns: '1fr 1fr', gap: 8 }}>
              <input {...register('expiry')} placeholder="MM/YY" style={{ padding: '8px 10px' }} maxLength={5} />
              <input {...register('cvv')} placeholder="CVV" style={{ padding: '8px 10px' }} maxLength={4} />
            </div>
          </div>
        )}

        <div style={{ display: 'flex', justifyContent: 'space-between', marginTop: '1.5rem' }}>
          {step > 0 && <button type="button" onClick={() => setStep(s => s - 1)}>← Back</button>}
          {step < 2 && <button type="button" onClick={nextStep} style={{ marginLeft: 'auto' }}>Next →</button>}
          {step === 2 && <button type="submit" disabled={isSubmitting}>{isSubmitting ? 'Processing...' : 'Complete'}</button>}
        </div>
      </form>
    </div>
  )
}
```
