## Goal
Learn how to manage complex state transitions with useReducer, producing predictable state updates through explicit actions.

## Background

### Why useReducer?
`useState` is great for simple, independent values. As state grows — multiple related fields, state transitions that depend on current state, or many event handlers updating the same state — `useReducer` provides a cleaner model. The reducer is a pure function: `(state, action) => newState`. All state logic lives in one place, making it easy to test and reason about.

### The Reducer Pattern
A reducer takes the current state and an action object, and returns the next state. The action object must have a `type` field (by convention a string) and any additional payload data. You dispatch actions from event handlers, and the reducer decides what the new state should be. This is the same pattern Redux uses — `useReducer` is essentially a local, built-in Redux.

### When to Use useReducer Over useState
Prefer `useReducer` when: state is an object with multiple fields that update together; the next state depends on the current state in a non-trivial way; you have many event handlers that all modify the same state; or you want to make all state transitions explicit and traceable. Use `useState` for simple boolean flags, single values, or state that doesn't interact with other state.

## Tasks

### Task 1 — Counter with useReducer

```tsx
// src/components/ReducerCounter.tsx
import { useReducer } from 'react'

// 1. Define the state shape
interface CounterState {
  count: number
  step: number
  history: number[]
}

// 2. Define all possible actions as a discriminated union
type CounterAction =
  | { type: 'INCREMENT' }
  | { type: 'DECREMENT' }
  | { type: 'RESET' }
  | { type: 'SET_STEP'; payload: number }
  | { type: 'UNDO' }

const initialState: CounterState = {
  count: 0,
  step: 1,
  history: [],
}

// 3. Write the reducer — pure function, no side effects
function counterReducer(state: CounterState, action: CounterAction): CounterState {
  switch (action.type) {
    case 'INCREMENT':
      return {
        ...state,
        count: state.count + state.step,
        history: [...state.history, state.count],
      }
    case 'DECREMENT':
      return {
        ...state,
        count: state.count - state.step,
        history: [...state.history, state.count],
      }
    case 'RESET':
      return initialState
    case 'SET_STEP':
      return { ...state, step: action.payload }
    case 'UNDO':
      if (state.history.length === 0) return state
      const prev = state.history[state.history.length - 1]
      return {
        ...state,
        count: prev,
        history: state.history.slice(0, -1),
      }
    default:
      return state
  }
}

export function ReducerCounter() {
  const [state, dispatch] = useReducer(counterReducer, initialState)

  return (
    <div style={{ maxWidth: 300 }}>
      <div style={{ fontSize: 48, textAlign: 'center', fontWeight: 'bold' }}>{state.count}</div>
      <div style={{ display: 'flex', gap: '0.5rem', justifyContent: 'center', marginBottom: '1rem' }}>
        <button onClick={() => dispatch({ type: 'DECREMENT' })}>-{state.step}</button>
        <button onClick={() => dispatch({ type: 'INCREMENT' })}>+{state.step}</button>
        <button onClick={() => dispatch({ type: 'UNDO' })} disabled={state.history.length === 0}>
          Undo
        </button>
        <button onClick={() => dispatch({ type: 'RESET' })}>Reset</button>
      </div>
      <label>
        Step: {state.step}
        <input
          type="range"
          min={1}
          max={10}
          value={state.step}
          onChange={e => dispatch({ type: 'SET_STEP', payload: Number(e.target.value) })}
        />
      </label>
      <p style={{ fontSize: 12, color: '#9ca3af' }}>
        History: [{state.history.join(', ')}]
      </p>
    </div>
  )
}
```

### Task 2 — Form State with useReducer

```tsx
// src/components/RegistrationForm.tsx
import { useReducer } from 'react'

interface FormState {
  values: {
    username: string
    email: string
    password: string
    confirmPassword: string
  }
  errors: Partial<Record<keyof FormState['values'], string>>
  touched: Partial<Record<keyof FormState['values'], boolean>>
  submitting: boolean
  submitError: string | null
  submitted: boolean
}

type FormField = keyof FormState['values']

type FormAction =
  | { type: 'SET_FIELD'; field: FormField; value: string }
  | { type: 'TOUCH_FIELD'; field: FormField }
  | { type: 'SET_ERRORS'; errors: FormState['errors'] }
  | { type: 'SUBMIT_START' }
  | { type: 'SUBMIT_SUCCESS' }
  | { type: 'SUBMIT_ERROR'; error: string }
  | { type: 'RESET' }

const initialFormState: FormState = {
  values: { username: '', email: '', password: '', confirmPassword: '' },
  errors: {},
  touched: {},
  submitting: false,
  submitError: null,
  submitted: false,
}

function validate(values: FormState['values']): FormState['errors'] {
  const errors: FormState['errors'] = {}
  if (!values.username) errors.username = 'Username is required'
  else if (values.username.length < 3) errors.username = 'At least 3 characters'
  if (!values.email || !values.email.includes('@')) errors.email = 'Valid email required'
  if (!values.password || values.password.length < 8) errors.password = 'At least 8 characters'
  if (values.password !== values.confirmPassword) errors.confirmPassword = 'Passwords do not match'
  return errors
}

function formReducer(state: FormState, action: FormAction): FormState {
  switch (action.type) {
    case 'SET_FIELD': {
      const values = { ...state.values, [action.field]: action.value }
      const errors = state.touched[action.field] ? validate(values) : state.errors
      return { ...state, values, errors }
    }
    case 'TOUCH_FIELD':
      return {
        ...state,
        touched: { ...state.touched, [action.field]: true },
        errors: validate(state.values),
      }
    case 'SET_ERRORS':
      return { ...state, errors: action.errors }
    case 'SUBMIT_START':
      return { ...state, submitting: true, submitError: null }
    case 'SUBMIT_SUCCESS':
      return { ...state, submitting: false, submitted: true }
    case 'SUBMIT_ERROR':
      return { ...state, submitting: false, submitError: action.error }
    case 'RESET':
      return initialFormState
    default:
      return state
  }
}

export function RegistrationForm() {
  const [state, dispatch] = useReducer(formReducer, initialFormState)

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    const errors = validate(state.values)
    if (Object.keys(errors).length > 0) {
      dispatch({ type: 'SET_ERRORS', errors })
      return
    }
    dispatch({ type: 'SUBMIT_START' })
    try {
      await new Promise(resolve => setTimeout(resolve, 1000))  // simulate API
      dispatch({ type: 'SUBMIT_SUCCESS' })
    } catch {
      dispatch({ type: 'SUBMIT_ERROR', error: 'Registration failed. Try again.' })
    }
  }

  if (state.submitted) {
    return (
      <div>
        <p>Welcome, {state.values.username}!</p>
        <button onClick={() => dispatch({ type: 'RESET' })}>Register another</button>
      </div>
    )
  }

  const fields: { name: FormField; label: string; type: string }[] = [
    { name: 'username', label: 'Username', type: 'text' },
    { name: 'email', label: 'Email', type: 'email' },
    { name: 'password', label: 'Password', type: 'password' },
    { name: 'confirmPassword', label: 'Confirm Password', type: 'password' },
  ]

  return (
    <form onSubmit={handleSubmit} style={{ maxWidth: 360, display: 'flex', flexDirection: 'column', gap: '1rem' }}>
      {fields.map(field => (
        <div key={field.name}>
          <label style={{ display: 'block', marginBottom: 4, fontWeight: 500 }}>{field.label}</label>
          <input
            type={field.type}
            value={state.values[field.name]}
            onChange={e => dispatch({ type: 'SET_FIELD', field: field.name, value: e.target.value })}
            onBlur={() => dispatch({ type: 'TOUCH_FIELD', field: field.name })}
            style={{
              width: '100%',
              padding: '8px 10px',
              border: `1px solid ${state.touched[field.name] && state.errors[field.name] ? '#ef4444' : '#d1d5db'}`,
              borderRadius: 4,
            }}
          />
          {state.touched[field.name] && state.errors[field.name] && (
            <p style={{ color: '#ef4444', fontSize: 12, margin: '4px 0 0' }}>
              {state.errors[field.name]}
            </p>
          )}
        </div>
      ))}
      {state.submitError && <p style={{ color: '#ef4444' }}>{state.submitError}</p>}
      <button type="submit" disabled={state.submitting}>
        {state.submitting ? 'Registering...' : 'Register'}
      </button>
    </form>
  )
}
```
