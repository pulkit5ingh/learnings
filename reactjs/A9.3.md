# A9.3 — Internationalization (i18n)

**Phase:** 9 — TypeScript, Accessibility & Production
**Topic:** react-i18next setup, useTranslation, Trans, pluralization, date/number formatting, language switcher
**Type:** Assignment

---

## Goal

Internationalise a React app so it can serve users in multiple languages. You will set up react-i18next, organise translation namespaces, handle pluralisation and formatted values, lazy-load locale files, and build a language switcher.

---

## Background

`react-i18next` is the most widely used i18n library for React. It wraps `i18next` and provides:
- `useTranslation(namespace)` hook — access translation strings
- `Trans` component — translate strings containing JSX (links, bold text)
- Pluralisation rules per language built in
- `Intl` API integration for dates and numbers

The translation files live in `public/locales/{lang}/{namespace}.json` and can be lazy-loaded via `i18next-http-backend`.

### 2. Configure i18n

```ts
// src/i18n.ts
import i18n from 'i18next';
import { initReactI18next } from 'react-i18next';
import HttpBackend from 'i18next-http-backend';
import LanguageDetector from 'i18next-browser-languagedetector';

i18n
  .use(HttpBackend)           // load translations from /public/locales
  .use(LanguageDetector)      // detect browser language
  .use(initReactI18next)      // bind to React
  .init({
    fallbackLng: 'en',
    supportedLngs: ['en', 'fr', 'de', 'es'],
    defaultNS: 'common',
    ns: ['common', 'dashboard', 'auth'],

    backend: {
      loadPath: '/locales/{{lng}}/{{ns}}.json',
    },

    detection: {
      order: ['querystring', 'localStorage', 'navigator'],
      caches: ['localStorage'],
    },

    interpolation: {
      escapeValue: false, // React already escapes
    },
  });

export default i18n;
```

```tsx
// src/main.tsx
import './i18n'; // import before App so translations are ready
import { Suspense } from 'react';
import { createRoot } from 'react-dom/client';
import App from './App';

createRoot(document.getElementById('root')!).render(
  <Suspense fallback={<div>Loading translations...</div>}>
    <App />
  </Suspense>
);
```

### 4. `useTranslation` hook

```tsx
// src/components/Navbar.tsx
import { useTranslation } from 'react-i18next';

export function Navbar() {
  const { t } = useTranslation('common'); // namespace

  return (
    <nav>
      <a href="/">{t('nav.home')}</a>
      <a href="/about">{t('nav.about')}</a>
      <a href="/dashboard">{t('nav.dashboard')}</a>
      <button>{t('nav.logout')}</button>
    </nav>
  );
}
```

```tsx
// src/pages/Dashboard.tsx
import { useTranslation } from 'react-i18next';

export function Dashboard({ user }: { user: { name: string } }) {
  // Multiple namespaces
  const { t: tc } = useTranslation('common');
  const { t: td } = useTranslation('dashboard');

  const userCount = 1;
  const revenue = 12345.67;
  const lastLoginDate = new Date();

  return (

      <h1>{td('title')}</h1>
```

### 5. Pluralisation

i18next uses `_other` as the plural suffix for English (following ICU rules).

```json
// public/locales/en/common.json (additions)
{
  "items": "{{count}} item",
  "items_other": "{{count}} items",
  "messages_zero": "No messages",
  "messages_one": "{{count}} message",
  "messages_other": "{{count}} messages"
}
```

```json
// public/locales/fr/common.json (additions)
{
  "items": "{{count}} élément",
  "items_other": "{{count}} éléments"
}
```

```tsx
function ItemCount({ count }: { count: number }) {
  const { t } = useTranslation('common');
  return <span>{t('items', { count })}</span>;
  // count=1 → "1 item", count=5 → "5 items"
}
```

### 6. `Trans` component for JSX-embedded strings

Use `Trans` when a translated string contains React elements (links, bold text).

```json
// public/locales/en/auth.json
{
  "agreeToTerms": "By signing up you agree to our <1>Terms of Service</1> and <3>Privacy Policy</3>.",
  "loginPrompt": "Already have an account? <lnk>Sign in</lnk>."
}
```

```tsx
import { Trans, useTranslation } from 'react-i18next';
import { Link } from 'react-router-dom';

function SignupFooter() {
  const { t } = useTranslation('auth');

  return (
    <p>
      <Trans
        i18nKey="agreeToTerms"
        ns="auth"
        components={[
          <span />,                              // index 0 — unused
          <Link to="/terms" />,                  // index 1
          <span />,                              // index 2 — unused
          <Link to="/privacy" />,               // index 3
        ]}
      />
    </p>
  );
}

// Alternative with named components (cleaner)
function LoginFooter() {
  return (
    <Trans
      i18nKey="auth:loginPrompt"
      components={{ lnk: <Link to="/login" /> }}
    />
  );
}
```

### 7. Date and number formatting with `Intl`

```tsx
// src/utils/format.ts
export function formatDate(
  date: Date,
  locale: string,
  options: Intl.DateTimeFormatOptions = { dateStyle: 'medium' }
): string {
  return new Intl.DateTimeFormat(locale, options).format(date);
}

export function formatCurrency(
  amount: number,
  locale: string,
  currency = 'USD'
): string {
  return new Intl.NumberFormat(locale, {
    style: 'currency',
    currency,
  }).format(amount);
}

export function formatRelativeTime(date: Date, locale: string): string {
  const diff = (date.getTime() - Date.now()) / 1000;
  const rtf = new Intl.RelativeTimeFormat(locale, { numeric: 'auto' });

  if (Math.abs(diff) < 60) return rtf.format(Math.round(diff), 'second');
  if (Math.abs(diff) < 3600) return rtf.format(Math.round(diff / 60), 'minute');
  if (Math.abs(diff) < 86400) return rtf.format(Math.round(diff / 3600), 'hour');
  return rtf.format(Math.round(diff / 86400), 'day');
}
```

```tsx
// src/components/DateDisplay.tsx
import { useTranslation } from 'react-i18next';
import { formatDate } from '../utils/format';

export function DateDisplay({ date }: { date: Date }) {
  const { i18n } = useTranslation();
  return <time dateTime={date.toISOString()}>{formatDate(date, i18n.language)}</time>;
}
```

### 8. Language switcher

```tsx
// src/components/LanguageSwitcher.tsx
import { useTranslation } from 'react-i18next';

const LANGUAGES = [
  { code: 'en', label: 'English' },
  { code: 'fr', label: 'Français' },
  { code: 'de', label: 'Deutsch' },
  { code: 'es', label: 'Español' },
];

export function LanguageSwitcher() {
  const { i18n } = useTranslation();

  const changeLanguage = async (lang: string) => {
    await i18n.changeLanguage(lang);
    // Update <html lang> for screen readers
    document.documentElement.lang = lang;
  };

  return (
    <select
      value={i18n.language}
      onChange={e => changeLanguage(e.target.value)}
      aria-label="Select language"
    >
      {LANGUAGES.map(({ code, label }) => (
        <option key={code} value={code}>{label}</option>
      ))}
    </select>
  );
}
```
