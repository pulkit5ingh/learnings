## Goal
Replace manual useEffect-based data fetching with React Query, which handles caching, background refetching, and loading states automatically.

## Background

### Server State vs Client State
Client state (selected tab, form input values, UI panel open/closed) lives in `useState` or `useReducer`. Server state (data from an API) is different: it's owned by the server, can be stale, needs to be refreshed, and may be shared across many components. React Query manages server state — caching it, re-fetching it when stale, and making it available without prop drilling.

### How React Query Works
You define a query with `useQuery({ queryKey, queryFn })`. The `queryKey` is an array that uniquely identifies the data — React Query uses it as the cache key. The `queryFn` is an async function that fetches and returns the data. React Query handles loading, error, and success states, plus background refetching when the user focuses the window.

### staleTime
After a query is fetched, it's "fresh" for `staleTime` milliseconds. During this window, React Query serves the cached data without refetching. After `staleTime`, the data is "stale" but still shown from cache — React Query refetches in the background. The default staleTime is 0 (immediately stale, refetch on every mount). Set `staleTime: 5 * 60 * 1000` (5 minutes) for data that doesn't change often.

```tsx
// src/main.tsx
import { StrictMode } from 'react'
import { createRoot } from 'react-dom/client'
import { QueryClient, QueryClientProvider } from '@tanstack/react-query'
import { ReactQueryDevtools } from '@tanstack/react-query-devtools'
import App from './App'

const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60,    // 1 minute
      retry: 2,                 // retry failed queries twice
      refetchOnWindowFocus: true,  // refetch when user returns to tab
    },
  },
})

createRoot(document.getElementById('root')!).render(
  <StrictMode>
    <QueryClientProvider client={queryClient}>
      <App />
      {/* Devtools panel — only in development */}
      <ReactQueryDevtools initialIsOpen={false} />
    </QueryClientProvider>
  </StrictMode>
)
```

### Task 2 — First useQuery

```tsx
// src/api/posts.ts
const BASE = 'https://jsonplaceholder.typicode.com'

export interface Post {
  id: number
  userId: number
  title: string
  body: string
}

export async function fetchPosts(): Promise<Post[]> {
  const res = await fetch(`${BASE}/posts?_limit=20`)
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
  return res.json()
}

export async function fetchPost(id: number): Promise<Post> {
  const res = await fetch(`${BASE}/posts/${id}`)
  if (!res.ok) throw new Error(res.status === 404 ? 'Post not found' : `HTTP ${res.status}`)
  return res.json()
}
```

```tsx
// src/components/PostList.tsx
import { useQuery } from '@tanstack/react-query'
import { fetchPosts } from '../api/posts'
import { Link } from 'react-router-dom'

export function PostList() {
  const {
    data: posts,
    isLoading,
    isError,
    error,
    isFetching,  // true during background refetch (data is still available)
  } = useQuery({
    queryKey: ['posts'],           // cache key
    queryFn: fetchPosts,
    staleTime: 1000 * 60 * 5,    // 5 min: don't refetch if fresh
  })

  if (isLoading) return <div>Loading posts...</div>
  if (isError) return <div style={{ color: '#ef4444' }}>Error: {error.message}</div>

  return (
    <div>
      <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
        <h2>Posts</h2>
        {isFetching && <span style={{ fontSize: 12, color: '#6b7280' }}>Refreshing...</span>}
      </div>

      <ul style={{ listStyle: 'none', padding: 0 }}>
        {posts.map(post => (
          <li key={post.id} style={{ borderBottom: '1px solid #f0f0f0', padding: '0.75rem 0' }}>
            <Link to={`/posts/${post.id}`} style={{ fontWeight: 500, color: '#1d4ed8', textDecoration: 'none' }}>
              {post.title}
            </Link>
            <p style={{ margin: '4px 0 0', fontSize: 13, color: '#6b7280' }}>{post.body.slice(0, 80)}…</p>
          </li>
        ))}
      </ul>
    </div>
  )
}
```

### Task 3 — Query with Parameters

```tsx
// src/components/PostDetail.tsx
import { useQuery } from '@tanstack/react-query'
import { useParams } from 'react-router-dom'
import { fetchPost } from '../api/posts'

export function PostDetail() {
  const { postId } = useParams<{ postId: string }>()
  const id = Number(postId)

  const { data: post, isLoading, isError, error } = useQuery({
    queryKey: ['posts', id],      // unique key per post ID
    queryFn: () => fetchPost(id),
    enabled: !isNaN(id) && id > 0,  // don't run if id is invalid
  })

  if (isLoading) return <div>Loading post {id}...</div>
  if (isError) return <div style={{ color: '#ef4444' }}>{error.message}</div>
  if (!post) return null

  return (
    <article>
      <h1>{post.title}</h1>
      <p style={{ color: '#6b7280', fontSize: 14 }}>User {post.userId}</p>
      <p>{post.body}</p>
    </article>
  )
}
```

### Task 4 — queryKey Conventions

```tsx
// Consistent queryKey conventions across your codebase:
// (define in src/api/queryKeys.ts)

export const queryKeys = {
  // Collections
  posts: () => ['posts'] as const,
  postsByUser: (userId: number) => ['posts', { userId }] as const,

  // Single items
  post: (id: number) => ['posts', id] as const,

  // With filters
  postsFiltered: (filters: { search?: string; page?: number }) =>
    ['posts', 'filtered', filters] as const,

  // Nested resources
  postComments: (postId: number) => ['posts', postId, 'comments'] as const,

  users: () => ['users'] as const,
  user: (id: number) => ['users', id] as const,
}

// Usage:
// useQuery({ queryKey: queryKeys.post(5), queryFn: () => fetchPost(5) })
// queryClient.invalidateQueries({ queryKey: queryKeys.posts() })
// → invalidates all queries starting with ['posts']
```
