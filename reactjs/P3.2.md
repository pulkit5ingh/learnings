# P3.2 — Search App with URL State

**Phase:** 3 — React Router & Navigation
**Topic:** useSearchParams, URL-driven filters/sort/pagination, shareable URLs, browser back/forward
**Type:** Project

---

## Goal

Build a search and filter UI where every piece of state (query, filters, sort, page) lives in the URL as query parameters. The page is fully bookmarkable and shareable. Browser back/forward navigation correctly restores the state.

---

## What to Build

- Search input synced to `?q=` parameter
- Category filter (multi-select checkboxes) synced to `?category=`
- Sort dropdown synced to `?sort=` parameter
- Pagination synced to `?page=` parameter
- Results count and "showing X of Y" text
- "Reset filters" clears all params and returns to page 1
- Results update from URL without reloading the page
- Works with static/mock data so no API key needed

### URL state hook

```ts
// src/hooks/useSearchState.ts
import { useSearchParams } from 'react-router-dom';
import { useCallback } from 'react';

export type SortOption = 'relevance' | 'rating-desc' | 'rating-asc' | 'year-desc' | 'year-asc';

export interface SearchState {
  query: string;
  categories: string[];
  sort: SortOption;
  page: number;
}

const PAGE_SIZE = 12;

export function useSearchState() {
  const [searchParams, setSearchParams] = useSearchParams();

  const state: SearchState = {
    query: searchParams.get('q') ?? '',
    categories: searchParams.getAll('category'),
    sort: (searchParams.get('sort') as SortOption) ?? 'relevance',
    page: Number(searchParams.get('page') ?? '1'),
  };

  const setQuery = useCallback((q: string) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      if (q) next.set('q', q); else next.delete('q');
      next.set('page', '1');
      return next;
    });
  }, [setSearchParams]);

  const toggleCategory = useCallback((category: string) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      const cats = next.getAll('category');
      if (cats.includes(category)) {
        next.delete('category');
        cats.filter(c => c !== category).forEach(c => next.append('category', c));
      } else {
        next.append('category', category);
      }
      next.set('page', '1');
      return next;
    });
  }, [setSearchParams]);

  const setSort = useCallback((sort: SortOption) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      next.set('sort', sort);
      next.set('page', '1');
      return next;
    });
  }, [setSearchParams]);

  const setPage = useCallback((page: number) => {
    setSearchParams(prev => {
      const next = new URLSearchParams(prev);
      next.set('page', String(page));
      return next;
    });
  }, [setSearchParams]);

  const resetFilters = useCallback(() => {
    setSearchParams({});
  }, [setSearchParams]);

  const hasActiveFilters = state.query || state.categories.length > 0 || state.sort !== 'relevance';

  return { state, setQuery, toggleCategory, setSort, setPage, resetFilters, hasActiveFilters, PAGE_SIZE };
}
```

### Filter utility

```ts
// src/utils/filter.ts
import type { Item } from '../data/items';
import type { SearchState, SortOption } from '../hooks/useSearchState';

export function filterAndSort(items: Item[], state: SearchState): Item[] {
  let result = [...items];

  if (state.query) {
    const q = state.query.toLowerCase();
    result = result.filter(
      item =>
        item.title.toLowerCase().includes(q) ||
        item.author.toLowerCase().includes(q) ||
        item.tags.some(t => t.includes(q))
    );
  }

  if (state.categories.length > 0) {
    result = result.filter(item => state.categories.includes(item.category));
  }

  switch (state.sort) {
    case 'rating-desc':
      result.sort((a, b) => b.rating - a.rating);
      break;
    case 'rating-asc':
      result.sort((a, b) => a.rating - b.rating);
      break;
    case 'year-desc':
      result.sort((a, b) => b.year - a.year);
      break;
    case 'year-asc':
      result.sort((a, b) => a.year - b.year);
      break;
    default:
      break;
  }

  return result;
}
```

### SearchInput

```tsx
// src/components/SearchInput.tsx
import { useEffect, useState } from 'react';

interface Props {
  value: string;
  onChange: (value: string) => void;
}

export function SearchInput({ value, onChange }: Props) {
  // Local state for the input — sync back to URL after debounce
  const [local, setLocal] = useState(value);

  useEffect(() => {
    setLocal(value);
  }, [value]);

  useEffect(() => {
    const timer = setTimeout(() => {
      if (local !== value) onChange(local);
    }, 400);
    return () => clearTimeout(timer);
  }, [local, value, onChange]);

  return (
    <div className="relative">
      <input
        type="search"
        value={local}
        onChange={e => setLocal(e.target.value)}
        placeholder="Search items..."
        className="w-full pl-10 pr-4 py-3 border rounded-xl focus:outline-none focus:ring-2 focus:ring-blue-400"
        aria-label="Search"
      />
      <span className="absolute left-3 top-1/2 -translate-y-1/2 text-gray-400">🔍</span>
    </div>
  );
}
```

### Pagination

```tsx
// src/components/Pagination.tsx
interface Props {
  currentPage: number;
  totalPages: number;
  onPageChange: (page: number) => void;
}

export function Pagination({ currentPage, totalPages, onPageChange }: Props) {
  if (totalPages <= 1) return null;

  const pages = Array.from({ length: Math.min(totalPages, 7) }, (_, i) => {
    if (totalPages <= 7) return i + 1;
    if (currentPage <= 4) return i + 1;
    if (currentPage >= totalPages - 3) return totalPages - 6 + i;
    return currentPage - 3 + i;
  });

  return (
    <nav aria-label="Pagination" className="flex items-center gap-1 justify-center">
      <button
        onClick={() => onPageChange(currentPage - 1)}
        disabled={currentPage === 1}
        className="px-3 py-2 rounded-lg border disabled:opacity-40"
      >
        ‹
      </button>
      {pages.map(p => (
        <button
          key={p}
          onClick={() => onPageChange(p)}
          aria-current={p === currentPage ? 'page' : undefined}
          className={`px-3 py-2 rounded-lg border ${
            p === currentPage ? 'bg-blue-500 text-white border-blue-500' : 'hover:bg-gray-50'
          }`}
        >
          {p}
        </button>
      ))}
      <button
        onClick={() => onPageChange(currentPage + 1)}
        disabled={currentPage === totalPages}
        className="px-3 py-2 rounded-lg border disabled:opacity-40"
      >
        ›
      </button>
    </nav>
  );
}
```

### App

```tsx
// src/App.tsx
import { useMemo } from 'react';
import { BrowserRouter, Routes, Route } from 'react-router-dom';
import { SearchInput } from './components/SearchInput';
import { FilterPanel } from './components/FilterPanel';
import { SortSelect } from './components/SortSelect';
import { ResultsGrid } from './components/ResultsGrid';
import { Pagination } from './components/Pagination';
import { useSearchState } from './hooks/useSearchState';
import { filterAndSort } from './utils/filter';
import { ITEMS, CATEGORIES } from './data/items';

function SearchPage() {
  const { state, setQuery, toggleCategory, setSort, setPage, resetFilters, hasActiveFilters, PAGE_SIZE } = useSearchState();

  const filtered = useMemo(() => filterAndSort(ITEMS, state), [state]);
  const totalPages = Math.ceil(filtered.length / PAGE_SIZE);
  const pageItems = filtered.slice((state.page - 1) * PAGE_SIZE, state.page * PAGE_SIZE);

  return (
    <div className="max-w-6xl mx-auto p-6">
      <h1 className="text-2xl font-bold mb-6">Search</h1>

      <div className="mb-6">
        <SearchInput value={state.query} onChange={setQuery} />
      </div>

      <div className="flex gap-6">
        <aside className="w-56 flex-shrink-0">
          <FilterPanel
            categories={CATEGORIES}
            selected={state.categories}
            onToggle={toggleCategory}
          />
          {hasActiveFilters && (
            <button
              onClick={resetFilters}
              className="mt-4 text-sm text-red-500 hover:text-red-700"
            >
              Reset all filters
            </button>
          )}
        </aside>

        <div className="flex-1 space-y-4">
          <div className="flex items-center justify-between">
            <p className="text-sm text-gray-500">
              {filtered.length} result{filtered.length !== 1 ? 's' : ''}
              {state.query && ` for "${state.query}"`}
            </p>
            <SortSelect value={state.sort} onChange={setSort} />
          </div>

          <ResultsGrid items={pageItems} />

          <Pagination
            currentPage={state.page}
            totalPages={totalPages}
            onPageChange={setPage}
          />
        </div>
      </div>
    </div>
  );
}

export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/" element={<SearchPage />} />
      </Routes>
    </BrowserRouter>
  );
}
```
