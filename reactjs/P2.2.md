# P2.2 — Kanban Board

**Phase:** 2 — Intermediate Hooks & Patterns
**Topic:** @dnd-kit drag-and-drop, useReducer for board state, custom column hooks
**Type:** Project

---

## Goal

Build a Trello-style Kanban board where cards can be dragged between columns. Board state is managed with `useReducer`, drag-and-drop is handled by `@dnd-kit`, and each column has a custom hook for its operations.

---

## What to Build

- Multiple columns (e.g. To Do, In Progress, Done)
- Cards with title and optional description
- Drag cards within a column (reorder) and between columns
- Add new card to any column (inline form)
- Edit card title inline (double-click)
- Delete a card
- Add / rename / delete columns
- Board state persists to localStorage

```ts
export interface Card {
  id: string;
  title: string;
  description?: string;
  createdAt: number;
}

export interface Column {
  id: string;
  title: string;
  cardIds: string[];
}

export interface BoardState {
  columns: Column[];
  cards: Record<string, Card>;
}

export type BoardAction =
  | { type: 'ADD_CARD'; columnId: string; title: string }
  | { type: 'DELETE_CARD'; cardId: string; columnId: string }
  | { type: 'EDIT_CARD'; cardId: string; title: string }
  | { type: 'MOVE_CARD'; fromColumn: string; toColumn: string; fromIndex: number; toIndex: number }
  | { type: 'ADD_COLUMN'; title: string }
  | { type: 'RENAME_COLUMN'; columnId: string; title: string }
  | { type: 'DELETE_COLUMN'; columnId: string };
```

### Reducer

```ts
// src/reducers/boardReducer.ts
import { nanoid } from 'nanoid';
import type { BoardState, BoardAction } from '../types/board';

export const initialState: BoardState = {
  columns: [
    { id: 'col-1', title: 'To Do', cardIds: [] },
    { id: 'col-2', title: 'In Progress', cardIds: [] },
    { id: 'col-3', title: 'Done', cardIds: [] },
  ],
  cards: {},
};

export function boardReducer(state: BoardState, action: BoardAction): BoardState {
  switch (action.type) {
    case 'ADD_CARD': {
      const id = nanoid();
      const card = { id, title: action.title, createdAt: Date.now() };
      return {
        cards: { ...state.cards, [id]: card },
        columns: state.columns.map(col =>
          col.id === action.columnId
            ? { ...col, cardIds: [...col.cardIds, id] }
            : col
        ),
      };
    }

    case 'DELETE_CARD': {
      const { [action.cardId]: _, ...restCards } = state.cards;
      return {
        cards: restCards,
        columns: state.columns.map(col =>
          col.id === action.columnId
            ? { ...col, cardIds: col.cardIds.filter(id => id !== action.cardId) }
            : col
        ),
      };
    }

    case 'EDIT_CARD':
      return {
        ...state,
        cards: {
          ...state.cards,
          [action.cardId]: { ...state.cards[action.cardId], title: action.title },
        },
      };

    case 'MOVE_CARD': {
      if (action.fromColumn === action.toColumn) {
        // Reorder within same column
        const col = state.columns.find(c => c.id === action.fromColumn)!;
        const newIds = [...col.cardIds];
        const [moved] = newIds.splice(action.fromIndex, 1);
        newIds.splice(action.toIndex, 0, moved);
        return {
          ...state,
          columns: state.columns.map(c =>
            c.id === action.fromColumn ? { ...c, cardIds: newIds } : c
          ),
        };
      }
      // Move to different column
      const fromCol = state.columns.find(c => c.id === action.fromColumn)!;
      const toCol = state.columns.find(c => c.id === action.toColumn)!;
      const fromIds = [...fromCol.cardIds];
      const [moved] = fromIds.splice(action.fromIndex, 1);
      const toIds = [...toCol.cardIds];
      toIds.splice(action.toIndex, 0, moved);
      return {
        ...state,
        columns: state.columns.map(c => {
          if (c.id === action.fromColumn) return { ...c, cardIds: fromIds };
          if (c.id === action.toColumn) return { ...c, cardIds: toIds };
          return c;
        }),
      };
    }

    case 'ADD_COLUMN':
      return {
        ...state,
        columns: [...state.columns, { id: nanoid(), title: action.title, cardIds: [] }],
      };

    case 'RENAME_COLUMN':
      return {
        ...state,
        columns: state.columns.map(c =>
          c.id === action.columnId ? { ...c, title: action.title } : c
        ),
      };

    case 'DELETE_COLUMN': {
      const col = state.columns.find(c => c.id === action.columnId)!;
      const remainingCards = { ...state.cards };
      col.cardIds.forEach(id => delete remainingCards[id]);
      return {
        cards: remainingCards,
        columns: state.columns.filter(c => c.id !== action.columnId),
      };
    }

    default:
      return state;
  }
}
```

### Board with DnD

```tsx
// src/components/Board.tsx
import {
  DndContext,
  DragEndEvent,
  DragOverlay,
  DragStartEvent,
  PointerSensor,
  useSensor,
  useSensors,
  closestCorners,
} from '@dnd-kit/core';
import { SortableContext, horizontalListSortingStrategy } from '@dnd-kit/sortable';
import { useState } from 'react';
import { Column } from './Column';
import { Card } from './Card';
import { useBoard } from '../hooks/useBoard';

export function Board() {
  const { state, moveCard, addColumn } = useBoard();
  const [activeCardId, setActiveCardId] = useState<string | null>(null);

  const sensors = useSensors(
    useSensor(PointerSensor, {
      activationConstraint: { distance: 8 },
    })
  );

  const handleDragStart = (event: DragStartEvent) => {
    setActiveCardId(event.active.id as string);
  };

  const handleDragEnd = (event: DragEndEvent) => {
    const { active, over } = event;
    setActiveCardId(null);

    if (!over || active.id === over.id) return;

    // Determine source column
    const fromColumn = state.columns.find(col =>
      col.cardIds.includes(active.id as string)
    );
    if (!fromColumn) return;

    // Determine destination — over could be a column or a card
    let toColumnId: string;
    let toIndex: number;

    const toColumn = state.columns.find(col => col.id === over.id);
    if (toColumn) {
      toColumnId = toColumn.id;
      toIndex = toColumn.cardIds.length;
    } else {
      const toColWithCard = state.columns.find(col =>
        col.cardIds.includes(over.id as string)
      );
      if (!toColWithCard) return;
      toColumnId = toColWithCard.id;
      toIndex = toColWithCard.cardIds.indexOf(over.id as string);
    }

    const fromIndex = fromColumn.cardIds.indexOf(active.id as string);

    moveCard({
      fromColumn: fromColumn.id,
      toColumn: toColumnId,
      fromIndex,
      toIndex,
    });
  };

  const activeCard = activeCardId ? state.cards[activeCardId] : null;

  return (
    <DndContext
      sensors={sensors}
      collisionDetection={closestCorners}
      onDragStart={handleDragStart}
      onDragEnd={handleDragEnd}
    >
      <div className="flex gap-4 overflow-x-auto p-4 min-h-screen bg-sky-900 items-start">
        <SortableContext
          items={state.columns.map(c => c.id)}
          strategy={horizontalListSortingStrategy}
        >
          {state.columns.map(column => (
            <Column key={column.id} column={column} />
          ))}
        </SortableContext>

        <button
          onClick={() => {
            const title = prompt('Column name:');
            if (title?.trim()) addColumn(title.trim());
          }}
          className="flex-shrink-0 w-64 px-4 py-3 rounded-xl bg-white/20 text-white hover:bg-white/30 transition"
        >
          + Add column
        </button>
      </div>

      <DragOverlay>
        {activeCard && (
          <div className="rotate-2 opacity-90">
            <Card card={activeCard} columnId="" isDragging />
          </div>
        )}
      </DragOverlay>
    </DndContext>
  );
}
```

### useBoard hook

```ts
// src/hooks/useBoard.ts
import { useReducer, useEffect } from 'react';
import { boardReducer, initialState } from '../reducers/boardReducer';

const STORAGE_KEY = 'kanban-board';

function loadState() {
  try {
    const saved = localStorage.getItem(STORAGE_KEY);
    return saved ? JSON.parse(saved) : initialState;
  } catch {
    return initialState;
  }
}

export function useBoard() {
  const [state, dispatch] = useReducer(boardReducer, undefined, loadState);

  useEffect(() => {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
  }, [state]);

  return {
    state,
    addCard: (columnId: string, title: string) =>
      dispatch({ type: 'ADD_CARD', columnId, title }),
    deleteCard: (cardId: string, columnId: string) =>
      dispatch({ type: 'DELETE_CARD', cardId, columnId }),
    editCard: (cardId: string, title: string) =>
      dispatch({ type: 'EDIT_CARD', cardId, title }),
    moveCard: (args: { fromColumn: string; toColumn: string; fromIndex: number; toIndex: number }) =>
      dispatch({ type: 'MOVE_CARD', ...args }),
    addColumn: (title: string) => dispatch({ type: 'ADD_COLUMN', title }),
    renameColumn: (columnId: string, title: string) =>
      dispatch({ type: 'RENAME_COLUMN', columnId, title }),
    deleteColumn: (columnId: string) => dispatch({ type: 'DELETE_COLUMN', columnId }),
  };
}
```
