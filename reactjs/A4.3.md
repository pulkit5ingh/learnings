## Goal
Handle complex data fetching patterns: infinite scroll, multiple simultaneous queries, queries that depend on previous query results, and prefetching for instant navigation.

## Background

### useInfiniteQuery
Paginated data where each page triggers a new request is handled by `useInfiniteQuery`. It maintains a list of pages rather than a single data object. You provide `getNextPageParam` to tell React Query what to pass as the cursor/page number for the next fetch. Call `fetchNextPage()` to load more. The `hasNextPage` flag tells you whether more pages exist.

### Parallel and Dependent Queries
When you need multiple pieces of data from different endpoints simultaneously, call `useQuery` multiple times — React Query runs them in parallel. When query B needs data from query A first (e.g., you need a user's ID to fetch their posts), set `enabled: !!userId` on query B so it waits until A provides the value.

### Prefetching
Call `queryClient.prefetchQuery()` to load data into the cache before the user navigates to a page. When they arrive, the data is already there — instant render. You can prefetch on hover over a link, or when you know the user is likely to navigate next.

### Task 1 — Infinite Scroll

```tsx
// src/api/posts.ts — add paginated fetch
export interface PaginatedPosts {
  posts: Post[]
  nextPage: number | null
  total: number
}

export async function fetchPostsPage(page: number, limit = 10): Promise<PaginatedPosts> {
  const res = await fetch(
    `https://jsonplaceholder.typicode.com/posts?_page=${page}&_limit=${limit}`
  )
  if (!res.ok) throw new Error(`HTTP ${res.status}`)
  const total = Number(res.headers.get('X-Total-Count') ?? 100)
  const posts: Post[] = await res.json()
  const hasMore = page * limit < total
  return { posts, nextPage: hasMore ? page + 1 : null, total }
}
```

```tsx
// src/components/InfinitePostList.tsx
import { useInfiniteQuery } from '@tanstack/react-query'
import { useRef, useEffect } from 'react'
import { fetchPostsPage } from '../api/posts'

export function InfinitePostList() {
  const {
    data,
    fetchNextPage,
    hasNextPage,
    isFetchingNextPage,
    isLoading,
  } = useInfiniteQuery({
    queryKey: ['posts', 'infinite'],
    queryFn: ({ pageParam }) => fetchPostsPage(pageParam),
    initialPageParam: 1,
    getNextPageParam: (lastPage) => lastPage.nextPage,
  })

  // Intersection Observer for automatic infinite scroll
  const sentinelRef = useRef<HTMLDivElement>(null)
  useEffect(() => {
    if (!sentinelRef.current) return
    const observer = new IntersectionObserver(
      ([entry]) => {
        if (entry.isIntersecting && hasNextPage && !isFetchingNextPage) {
          fetchNextPage()
        }
      },
      { threshold: 0.5 }
    )
    observer.observe(sentinelRef.current)
    return () => observer.disconnect()
  }, [fetchNextPage, hasNextPage, isFetchingNextPage])

  if (isLoading) return <p>Loading...</p>

  // data.pages is an array of page results
  const allPosts = data?.pages.flatMap(page => page.posts) ?? []

  return (
    <div>
      <h2>Posts ({data?.pages[0].total} total)</h2>
      <ul style={{ listStyle: 'none', padding: 0 }}>
        {allPosts.map(post => (
          <li key={post.id} style={{ borderBottom: '1px solid #f0f0f0', padding: '0.75rem 0' }}>
            <strong>{post.title}</strong>
            <p style={{ fontSize: 13, color: '#6b7280', margin: '4px 0 0' }}>{post.body.slice(0, 80)}…</p>
          </li>
        ))}
      </ul>

      {/* Sentinel element — triggers next page when scrolled into view */}
      <div ref={sentinelRef} style={{ height: 20 }} />

      {isFetchingNextPage && <p style={{ textAlign: 'center', color: '#6b7280' }}>Loading more...</p>}
      {!hasNextPage && allPosts.length > 0 && (
        <p style={{ textAlign: 'center', color: '#9ca3af', fontSize: 14 }}>You've reached the end</p>
      )}
    </div>
  )
}
```

### Task 2 — Parallel Queries

```tsx
// src/components/UserDashboard.tsx
import { useQueries } from '@tanstack/react-query'

const USER_IDS = [1, 2, 3]

export function MultiUserDashboard() {
  // useQueries runs all queries in parallel
  const results = useQueries({
    queries: USER_IDS.map(id => ({
      queryKey: ['users', id],
      queryFn: () => fetch(`https://jsonplaceholder.typicode.com/users/${id}`).then(r => r.json()),
      staleTime: 1000 * 60 * 10,
    })),
  })

  const loading = results.some(r => r.isLoading)
  const error = results.find(r => r.isError)

  if (loading) return <p>Loading users...</p>
  if (error) return <p style={{ color: '#ef4444' }}>Error loading users</p>

  return (
    <div style={{ display: 'grid', gridTemplateColumns: 'repeat(3, 1fr)', gap: '1rem' }}>
      {results.map((result, i) => (
        <div key={USER_IDS[i]} style={{ padding: '1rem', border: '1px solid #e5e7eb', borderRadius: 8 }}>
          <strong>{result.data?.name}</strong>
          <p style={{ fontSize: 13, color: '#6b7280', margin: '4px 0 0' }}>{result.data?.email}</p>
        </div>
      ))}
    </div>
  )
}
```

### Task 3 — Dependent Queries

```tsx
// src/components/UserPostsDependentQuery.tsx
import { useQuery } from '@tanstack/react-query'

export function UserPosts({ email }: { email: string }) {
  // Query 1: Find user by email
  const { data: user, isLoading: loadingUser } = useQuery({
    queryKey: ['users', { email }],
    queryFn: async () => {
      const res = await fetch(`https://jsonplaceholder.typicode.com/users?email=${email}`)
      const users = await res.json()
      return users[0] ?? null
    },
  })

  // Query 2: Fetch posts for that user — only runs when user.id is available
  const { data: posts, isLoading: loadingPosts } = useQuery({
    queryKey: ['posts', { userId: user?.id }],
    queryFn: () =>
      fetch(`https://jsonplaceholder.typicode.com/posts?userId=${user!.id}`)
        .then(r => r.json()),
    enabled: !!user?.id,  // dependent: wait for user query to succeed
  })

  if (loadingUser) return <p>Looking up user by email...</p>
  if (!user) return <p>No user found with email: {email}</p>
  if (loadingPosts) return <p>Loading posts for {user.name}...</p>

  return (
    <div>
      <h3>Posts by {user.name}</h3>
      <ul>
        {posts?.map((p: Post) => <li key={p.id}>{p.title}</li>)}
      </ul>
    </div>
  )
}
```

### Task 4 — Prefetching on Hover

```tsx
// src/components/PrefetchLink.tsx
import { Link } from 'react-router-dom'
import { useQueryClient } from '@tanstack/react-query'
import { fetchPost } from '../api/posts'

interface PrefetchLinkProps {
  postId: number
  title: string
}

export function PostLink({ postId, title }: PrefetchLinkProps) {
  const queryClient = useQueryClient()

  function handleMouseEnter() {
    // Prefetch when user hovers — usually loads before they click
    queryClient.prefetchQuery({
      queryKey: ['posts', postId],
      queryFn: () => fetchPost(postId),
      staleTime: 1000 * 60,  // don't prefetch if cache is fresh
    })
  }

  return (
    <Link
      to={`/posts/${postId}`}
      onMouseEnter={handleMouseEnter}
      style={{ color: '#1d4ed8', textDecoration: 'none' }}
    >
      {title}
    </Link>
  )
}
```
