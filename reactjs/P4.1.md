# P4.1 — GitHub Profile Explorer

**Phase:** 4 — Data Fetching & Server State
**Topic:** React Query, pagination, dependent queries, search, loading/error states
**Type:** Project

---

## Goal

Build a GitHub explorer that lets users search for GitHub profiles, view their repositories, and browse commit history. All data fetching uses React Query (TanStack Query v5). Practice dependent queries, pagination, and cache management.

---

## What to Build

- User search with username input
- User profile card (avatar, name, bio, followers, following, public repos)
- Repository list with pagination (30 per page)
- Commit list for selected repository (last 30 commits)
- Loading skeletons for each section
- Error states (user not found, API rate limit)
- Stale-while-revalidate behaviour visible on revisit

## Implementation

### QueryClient

```ts
// src/lib/queryClient.ts
import { QueryClient } from '@tanstack/react-query';

export const queryClient = new QueryClient({
  defaultOptions: {
    queries: {
      staleTime: 1000 * 60 * 5,     // 5 minutes
      gcTime: 1000 * 60 * 30,        // 30 minutes
      retry: (failureCount, error: unknown) => {
        // Don't retry 404 or 403 (rate limit)
        const status = (error as { status?: number })?.status;
        if (status === 404 || status === 403) return false;
        return failureCount < 2;
      },
    },
  },
});
```

### API helper

```ts
// src/lib/github.ts
import type { GitHubUser, GitHubRepo, GitHubCommit } from '../types/github';

const BASE = 'https://api.github.com';

async function githubFetch<T>(path: string): Promise<T> {
  const res = await fetch(`${BASE}${path}`, {
    headers: {
      Accept: 'application/vnd.github.v3+json',
      // Add Authorization if you have a PAT:
      // Authorization: `Bearer ${import.meta.env.VITE_GITHUB_TOKEN}`,
    },
  });

  if (!res.ok) {
    const err = { status: res.status, message: `GitHub API error: ${res.status}` };
    throw err;
  }

  return res.json();
}

export const githubApi = {
  getUser: (username: string) =>
    githubFetch<GitHubUser>(`/users/${username}`),

  getUserRepos: (username: string, page = 1) =>
    githubFetch<GitHubRepo[]>(
      `/users/${username}/repos?sort=updated&per_page=10&page=${page}`
    ),

  getRepoCommits: (owner: string, repo: string, page = 1) =>
    githubFetch<GitHubCommit[]>(
      `/repos/${owner}/${repo}/commits?per_page=20&page=${page}`
    ),
};
```

### Query hooks

```ts
// src/hooks/useGitHubUser.ts
import { useQuery } from '@tanstack/react-query';
import { githubApi } from '../lib/github';

export function useGitHubUser(username: string) {
  return useQuery({
    queryKey: ['github', 'user', username],
    queryFn: () => githubApi.getUser(username),
    enabled: username.length > 0,
    staleTime: 1000 * 60 * 5,
  });
}
```

```ts
// src/hooks/useUserRepos.ts
import { useQuery } from '@tanstack/react-query';
import { githubApi } from '../lib/github';

export function useUserRepos(username: string, page: number) {
  return useQuery({
    queryKey: ['github', 'repos', username, page],
    queryFn: () => githubApi.getUserRepos(username, page),
    enabled: username.length > 0,
    // Keep previous page data while fetching next
    placeholderData: previousData => previousData,
  });
}
```

```ts
// src/hooks/useRepoCommits.ts
import { useQuery } from '@tanstack/react-query';
import { githubApi } from '../lib/github';

export function useRepoCommits(owner: string, repo: string | null, page: number) {
  return useQuery({
    queryKey: ['github', 'commits', owner, repo, page],
    queryFn: () => githubApi.getRepoCommits(owner, repo!, page),
    // Dependent query — only runs when a repo is selected
    enabled: !!repo,
    placeholderData: previousData => previousData,
  });
}
```

### App

```tsx
// src/App.tsx
import { useState } from 'react';
import { QueryClientProvider } from '@tanstack/react-query';
import { ReactQueryDevtools } from '@tanstack/react-query-devtools';
import { queryClient } from './lib/queryClient';
import { UserSearch } from './components/UserSearch';
import { UserProfile } from './components/UserProfile';
import { RepoList } from './components/RepoList';
import { CommitList } from './components/CommitList';
import { useGitHubUser } from './hooks/useGitHubUser';
import { useUserRepos } from './hooks/useUserRepos';
import { useRepoCommits } from './hooks/useRepoCommits';

function Explorer() {
  const [username, setUsername] = useState('');
  const [reposPage, setReposPage] = useState(1);
  const [selectedRepo, setSelectedRepo] = useState<string | null>(null);
  const [commitsPage, setCommitsPage] = useState(1);

  const userQuery = useGitHubUser(username);
  const reposQuery = useUserRepos(username, reposPage);
  const commitsQuery = useRepoCommits(username, selectedRepo, commitsPage);

  const handleUserSearch = (u: string) => {
    setUsername(u);
    setReposPage(1);
    setSelectedRepo(null);
  };

  return (
    <div className="max-w-4xl mx-auto p-6 space-y-6">
      <h1 className="text-2xl font-bold">GitHub Explorer</h1>

      <UserSearch onSearch={handleUserSearch} />

      {userQuery.isError && (
        <p className="text-red-500">
          {(userQuery.error as { status: number }).status === 404
            ? `User "${username}" not found.`
            : 'Failed to load user. Check your rate limit.'}
        </p>
      )}

      {userQuery.isLoading && username && <div className="animate-pulse h-32 bg-gray-200 rounded-2xl" />}

      {userQuery.data && (
        <UserProfile user={userQuery.data} />
      )}

      {reposQuery.data && (
        <RepoList
          repos={reposQuery.data}
          page={reposPage}
          onPageChange={p => { setReposPage(p); setSelectedRepo(null); }}
          selectedRepo={selectedRepo}
          onSelectRepo={name => { setSelectedRepo(name); setCommitsPage(1); }}
          isFetching={reposQuery.isFetching}
        />
      )}

      {selectedRepo && commitsQuery.data && (
        <CommitList
          commits={commitsQuery.data}
          repo={selectedRepo}
          page={commitsPage}
          onPageChange={setCommitsPage}
          isFetching={commitsQuery.isFetching}
        />
      )}
    </div>
  );
}

export default function App() {
  return (
    <QueryClientProvider client={queryClient}>
      <Explorer />
      <ReactQueryDevtools initialIsOpen={false} />
    </QueryClientProvider>
  );
}
```
