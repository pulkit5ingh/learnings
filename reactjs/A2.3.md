# A2.3 — useRef & useMemo/useCallback

**Phase:** 2 — Intermediate Hooks & Patterns
**Topic:** Ref for DOM access, ref for mutable values, memo for expensive computation, callback memoization
**Type:** Assignment

---

## Goal
Learn when and how to use useRef for imperative DOM access and mutable values, and useMemo/useCallback to avoid unnecessary recomputation and re-renders.

## Background

### useRef — Two Use Cases
`useRef` creates a mutable container whose `.current` property persists across renders without triggering a re-render when changed. Use case 1: DOM access — attach a ref to a JSX element via the `ref` prop to get direct access to the underlying DOM node (for focus, scrolling, measuring, integrating with non-React libraries). Use case 2: mutable instance variable — store a value that needs to persist across renders but shouldn't trigger re-renders (previous value, timer IDs, abort controllers, flags).

### useMemo — Expensive Computations
`useMemo(fn, deps)` memoizes the result of a computation, re-running only when dependencies change. Use it when you have a genuinely expensive calculation (sorting/filtering large arrays, complex transforms) that runs on every render. Do not use it for simple computations — the overhead of memoization itself can be more expensive than just recomputing.

### useCallback — Stable Function References
`useCallback(fn, deps)` memoizes a function reference. The function is recreated only when dependencies change. This matters when you pass callbacks to child components wrapped in `React.memo`, or when a callback is a dependency of another `useEffect`. Without `useCallback`, a new function reference is created every render, causing `React.memo` children to re-render unnecessarily.

### Task 1 — useRef for DOM Access

```tsx
// src/components/AutoFocusInput.tsx
import { useRef, useEffect } from 'react'

export function AutoFocusInput() {
  const inputRef = useRef<HTMLInputElement>(null)

  useEffect(() => {
    // Focus the input on mount
    inputRef.current?.focus()
  }, [])

  return (
    <div>
      <label>
        Search (auto-focused):
        <input
          ref={inputRef}
          type="search"
          placeholder="Start typing..."
          style={{ marginLeft: 8, padding: '6px 10px' }}
        />
      </label>
      <button
        onClick={() => {
          inputRef.current?.focus()
          inputRef.current?.select()  // select all text
        }}
      >
        Re-focus
      </button>
    </div>
  )
}
```

// src/components/ScrollableList.tsx
import { useRef } from 'react'

const items = Array.from({ length: 50 }, (_, i) => ({ id: i + 1, text: `Item ${i + 1}` }))

export function ScrollableList() {
  const listRef = useRef<HTMLUListElement>(null)
  const bottomRef = useRef<HTMLLIElement>(null)

  function scrollToTop() {
    listRef.current?.scrollTo({ top: 0, behavior: 'smooth' })
  }

  function scrollToBottom() {
    bottomRef.current?.scrollIntoView({ behavior: 'smooth' })
  }

  return (
    <div>
      <div style={{ marginBottom: 8, display: 'flex', gap: 8 }}>
        <button onClick={scrollToTop}>Top</button>
        <button onClick={scrollToBottom}>Bottom</button>
      </div>
      <ul
        ref={listRef}
        style={{ height: 200, overflowY: 'auto', border: '1px solid #e5e7eb', padding: '0.5rem', listStyle: 'none', margin: 0 }}
      >
        {items.map(item => (
          <li key={item.id} style={{ padding: '4px 0', borderBottom: '1px solid #f0f0f0' }}>
            {item.text}
          </li>
        ))}
        <li ref={bottomRef} />
      </ul>
    </div>
  )
}

### Task 2 — useRef as Instance Variable

```tsx
// src/components/Stopwatch.tsx
import { useState, useRef, useEffect } from 'react'

export function StopwatchWithRef() {
  const [elapsed, setElapsed] = useState(0)
  const [running, setRunning] = useState(false)

  // Store interval ID in a ref — changing it does not trigger re-render
  const intervalRef = useRef<ReturnType<typeof setInterval> | null>(null)
  const startTimeRef = useRef<number>(0)

  function start() {
    startTimeRef.current = Date.now() - elapsed
    intervalRef.current = setInterval(() => {
      setElapsed(Date.now() - startTimeRef.current)
    }, 10)
    setRunning(true)
  }

  function pause() {
    if (intervalRef.current) clearInterval(intervalRef.current)
    setRunning(false)
  }

  function reset() {
    if (intervalRef.current) clearInterval(intervalRef.current)
    setElapsed(0)
    setRunning(false)
  }

  // Cleanup on unmount
  useEffect(() => {
    return () => {
      if (intervalRef.current) clearInterval(intervalRef.current)
    }
  }, [])

  const seconds = (elapsed / 1000).toFixed(2)

  return (
    <div>
      <div style={{ fontFamily: 'monospace', fontSize: 40 }}>{seconds}s</div>
      <div style={{ display: 'flex', gap: 8 }}>
        {running ? (
          <button onClick={pause}>Pause</button>
        ) : (
          <button onClick={start}>{elapsed > 0 ? 'Resume' : 'Start'}</button>
        )}
        <button onClick={reset} disabled={elapsed === 0 && !running}>Reset</button>
      </div>
    </div>
  )
}
```

### Task 3 — useMemo for Expensive Computation

```tsx
// src/components/FilteredList.tsx
import { useState, useMemo } from 'react'

// Simulate a large dataset
const allProducts = Array.from({ length: 5000 }, (_, i) => ({
  id: i + 1,
  name: `Product ${i + 1}`,
  category: ['Electronics', 'Clothing', 'Books', 'Food'][i % 4],
  price: Math.round(Math.random() * 500 * 100) / 100,
  rating: Math.round(Math.random() * 50) / 10,
}))

export function FilteredProductList() {
  const [search, setSearch] = useState('')
  const [category, setCategory] = useState('All')
  const [sortBy, setSortBy] = useState<'price' | 'rating' | 'name'>('name')

  // Expensive: filter + sort 5000 items
  // Without useMemo, this runs on every render (even unrelated state changes)
  const filtered = useMemo(() => {
    console.log('Recomputing filtered list...')  // watch the console
    const lowerSearch = search.toLowerCase()
    return allProducts
      .filter(p => {
        const matchesSearch = p.name.toLowerCase().includes(lowerSearch)
        const matchesCategory = category === 'All' || p.category === category
        return matchesSearch && matchesCategory
      })
      .sort((a, b) => {
        if (sortBy === 'price') return a.price - b.price
        if (sortBy === 'rating') return b.rating - a.rating
        return a.name.localeCompare(b.name)
      })
      .slice(0, 20)  // show first 20 results
  }, [search, category, sortBy])  // only recompute when these change

  return (
    <div style={{ maxWidth: 600 }}>
      <div style={{ display: 'flex', gap: 8, marginBottom: 12, flexWrap: 'wrap' }}>
        <input
          type="search"
          value={search}
          onChange={e => setSearch(e.target.value)}
          placeholder="Search products..."
          style={{ flex: 1, padding: '6px 10px' }}
        />
        <select value={category} onChange={e => setCategory(e.target.value)}>
          {['All', 'Electronics', 'Clothing', 'Books', 'Food'].map(c => (
            <option key={c}>{c}</option>
          ))}
        </select>
        <select value={sortBy} onChange={e => setSortBy(e.target.value as typeof sortBy)}>
          <option value="name">Name</option>
          <option value="price">Price</option>
          <option value="rating">Rating</option>
        </select>
      </div>

      <p style={{ color: '#6b7280', fontSize: 14 }}>{filtered.length} results</p>

      <table style={{ width: '100%', borderCollapse: 'collapse', fontSize: 14 }}>
        <thead>
          <tr style={{ background: '#f9fafb' }}>
            <th style={{ textAlign: 'left', padding: '8px', borderBottom: '1px solid #e5e7eb' }}>Name</th>
            <th style={{ padding: '8px', borderBottom: '1px solid #e5e7eb' }}>Category</th>
            <th style={{ padding: '8px', borderBottom: '1px solid #e5e7eb' }}>Price</th>
            <th style={{ padding: '8px', borderBottom: '1px solid #e5e7eb' }}>Rating</th>
          </tr>
        </thead>
        <tbody>
          {filtered.map(p => (
            <tr key={p.id} style={{ borderBottom: '1px solid #f0f0f0' }}>
              <td style={{ padding: '8px' }}>{p.name}</td>
              <td style={{ padding: '8px', textAlign: 'center' }}>{p.category}</td>
              <td style={{ padding: '8px', textAlign: 'right' }}>${p.price.toFixed(2)}</td>
              <td style={{ padding: '8px', textAlign: 'center' }}>{p.rating}★</td>
            </tr>
          ))}
        </tbody>
      </table>
    </div>
  )
}
```

### Task 4 — useCallback with React.memo

```tsx
// src/components/MemoDemo.tsx
import { useState, useCallback, memo } from 'react'

// Without memo, this re-renders every time parent renders
// With memo, it only re-renders when its props change
const ExpensiveChild = memo(function ExpensiveChild({
  label,
  onClick,
}: {
  label: string
  onClick: () => void
}) {
  console.log(`Rendering: ${label}`)  // watch the console
  return (
    <button
      onClick={onClick}
      style={{ padding: '8px 16px', margin: 4, background: '#f0f9ff', border: '1px solid #bae6fd', borderRadius: 4 }}
    >
      {label}
    </button>
  )
})

export function MemoDemo() {
  const [count, setCount] = useState(0)
  const [otherState, setOtherState] = useState(0)

  // WITHOUT useCallback: new function reference on every render
  // → memo'd child re-renders even when count didn't change
  const handleCountClick = useCallback(() => {
    setCount(c => c + 1)
  }, [])  // empty deps: function never changes

  // WITH useCallback: stable reference → memo'd child doesn't re-render
  const handleOtherClick = useCallback(() => {
    setOtherState(s => s + 1)
  }, [])

  return (
    <div>
      <p>Count: {count} | Other: {otherState}</p>
      <p style={{ fontSize: 13, color: '#6b7280' }}>Open console to see which children re-render</p>

      <ExpensiveChild label="Increment count" onClick={handleCountClick} />
      <ExpensiveChild label="Change other state" onClick={handleOtherClick} />

      <button onClick={() => setCount(c => c + 1)} style={{ marginLeft: 8 }}>
        Force parent re-render
      </button>
    </div>
  )
}
```
