## Goal
Navigate programmatically — after a form submission, after login/logout, or in response to any non-link user action.

## Background

### useNavigate
`useNavigate()` returns a function you call to navigate. Pass a path string to go to that route, or pass a number (`-1`, `-2`) to go back/forward in history. The second argument is an options object: `{ replace: true }` replaces the current history entry instead of pushing a new one (so the user can't go back to where they were), and `{ state: {...} }` passes data to the target route without encoding it in the URL.

### When to Use navigate() vs Link
Use `<Link>` for navigation the user can see and initiate directly (navbar links, "View details" buttons). Use `navigate()` for navigation that happens as a side effect of an action: after submitting a form, after a delete operation, after logging in, after a timeout.

### Replace vs Push
When you redirect a user because they're not authenticated, use `replace: true`. If you use push, they'll see the protected page in their history — pressing back takes them to the protected page and they get redirected again. With replace, the redirect overwrites the protected page in history.

### Task 1 — Navigate After Form Submit

```tsx
// src/pages/NewPostPage.tsx
import { useState } from 'react'
import { useNavigate } from 'react-router-dom'

export function NewPostPage() {
  const navigate = useNavigate()
  const [form, setForm] = useState({ title: '', body: '' })
  const [submitting, setSubmitting] = useState(false)
  const [error, setError] = useState('')

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setSubmitting(true)
    setError('')

    try {
      const res = await fetch('https://jsonplaceholder.typicode.com/posts', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ ...form, userId: 1 }),
      })
      const data = await res.json()

      // Navigate to the new post's detail page after successful creation
      navigate(`/posts/${data.id}`, {
        state: { newPost: true, title: data.title },  // pass data to destination
      })
    } catch {
      setError('Failed to create post. Please try again.')
      setSubmitting(false)
    }
  }

  return (
    <div style={{ maxWidth: 500 }}>
      <h1>New Post</h1>
      {error && <p style={{ color: '#ef4444' }}>{error}</p>}
      <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '1rem' }}>
        <label>
          Title
          <input
            type="text"
            value={form.title}
            onChange={e => setForm(f => ({ ...f, title: e.target.value }))}
            required
            style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}
          />
        </label>
        <label>
          Body
          <textarea
            value={form.body}
            onChange={e => setForm(f => ({ ...f, body: e.target.value }))}
            rows={5}
            required
            style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}
          />
        </label>
        <div style={{ display: 'flex', gap: 8 }}>
          <button type="submit" disabled={submitting}>
            {submitting ? 'Creating...' : 'Create Post'}
          </button>
          <button type="button" onClick={() => navigate(-1)}>Cancel</button>
        </div>
      </form>
    </div>
  )
}
```

### Task 2 — Reading Navigation State

```tsx
// src/pages/PostDetailPage.tsx
import { useParams, useLocation, Link, useNavigate } from 'react-router-dom'

export function PostDetailPage() {
  const { postId } = useParams()
  const location = useLocation()
  const navigate = useNavigate()

  // Read state passed from navigate()
  const locationState = location.state as { newPost?: boolean; title?: string } | null

  return (
    <div>
      {locationState?.newPost && (
        <div style={{ background: '#dcfce7', border: '1px solid #bbf7d0', borderRadius: 6, padding: '10px 16px', marginBottom: '1rem' }}>
          ✓ Post created successfully!
        </div>
      )}
      <h1>Post {postId}</h1>
      <div style={{ display: 'flex', gap: 8 }}>
        <button onClick={() => navigate(-1)}>← Back</button>
        <Link to="/posts">All Posts</Link>
      </div>
    </div>
  )
}
```

### Task 3 — Auth Guard with Redirect

```tsx
// src/components/RequireAuth.tsx
import { Navigate, useLocation, Outlet } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'

export function RequireAuth() {
  const { isAuthenticated } = useAuth()
  const location = useLocation()

  if (!isAuthenticated) {
    // Replace: true — so back button doesn't loop back to protected page
    // state.from: remember where the user was trying to go
    return <Navigate to="/login" state={{ from: location.pathname }} replace />
  }

  return <Outlet />
}
```

```tsx
// src/pages/LoginPage.tsx
import { useState } from 'react'
import { useNavigate, useLocation } from 'react-router-dom'
import { useAuth } from '../context/AuthContext'

export function LoginPage() {
  const navigate = useNavigate()
  const location = useLocation()
  const { login, loading } = useAuth()
  const [email, setEmail] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState('')

  // Where did the user come from?
  const from = (location.state as { from?: string })?.from ?? '/'

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setError('')
    try {
      await login(email, password)
      // Send them back to where they were trying to go
      navigate(from, { replace: true })
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Login failed')
    }
  }

  return (
    <div style={{ maxWidth: 360, margin: '4rem auto' }}>
      <h1>Sign In</h1>
      {from !== '/' && (
        <p style={{ color: '#92400e', background: '#fffbeb', padding: '8px 12px', borderRadius: 4, fontSize: 14 }}>
          Please sign in to access {from}
        </p>
      )}
      {error && <p style={{ color: '#ef4444' }}>{error}</p>}
      <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '1rem' }}>
        <input type="email" value={email} onChange={e => setEmail(e.target.value)} placeholder="Email" required style={{ padding: '8px 10px' }} />
        <input type="password" value={password} onChange={e => setPassword(e.target.value)} placeholder="Password" required style={{ padding: '8px 10px' }} />
        <button type="submit" disabled={loading}>{loading ? 'Signing in...' : 'Sign In'}</button>
      </form>
      <p style={{ fontSize: 13, color: '#6b7280', marginTop: '1rem' }}>
        Hint: use any email + password "password123"
      </p>
    </div>
  )
}
```

### Task 5 — Navigate with Go Back / Go Forward

```tsx
// src/components/NavigationControls.tsx
import { useNavigate } from 'react-router-dom'

export function NavigationControls() {
  const navigate = useNavigate()

  return (
    <div style={{ display: 'flex', gap: 4 }}>
      <button onClick={() => navigate(-1)} title="Back">←</button>
      <button onClick={() => navigate(1)} title="Forward">→</button>
      <button onClick={() => navigate('/', { replace: true })} title="Home">⌂</button>
    </div>
  )
}
```
