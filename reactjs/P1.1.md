# P1.1 — Weather App

**Phase:** 1 — React Fundamentals
**Topic:** Fetching APIs, conditional rendering, loading/error states, component composition
**Type:** Project

---

## Goal

Build a weather app that fetches current conditions and a 5-day forecast from the OpenWeatherMap API. Practice composing components, managing loading and error states with `useState` + `useEffect`, and rendering lists with keys.

---

## What to Build

- City search input with submit
- Current weather card (temperature, feels-like, humidity, wind speed, icon)
- 5-day forecast strip (one card per day)
- Loading skeleton while fetching
- Error banner for city-not-found or network errors
- Celsius / Fahrenheit toggle

### Custom hook

```ts
// src/hooks/useWeather.ts
import { useState, useCallback } from 'react';
import type { CurrentWeather, ForecastDay, Unit } from '../types/weather';

const API_KEY = import.meta.env.VITE_OPENWEATHER_KEY;
const BASE = 'https://api.openweathermap.org/data/2.5';

interface WeatherState {
  current: CurrentWeather | null;
  forecast: ForecastDay[];
  loading: boolean;
  error: string | null;
}

export function useWeather(unit: Unit) {
  const [state, setState] = useState<WeatherState>({
    current: null,
    forecast: [],
    loading: false,
    error: null,
  });

  const search = useCallback(async (city: string) => {
    setState(s => ({ ...s, loading: true, error: null }));

    try {
      const [weatherRes, forecastRes] = await Promise.all([
        fetch(`${BASE}/weather?q=${encodeURIComponent(city)}&units=${unit}&appid=${API_KEY}`),
        fetch(`${BASE}/forecast?q=${encodeURIComponent(city)}&units=${unit}&cnt=40&appid=${API_KEY}`),
      ]);

      if (weatherRes.status === 404) throw new Error(`City "${city}" not found.`);
      if (!weatherRes.ok) throw new Error('Failed to fetch weather data.');

      const w = await weatherRes.json();
      const f = await forecastRes.json();

      const current: CurrentWeather = {
        city: w.name,
        country: w.sys.country,
        temp: Math.round(w.main.temp),
        feelsLike: Math.round(w.main.feels_like),
        humidity: w.main.humidity,
        windSpeed: Math.round(w.wind.speed),
        condition: w.weather[0],
        sunrise: w.sys.sunrise,
        sunset: w.sys.sunset,
      };

      // One entry per day (take the noon reading)
      const days = new Map<string, ForecastDay>();
      for (const item of f.list) {
        const date = new Date(item.dt * 1000);
        const key = date.toDateString();
        if (!days.has(key)) {
          days.set(key, {
            date: item.dt,
            tempMin: item.main.temp_min,
            tempMax: item.main.temp_max,
            condition: item.weather[0],
          });
        } else {
          const d = days.get(key)!;
          d.tempMin = Math.min(d.tempMin, item.main.temp_min);
          d.tempMax = Math.max(d.tempMax, item.main.temp_max);
        }
      }

      setState({
        current,
        forecast: Array.from(days.values()).slice(0, 5),
        loading: false,
        error: null,
      });
    } catch (err) {
      setState(s => ({
        ...s,
        loading: false,
        error: err instanceof Error ? err.message : 'Unknown error',
      }));
    }
  }, [unit]);

  return { ...state, search };
}
```

### SearchBar

```tsx
// src/components/SearchBar.tsx
import { useState } from 'react';

interface SearchBarProps {
  onSearch: (city: string) => void;
  loading: boolean;
}

export function SearchBar({ onSearch, loading }: SearchBarProps) {
  const [value, setValue] = useState('');

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    const trimmed = value.trim();
    if (trimmed) onSearch(trimmed);
  };

  return (
    <form onSubmit={handleSubmit} className="flex gap-2">
      <input
        type="search"
        value={value}
        onChange={e => setValue(e.target.value)}
        placeholder="Enter city name..."
        className="flex-1 px-4 py-2 rounded-lg border border-gray-300 focus:outline-none focus:ring-2 focus:ring-blue-400"
        disabled={loading}
        aria-label="City name"
      />
      <button
        type="submit"
        disabled={loading || !value.trim()}
        className="px-6 py-2 bg-blue-500 text-white rounded-lg disabled:opacity-50 hover:bg-blue-600 transition"
      >
        {loading ? 'Searching...' : 'Search'}
      </button>
    </form>
  );
}
```

```tsx
import type { CurrentWeather as ICurrentWeather, Unit } from '../types/weather';
import { formatTime } from '../utils/formatters';

interface Props {
  data: ICurrentWeather;
  unit: Unit;
}

export function CurrentWeather({ data, unit }: Props) {
  const unitSymbol = unit === 'metric' ? '°C' : '°F';
  const windUnit = unit === 'metric' ? 'm/s' : 'mph';

  return (
    <div className="bg-gradient-to-br from-blue-500 to-blue-700 text-white rounded-2xl p-6 shadow-lg">
      <div className="flex justify-between items-start">
        <div>
          <h2 className="text-3xl font-bold">{data.city}, {data.country}</h2>
          <p className="text-blue-100 capitalize">{data.condition.description}</p>
        </div>
        <img
          src={`https://openweathermap.org/img/wn/${data.condition.icon}@2x.png`}
          alt={data.condition.description}
          className="w-20 h-20"
        />
      </div>

      <div className="text-6xl font-thin my-4">
        {data.temp}{unitSymbol}
      </div>

      <div className="grid grid-cols-3 gap-4 text-sm text-blue-100 mt-4">
        <div>
          <div className="font-semibold text-white">Feels like</div>
          {data.feelsLike}{unitSymbol}
        </div>
        <div>
          <div className="font-semibold text-white">Humidity</div>
          {data.humidity}%
        </div>
        <div>
          <div className="font-semibold text-white">Wind</div>
          {data.windSpeed} {windUnit}
        </div>
      </div>

      <div className="flex gap-6 mt-4 text-sm text-blue-100">
        <span>Sunrise: {formatTime(data.sunrise)}</span>
        <span>Sunset: {formatTime(data.sunset)}</span>
      </div>
    </div>
  );
}
```

```tsx
// src/components/ForecastCard.tsx
import type { ForecastDay, Unit } from '../types/weather';

interface Props {
  day: ForecastDay;
  unit: Unit;
}

export function ForecastCard({ day, unit }: Props) {
  const symbol = unit === 'metric' ? '°C' : '°F';
  const date = new Date(day.date * 1000);
  const dayName = date.toLocaleDateString('en-US', { weekday: 'short' });

  return (
    <div className="bg-white rounded-xl p-3 shadow text-center">
      <div className="font-semibold text-gray-600">{dayName}</div>
      <img
        src={`https://openweathermap.org/img/wn/${day.condition.icon}.png`}
        alt={day.condition.description}
        className="w-10 h-10 mx-auto"
      />
      <div className="text-sm">
        <span className="font-bold text-gray-800">{Math.round(day.tempMax)}{symbol}</span>
        <span className="text-gray-400 ml-1">{Math.round(day.tempMin)}{symbol}</span>
      </div>
    </div>
  );
}
```

### App

```tsx
// src/App.tsx
import { useState } from 'react';
import { SearchBar } from './components/SearchBar';
import { CurrentWeather } from './components/CurrentWeather';
import { ForecastStrip } from './components/ForecastStrip';
import { ErrorBanner } from './components/ErrorBanner';
import { useWeather } from './hooks/useWeather';
import type { Unit } from './types/weather';

export default function App() {
  const [unit, setUnit] = useState<Unit>('metric');
  const { current, forecast, loading, error, search } = useWeather(unit);

  return (
    <div className="min-h-screen bg-gray-100 p-6">
      <div className="max-w-2xl mx-auto space-y-6">
        <div className="flex items-center justify-between">
          <h1 className="text-2xl font-bold text-gray-800">Weather App</h1>
          <button
            onClick={() => setUnit(u => u === 'metric' ? 'imperial' : 'metric')}
            className="px-3 py-1 border rounded-full text-sm text-gray-600 hover:bg-gray-200 transition"
          >
            {unit === 'metric' ? '°C → °F' : '°F → °C'}
          </button>
        </div>

        <SearchBar onSearch={search} loading={loading} />

        {error && <ErrorBanner message={error} />}

        {loading && (
          <div className="text-center text-gray-500 py-10">Loading weather data...</div>
        )}

        {!loading && current && (
          <>
            <CurrentWeather data={current} unit={unit} />
            {forecast.length > 0 && <ForecastStrip days={forecast} unit={unit} />}
          </>
        )}

        {!loading && !current && !error && (
          <p className="text-center text-gray-400 py-10">
            Search for a city to see the weather.
          </p>
        )}
      </div>
    </div>
  );
}
```
