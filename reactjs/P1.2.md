# P1.2 — Movie Search

**Phase:** 1 — React Fundamentals
**Topic:** API fetching, component composition, localStorage persistence, conditional rendering
**Type:** Project

---

## Goal

Build a movie search app using the OMDB API. Users can search for movies, browse a results grid, view a detail page (in a modal or separate view), and save/remove favorites that persist in localStorage.

---

## What to Build

- Search input with debounced requests
- Results grid of movie cards (poster, title, year, type)
- Movie detail view with full plot, ratings, director, cast
- Favorites list with localStorage persistence
- Loading and error states
- Empty state when no results

### Hooks

```ts
// src/hooks/useMovieSearch.ts
import { useState, useEffect } from 'react';
import type { MovieSummary } from '../types/movie';

const API_KEY = import.meta.env.VITE_OMDB_KEY;
const BASE = 'https://www.omdbapi.com';

export function useMovieSearch(query: string) {
  const [movies, setMovies] = useState<MovieSummary[]>([]);
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);
  const [totalResults, setTotalResults] = useState(0);

  useEffect(() => {
    if (!query.trim()) {
      setMovies([]);
      setTotalResults(0);
      return;
    }

    const controller = new AbortController();

    const fetchMovies = async () => {
      setLoading(true);
      setError(null);

      try {
        const res = await fetch(
          `${BASE}/?apikey=${API_KEY}&s=${encodeURIComponent(query)}&type=movie`,
          { signal: controller.signal }
        );
        const data = await res.json();

        if (data.Response === 'False') {
          setMovies([]);
          setError(data.Error);
          setTotalResults(0);
        } else {
          setMovies(data.Search);
          setTotalResults(Number(data.totalResults));
        }
      } catch (err) {
        if ((err as Error).name !== 'AbortError') {
          setError('Network error. Please try again.');
        }
      } finally {
        setLoading(false);
      }
    };

    fetchMovies();
    return () => controller.abort();
  }, [query]);

  return { movies, loading, error, totalResults };
}
```

```ts
// src/hooks/useMovieDetail.ts
import { useState, useEffect } from 'react';
import type { MovieDetail } from '../types/movie';

const API_KEY = import.meta.env.VITE_OMDB_KEY;

export function useMovieDetail(imdbID: string | null) {
  const [detail, setDetail] = useState<MovieDetail | null>(null);
  const [loading, setLoading] = useState(false);

  useEffect(() => {
    if (!imdbID) return;

    setLoading(true);
    fetch(`https://www.omdbapi.com/?apikey=${API_KEY}&i=${imdbID}&plot=full`)
      .then(r => r.json())
      .then(data => {
        if (data.Response !== 'False') setDetail(data);
      })
      .finally(() => setLoading(false));
  }, [imdbID]);

  return { detail, loading };
}
```

```ts
// src/hooks/useFavorites.ts
import { useState, useCallback } from 'react';
import type { MovieSummary } from '../types/movie';

const STORAGE_KEY = 'movie-favorites';

function loadFavorites(): MovieSummary[] {
  try {
    return JSON.parse(localStorage.getItem(STORAGE_KEY) ?? '[]');
  } catch {
    return [];
  }
}

export function useFavorites() {
  const [favorites, setFavorites] = useState<MovieSummary[]>(loadFavorites);

  const toggle = useCallback((movie: MovieSummary) => {
    setFavorites(prev => {
      const exists = prev.some(m => m.imdbID === movie.imdbID);
      const next = exists
        ? prev.filter(m => m.imdbID !== movie.imdbID)
        : [...prev, movie];
      localStorage.setItem(STORAGE_KEY, JSON.stringify(next));
      return next;
    });
  }, []);

  const isFavorite = useCallback(
    (imdbID: string) => favorites.some(m => m.imdbID === imdbID),
    [favorites]
  );

  return { favorites, toggle, isFavorite };
}
```

```tsx
// src/components/MovieCard.tsx
import type { MovieSummary } from '../types/movie';

interface Props {
  movie: MovieSummary;
  isFavorite: boolean;
  onSelect: (id: string) => void;
  onToggleFavorite: (movie: MovieSummary) => void;
}

const PLACEHOLDER = 'https://via.placeholder.com/300x445?text=No+Poster';

export function MovieCard({ movie, isFavorite, onSelect, onToggleFavorite }: Props) {
  return (
    <div className="bg-white rounded-xl shadow hover:shadow-md transition cursor-pointer group overflow-hidden">
      <div className="relative" onClick={() => onSelect(movie.imdbID)}>
        <img
          src={movie.Poster !== 'N/A' ? movie.Poster : PLACEHOLDER}
          alt={`${movie.Title} poster`}
          className="w-full h-64 object-cover"
          loading="lazy"
        />
        <button
          aria-label={isFavorite ? `Remove ${movie.Title} from favorites` : `Add ${movie.Title} to favorites`}
          onClick={e => {
            e.stopPropagation();
            onToggleFavorite(movie);
          }}
          className="absolute top-2 right-2 text-2xl"
        >
          {isFavorite ? '❤️' : '🤍'}
        </button>
      </div>
      <div className="p-3" onClick={() => onSelect(movie.imdbID)}>
        <h3 className="font-semibold text-gray-800 line-clamp-2">{movie.Title}</h3>
        <p className="text-sm text-gray-500">{movie.Year}</p>
      </div>
    </div>
  );
}
```

### App

```tsx
// src/App.tsx
import { useState } from 'react';
import { useMovieSearch } from './hooks/useMovieSearch';
import { useMovieDetail } from './hooks/useMovieDetail';
import { useFavorites } from './hooks/useFavorites';
import { SearchBar } from './components/SearchBar';
import { MovieGrid } from './components/MovieGrid';
import { MovieDetail } from './components/MovieDetail';
import { FavoritesList } from './components/FavoritesList';
import { useDebounce } from './hooks/useDebounce';

export default function App() {
  const [query, setQuery] = useState('');
  const [selectedId, setSelectedId] = useState<string | null>(null);
  const [tab, setTab] = useState<'search' | 'favorites'>('search');

  const debouncedQuery = useDebounce(query, 400);
  const { movies, loading, error } = useMovieSearch(debouncedQuery);
  const { detail, loading: detailLoading } = useMovieDetail(selectedId);
  const { favorites, toggle, isFavorite } = useFavorites();

  return (
    <div className="min-h-screen bg-gray-50">
      <header className="bg-gray-900 text-white py-4 px-6">
        <div className="max-w-5xl mx-auto flex items-center justify-between">
          <h1 className="text-xl font-bold">🎬 Movie Search</h1>
          <div className="flex gap-3">
            <button
              onClick={() => setTab('search')}
              className={tab === 'search' ? 'text-yellow-400 font-semibold' : 'text-gray-300'}
            >
              Search
            </button>
            <button
              onClick={() => setTab('favorites')}
              className={tab === 'favorites' ? 'text-yellow-400 font-semibold' : 'text-gray-300'}
            >
              Favorites ({favorites.length})
            </button>
          </div>
        </div>
      </header>

      <main className="max-w-5xl mx-auto p-6 space-y-6">
        {tab === 'search' && (
          <>
            <SearchBar value={query} onChange={setQuery} />
            {error && <p className="text-red-500">{error}</p>}
            <MovieGrid
              movies={movies}
              loading={loading}
              isFavorite={isFavorite}
              onSelect={setSelectedId}
              onToggleFavorite={toggle}
            />
          </>
        )}

        {tab === 'favorites' && (
          <FavoritesList
            favorites={favorites}
            onSelect={setSelectedId}
            onToggleFavorite={toggle}
          />
        )}
      </main>

      {selectedId && detail && !detailLoading && (
        <MovieDetail
          detail={detail}
          isFavorite={isFavorite(detail.imdbID)}
          onToggleFavorite={() => toggle(detail)}
          onClose={() => setSelectedId(null)}
        />
      )}
    </div>
  );
}
```
