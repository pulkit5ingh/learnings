# P7.2 — Virtual List

**Phase:** 7 — Performance
**Topic:** TanStack Virtual, 100 k-row table, sort/filter, bundle analysis, LCP measurement
**Type:** Project

---

## Goal

Build a 100,000-row virtualized data table using TanStack Virtual. Add sort and filter without blocking the main thread, analyse the final bundle with `rollup-plugin-visualizer`, and measure Largest Contentful Paint (LCP).

---

## What to Build

- Virtual table of 100,000 rows — only DOM nodes in the viewport exist
- Column headers with sortable click (ascending / descending)
- Filter input per column (debounced)
- Global search input
- Row count indicator ("Showing X of 100,000")
- Scroll-to-top button
- Sticky column headers during scroll
- Bundle size report (dist/stats.html)

### Generate 100k rows

```ts
// src/data/generateRows.ts
export interface Row {
  id: number;
  name: string;
  email: string;
  department: string;
  salary: number;
  status: 'active' | 'inactive' | 'pending';
  joinDate: string;
}

const departments = ['Engineering', 'Marketing', 'Sales', 'HR', 'Finance', 'Design', 'Legal'];
const statuses: Row['status'][] = ['active', 'inactive', 'pending'];
const firstNames = ['Alice', 'Bob', 'Carol', 'Dave', 'Eve', 'Frank', 'Grace', 'Henry'];
const lastNames = ['Smith', 'Jones', 'Williams', 'Brown', 'Davis', 'Miller', 'Wilson', 'Moore'];

export function generateRows(count = 100_000): Row[] {
  return Array.from({ length: count }, (_, i) => ({
    id: i + 1,
    name: `${firstNames[i % firstNames.length]} ${lastNames[Math.floor(i / firstNames.length) % lastNames.length]}`,
    email: `user${i + 1}@company.com`,
    department: departments[i % departments.length],
    salary: 40_000 + ((i * 7) % 120_000),
    status: statuses[i % 3],
    joinDate: new Date(2018 + (i % 6), i % 12, (i % 28) + 1).toISOString().slice(0, 10),
  }));
}

// Generate once at module level — not inside a component
export const ALL_ROWS: Row[] = generateRows();
```

### Table filter hook

```ts
// src/hooks/useTableFilter.ts
import { useMemo, useState, useCallback, useTransition } from 'react';
import type { Row } from '../data/generateRows';

export type SortKey = keyof Row;
export type SortDir = 'asc' | 'desc';

export function useTableFilter(rows: Row[]) {
  const [search, setSearch] = useState('');
  const [columnFilters, setColumnFilters] = useState<Partial<Record<SortKey, string>>>({});
  const [sortKey, setSortKey] = useState<SortKey>('id');
  const [sortDir, setSortDir] = useState<SortDir>('asc');
  const [, startTransition] = useTransition();

  const handleSearch = useCallback((value: string) => {
    startTransition(() => setSearch(value));
  }, []);

  const handleColumnFilter = useCallback((key: SortKey, value: string) => {
    startTransition(() => {
      setColumnFilters(prev => ({ ...prev, [key]: value }));
    });
  }, []);

  const handleSort = useCallback((key: SortKey) => {
    startTransition(() => {
      setSortDir(prev => sortKey === key && prev === 'asc' ? 'desc' : 'asc');
      setSortKey(key);
    });
  }, [sortKey]);

  const filtered = useMemo(() => {
    let result = rows;

    if (search) {
      const q = search.toLowerCase();
      result = result.filter(r =>
        r.name.toLowerCase().includes(q) ||
        r.email.toLowerCase().includes(q) ||
        r.department.toLowerCase().includes(q)
      );
    }

    for (const [key, value] of Object.entries(columnFilters)) {
      if (!value) continue;
      const v = value.toLowerCase();
      result = result.filter(r => String(r[key as SortKey]).toLowerCase().includes(v));
    }

    result = [...result].sort((a, b) => {
      const va = a[sortKey];
      const vb = b[sortKey];
      const cmp = typeof va === 'number' && typeof vb === 'number'
        ? va - vb
        : String(va).localeCompare(String(vb));
      return sortDir === 'asc' ? cmp : -cmp;
    });

    return result;
  }, [rows, search, columnFilters, sortKey, sortDir]);

  return {
    filtered,
    search,
    columnFilters,
    sortKey,
    sortDir,
    handleSearch,
    handleColumnFilter,
    handleSort,
  };
}
```

### VirtualTable

```tsx
// src/components/VirtualTable.tsx
import { useRef, useCallback } from 'react';
import { useVirtualizer } from '@tanstack/react-virtual';
import type { Row } from '../data/generateRows';
import type { SortKey, SortDir } from '../hooks/useTableFilter';

interface Column {
  key: SortKey;
  header: string;
  width: number;
  render?: (value: Row[SortKey]) => React.ReactNode;
}

interface Props {
  rows: Row[];
  sortKey: SortKey;
  sortDir: SortDir;
  onSort: (key: SortKey) => void;
  columnFilters: Partial<Record<SortKey, string>>;
  onColumnFilter: (key: SortKey, value: string) => void;
}

const COLUMNS: Column[] = [
  { key: 'id', header: 'ID', width: 80 },
  { key: 'name', header: 'Name', width: 200 },
  { key: 'email', header: 'Email', width: 240 },
  { key: 'department', header: 'Department', width: 160 },
  {
    key: 'salary',
    header: 'Salary',
    width: 120,
    render: v => `$${Number(v).toLocaleString()}`,
  },
  {
    key: 'status',
    header: 'Status',
    width: 100,
    render: v => (
      <span className={`px-2 py-0.5 rounded-full text-xs font-semibold ${
        v === 'active' ? 'bg-green-100 text-green-700' :
        v === 'inactive' ? 'bg-gray-100 text-gray-600' :
        'bg-yellow-100 text-yellow-700'
      }`}>
        {String(v)}
      </span>
    ),
  },
  { key: 'joinDate', header: 'Join Date', width: 120 },
];

const ROW_HEIGHT = 40;

export function VirtualTable({ rows, sortKey, sortDir, onSort, columnFilters, onColumnFilter }: Props) {
  const parentRef = useRef<HTMLDivElement>(null);

  const virtualizer = useVirtualizer({
    count: rows.length,
    getScrollElement: () => parentRef.current,
    estimateSize: useCallback(() => ROW_HEIGHT, []),
    overscan: 20, // Render 20 extra rows above/below viewport
  });

  const totalWidth = COLUMNS.reduce((sum, c) => sum + c.width, 0);

  return (
    <div className="border rounded-xl overflow-hidden flex flex-col">
      {/* Sticky header */}
      <div className="border-b bg-gray-50 flex-shrink-0" style={{ minWidth: totalWidth }}>
        {/* Sort row */}
        <div className="flex">
          {COLUMNS.map(col => (
            <div
              key={col.key}
              style={{ width: col.width, minWidth: col.width }}
              className="px-3 py-2 font-semibold text-xs text-gray-600 uppercase tracking-wide cursor-pointer hover:bg-gray-100 select-none flex items-center gap-1"
              onClick={() => onSort(col.key)}
            >
              {col.header}
              {sortKey === col.key && (sortDir === 'asc' ? ' ↑' : ' ↓')}
            </div>
          ))}
        </div>
        {/* Filter row */}
        <div className="flex border-t">
          {COLUMNS.map(col => (
            <div key={col.key} style={{ width: col.width, minWidth: col.width }} className="px-1 py-1">
              <input
                value={columnFilters[col.key] ?? ''}
                onChange={e => onColumnFilter(col.key, e.target.value)}
                placeholder="Filter..."
                className="w-full text-xs border rounded px-2 py-0.5 focus:outline-none focus:ring-1 focus:ring-blue-400"
              />
            </div>
          ))}
        </div>
      </div>

      {/* Virtual scroll container */}
      <div
        ref={parentRef}
        className="flex-1 overflow-auto"
        style={{ height: 500 }}
      >
        {/* Total height spacer */}
        <div style={{ height: virtualizer.getTotalSize(), position: 'relative' }}>
          {virtualizer.getVirtualItems().map(virtualRow => {
            const row = rows[virtualRow.index];
            return (
              <div
                key={row.id}
                style={{
                  position: 'absolute',
                  top: virtualRow.start,
                  height: ROW_HEIGHT,
                  minWidth: totalWidth,
                }}
                className={`flex items-center ${virtualRow.index % 2 === 0 ? 'bg-white' : 'bg-gray-50'} hover:bg-blue-50 transition-colors`}
              >
                {COLUMNS.map(col => (
                  <div
                    key={col.key}
                    style={{ width: col.width, minWidth: col.width }}
                    className="px-3 text-sm text-gray-700 truncate"
                  >
                    {col.render ? col.render(row[col.key]) : String(row[col.key])}
                  </div>
                ))}
              </div>
            );
          })}
        </div>
      </div>
    </div>
  );
}
```

### App

```tsx
// src/App.tsx
import { useRef } from 'react';
import { ALL_ROWS } from './data/generateRows';
import { useTableFilter } from './hooks/useTableFilter';
import { VirtualTable } from './components/VirtualTable';
import { ScrollToTop } from './components/ScrollToTop';
import { useDebounce } from './hooks/useDebounce';

export default function App() {
  const {
    filtered,
    search,
    columnFilters,
    sortKey,
    sortDir,
    handleSearch,
    handleColumnFilter,
    handleSort,
  } = useTableFilter(ALL_ROWS);

  return (
    <div className="min-h-screen bg-gray-100 p-6">
      <div className="max-w-7xl mx-auto space-y-4">
        <div className="flex items-center justify-between">
          <h1 className="text-2xl font-bold text-gray-900">Employee Directory</h1>
          <p className="text-sm text-gray-500">
            Showing {filtered.length.toLocaleString()} of {ALL_ROWS.length.toLocaleString()} employees
          </p>
        </div>

        <input
          type="search"
          value={search}
          onChange={e => handleSearch(e.target.value)}
          placeholder="Search by name, email, or department..."
          className="w-full border rounded-xl px-4 py-2 focus:outline-none focus:ring-2 focus:ring-blue-400"
        />

        <VirtualTable
          rows={filtered}
          sortKey={sortKey}
          sortDir={sortDir}
          onSort={handleSort}
          columnFilters={columnFilters}
          onColumnFilter={handleColumnFilter}
        />

        <ScrollToTop />
      </div>
    </div>
  );
}
```
