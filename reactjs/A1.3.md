## Goal
Learn how React components hold and update dynamic data using useState, and how to wire user interactions through event handlers.

## Background

### useState — Local Component State
`useState` is a hook that lets a functional component remember values between renders. It returns a tuple: the current value and a setter function. When you call the setter, React schedules a re-render with the new value. State updates are asynchronous — the new value won't be visible until the next render.

Never mutate state directly (e.g., `state.push(item)` or `state.count++`). Always call the setter with a new value. For complex updates that depend on the previous state, use the **functional update form**: `setValue(prev => prev + 1)`. This is safe even when multiple updates are batched together.

### Synthetic Events
React wraps native DOM events in `SyntheticEvent` objects. This normalizes behavior across browsers. The most important thing to know: you cannot use a synthetic event asynchronously without calling `e.persist()` (though this is less relevant in React 17+ where events are no longer pooled). TypeScript event types are `React.ChangeEvent<HTMLInputElement>`, `React.FormEvent<HTMLFormElement>`, `React.MouseEvent<HTMLButtonElement>`, etc.

### Controlled Inputs
An input is "controlled" when its `value` prop is driven by React state. The `onChange` handler updates the state, which causes a re-render, which updates the input value. This gives React full control over the form field. Uncontrolled inputs use a ref to read the DOM value directly — useful in Phase 5 with React Hook Form, but controlled inputs are the React-native pattern.

### Task 1 — Basic State and Functional Updates

```tsx
// src/components/Counter.tsx
import { useState } from 'react'

export function Counter() {
  const [count, setCount] = useState(0)
  const [step, setStep] = useState(1)

  // Functional update — safe when new value depends on previous
  const increment = () => setCount(prev => prev + step)
  const decrement = () => setCount(prev => prev - step)
  const reset = () => setCount(0)

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem', maxWidth: 200 }}>
      <div style={{ fontSize: 48, fontWeight: 'bold', textAlign: 'center' }}>{count}</div>

      <div style={{ display: 'flex', gap: '0.5rem' }}>
        <button onClick={decrement}>-{step}</button>
        <button onClick={increment}>+{step}</button>
        <button onClick={reset}>Reset</button>
      </div>

      <label style={{ display: 'flex', flexDirection: 'column', fontSize: 14 }}>
        Step size:
        <input
          type="range"
          min={1}
          max={10}
          value={step}
          onChange={e => setStep(Number(e.target.value))}
        />
        <span>{step}</span>
      </label>
    </div>
  )
}
```

### Task 2 — Object State

```tsx
// src/components/UserForm.tsx
import { useState } from 'react'

interface UserProfile {
  name: string
  email: string
  bio: string
  newsletter: boolean
}

const initialProfile: UserProfile = {
  name: '',
  email: '',
  bio: '',
  newsletter: false,
}

export function UserForm() {
  const [profile, setProfile] = useState<UserProfile>(initialProfile)
  const [submitted, setSubmitted] = useState(false)

  // Generic field updater — avoids writing one handler per field
  function handleChange(field: keyof UserProfile, value: string | boolean) {
    setProfile(prev => ({ ...prev, [field]: value }))
    // Note: spread first, then override — this is how you update one field
    // without losing the rest of the object
  }

  function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()  // prevent page reload
    setSubmitted(true)
  }

  if (submitted) {
    return (
      <div>
        <h3>Submitted!</h3>
        <pre>{JSON.stringify(profile, null, 2)}</pre>
        <button onClick={() => { setProfile(initialProfile); setSubmitted(false) }}>
          Reset
        </button>
      </div>
    )
  }

  return (
    <form onSubmit={handleSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '1rem', maxWidth: 400 }}>
      <label>
        Name
        <input
          type="text"
          value={profile.name}
          onChange={e => handleChange('name', e.target.value)}
          placeholder="Your name"
          required
          style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px' }}
        />
      </label>

      <label>
        Email
        <input
          type="email"
          value={profile.email}
          onChange={e => handleChange('email', e.target.value)}
          placeholder="you@example.com"
          required
          style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px' }}
        />
      </label>

      <label>
        Bio
        <textarea
          value={profile.bio}
          onChange={e => handleChange('bio', e.target.value)}
          rows={3}
          placeholder="Tell us about yourself"
          style={{ display: 'block', width: '100%', marginTop: 4, padding: '6px 10px' }}
        />
      </label>

      <label style={{ display: 'flex', gap: '0.5rem', alignItems: 'center' }}>
        <input
          type="checkbox"
          checked={profile.newsletter}
          onChange={e => handleChange('newsletter', e.target.checked)}
        />
        Subscribe to newsletter
      </label>

      <button type="submit">Save Profile</button>
    </form>
  )
}
```

### Task 3 — Array State (Todo List)

```tsx
// src/components/TodoList.tsx
import { useState } from 'react'

interface Todo {
  id: number
  text: string
  done: boolean
}

let nextId = 1

export function TodoList() {
  const [todos, setTodos] = useState<Todo[]>([])
  const [input, setInput] = useState('')

  function addTodo() {
    const text = input.trim()
    if (!text) return
    setTodos(prev => [...prev, { id: nextId++, text, done: false }])
    setInput('')
  }

  function toggleTodo(id: number) {
    setTodos(prev =>
      prev.map(todo =>
        todo.id === id ? { ...todo, done: !todo.done } : todo
      )
    )
  }

  function deleteTodo(id: number) {
    setTodos(prev => prev.filter(todo => todo.id !== id))
  }

  function handleKeyDown(e: React.KeyboardEvent<HTMLInputElement>) {
    if (e.key === 'Enter') addTodo()
  }

  const pending = todos.filter(t => !t.done).length

  return (
    <div style={{ maxWidth: 400 }}>
      <h3>Todos ({pending} remaining)</h3>

      <div style={{ display: 'flex', gap: '0.5rem', marginBottom: '1rem' }}>
        <input
          type="text"
          value={input}
          onChange={e => setInput(e.target.value)}
          onKeyDown={handleKeyDown}
          placeholder="Add a todo..."
          style={{ flex: 1, padding: '6px 10px' }}
        />
        <button onClick={addTodo} disabled={!input.trim()}>
          Add
        </button>
      </div>

      {todos.length === 0 && (
        <p style={{ color: '#9ca3af', textAlign: 'center' }}>No todos yet. Add one above.</p>
      )}

      <ul style={{ listStyle: 'none', padding: 0, margin: 0 }}>
        {todos.map(todo => (
          <li
            key={todo.id}
            style={{
              display: 'flex',
              alignItems: 'center',
              gap: '0.5rem',
              padding: '0.5rem 0',
              borderBottom: '1px solid #f0f0f0',
            }}
          >
            <input
              type="checkbox"
              checked={todo.done}
              onChange={() => toggleTodo(todo.id)}
            />
            <span
              style={{
                flex: 1,
                textDecoration: todo.done ? 'line-through' : 'none',
                color: todo.done ? '#9ca3af' : 'inherit',
              }}
            >
              {todo.text}
            </span>
            <button
              onClick={() => deleteTodo(todo.id)}
              style={{ background: 'none', border: 'none', cursor: 'pointer', color: '#ef4444' }}
              aria-label={`Delete ${todo.text}`}
            >
              ✕
            </button>
          </li>
        ))}
      </ul>

      {todos.some(t => t.done) && (
        <button
          onClick={() => setTodos(prev => prev.filter(t => !t.done))}
          style={{ marginTop: '0.75rem', fontSize: 13, color: '#6b7280' }}
        >
          Clear completed
        </button>
      )}
    </div>
  )
}
```

### Task 4 — Event Types Reference

```tsx
// src/components/EventTypesDemo.tsx
// This component demonstrates the TypeScript types for common DOM events

export function EventTypesDemo() {
  // Change event on text input
  function onTextChange(e: React.ChangeEvent<HTMLInputElement>) {
    console.log('text:', e.target.value)
  }

  // Change event on select
  function onSelectChange(e: React.ChangeEvent<HTMLSelectElement>) {
    console.log('selected:', e.target.value)
  }

  // Click event — you can read the button element
  function onButtonClick(e: React.MouseEvent<HTMLButtonElement>) {
    console.log('clicked button:', e.currentTarget.textContent)
    e.preventDefault()  // works on anchor clicks too
  }

  // Form submit
  function onSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault()  // always call this on form submit
    const formData = new FormData(e.currentTarget)
    console.log('name:', formData.get('name'))
  }

  // Keyboard event
  function onKeyDown(e: React.KeyboardEvent<HTMLInputElement>) {
    console.log('key:', e.key, 'ctrl:', e.ctrlKey, 'shift:', e.shiftKey)
    if (e.key === 'Escape') e.currentTarget.blur()
  }

  // Focus / blur
  function onFocus(e: React.FocusEvent<HTMLInputElement>) {
    e.currentTarget.style.outline = '2px solid #3b82f6'
  }
  function onBlur(e: React.FocusEvent<HTMLInputElement>) {
    e.currentTarget.style.outline = ''
  }

  return (
    <form onSubmit={onSubmit} style={{ display: 'flex', flexDirection: 'column', gap: '0.75rem', maxWidth: 300 }}>
      <input name="name" type="text" onChange={onTextChange} onKeyDown={onKeyDown} onFocus={onFocus} onBlur={onBlur} placeholder="Text input" />
      <select onChange={onSelectChange}>
        <option value="">Select option</option>
        <option value="a">Option A</option>
        <option value="b">Option B</option>
      </select>
      <button type="button" onClick={onButtonClick}>Click me</button>
      <button type="submit">Submit form</button>
    </form>
  )
}
```
