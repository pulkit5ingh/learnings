## Goal
Build forms that are performant, accessible, and easy to validate using React Hook Form's uncontrolled input approach.

## Background

### Why React Hook Form?
Controlled inputs (value + onChange) re-render the component on every keystroke. For large forms, this causes performance issues and complex state management. React Hook Form uses uncontrolled inputs — it reads values from the DOM directly via refs. This means zero re-renders on typing. Only error state changes and explicit `watch` calls trigger re-renders.

### Core API
`useForm()` returns a set of functions. `register(name)` returns props to spread onto an input — it registers the field. `handleSubmit(fn)` wraps your submit function and provides the validated values. `formState` contains `errors`, `isSubmitting`, `isDirty`, `isValid`, and more. `watch(name)` subscribes to a field's value. `setValue(name, value)` imperatively sets a field. `reset(values)` resets the form to its initial values (or new values you provide).

### Validation
Pass a validation rules object as the second argument to `register`: `register('email', { required: 'Email is required', pattern: { value: /.../, message: '...' } })`. Or use a schema validator like Zod with `zodResolver` (covered in A5.2).

## Tasks

### Task 1 — Basic Form

```tsx
// src/components/LoginForm.tsx
import { useForm } from 'react-hook-form'

interface LoginValues {
  email: string
  password: string
  rememberMe: boolean
}

export function LoginForm() {
  const {
    register,
    handleSubmit,
    formState: { errors, isSubmitting },
    reset,
  } = useForm<LoginValues>({
    defaultValues: { email: '', password: '', rememberMe: false },
  })

  async function onSubmit(data: LoginValues) {
    await new Promise(r => setTimeout(r, 1000))  // simulate API
    console.log('Login:', data)
    reset()
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ display: 'flex', flexDirection: 'column', gap: '1rem', maxWidth: 360 }}>
      <div>
        <label htmlFor="email">Email</label>
        <input
          id="email"
          type="email"
          style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px', border: `1px solid ${errors.email ? '#ef4444' : '#d1d5db'}` }}
          {...register('email', {
            required: 'Email is required',
            pattern: { value: /^\S+@\S+\.\S+$/, message: 'Enter a valid email' },
          })}
        />
        {errors.email && <p style={{ color: '#ef4444', fontSize: 12, margin: '4px 0 0' }}>{errors.email.message}</p>}
      </div>

      <div>
        <label htmlFor="password">Password</label>
        <input
          id="password"
          type="password"
          style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px', border: `1px solid ${errors.password ? '#ef4444' : '#d1d5db'}` }}
          {...register('password', {
            required: 'Password is required',
            minLength: { value: 8, message: 'At least 8 characters' },
          })}
        />
        {errors.password && <p style={{ color: '#ef4444', fontSize: 12, margin: '4px 0 0' }}>{errors.password.message}</p>}
      </div>

      <label style={{ display: 'flex', gap: '0.5rem', alignItems: 'center' }}>
        <input type="checkbox" {...register('rememberMe')} />
        Remember me
      </label>

      <button type="submit" disabled={isSubmitting}>
        {isSubmitting ? 'Signing in...' : 'Sign In'}
      </button>
    </form>
  )
}
```

### Task 2 — watch and Conditional Fields

```tsx
// src/components/ShippingForm.tsx
import { useForm } from 'react-hook-form'

interface ShippingValues {
  shippingType: 'standard' | 'express' | 'pickup'
  address: string
  city: string
  pickupLocation: string
  giftWrap: boolean
  giftMessage: string
}

export function ShippingForm() {
  const { register, handleSubmit, watch, formState: { errors } } = useForm<ShippingValues>({
    defaultValues: { shippingType: 'standard', giftWrap: false },
  })

  // Watch specific fields — these cause re-renders when their value changes
  const shippingType = watch('shippingType')
  const giftWrap = watch('giftWrap')

  function onSubmit(data: ShippingValues) {
    console.log('Shipping:', data)
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ display: 'flex', flexDirection: 'column', gap: '1rem', maxWidth: 400 }}>
      <div>
        <label>Shipping Type</label>
        <select {...register('shippingType')} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}>
          <option value="standard">Standard (3-5 days)</option>
          <option value="express">Express (1-2 days)</option>
          <option value="pickup">Store Pickup</option>
        </select>
      </div>

      {/* Conditional: show address fields only for non-pickup */}
      {shippingType !== 'pickup' && (
        <>
          <div>
            <label>Address</label>
            <input
              {...register('address', { required: shippingType !== 'pickup' ? 'Address required' : false })}
              style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}
              placeholder="123 Main St"
            />
            {errors.address && <p style={{ color: '#ef4444', fontSize: 12 }}>{errors.address.message}</p>}
          </div>
          <div>
            <label>City</label>
            <input {...register('city', { required: 'City required' })} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }} />
          </div>
        </>
      )}

      {shippingType === 'pickup' && (
        <div>
          <label>Pickup Location</label>
          <select {...register('pickupLocation', { required: 'Select a location' })} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}>
            <option value="">Select location</option>
            <option value="downtown">Downtown Store</option>
            <option value="mall">Mall Location</option>
          </select>
        </div>
      )}

      <label style={{ display: 'flex', gap: '0.5rem', alignItems: 'center' }}>
        <input type="checkbox" {...register('giftWrap')} />
        Gift wrap (+$5)
      </label>

      {giftWrap && (
        <div>
          <label>Gift Message</label>
          <textarea
            {...register('giftMessage', { maxLength: { value: 200, message: 'Max 200 characters' } })}
            rows={3}
            style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }}
            placeholder="Your message..."
          />
        </div>
      )}

      <button type="submit">Continue to Payment</button>
    </form>
  )
}
```

### Task 3 — setValue and Controlled Integration

```tsx
// src/components/ProfileForm.tsx
import { useForm } from 'react-hook-form'
import { useState } from 'react'

interface ProfileValues {
  displayName: string
  username: string
  bio: string
  skills: string[]
}

const ALL_SKILLS = ['React', 'TypeScript', 'Node.js', 'CSS', 'Testing', 'PostgreSQL']

export function ProfileForm() {
  const { register, handleSubmit, setValue, watch, formState: { isDirty, isSubmitting } } = useForm<ProfileValues>({
    defaultValues: { displayName: '', username: '', bio: '', skills: [] },
  })

  const selectedSkills = watch('skills') ?? []

  function toggleSkill(skill: string) {
    const next = selectedSkills.includes(skill)
      ? selectedSkills.filter(s => s !== skill)
      : [...selectedSkills, skill]
    setValue('skills', next, { shouldDirty: true })  // mark form as dirty
  }

  function onSubmit(data: ProfileValues) {
    console.log('Profile:', data)
  }

  return (
    <form onSubmit={handleSubmit(onSubmit)} style={{ display: 'flex', flexDirection: 'column', gap: '1rem', maxWidth: 400 }}>
      <div>
        <label>Display Name</label>
        <input {...register('displayName', { required: 'Display name required' })} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }} />
      </div>
      <div>
        <label>Username</label>
        <input {...register('username', { required: true, pattern: /^[a-z0-9_]+$/ })} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }} placeholder="lowercase_only" />
      </div>
      <div>
        <label>Bio</label>
        <textarea {...register('bio', { maxLength: 300 })} rows={3} style={{ display: 'block', width: '100%', marginTop: 4, padding: '8px 10px' }} />
      </div>
      <div>
        <label style={{ display: 'block', marginBottom: 8 }}>Skills</label>
        <div style={{ display: 'flex', flexWrap: 'wrap', gap: 8 }}>
          {ALL_SKILLS.map(skill => (
            <button
              key={skill}
              type="button"
              onClick={() => toggleSkill(skill)}
              style={{
                padding: '4px 12px',
                borderRadius: 16,
                border: '1px solid',
                cursor: 'pointer',
                background: selectedSkills.includes(skill) ? '#3b82f6' : 'white',
                borderColor: selectedSkills.includes(skill) ? '#3b82f6' : '#d1d5db',
                color: selectedSkills.includes(skill) ? 'white' : 'inherit',
              }}
            >
              {skill}
            </button>
          ))}
        </div>
      </div>
      <button type="submit" disabled={!isDirty || isSubmitting}>
        {isSubmitting ? 'Saving...' : isDirty ? 'Save Changes' : 'No Changes'}
      </button>
    </form>
  )
}
```
