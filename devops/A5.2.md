# A5.2 — Terraform Modules and Remote State

**Phase:** 5 — Infrastructure as Code  
**Topic:** Terraform modules, state management, workspaces  
**Type:** Assignment

---

## Goal
Structure Terraform code as reusable modules and manage state remotely — the pattern used by teams to avoid "works on my machine" infrastructure.

---

## Task

### Part 1 — Why Modules?
Without modules, you copy-paste infrastructure code. With modules, you define once and reuse:

```hcl
# Without modules: copy-paste nightmare
resource "aws_vpc" "dev_vpc" { ... }
resource "aws_vpc" "staging_vpc" { ... }   # copy-paste
resource "aws_vpc" "prod_vpc" { ... }      # copy-paste again

# With modules: call once per environment
module "dev_vpc" {
  source       = "./modules/vpc"
  environment  = "dev"
  cidr_block   = "10.0.0.0/16"
}

module "prod_vpc" {
  source       = "./modules/vpc"
  environment  = "prod"
  cidr_block   = "10.1.0.0/16"
}
```

### Part 2 — Create a VPC Module
```
modules/
└── vpc/
    ├── main.tf
    ├── variables.tf
    ├── outputs.tf
    └── README.md
```

```hcl
# modules/vpc/variables.tf
variable "project_name" {
  type = string
}

variable "environment" {
  type = string
}

variable "vpc_cidr" {
  type    = string
  default = "10.0.0.0/16"
}

variable "public_subnet_count" {
  type    = number
  default = 2
}

variable "private_subnet_count" {
  type    = number
  default = 2
}

variable "enable_nat_gateway" {
  type    = bool
  default = true
}
```

```hcl
# modules/vpc/outputs.tf
output "vpc_id" { value = aws_vpc.main.id }
output "public_subnet_ids" { value = aws_subnet.public[*].id }
output "private_subnet_ids" { value = aws_subnet.private[*].id }
output "nat_gateway_ip" { value = var.enable_nat_gateway ? aws_eip.nat[0].public_ip : null }
```

### Part 3 — Create an EC2 Module
```
modules/
└── ec2/
    ├── main.tf       # instance, security group, key pair
    ├── variables.tf  # ami, instance_type, subnet_id, etc.
    └── outputs.tf    # instance_id, public_ip, private_ip
```

### Part 4 — Root Module Composes Everything
```hcl
# environments/dev/main.tf
terraform {
  required_version = ">= 1.5"
  backend "s3" {
    bucket = "myorg-terraform-state"
    key    = "dev/terraform.tfstate"
    region = "us-east-1"
    dynamodb_table = "terraform-state-lock"  # prevents concurrent applies
    encrypt = true
  }
}

module "vpc" {
  source              = "../../modules/vpc"
  project_name        = "myapp"
  environment         = "dev"
  vpc_cidr            = "10.0.0.0/16"
  enable_nat_gateway  = false  # save cost in dev
}

module "web_server" {
  source        = "../../modules/ec2"
  project_name  = "myapp"
  environment   = "dev"
  subnet_id     = module.vpc.public_subnet_ids[0]
  vpc_id        = module.vpc.vpc_id
  instance_type = "t3.micro"
}
```

### Part 5 — Remote State Backend with S3
```hcl
# First, create the S3 bucket and DynamoDB table MANUALLY (bootstrap problem)
# Or use a separate "bootstrap" terraform config:

resource "aws_s3_bucket" "terraform_state" {
  bucket = "myorg-terraform-state-${random_id.suffix.hex}"

  lifecycle {
    prevent_destroy = true  # never accidentally delete state
  }
}

resource "aws_s3_bucket_versioning" "state" {
  bucket = aws_s3_bucket.terraform_state.id
  versioning_configuration {
    status = "Enabled"
  }
}

resource "aws_s3_bucket_server_side_encryption_configuration" "state" {
  bucket = aws_s3_bucket.terraform_state.id
  rule {
    apply_server_side_encryption_by_default {
      sse_algorithm = "AES256"
    }
  }
}

resource "aws_dynamodb_table" "state_lock" {
  name         = "terraform-state-lock"
  billing_mode = "PAY_PER_REQUEST"
  hash_key     = "LockID"

  attribute {
    name = "LockID"
    type = "S"
  }
}
```

### Part 6 — Workspaces for Environment Management
```bash
# Workspaces give you separate state per environment
terraform workspace new dev
terraform workspace new staging
terraform workspace new prod

terraform workspace list
terraform workspace select dev

# Use workspace name in resources
resource "aws_instance" "web" {
  instance_type = terraform.workspace == "prod" ? "t3.medium" : "t3.micro"
  tags = {
    Environment = terraform.workspace
  }
}
```

Alternatively, use separate state files per environment (more common in large orgs):
```
environments/
├── dev/
│   ├── main.tf
│   └── dev.tfvars
├── staging/
│   ├── main.tf
│   └── staging.tfvars
└── prod/
    ├── main.tf
    └── prod.tfvars
```

### Part 7 — Data Sources and State Sharing
```hcl
# Read outputs from another Terraform state
data "terraform_remote_state" "vpc" {
  backend = "s3"
  config = {
    bucket = "myorg-terraform-state"
    key    = "dev/vpc/terraform.tfstate"
    region = "us-east-1"
  }
}

# Use the VPC ID from the other state
resource "aws_instance" "app" {
  subnet_id = data.terraform_remote_state.vpc.outputs.public_subnet_ids[0]
}
```
