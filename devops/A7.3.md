## Goal
Implement distributed tracing to understand how requests flow through a multi-service system. When you have metrics saying "p99 is 3 seconds" and logs saying "error in service X", traces tell you exactly where the 3 seconds went.

---

## Task

### Part 1 — The Observability Three Pillars
- **Metrics**: aggregated numbers (request rate, error rate, latency)
- **Logs**: discrete events with context
- **Traces**: request journey across services with timing per step

A single user request might go:
```
Browser → nginx → API → auth-service → database → cache
```
A trace shows you exactly how long each step took.

### Part 3 — Instrument a Python Service
```python
# tracing.py — setup module
from opentelemetry import trace
from opentelemetry.sdk.trace import TracerProvider
from opentelemetry.sdk.trace.export import BatchSpanProcessor
from opentelemetry.exporter.otlp.proto.grpc.trace_exporter import OTLPSpanExporter
from opentelemetry.sdk.resources import Resource

def setup_tracing(service_name: str):
    resource = Resource.create({"service.name": service_name})
    
    provider = TracerProvider(resource=resource)
    
    exporter = OTLPSpanExporter(
        endpoint="http://otel-collector:4317",
        insecure=True
    )
    provider.add_span_processor(BatchSpanProcessor(exporter))
    
    trace.set_tracer_provider(provider)
    return trace.get_tracer(service_name)
```

```python
# api/main.py
from fastapi import FastAPI, Request
from opentelemetry.instrumentation.fastapi import FastAPIInstrumentor
from opentelemetry.instrumentation.httpx import HTTPXClientInstrumentor
from opentelemetry.instrumentation.sqlalchemy import SQLAlchemyInstrumentor
from tracing import setup_tracing

tracer = setup_tracing("api-service")

app = FastAPI()
FastAPIInstrumentor.instrument_app(app)  # auto-instrument all routes
HTTPXClientInstrumentor().instrument()    # auto-instrument outbound HTTP
SQLAlchemyInstrumentor().instrument()     # auto-instrument database queries

@app.get("/api/orders/{order_id}")
async def get_order(order_id: str):
    # Auto-instrumented — a span is created for this endpoint
    
    # Create a custom child span for business logic
    with tracer.start_as_current_span("fetch_order_details") as span:
        span.set_attribute("order.id", order_id)
        span.set_attribute("order.source", "database")
        
        order = await db.get_order(order_id)
        
        if not order:
            span.set_attribute("order.found", False)
            raise HTTPException(status_code=404)
        
        span.set_attribute("order.found", True)
        span.set_attribute("order.total", float(order.total))
    
    # Call another service — trace propagates automatically
    async with httpx.AsyncClient() as client:
        inventory = await client.get(f"http://inventory-service/items/{order.item_id}")
    
    return {"order": order, "inventory": inventory.json()}
```

### Part 4 — Trace Context Propagation

### Part 6 — Custom Spans for Business Logic
```python
# Track business-specific events
with tracer.start_as_current_span("payment.process") as span:
    span.set_attribute("payment.amount", amount)
    span.set_attribute("payment.currency", "USD")
    span.set_attribute("payment.method", "card")
    
    try:
        result = payment_gateway.charge(amount)
        span.set_attribute("payment.status", "success")
        span.set_attribute("payment.transaction_id", result.id)
    except PaymentDeclined as e:
        span.set_attribute("payment.status", "declined")
        span.set_attribute("payment.decline_reason", str(e))
        span.record_exception(e)
        span.set_status(trace.Status(trace.StatusCode.ERROR, str(e)))
        raise
```
