## Task

### Part 1 — Pods (The Basic Unit)
```yaml
# pod.yaml
apiVersion: v1
kind: Pod
metadata:
  name: nginx-pod
  labels:
    app: nginx
spec:
  containers:
    - name: nginx
      image: nginx:1.25-alpine
      ports:
        - containerPort: 80
      resources:
        requests:
          memory: "64Mi"
          cpu: "100m"
        limits:
          memory: "128Mi"
          cpu: "200m"
      livenessProbe:
        httpGet:
          path: /
          port: 80
        initialDelaySeconds: 5
        periodSeconds: 10
      readinessProbe:
        httpGet:
          path: /
          port: 80
        initialDelaySeconds: 3
        periodSeconds: 5
```

```bash
kubectl apply -f pod.yaml
kubectl get pods
kubectl describe pod nginx-pod
kubectl logs nginx-pod
kubectl exec -it nginx-pod -- sh

# Port forward to access locally
kubectl port-forward pod/nginx-pod 8080:80
curl http://localhost:8080

# Delete and watch it NOT come back (pods are not self-healing)
kubectl delete pod nginx-pod
kubectl get pods  # gone
```

Explain: what is the difference between `liveness` and `readiness` probes?

### Part 2 — Deployments (Self-Healing)
```yaml
# deployment.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: nginx-deploy
  labels:
    app: nginx
spec:
  replicas: 3
  selector:
    matchLabels:
      app: nginx
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1          # can temporarily have 4 pods during update
      maxUnavailable: 0    # never go below 3 pods (zero downtime)
  template:
    metadata:
      labels:
        app: nginx
    spec:
      containers:
        - name: nginx
          image: nginx:1.25-alpine
          ports:
            - containerPort: 80
          resources:
            requests:
              memory: "64Mi"
              cpu: "100m"
            limits:
              memory: "128Mi"
              cpu: "200m"
```

```bash
kubectl apply -f deployment.yaml
kubectl get deployments
kubectl get replicasets
kubectl get pods

# Delete a pod — watch the Deployment recreate it immediately
kubectl delete pod <pod-name>
kubectl get pods -w  # watch flag

# Scale
kubectl scale deployment nginx-deploy --replicas=5
kubectl get pods  # 5 pods now
```

### Part 3 — Services (Stable Networking)
Pods get new IPs every time they restart. Services provide a stable IP/DNS name.

```yaml
# service.yaml
apiVersion: v1
kind: Service
metadata:
  name: nginx-service
spec:
  selector:
    app: nginx   # selects pods with this label
  type: ClusterIP  # only accessible within cluster
  ports:
    - port: 80
      targetPort: 80
```

```yaml
# For external access during development:
spec:
  type: NodePort
  ports:
    - port: 80
      targetPort: 80
      nodePort: 30080  # access via <node-ip>:30080
```

```bash
kubectl apply -f service.yaml
kubectl get services
kubectl describe service nginx-service

# Test DNS resolution inside a pod
kubectl run -it --rm debug --image=busybox -- sh
wget -qO- http://nginx-service  # resolves by service name
nslookup nginx-service
# Full DNS: nginx-service.default.svc.cluster.local
```

### Part 4 — Rolling Updates and Rollbacks
```bash
# Update the image version
kubectl set image deployment/nginx-deploy nginx=nginx:1.26-alpine

# Watch the rollout
kubectl rollout status deployment/nginx-deploy

# See rollout history
kubectl rollout history deployment/nginx-deploy

# Rollback to previous version
kubectl rollout undo deployment/nginx-deploy

# Rollback to specific revision
kubectl rollout undo deployment/nginx-deploy --to-revision=1

# Pause and resume a rollout
kubectl rollout pause deployment/nginx-deploy
kubectl rollout resume deployment/nginx-deploy
```

### Part 5 — Labels and Selectors
```bash
# Labels are key/value pairs attached to objects
kubectl label pod <pod-name> environment=staging
kubectl label node <node-name> disktype=ssd

# Select by label
kubectl get pods -l app=nginx
kubectl get pods -l "app=nginx,environment=production"
kubectl get pods -l "environment in (staging, production)"

# Use node selectors to schedule pods on specific nodes
spec:
  nodeSelector:
    disktype: ssd
```
