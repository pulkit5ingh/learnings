# P9.3 — Internal Developer Platform with Self-Service App Deployment

**Phase:** 9 — Advanced: GitOps, SRE & Platform Engineering  
**Topic:** Platform engineering, developer experience  
**Type:** Project

---

## Goal
Build a self-service Internal Developer Platform (IDP) where developers can provision environments, deploy applications, and manage their services — without touching Kubernetes, Terraform, or the platform team.

---

## What to Build
A web portal where developers can:
- Create a new microservice with one click
- Deploy to dev/staging/production
- View their service's health metrics
- Manage environment variables (secrets)
- See deployment history

## Architecture
```
Developer → IDP Web Portal (React + FastAPI)
                    ↓
            Platform API (orchestrates)
            ├── GitHub (create repo, trigger workflows)
            ├── ArgoCD (deployment status)
            ├── Vault (secrets management)
            ├── Kubernetes (namespace, RBAC)
            └── Grafana (metrics links)
```

## Core Platform API
```python
# platform-api/main.py
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
import subprocess, json, httpx

app = FastAPI(title="Internal Developer Platform API")

class ServiceRequest(BaseModel):
    name: str               # e.g., "payment-service"
    team: str               # e.g., "payments-team"
    language: str           # python | nodejs | go
    description: str

class DeployRequest(BaseModel):
    service_name: str
    environment: str        # dev | staging | production
    image_tag: str

@app.post("/services", status_code=201)
async def create_service(req: ServiceRequest):
    """
    Creates a new microservice:
    1. Creates GitHub repo from template
    2. Creates Kubernetes namespace
    3. Creates ArgoCD application
    4. Creates Vault policy
    5. Sets up Slack notifications
    """
    steps = []
    
    # 1. Create GitHub repo from template
    repo_url = await create_github_repo(
        name=req.name,
        template=f"{req.language}-service-template",
        description=req.description
    )
    steps.append(f"GitHub repo: {repo_url}")
    
    # 2. Create Kubernetes namespace and RBAC
    await create_namespace(req.name, req.team)
    steps.append(f"Kubernetes namespace: {req.name}")
    
    # 3. Create ArgoCD application
    await create_argocd_app(req.name)
    steps.append("ArgoCD application created")
    
    # 4. Create Vault policy
    await create_vault_policy(req.name, req.team)
    steps.append("Vault policy created")
    
    return {
        "service": req.name,
        "status": "created",
        "steps": steps,
        "links": {
            "github": repo_url,
            "argocd": f"https://argocd.internal/applications/{req.name}-dev",
            "grafana": f"https://grafana.internal/d/service?var-service={req.name}",
            "docs": f"https://docs.internal/services/{req.name}"
        }
    }

@app.post("/deployments")
async def deploy_service(req: DeployRequest):
    """
    Deploys a service to an environment by updating GitOps repo
    """
    if req.environment == "production":
        # Check error budget before allowing production deploy
        budget = await check_error_budget(req.service_name)
        if budget < 10:  # less than 10% remaining
            raise HTTPException(
                status_code=409,
                detail=f"Error budget too low ({budget:.1f}% remaining). Fix reliability first."
            )
    
    # Update image tag in GitOps repo
    await update_gitops_image(
        service=req.service_name,
        environment=req.environment,
        tag=req.image_tag
    )
    
    return {
        "deployment_id": generate_id(),
        "service": req.service_name,
        "environment": req.environment,
        "image_tag": req.image_tag,
        "status": "initiated",
        "argocd_url": f"https://argocd.internal/applications/{req.service_name}-{req.environment}"
    }

@app.get("/services/{name}/status")
async def get_service_status(name: str):
    """
    Returns complete status: health, deployments, metrics, alerts
    """
    argocd_status = await get_argocd_status(name)
    metrics = await get_prometheus_metrics(name)
    alerts = await get_active_alerts(name)
    
    return {
        "service": name,
        "environments": {
            "dev": argocd_status.get("dev"),
            "staging": argocd_status.get("staging"),
            "production": argocd_status.get("production")
        },
        "slo": {
            "availability_30d": metrics.get("availability"),
            "error_budget_remaining": metrics.get("error_budget_pct"),
            "p99_latency_ms": metrics.get("p99_ms")
        },
        "active_alerts": alerts,
        "links": {
            "grafana": f"https://grafana.internal/d/service?var={name}",
            "logs": f"https://grafana.internal/d/logs?var={name}",
            "traces": f"https://grafana.internal/d/traces?var={name}"
        }
    }
```

// DeployModal.jsx — deploy with a form
function DeployModal({ service }) {
  const [env, setEnv] = useState('dev');
  const [tag, setTag] = useState('');
  
  const handleDeploy = async () => {
    if (env === 'production') {
      const confirmed = await showConfirmation(
        `Deploy ${service} to PRODUCTION with tag ${tag}?`
      );
      if (!confirmed) return;
    }
    
    const result = await api.deploy({ service, env, tag });
    showToast(`Deployment initiated: ${result.argocd_url}`);
  };
  
  return (
    <Modal>
      <Select label="Environment" value={env} onChange={setEnv}
        options={['dev', 'staging', 'production']} />
      <Input label="Image Tag" value={tag} onChange={setTag} 
        placeholder="sha-abc123 or v1.2.3" />
      <Button onClick={handleDeploy}>Deploy</Button>
    </Modal>
  );
}

## Self-Service Secrets Management
```python
@app.post("/services/{name}/secrets")
async def set_secret(name: str, req: SecretRequest, user: User = Depends(get_current_user)):
    """
    Developer sets a secret for their service.
    The platform writes it to Vault, never to the developer.
    """
    # Verify user owns this service
    if not await user_owns_service(user, name):
        raise HTTPException(403, "Access denied")
    
    # Write to Vault
    vault.write(f"secret/data/services/{name}/{req.environment}",
                {req.key: req.value})
    
    # Trigger ArgoCD sync to pick up new secret
    await trigger_argocd_sync(f"{name}-{req.environment}")
    
    return {"status": "set", "key": req.key, "environment": req.environment}
```
