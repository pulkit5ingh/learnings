### Part 1 — Lambda Basics
```python
# lambda_function.py
import json
import logging
import os

logger = logging.getLogger()
logger.setLevel(logging.INFO)

def handler(event, context):
    """
    event: the input (from API Gateway, SQS, S3, etc.)
    context: Lambda runtime info (function name, memory, timeout remaining)
    """
    logger.info("Event: %s", json.dumps(event))
    logger.info("Function: %s, Memory: %dMB, Timeout remaining: %dms",
                context.function_name,
                context.memory_limit_in_mb,
                context.get_remaining_time_in_millis())

    body = json.loads(event.get('body', '{}'))
    name = body.get('name', 'World')

    return {
        'statusCode': 200,
        'headers': {'Content-Type': 'application/json'},
        'body': json.dumps({'message': f'Hello, {name}!'})
    }
```

```bash
# Deploy using AWS CLI
zip function.zip lambda_function.py

aws lambda create-function \
    --function-name my-hello-function \
    --runtime python3.11 \
    --role arn:aws:iam::ACCOUNT_ID:role/lambda-execution-role \
    --handler lambda_function.handler \
    --zip-file fileb://function.zip \
    --timeout 30 \
    --memory-size 256 \
    --environment Variables='{ENVIRONMENT=production}'

# Invoke the function
aws lambda invoke \
    --function-name my-hello-function \
    --payload '{"body": "{\"name\": \"DevOps\"}"}' \
    --cli-binary-format raw-in-base64-out \
    response.json && cat response.json

# Update function code
zip function.zip lambda_function.py
aws lambda update-function-code \
    --function-name my-hello-function \
    --zip-file fileb://function.zip
```

### Part 2 — Lambda Execution Role
```bash
# Trust policy: Lambda can assume this role
cat trust-policy.json
{
  "Version": "2012-10-17",
  "Statement": [{
    "Effect": "Allow",
    "Principal": {"Service": "lambda.amazonaws.com"},
    "Action": "sts:AssumeRole"
  }]
}

aws iam create-role \
    --role-name lambda-execution-role \
    --assume-role-policy-document file://trust-policy.json

# Attach basic execution policy (allows CloudWatch Logs)
aws iam attach-role-policy \
    --role-name lambda-execution-role \
    --policy-arn arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole

# Add SQS permissions for later
aws iam attach-role-policy \
    --role-name lambda-execution-role \
    --policy-arn arn:aws:iam::aws:policy/AmazonSQSFullAccess
```

### Part 3 — API Gateway
```bash
# Create REST API
API_ID=$(aws apigateway create-rest-api \
    --name "my-api" \
    --query 'id' --output text)

ROOT_ID=$(aws apigateway get-resources \
    --rest-api-id $API_ID \
    --query 'items[0].id' --output text)

# Create resource /hello
RESOURCE_ID=$(aws apigateway create-resource \
    --rest-api-id $API_ID \
    --parent-id $ROOT_ID \
    --path-part hello \
    --query 'id' --output text)

# Create POST method
aws apigateway put-method \
    --rest-api-id $API_ID \
    --resource-id $RESOURCE_ID \
    --http-method POST \
    --authorization-type NONE

# Integrate with Lambda
LAMBDA_ARN=arn:aws:lambda:us-east-1:ACCOUNT_ID:function:my-hello-function
aws apigateway put-integration \
    --rest-api-id $API_ID \
    --resource-id $RESOURCE_ID \
    --http-method POST \
    --type AWS_PROXY \
    --integration-http-method POST \
    --uri "arn:aws:apigateway:us-east-1:lambda:path/2015-03-31/functions/$LAMBDA_ARN/invocations"

# Grant API Gateway permission to invoke Lambda
aws lambda add-permission \
    --function-name my-hello-function \
    --statement-id api-gw \
    --action lambda:InvokeFunction \
    --principal apigateway.amazonaws.com

# Deploy to a stage
aws apigateway create-deployment \
    --rest-api-id $API_ID \
    --stage-name prod

# Test
curl -X POST \
    https://$API_ID.execute-api.us-east-1.amazonaws.com/prod/hello \
    -H "Content-Type: application/json" \
    -d '{"name": "DevOps"}'
```

### Part 4 — SQS Queue for Async Processing
```bash
# Create SQS queue
QUEUE_URL=$(aws sqs create-queue \
    --queue-name image-processing \
    --attributes '{
        "VisibilityTimeout": "60",
        "MessageRetentionPeriod": "86400",
        "ReceiveMessageWaitTimeSeconds": "20"
    }' \
    --query 'QueueUrl' --output text)

# Create Dead Letter Queue for failed messages
DLQ_ARN=$(aws sqs create-queue \
    --queue-name image-processing-dlq \
    --query 'QueueUrl' --output text | xargs aws sqs get-queue-attributes \
    --attribute-names QueueArn --query 'Attributes.QueueArn' --output text)

# Attach DLQ to main queue (after 3 failures, send to DLQ)
aws sqs set-queue-attributes \
    --queue-url $QUEUE_URL \
    --attributes "{
        \"RedrivePolicy\": \"{\\\"maxReceiveCount\\\":\\\"3\\\",\\\"deadLetterTargetArn\\\":\\\"$DLQ_ARN\\\"}\"
    }"
```

### Part 5 — Lambda SQS Trigger Pipeline
Architecture: API Gateway → Lambda (enqueue) → SQS → Lambda (process)

```python
# enqueue_lambda.py (triggered by API Gateway)
import boto3, json, uuid

sqs = boto3.client('sqs')
QUEUE_URL = os.environ['QUEUE_URL']

def handler(event, context):
    body = json.loads(event['body'])
    
    response = sqs.send_message(
        QueueUrl=QUEUE_URL,
        MessageBody=json.dumps(body),
        MessageGroupId='image-processing',  # for FIFO queues
        MessageDeduplicationId=str(uuid.uuid4())
    )
    
    return {
        'statusCode': 202,
        'body': json.dumps({'messageId': response['MessageId'], 'status': 'queued'})
    }
```
