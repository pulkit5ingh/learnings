# A8.4 — Policy as Code with OPA and Conftest

**Phase:** 8 — DevSecOps & Compliance  
**Topic:** Policy enforcement, OPA, Conftest, Rego  
**Type:** Assignment

---

## Goal
Write policies as code to automatically enforce security and compliance rules across your infrastructure. Manual reviews miss things — automated policy enforcement doesn't.

---

## Task

### Part 1 — Open Policy Agent (OPA)
OPA is a general-purpose policy engine. You write policies in Rego (a declarative language), and OPA evaluates them against any JSON/YAML data.

```bash
# Install OPA
brew install opa

# Install Conftest (OPA wrapper for config files)
brew install conftest
```

### Part 2 — Rego Basics
```rego
# policy.rego
package main

# Default: deny everything
default allow = false

# Allow if user is admin
allow {
    input.user.role == "admin"
}

# Allow if resource belongs to user
allow {
    input.user.id == input.resource.owner_id
}

# Deny if user is suspended
deny[msg] {
    input.user.suspended == true
    msg := sprintf("User %v is suspended", [input.user.id])
}
```

```bash
# Evaluate a policy
echo '{"user": {"role": "admin"}, "resource": {"owner_id": "123"}}' | \
    opa eval -d policy.rego -I 'data.main.allow'

# Run all rules
opa eval -d policy.rego \
    -i input.json \
    'data.main'
```

### Part 3 — Dockerfile Policies with Conftest
```rego
# policies/dockerfile.rego
package main

# Rule: Dockerfile must not use :latest tag
deny[msg] {
    input[i].Cmd == "from"
    val := input[i].Value
    contains(val[0], ":latest")
    msg := sprintf("Do not use :latest tag. Found: %s", [val[0]])
}

# Rule: Must have a non-root USER instruction
deny[msg] {
    not has_user_instruction
    msg := "Dockerfile must have a USER instruction (not root)"
}

has_user_instruction {
    input[i].Cmd == "user"
    val := input[i].Value[0]
    val != "root"
    not contains(val, "0")
}

# Rule: Must have HEALTHCHECK
deny[msg] {
    not has_healthcheck
    msg := "Dockerfile must have a HEALTHCHECK instruction"
}

has_healthcheck {
    input[i].Cmd == "healthcheck"
}

# Rule: Must not use ADD (use COPY instead)
warn[msg] {
    input[i].Cmd == "add"
    msg := "Use COPY instead of ADD unless you need URL or tar extraction"
}
```

```bash
# Test against your Dockerfile
conftest test Dockerfile --policy policies/
```

### Part 4 — Kubernetes Manifest Policies
```rego
# policies/kubernetes.rego
package main

# Every container must have resource limits
deny[msg] {
    container := input.spec.containers[_]
    not container.resources.limits
    msg := sprintf("Container '%v' must have resource limits", [container.name])
}

# No privileged containers
deny[msg] {
    container := input.spec.containers[_]
    container.securityContext.privileged == true
    msg := sprintf("Container '%v' must not run as privileged", [container.name])
}

# Must not run as root
deny[msg] {
    container := input.spec.containers[_]
    container.securityContext.runAsUser == 0
    msg := sprintf("Container '%v' must not run as root (UID 0)", [container.name])
}

# Must set runAsNonRoot
deny[msg] {
    container := input.spec.containers[_]
    not container.securityContext.runAsNonRoot
    msg := sprintf("Container '%v' must set runAsNonRoot: true", [container.name])
}

# Must not allow privilege escalation
deny[msg] {
    container := input.spec.containers[_]
    container.securityContext.allowPrivilegeEscalation == true
    msg := sprintf("Container '%v' must not allow privilege escalation", [container.name])
}

# Must have readOnlyRootFilesystem
warn[msg] {
    container := input.spec.containers[_]
    not container.securityContext.readOnlyRootFilesystem
    msg := sprintf("Container '%v' should have readOnlyRootFilesystem: true", [container.name])
}

# Required labels
deny[msg] {
    required_labels := {"app", "version", "team"}
    existing_labels := {label | input.metadata.labels[label]}
    missing := required_labels - existing_labels
    count(missing) > 0
    msg := sprintf("Missing required labels: %v", [missing])
}
```

```bash
# Test against Kubernetes manifests
conftest test k8s/deployment.yaml --policy policies/
conftest test k8s/ --policy policies/ --all-namespaces
```

### Part 5 — Terraform Policies
```rego
# policies/terraform.rego
package main

# S3 buckets must have versioning enabled
deny[msg] {
    resource := input.resource.aws_s3_bucket[name]
    not resource.versioning[_].enabled
    msg := sprintf("S3 bucket '%v' must have versioning enabled", [name])
}

# Security groups must not allow 0.0.0.0/0 on SSH (port 22)
deny[msg] {
    resource := input.resource.aws_security_group[name]
    rule := resource.ingress[_]
    rule.from_port <= 22
    rule.to_port >= 22
    rule.cidr_blocks[_] == "0.0.0.0/0"
    msg := sprintf("Security group '%v' allows SSH from 0.0.0.0/0 — restrict to specific IPs", [name])
}

# EC2 instances must not use t2 (deprecated, use t3)
warn[msg] {
    resource := input.resource.aws_instance[name]
    startswith(resource.instance_type, "t2.")
    msg := sprintf("EC2 instance '%v' uses deprecated t2 family, use t3 instead", [name])
}

# All resources must have required tags
deny[msg] {
    resource := input.resource[resource_type][name]
    required_tags := {"Environment", "Project", "Owner"}
    existing_tags := {tag | resource.tags[tag]}
    missing := required_tags - existing_tags
    count(missing) > 0
    msg := sprintf("Resource '%v/%v' missing required tags: %v", [resource_type, name, missing])
}
```

```bash
# Test against Terraform plan output
terraform show -json tfplan > tfplan.json
conftest test tfplan.json --policy policies/
```

### Part 6 — OPA as Kubernetes Admission Controller
OPA Gatekeeper enforces policies at admission time — before resources are created:

```yaml
# Install Gatekeeper
kubectl apply -f https://raw.githubusercontent.com/open-policy-agent/gatekeeper/release-3.14/deploy/gatekeeper.yaml

# ConstraintTemplate: define a reusable policy
apiVersion: templates.gatekeeper.sh/v1
kind: ConstraintTemplate
metadata:
  name: k8srequiredlabels
spec:
  crd:
    spec:
      names:
        kind: K8sRequiredLabels
      validation:
        openAPIV3Schema:
          properties:
            labels:
              type: array
              items:
                type: string
  targets:
    - target: admission.k8s.gatekeeper.sh
      rego: |
        package k8srequiredlabels
        violation[{"msg": msg}] {
          missing := {label | required := input.parameters.labels[_]; label := required; not input.review.object.metadata.labels[label]}
          count(missing) > 0
          msg := sprintf("Missing required labels: %v", [missing])
        }

# Constraint: enforce the template
apiVersion: constraints.gatekeeper.sh/v1beta1
kind: K8sRequiredLabels
metadata:
  name: require-app-labels
spec:
  match:
    kinds:
      - apiGroups: ["apps"]
        kinds: ["Deployment"]
  parameters:
    labels: ["app", "team", "version"]
```
