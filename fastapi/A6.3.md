## Goal
Build a consistent, informative error handling system following RFC 7807 Problem Details standard.

## Background

### Consistent Error Responses
Nothing frustrates API consumers more than inconsistent error formats: sometimes `{"error": "..."}`, sometimes `{"message": "..."}`, sometimes a plain string. Pick one format and use it everywhere. RFC 7807 "Problem Details for HTTP APIs" is the standard: a JSON object with `type`, `title`, `status`, `detail`, and `instance` fields. FastAPI's default validation errors use a different format — you should customize it.

### Exception Hierarchy
Structure your application exceptions as a hierarchy. A base `AppException` can carry HTTP status, error code, and message. Subclasses add specificity: `NotFoundError`, `ConflictError`, `ValidationError`, `UnauthorizedError`. Register one handler that catches `AppException` and formats it uniformly.

### FastAPI's Default Validation Error
When Pydantic validation fails, FastAPI raises `RequestValidationError` and returns a 422 with a detailed error list. The default format is not RFC 7807 compliant. Override the handler to return your standard format.

### Task 1: Application Exception Hierarchy
```python
# app/exceptions.py
from typing import Any

class AppException(Exception):
    """Base exception for all application errors."""
    
    def __init__(
        self,
        status_code: int,
        error_code: str,
        detail: str,
        extra: dict[str, Any] | None = None,
    ):
        self.status_code = status_code
        self.error_code = error_code
        self.detail = detail
        self.extra = extra or {}
        super().__init__(detail)

class NotFoundError(AppException):
    def __init__(self, resource: str, resource_id: Any):
        super().__init__(
            status_code=404,
            error_code="NOT_FOUND",
            detail=f"{resource} with id '{resource_id}' not found",
            extra={"resource": resource, "id": str(resource_id)},
        )

class ConflictError(AppException):
    def __init__(self, detail: str, extra: dict | None = None):
        super().__init__(status_code=409, error_code="CONFLICT", detail=detail, extra=extra)

class UnauthorizedError(AppException):
    def __init__(self, detail: str = "Authentication required"):
        super().__init__(status_code=401, error_code="UNAUTHORIZED", detail=detail)

class ForbiddenError(AppException):
    def __init__(self, detail: str = "Insufficient permissions"):
        super().__init__(status_code=403, error_code="FORBIDDEN", detail=detail)

class BadRequestError(AppException):
    def __init__(self, detail: str, extra: dict | None = None):
        super().__init__(status_code=400, error_code="BAD_REQUEST", detail=detail, extra=extra)

class UnprocessableError(AppException):
    def __init__(self, detail: str, fields: dict | None = None):
        super().__init__(
            status_code=422,
            error_code="VALIDATION_ERROR",
            detail=detail,
            extra={"fields": fields or {}},
        )
```

```python
# app/error_responses.py
from pydantic import BaseModel, Field
from typing import Any
import uuid

class ProblemDetail(BaseModel):
    """RFC 7807 Problem Details response."""
    type: str = Field(
        default="about:blank",
        description="URI reference identifying the problem type",
        examples=["https://api.example.com/errors/not-found"],
    )
    title: str = Field(description="Short, human-readable summary")
    status: int = Field(description="HTTP status code")
    detail: str = Field(description="Human-readable explanation")
    instance: str = Field(
        default_factory=lambda: f"/errors/{uuid.uuid4()}",
        description="Unique URI for this specific occurrence",
    )
    error_code: str | None = None
    extra: dict[str, Any] = {}

ERROR_TYPE_MAP = {
    "NOT_FOUND": "https://api.example.com/errors/not-found",
    "CONFLICT": "https://api.example.com/errors/conflict",
    "UNAUTHORIZED": "https://api.example.com/errors/unauthorized",
    "FORBIDDEN": "https://api.example.com/errors/forbidden",
    "BAD_REQUEST": "https://api.example.com/errors/bad-request",
    "VALIDATION_ERROR": "https://api.example.com/errors/validation",
    "INTERNAL_ERROR": "https://api.example.com/errors/internal",
}

HTTP_TITLES = {
    400: "Bad Request",
    401: "Unauthorized",
    403: "Forbidden",
    404: "Not Found",
    409: "Conflict",
    422: "Unprocessable Entity",
    429: "Too Many Requests",
    500: "Internal Server Error",
}
```

### Task 3: Custom Exception Handlers
```python
# app/main.py
from fastapi import FastAPI, Request, status
from fastapi.responses import JSONResponse
from fastapi.exceptions import RequestValidationError
from app.exceptions import AppException
from app.error_responses import ProblemDetail, ERROR_TYPE_MAP, HTTP_TITLES

app = FastAPI()

@app.exception_handler(AppException)
async def app_exception_handler(request: Request, exc: AppException) -> JSONResponse:
    problem = ProblemDetail(
        type=ERROR_TYPE_MAP.get(exc.error_code, "about:blank"),
        title=HTTP_TITLES.get(exc.status_code, "Error"),
        status=exc.status_code,
        detail=exc.detail,
        error_code=exc.error_code,
        extra=exc.extra,
    )
    return JSONResponse(
        status_code=exc.status_code,
        content=problem.model_dump(),
        headers={"Content-Type": "application/problem+json"},
    )

@app.exception_handler(RequestValidationError)
async def validation_exception_handler(
    request: Request, exc: RequestValidationError
) -> JSONResponse:
    """Override FastAPI's default 422 handler to use RFC 7807 format."""
    # Convert FastAPI error format to our format
    field_errors = {}
    for error in exc.errors():
        field_path = ".".join(str(loc) for loc in error["loc"] if loc != "body")
        field_errors[field_path] = error["msg"]
    
    problem = ProblemDetail(
        type=ERROR_TYPE_MAP["VALIDATION_ERROR"],
        title="Validation Error",
        status=422,
        detail="Request validation failed. Check 'extra.fields' for details.",
        error_code="VALIDATION_ERROR",
        extra={"fields": field_errors},
    )
    return JSONResponse(
        status_code=422,
        content=problem.model_dump(),
        headers={"Content-Type": "application/problem+json"},
    )

@app.exception_handler(Exception)
async def unhandled_exception_handler(request: Request, exc: Exception) -> JSONResponse:
    """Catch-all for unhandled exceptions — never expose internals."""
    import logging
    logging.getLogger(__name__).exception(f"Unhandled error: {exc}")
    
    problem = ProblemDetail(
        type=ERROR_TYPE_MAP["INTERNAL_ERROR"],
        title="Internal Server Error",
        status=500,
        detail="An unexpected error occurred. Please try again later.",
        error_code="INTERNAL_ERROR",
    )
    return JSONResponse(
        status_code=500,
        content=problem.model_dump(),
        headers={"Content-Type": "application/problem+json"},
    )
```

### Task 5: Custom Validation with Useful Error Messages
```python
# app/items/schemas.py
from pydantic import BaseModel, Field, field_validator, model_validator
from typing import Self
import re

class ItemCreate(BaseModel):
    name: str = Field(min_length=1, max_length=200)
    price: float = Field(gt=0)
    slug: str = Field(min_length=1, max_length=100)
    category_id: int = Field(gt=0)
    discount_percent: float = Field(ge=0, le=100, default=0)
    sale_price: float | None = None

    @field_validator("slug")
    @classmethod
    def slug_format(cls, v: str) -> str:
        if not re.match(r"^[a-z0-9]+(?:-[a-z0-9]+)*$", v):
            raise ValueError(
                "Slug must contain only lowercase letters, numbers, and hyphens. "
                "Cannot start or end with a hyphen. Example: 'my-product-name'"
            )
        return v

    @model_validator(mode="after")
    def sale_price_consistency(self) -> Self:
        if self.sale_price is not None and self.sale_price >= self.price:
            raise ValueError(
                f"sale_price ({self.sale_price}) must be less than price ({self.price})"
            )
        return self
```

### Task 6: Testing Error Responses
```python
# tests/test_error_handling.py
import pytest
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)

def test_not_found_returns_problem_details():
    response = client.get("/items/99999")
    assert response.status_code == 404
    body = response.json()
    assert body["status"] == 404
    assert body["error_code"] == "NOT_FOUND"
    assert "title" in body
    assert "detail" in body
    assert "type" in body
    assert "instance" in body
    assert response.headers["content-type"] == "application/problem+json"

def test_validation_error_shows_field_details():
    response = client.post("/items/", json={"name": "", "price": -1, "slug": "INVALID!!!"})
    assert response.status_code == 422
    body = response.json()
    assert body["error_code"] == "VALIDATION_ERROR"
    fields = body["extra"]["fields"]
    assert "name" in fields or "price" in fields or "slug" in fields

def test_internal_error_does_not_expose_stack_trace():
    # Trigger an unhandled exception (mock it)
    with pytest.raises(Exception):
        response = client.get("/items/trigger-error")
        if response.status_code == 500:
            body = response.json()
            assert "traceback" not in str(body).lower()
            assert "sqlalchemy" not in str(body).lower()
```
