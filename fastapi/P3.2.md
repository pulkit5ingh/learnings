# P3.2 — Multi-tenant SaaS API

**Phase:** 3 — Authentication & Authorization
**Topic:** Multi-tenancy, org-scoped JWT, RBAC per organization
**Type:** Project

---

## Goal

Build a multi-tenant SaaS API where users can belong to multiple organizations with different roles per org, and JWTs carry org context.

## What to Build

- User registration and login (global)
- Create/manage **organizations**
- Invite users to an organization with a role (owner / admin / member)
- `POST /auth/switch-org/{org_id}` — issue new token scoped to that org
- All resource endpoints enforce org membership and role
- Resources (e.g., projects) are scoped to an organization

### app/models/membership.py

```python
from sqlalchemy import ForeignKey, String, Enum as SAEnum
from sqlalchemy.orm import Mapped, mapped_column, relationship
from app.models.base import Base, TimestampMixin
import enum

class OrgRole(str, enum.Enum):
    owner = "owner"
    admin = "admin"
    member = "member"

class Membership(Base, TimestampMixin):
    __tablename__ = "memberships"

    id: Mapped[int] = mapped_column(primary_key=True)
    user_id: Mapped[int] = mapped_column(ForeignKey("users.id", ondelete="CASCADE"))
    org_id: Mapped[int] = mapped_column(ForeignKey("organizations.id", ondelete="CASCADE"))
    role: Mapped[OrgRole] = mapped_column(default=OrgRole.member)
    is_active: Mapped[bool] = mapped_column(default=True)

    user: Mapped["User"] = relationship(lazy="selectin")
    org: Mapped["Organization"] = relationship(lazy="selectin")
```

### app/security/jwt.py

```python
from jose import jwt, JWTError
from datetime import datetime, timedelta, UTC
from app.config import settings

ALGORITHM = "HS256"

def create_token(user_id: int, org_id: int | None = None, role: str | None = None) -> str:
    payload = {
        "sub": str(user_id),
        "exp": datetime.now(UTC) + timedelta(minutes=60),
        "type": "access",
    }
    if org_id is not None:
        payload["org_id"] = org_id
        payload["role"] = role
    return jwt.encode(payload, settings.secret_key, algorithm=ALGORITHM)

def decode_token(token: str) -> dict:
    try:
        return jwt.decode(token, settings.secret_key, algorithms=[ALGORITHM])
    except JWTError as e:
        raise ValueError(str(e))
```

### app/security/dependencies.py

```python
from fastapi import Depends, HTTPException, Security
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.security.jwt import decode_token
from app.models.user import User
from app.models.membership import Membership, OrgRole
from app.database import get_session
from dataclasses import dataclass

bearer_scheme = HTTPBearer()

@dataclass
class CurrentUser:
    user: User
    org_id: int | None
    role: OrgRole | None

async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Security(bearer_scheme),
    session: AsyncSession = Depends(get_session),
) -> CurrentUser:
    try:
        payload = decode_token(credentials.credentials)
    except ValueError:
        raise HTTPException(401, "Invalid token")

    user = await session.get(User, int(payload["sub"]))
    if not user:
        raise HTTPException(401, "User not found")

    return CurrentUser(
        user=user,
        org_id=payload.get("org_id"),
        role=OrgRole(payload["role"]) if payload.get("role") else None,
    )

def require_org(minimum_role: OrgRole = OrgRole.member):
    """Dependency factory: ensures the token has an org context and sufficient role."""
    role_order = [OrgRole.member, OrgRole.admin, OrgRole.owner]

    def _check(current: CurrentUser = Depends(get_current_user)) -> CurrentUser:
        if current.org_id is None:
            raise HTTPException(403, "No organization context in token. Call /auth/switch-org first.")
        if role_order.index(current.role) < role_order.index(minimum_role):
            raise HTTPException(403, f"Requires {minimum_role.value} role or higher")
        return current

    return _check
```

### app/routers/auth.py

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.database import get_session
from app.models.membership import Membership
from app.security.jwt import create_token
from app.security.dependencies import get_current_user, CurrentUser

router = APIRouter(prefix="/auth", tags=["auth"])

@router.post("/switch-org/{org_id}")
async def switch_org(
    org_id: int,
    current: CurrentUser = Depends(get_current_user),
    session: AsyncSession = Depends(get_session),
):
    result = await session.execute(
        select(Membership).where(
            Membership.user_id == current.user.id,
            Membership.org_id == org_id,
            Membership.is_active == True,
        )
    )
    membership = result.scalar_one_or_none()
    if not membership:
        raise HTTPException(403, "Not a member of this organization")

    token = create_token(current.user.id, org_id=org_id, role=membership.role.value)
    return {"access_token": token, "token_type": "bearer", "org_id": org_id, "role": membership.role}
```

### app/routers/projects.py

```python
from fastapi import APIRouter, Depends, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import get_session
from app.security.dependencies import require_org, CurrentUser, OrgRole
from app.models.project import Project
from sqlalchemy import select

router = APIRouter(prefix="/projects", tags=["projects"])

@router.get("")
async def list_projects(
    current: CurrentUser = Depends(require_org(OrgRole.member)),
    session: AsyncSession = Depends(get_session),
):
    result = await session.execute(
        select(Project).where(Project.org_id == current.org_id)
    )
    return result.scalars().all()

@router.delete("/{project_id}", status_code=204)
async def delete_project(
    project_id: int,
    current: CurrentUser = Depends(require_org(OrgRole.admin)),  # admin+
    session: AsyncSession = Depends(get_session),
):
    project = await session.get(Project, project_id)
    if not project or project.org_id != current.org_id:
        raise HTTPException(404)
    await session.delete(project)
    await session.commit()
```

## Deliverables

- [ ] User can belong to multiple orgs with different roles
- [ ] `POST /auth/switch-org/{org_id}` fails with 403 if user is not a member
- [ ] Org-scoped token contains `org_id` and `role` claims
- [ ] `require_org(OrgRole.admin)` dependency blocks members from admin-only routes
- [ ] Project CRUD is fully scoped — users cannot see/modify other orgs' projects
- [ ] Invitation flow: `POST /orgs/{org_id}/invite` creates pending membership
- [ ] Owner role can promote/demote other members
