## Goal
Build real-time bidirectional communication using WebSockets with rooms, connection management, and JWT authentication.

## Background

### WebSockets vs HTTP
HTTP is request-response: client asks, server answers. WebSockets are persistent, bidirectional TCP connections. The client connects once and both sides can send messages at any time. This enables real-time features: chat, live notifications, collaborative editing, live dashboards.

### FastAPI WebSocket Support
FastAPI provides a `WebSocket` class as a route parameter. The route stays open as long as the connection is alive. You call `await websocket.receive_text()` to wait for a message and `await websocket.send_text()` to push data. For multiple concurrent connections, you manage them yourself (FastAPI does not do this for you).

### Scaling WebSockets
Each WebSocket connection lives on one server instance. For horizontal scaling (multiple instances), you need a pub/sub layer (Redis) to broadcast messages across instances. When instance A receives a message, it publishes to Redis, and instances B and C receive it and forward to their connected clients.

# app/websockets/basic.py
from fastapi import FastAPI, WebSocket, WebSocketDisconnect

app = FastAPI()

@app.websocket("/ws")
async def websocket_endpoint(websocket: WebSocket):
    await websocket.accept()
    try:
        while True:
            data = await websocket.receive_text()

### Task 2: Connection Manager
```python
# app/websockets/manager.py
from fastapi import WebSocket
from typing import Any
import json
import logging

logger = logging.getLogger(__name__)

class ConnectionManager:
    def __init__(self):
        # Map of connection_id → WebSocket
        self.active_connections: dict[str, WebSocket] = {}
        # Map of room_id → set of connection_ids
        self.rooms: dict[str, set[str]] = {}

    async def connect(self, websocket: WebSocket, connection_id: str) -> None:
        await websocket.accept()
        self.active_connections[connection_id] = websocket
        logger.info(f"Connected: {connection_id}. Total: {len(self.active_connections)}")

    def disconnect(self, connection_id: str) -> None:
        self.active_connections.pop(connection_id, None)
        # Remove from all rooms
        for room_members in self.rooms.values():
            room_members.discard(connection_id)
        logger.info(f"Disconnected: {connection_id}. Total: {len(self.active_connections)}")

    def join_room(self, room_id: str, connection_id: str) -> None:
        if room_id not in self.rooms:
            self.rooms[room_id] = set()
        self.rooms[room_id].add(connection_id)

    def leave_room(self, room_id: str, connection_id: str) -> None:
        if room_id in self.rooms:
            self.rooms[room_id].discard(connection_id)

    async def send_personal(self, connection_id: str, message: Any) -> None:
        ws = self.active_connections.get(connection_id)
        if ws:
            text = json.dumps(message) if not isinstance(message, str) else message
            await ws.send_text(text)

    async def broadcast_to_room(self, room_id: str, message: Any, exclude: str | None = None) -> None:
        member_ids = self.rooms.get(room_id, set()).copy()
        text = json.dumps(message) if not isinstance(message, str) else message
        
        disconnected = []
        for conn_id in member_ids:
            if conn_id == exclude:
                continue
            ws = self.active_connections.get(conn_id)
            if ws:
                try:
                    await ws.send_text(text)
                except Exception:
                    disconnected.append(conn_id)
        
        for conn_id in disconnected:
            self.disconnect(conn_id)

    async def broadcast_all(self, message: Any) -> None:
        text = json.dumps(message) if not isinstance(message, str) else message
        for conn_id, ws in list(self.active_connections.items()):
            try:
                await ws.send_text(text)
            except Exception:
                self.disconnect(conn_id)

    @property
    def connection_count(self) -> int:
        return len(self.active_connections)

# Global instance
manager = ConnectionManager()
```

### Task 3: Chat Room Endpoint
```python
# app/websockets/chat.py
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Query
from app.websockets.manager import manager
import json
import uuid

router = APIRouter()

@router.websocket("/ws/chat/{room_id}")
async def chat_endpoint(
    websocket: WebSocket,
    room_id: str,
    username: str = Query(...),
):
    connection_id = str(uuid.uuid4())
    
    await manager.connect(websocket, connection_id)
    manager.join_room(room_id, connection_id)
    
    # Notify room of new member
    await manager.broadcast_to_room(
        room_id,
        {"type": "system", "message": f"{username} joined the room"},
    )
    
    try:
        while True:
            raw = await websocket.receive_text()
            
            try:
                data = json.loads(raw)
            except json.JSONDecodeError:
                data = {"text": raw}
            
            message = {
                "type": "message",
                "room": room_id,
                "from": username,
                "text": data.get("text", ""),
                "connection_id": connection_id,
            }
            
            # Broadcast to everyone in the room (including sender)
            await manager.broadcast_to_room(room_id, message)
    
    except WebSocketDisconnect:
        manager.disconnect(connection_id)
        await manager.broadcast_to_room(
            room_id,
            {"type": "system", "message": f"{username} left the room"},
        )
```

### Task 4: WebSocket Authentication
```python
# app/websockets/auth.py
from fastapi import WebSocket, WebSocketDisconnect, Query, HTTPException
from app.auth.tokens import decode_access_token

async def get_ws_user(
    websocket: WebSocket,
    token: str = Query(...),
) -> dict:
    """
    Authenticate a WebSocket connection via query parameter token.
    (Headers are not easily accessible during WebSocket handshake in browsers.)
    """
    try:
        payload = decode_access_token(token)
        return {"user_id": int(payload["sub"]), "roles": payload.get("roles", [])}
    except ValueError:
        await websocket.close(code=4001, reason="Invalid or expired token")
        raise WebSocketDisconnect(code=4001)

# Usage:
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Depends
from typing import Annotated

router = APIRouter()

@router.websocket("/ws/notifications")
async def notifications_endpoint(
    websocket: WebSocket,
    user: Annotated[dict, Depends(get_ws_user)],
):
    user_id = user["user_id"]
    conn_id = str(user_id)
    
    await manager.connect(websocket, conn_id)
    
    try:
        # Send pending notifications on connect
        await manager.send_personal(
            conn_id,
            {"type": "connected", "user_id": user_id},
        )
        
        # Keep connection alive, waiting for messages or pings
        while True:
            data = await websocket.receive_text()
            if data == "ping":
                await websocket.send_text("pong")
    
    except WebSocketDisconnect:
        manager.disconnect(conn_id)
```

### Task 5: Redis Pub/Sub for Multi-Instance Scaling
```python
# app/websockets/redis_pubsub.py
import asyncio
import json
import redis.asyncio as aioredis
from app.cache import get_redis

CHANNEL = "ws_broadcast"

async def publish_to_all_instances(message: dict) -> None:
    """Publish a message that all server instances should forward."""
    redis = await get_redis()
    await redis.publish(CHANNEL, json.dumps(message))

async def subscribe_and_forward() -> None:
    """Subscribe to Redis channel and forward to local WebSocket connections."""
    redis = await get_redis()
    pubsub = redis.pubsub()
    await pubsub.subscribe(CHANNEL)
    
    async for raw_message in pubsub.listen():
        if raw_message["type"] != "message":
            continue
        
        message = json.loads(raw_message["data"])
        
        if message.get("target") == "all":
            await manager.broadcast_all(message["payload"])
        elif room_id := message.get("room"):
            await manager.broadcast_to_room(room_id, message["payload"])
        elif conn_id := message.get("connection_id"):
            await manager.send_personal(conn_id, message["payload"])

```
