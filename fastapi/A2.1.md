## Goal

## Background

### SQLAlchemy Core vs ORM
SQLAlchemy has two layers: Core and ORM. Core is the low-level layer — it provides a Pythonic SQL expression language that generates SQL statements. ORM is built on top of Core and maps Python classes to tables. Core is useful when you need fine-grained control over SQL, want to work with existing databases without mapping every table to a class, or need maximum performance for bulk operations.

### Engine and Connection
The `Engine` is a factory for database connections. It manages a connection pool, so you don't create a new TCP connection for every query. A `Connection` is a single active connection from the pool. In SQLAlchemy 2.0, you use context managers to get a connection: `with engine.connect() as conn:`. This ensures the connection is returned to the pool when you're done.

### Transactions
By default in SQLAlchemy 2.0, `engine.connect()` starts a transaction that must be explicitly committed with `conn.commit()` or rolled back with `conn.rollback()`. Alternatively, use `engine.begin()` which auto-commits on exit and rolls back on exception. Understanding transaction boundaries is critical for data integrity.

### Task 1: Engine and MetaData Setup
```python
# core_setup.py
from sqlalchemy import create_engine, MetaData, Table, Column, Integer, String, Float, Boolean, DateTime, ForeignKey, text
from sqlalchemy.sql import func
import os

DATABASE_URL = os.getenv("DATABASE_URL", "postgresql://user:password@localhost/testdb")

# echo=True logs all SQL — great for development
engine = create_engine(DATABASE_URL, echo=True, pool_size=5, max_overflow=10)

metadata = MetaData()

# Define table schema
users = Table(
    "users",
    metadata,
    Column("id", Integer, primary_key=True, autoincrement=True),
    Column("name", String(100), nullable=False),
    Column("email", String(255), unique=True, nullable=False),
    Column("is_active", Boolean, default=True),
    Column("created_at", DateTime, server_default=func.now()),
)

posts = Table(
    "posts",
    metadata,
    Column("id", Integer, primary_key=True, autoincrement=True),
    Column("title", String(200), nullable=False),
    Column("body", String, nullable=False),
    Column("user_id", Integer, ForeignKey("users.id", ondelete="CASCADE"), nullable=False),
    Column("created_at", DateTime, server_default=func.now()),
)

# Create all tables
metadata.create_all(engine)
print("Tables created!")
```

### Task 2: INSERT and SELECT
```python
# crud_operations.py
from sqlalchemy import insert, select, update, delete
from core_setup import engine, users, posts

def seed_users():
    with engine.begin() as conn:  # auto-commit on success
        # Insert one
        result = conn.execute(
            insert(users).values(name="Alice", email="alice@example.com")
        )
        alice_id = result.inserted_primary_key[0]
        print(f"Alice inserted with ID: {alice_id}")

        # Insert many
        conn.execute(
            insert(users),
            [
                {"name": "Bob", "email": "bob@example.com"},
                {"name": "Charlie", "email": "charlie@example.com"},
            ],
        )
    print("Users seeded.")

def fetch_users():
    with engine.connect() as conn:
        # Select all
        stmt = select(users).order_by(users.c.name)
        result = conn.execute(stmt)
        for row in result:
            print(dict(row._mapping))

        # Select with WHERE
        stmt = select(users).where(users.c.email.like("%@example.com"))
        result = conn.execute(stmt)
        print(f"Found {result.rowcount} users")  # rowcount only for DML

        # Select specific columns
        stmt = select(users.c.id, users.c.name).where(users.c.is_active == True)
        result = conn.execute(stmt)
        print(result.all())
```

### Task 3: UPDATE and DELETE
```python
def update_user(user_id: int, name: str):
    with engine.begin() as conn:
        stmt = (
            update(users)
            .where(users.c.id == user_id)
            .values(name=name)
            .returning(users.c.id, users.c.name, users.c.email)
        )
        result = conn.execute(stmt)
        updated = result.fetchone()
        if updated is None:
            print(f"User {user_id} not found")
        else:
            print(f"Updated: {dict(updated._mapping)}")

def deactivate_user(user_id: int):
    with engine.begin() as conn:
        conn.execute(
            update(users)
            .where(users.c.id == user_id)
            .values(is_active=False)
        )

def delete_user(user_id: int):
    with engine.begin() as conn:
        result = conn.execute(
            delete(users).where(users.c.id == user_id)
        )
        print(f"Deleted {result.rowcount} rows")
```

### Task 4: Joins and Aggregations
```python
from sqlalchemy import join, func, case

def user_post_counts():
    with engine.connect() as conn:
        # JOIN with aggregation
        stmt = (
            select(
                users.c.id,
                users.c.name,
                func.count(posts.c.id).label("post_count"),
            )
            .select_from(users.outerjoin(posts, users.c.id == posts.c.user_id))
            .group_by(users.c.id, users.c.name)
            .order_by(func.count(posts.c.id).desc())
        )
        result = conn.execute(stmt)
        for row in result:
            print(dict(row._mapping))

def get_users_with_posts():
    with engine.connect() as conn:
        stmt = (
            select(users, posts)
            .join(posts, users.c.id == posts.c.user_id)
            .where(users.c.is_active == True)
        )
        result = conn.execute(stmt)
        return [dict(row._mapping) for row in result]
```

### Task 5: Transactions with Explicit Control

```python
def transfer_example():
    """Example showing manual transaction control."""
    with engine.connect() as conn:
        trans = conn.begin()
        try:
            # Operation 1
            conn.execute(update(users).where(users.c.id == 1).values(name="Alice Updated"))
            
            # Simulate an error condition
            if True:  # change to False to let it succeed
                raise ValueError("Simulated failure!")
            
            # Operation 2
            conn.execute(update(users).where(users.c.id == 2).values(name="Bob Updated"))
            
            trans.commit()
            print("Transaction committed")
        except Exception as e:
            trans.rollback()
            print(f"Transaction rolled back due to: {e}")

def nested_transaction_example():
    """Savepoints for nested transactions."""
    with engine.begin() as conn:
        conn.execute(update(users).where(users.c.id == 1).values(name="Step 1"))
        
        savepoint = conn.begin_nested()  # SAVEPOINT
        try:
            conn.execute(update(users).where(users.c.id == 2).values(name="Step 2"))
            raise ValueError("Inner failure")
            savepoint.commit()
        except ValueError:
            savepoint.rollback()  # rolls back to savepoint, not full transaction
            print("Inner transaction rolled back, outer continues")
        
        conn.execute(update(users).where(users.c.id == 3).values(name="Step 3"))
    # Outer transaction committed here
```

```python
def raw_sql_example():
    with engine.connect() as conn:
        # Always use text() for raw SQL — never use f-strings
        stmt = text("SELECT * FROM users WHERE email = :email")
        result = conn.execute(stmt, {"email": "alice@example.com"})
        user = result.fetchone()
        print(dict(user._mapping) if user else "Not found")
        
        # Execute DDL
        conn.execute(text("CREATE INDEX IF NOT EXISTS idx_users_email ON users(email)"))
        conn.commit()
```
