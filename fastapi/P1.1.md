## Goal

Build a URL shortening service with full CRUD operations backed by an in-memory dictionary.

## What to Build

- `POST /shorten` — accept a long URL, return a short code
- `GET /{code}` — redirect (307) to the original URL
- `GET /links` — list all shortened links with click counts
- `GET /links/{code}` — get stats for a single link
- `DELETE /links/{code}` — delete a link
- Custom 404 error when a code is not found

### app/models.py

```python
from pydantic import BaseModel, HttpUrl, Field
from datetime import datetime

class ShortenRequest(BaseModel):
    url: HttpUrl
    custom_code: str | None = Field(default=None, min_length=3, max_length=20)

class LinkResponse(BaseModel):
    code: str
    original_url: str
    short_url: str
    clicks: int
    created_at: datetime

class ShortenResponse(BaseModel):
    code: str
    short_url: str
```

### app/store.py

```python
from datetime import datetime, UTC
from dataclasses import dataclass, field

@dataclass
class LinkRecord:
    code: str
    original_url: str
    clicks: int = 0
    created_at: datetime = field(default_factory=lambda: datetime.now(UTC))

class LinkStore:
    def __init__(self):
        self._data: dict[str, LinkRecord] = {}

    def save(self, code: str, url: str) -> LinkRecord:
        record = LinkRecord(code=code, original_url=url)
        self._data[code] = record
        return record

    def get(self, code: str) -> LinkRecord | None:
        return self._data.get(code)

    def increment_clicks(self, code: str) -> None:
        if code in self._data:
            self._data[code].clicks += 1

    def delete(self, code: str) -> bool:
        return self._data.pop(code, None) is not None

    def all(self) -> list[LinkRecord]:
        return list(self._data.values())

    def code_exists(self, code: str) -> bool:
        return code in self._data

link_store = LinkStore()
```

### app/utils.py

```python
import secrets
import string

ALPHABET = string.ascii_letters + string.digits

def generate_code(length: int = 7) -> str:
    return "".join(secrets.choice(ALPHABET) for _ in range(length))
```

### app/router.py

```python
from fastapi import APIRouter, HTTPException, Request
from fastapi.responses import RedirectResponse
from app.models import ShortenRequest, ShortenResponse, LinkResponse
from app.store import link_store
from app.utils import generate_code

router = APIRouter()

BASE_URL = "http://localhost:8000"

@router.post("/shorten", response_model=ShortenResponse, status_code=201)
async def shorten_url(body: ShortenRequest):
    code = body.custom_code or generate_code()
    if link_store.code_exists(code):
        raise HTTPException(409, detail=f"Code '{code}' already taken")
    link_store.save(code, str(body.url))
    return ShortenResponse(code=code, short_url=f"{BASE_URL}/{code}")

@router.get("/{code}")
async def redirect(code: str):
    record = link_store.get(code)
    if not record:
        raise HTTPException(404, detail="Short link not found")
    link_store.increment_clicks(code)
    return RedirectResponse(url=record.original_url, status_code=307)

@router.get("/links", response_model=list[LinkResponse])
async def list_links(request: Request):
    base = str(request.base_url).rstrip("/")
    return [
        LinkResponse(
            code=r.code,
            original_url=r.original_url,
            short_url=f"{base}/{r.code}",
            clicks=r.clicks,
            created_at=r.created_at,
        )
        for r in link_store.all()
    ]

@router.get("/links/{code}", response_model=LinkResponse)
async def get_link(code: str, request: Request):
    record = link_store.get(code)
    if not record:
        raise HTTPException(404, detail="Short link not found")
    base = str(request.base_url).rstrip("/")
    return LinkResponse(
        code=record.code,
        original_url=record.original_url,
        short_url=f"{base}/{record.code}",
        clicks=record.clicks,
        created_at=record.created_at,
    )

@router.delete("/links/{code}", status_code=204)
async def delete_link(code: str):
    if not link_store.delete(code):
        raise HTTPException(404, detail="Short link not found")
```
