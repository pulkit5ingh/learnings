### tests/unit/test_task_service.py

```python
import pytest
from unittest.mock import AsyncMock, MagicMock
from datetime import date, timedelta
from app.services.task_service import TaskService
from app.models.task import Task, TaskStatus

@pytest.mark.unit
class TestTaskService:
    def setup_method(self):
        self.mock_session = MagicMock()
        self.svc = TaskService(self.mock_session)

    def test_is_overdue_with_past_due_date(self):
        task = Task(due_date=date.today() - timedelta(days=1), status=TaskStatus.todo)
        assert self.svc.is_overdue(task) is True

    def test_is_overdue_with_future_due_date(self):
        task = Task(due_date=date.today() + timedelta(days=1), status=TaskStatus.todo)
        assert self.svc.is_overdue(task) is False

    def test_is_overdue_returns_false_for_done_task(self):
        task = Task(due_date=date.today() - timedelta(days=5), status=TaskStatus.done)
        assert self.svc.is_overdue(task) is False

    @pytest.mark.parametrize("from_status,to_status,valid", [
        (TaskStatus.todo, TaskStatus.in_progress, True),
        (TaskStatus.in_progress, TaskStatus.done, True),
        (TaskStatus.todo, TaskStatus.done, False),  # must go through in_progress
        (TaskStatus.done, TaskStatus.todo, False),  # can't reopen
    ])
    def test_status_transitions(self, from_status, to_status, valid):
        if valid:
            assert self.svc.validate_transition(from_status, to_status) is True
        else:
            with pytest.raises(ValueError):
                self.svc.validate_transition(from_status, to_status)
```

```python
import pytest
from datetime import date
from app.services.task_service import TaskService
from app.schemas.task import TaskCreate

@pytest.mark.integration

        data = TaskCreate(title="Buy groceries", due_date=date(2025, 12, 31))
        task = await svc.create(data)
        assert task.id is not None

    async def test_list_filters_by_status(self, session):
        svc = TaskService(session)
        await svc.create(TaskCreate(title="Task A"))
        t = await svc.create(TaskCreate(title="Task B"))
        await svc.update_status(t.id, "in_progress")

        in_progress = await svc.list_tasks(status="in_progress")
        assert all(t.status.value == "in_progress" for t in in_progress)

        svc = TaskService(session)
        task = await svc.create(TaskCreate(title="Temp task"))
        await svc.delete(task.id)
        assert await svc.get(task.id) is None
```

```python
import pytest

@pytest.mark.e2e
class TestTasksAPI:
    async def test_create_task(self, client):
        resp = await client.post("/tasks", json={"title": "Write tests"})
        assert resp.status_code == 201
        data = resp.json()
        assert data["title"] == "Write tests"
        assert data["status"] == "todo"

    async def test_get_nonexistent_task(self, client):
        resp = await client.get("/tasks/99999")
        assert resp.status_code == 404

    async def test_status_transition_flow(self, client):
        resp = await client.post("/tasks", json={"title": "Pipeline task"})
        task_id = resp.json()["id"]

        resp = await client.patch(f"/tasks/{task_id}/status", json={"status": "in_progress"})
        assert resp.status_code == 200

        resp = await client.patch(f"/tasks/{task_id}/status", json={"status": "done"})
        assert resp.status_code == 200

    async def test_invalid_status_transition(self, client):
        resp = await client.post("/tasks", json={"title": "Bad transition"})
        task_id = resp.json()["id"]

        # Skip in_progress, go directly to done
        resp = await client.patch(f"/tasks/{task_id}/status", json={"status": "done"})
        assert resp.status_code == 422

    async def test_list_with_pagination(self, client):
        for i in range(5):
            await client.post("/tasks", json={"title": f"Task {i}"})
        resp = await client.get("/tasks?limit=3&offset=0")
        assert resp.status_code == 200
        assert len(resp.json()["items"]) == 3
```

### app/services/task_service.py

```python
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from fastapi import HTTPException
from app.models.task import Task, TaskStatus
from app.schemas.task import TaskCreate, TaskUpdate
from datetime import date

VALID_TRANSITIONS = {
    TaskStatus.todo: {TaskStatus.in_progress},
    TaskStatus.in_progress: {TaskStatus.done},
    TaskStatus.done: set(),
}

class TaskService:
    def __init__(self, session: AsyncSession):
        self.session = session

    def is_overdue(self, task: Task) -> bool:
        if task.status == TaskStatus.done:
            return False
        return task.due_date is not None and task.due_date < date.today()

    def validate_transition(self, from_status: TaskStatus, to_status: TaskStatus) -> bool:
        if to_status not in VALID_TRANSITIONS[from_status]:
            raise ValueError(f"Cannot transition from {from_status} to {to_status}")
        return True

    async def create(self, data: TaskCreate) -> Task:
        task = Task(**data.model_dump())
        self.session.add(task)
        await self.session.flush()
        return task

    async def get(self, task_id: int) -> Task | None:
        return await self.session.get(Task, task_id)

    async def list_tasks(self, status: str | None = None, offset=0, limit=20):
        q = select(Task)
        if status:
            q = q.where(Task.status == status)
        result = await self.session.execute(q.offset(offset).limit(limit))
        return result.scalars().all()

    async def update_status(self, task_id: int, new_status: str) -> Task:
        task = await self.get(task_id)
        if not task:
            raise HTTPException(404)
        self.validate_transition(task.status, TaskStatus(new_status))
        task.status = TaskStatus(new_status)
        await self.session.flush()
        return task

    async def delete(self, task_id: int) -> None:
        task = await self.get(task_id)
        if task:
            await self.session.delete(task)
            await self.session.flush()
```
