## Goal

Build a blog API with posts, tags, and authors using async SQLAlchemy and Alembic-managed migrations.

## What to Build

- Authors: CRUD
- Posts: CRUD with author FK, many-to-many tags, published/draft status
- Tags: CRUD, list posts by tag
- `GET /posts` with pagination and filter by `author_id`, `tag`, `status`
- Alembic migrations from scratch (no `--autogenerate` shortcut — write them by hand first)

    updated_at: Mapped[datetime] = mapped_column(
        server_default=func.now(), onupdate=func.now()

### app/models/post.py

```python
from sqlalchemy import String, Text, ForeignKey, Table, Column, Enum as SAEnum
from sqlalchemy.orm import Mapped, mapped_column, relationship
from app.models.base import Base, TimestampMixin
import enum

post_tag = Table(
    "post_tag",
    Base.metadata,
    Column("post_id", ForeignKey("posts.id", ondelete="CASCADE"), primary_key=True),
    Column("tag_id", ForeignKey("tags.id", ondelete="CASCADE"), primary_key=True),
)

class PostStatus(str, enum.Enum):
    draft = "draft"
    published = "published"

class Post(Base, TimestampMixin):
    __tablename__ = "posts"

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    body: Mapped[str] = mapped_column(Text)
    status: Mapped[PostStatus] = mapped_column(default=PostStatus.draft)
    author_id: Mapped[int] = mapped_column(ForeignKey("authors.id", ondelete="CASCADE"))

    author: Mapped["Author"] = relationship(back_populates="posts", lazy="selectin")
    tags: Mapped[list["Tag"]] = relationship(
        secondary=post_tag, back_populates="posts", lazy="selectin"
    )
```

### app/repositories/post_repo.py

```python
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func
from sqlalchemy.orm import selectinload
from app.models.post import Post, PostStatus
from app.models.tag import Tag

class PostRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def create(self, title: str, body: str, author_id: int, tag_ids: list[int]) -> Post:
        post = Post(title=title, body=body, author_id=author_id)
        if tag_ids:
            result = await self.session.execute(select(Tag).where(Tag.id.in_(tag_ids)))
            post.tags = result.scalars().all()
        self.session.add(post)
        await self.session.commit()
        await self.session.refresh(post)
        return post

    async def list_posts(
        self,
        author_id: int | None = None,
        tag: str | None = None,
        status: PostStatus | None = None,
        offset: int = 0,
        limit: int = 20,
    ) -> tuple[list[Post], int]:
        q = select(Post).options(selectinload(Post.tags), selectinload(Post.author))
        if author_id:
            q = q.where(Post.author_id == author_id)
        if status:
            q = q.where(Post.status == status)
        if tag:
            q = q.join(Post.tags).where(Tag.name == tag)

        total_q = select(func.count()).select_from(q.subquery())
        total = (await self.session.execute(total_q)).scalar_one()
        posts = (await self.session.execute(q.offset(offset).limit(limit))).scalars().all()
        return posts, total

    async def get(self, post_id: int) -> Post | None:
        return await self.session.get(Post, post_id)

    async def delete(self, post: Post) -> None:
        await self.session.delete(post)
        await self.session.commit()
```
