**Phase:** 1 — Python & FastAPI Fundamentals
**Topic:** FastAPI, Pydantic v2, dependency injection, in-memory store

## Goal

Build a personal finance tracker that manages categories, transactions, and generates monthly summaries.

## What to Build

- CRUD for **categories** (income / expense)
- CRUD for **transactions** (amount, date, category, note)
- `GET /summary/{year}/{month}` — income vs. expense totals and per-category breakdown
- Dependency-injected store (overrideable in tests)
- Filtering transactions by category and date range

## Implementation

### app/models.py

```python
from pydantic import BaseModel, Field, model_validator
from datetime import date
from enum import Enum
from decimal import Decimal

class CategoryType(str, Enum):
    income = "income"
    expense = "expense"

class CategoryCreate(BaseModel):
    name: str = Field(min_length=1, max_length=50)
    type: CategoryType

class CategoryResponse(CategoryCreate):
    id: int

class TransactionCreate(BaseModel):
    amount: Decimal = Field(gt=0, decimal_places=2)
    date: date
    category_id: int
    note: str = ""

    @model_validator(mode="after")
    def date_not_future(self) -> "TransactionCreate":
        if self.date > date.today():
            raise ValueError("Transaction date cannot be in the future")
        return self

class TransactionResponse(TransactionCreate):
    id: int
    category_name: str
    category_type: CategoryType

class MonthlySummary(BaseModel):
    year: int
    month: int
    total_income: Decimal
    total_expense: Decimal
    net: Decimal
    by_category: dict[str, Decimal]
```

### app/store.py

```python
from app.models import CategoryCreate, TransactionCreate
from dataclasses import dataclass, field
from decimal import Decimal

@dataclass
class Category:
    id: int
    name: str
    type: str

@dataclass
class Transaction:
    id: int
    amount: Decimal
    date: object
    category_id: int
    note: str

class FinanceStore:
    def __init__(self):
        self._categories: dict[int, Category] = {}
        self._transactions: dict[int, Transaction] = {}
        self._cat_counter = 0
        self._txn_counter = 0

    # Categories
    def create_category(self, data: CategoryCreate) -> Category:
        self._cat_counter += 1
        cat = Category(id=self._cat_counter, name=data.name, type=data.type.value)
        self._categories[cat.id] = cat
        return cat

    def get_category(self, cat_id: int) -> Category | None:
        return self._categories.get(cat_id)

    def list_categories(self) -> list[Category]:
        return list(self._categories.values())

    def delete_category(self, cat_id: int) -> bool:
        return self._categories.pop(cat_id, None) is not None

    # Transactions
    def create_transaction(self, data: TransactionCreate) -> Transaction:
        self._txn_counter += 1
        txn = Transaction(
            id=self._txn_counter,
            amount=data.amount,
            date=data.date,
            category_id=data.category_id,
            note=data.note,
        )
        self._transactions[txn.id] = txn
        return txn

    def list_transactions(
        self,
        category_id: int | None = None,
        from_date=None,
        to_date=None,
    ) -> list[Transaction]:
        txns = list(self._transactions.values())
        if category_id:
            txns = [t for t in txns if t.category_id == category_id]
        if from_date:
            txns = [t for t in txns if t.date >= from_date]
        if to_date:
            txns = [t for t in txns if t.date <= to_date]
        return txns

    def get_transaction(self, txn_id: int) -> Transaction | None:
        return self._transactions.get(txn_id)

    def delete_transaction(self, txn_id: int) -> bool:
        return self._transactions.pop(txn_id, None) is not None
```

### app/routers/summary.py

```python
from fastapi import APIRouter, Depends, HTTPException
from decimal import Decimal
from app.models import MonthlySummary, CategoryType
from app.store import FinanceStore
from app.dependencies import get_store

router = APIRouter(prefix="/summary", tags=["summary"])

@router.get("/{year}/{month}", response_model=MonthlySummary)
async def monthly_summary(
    year: int,
    month: int,
    store: FinanceStore = Depends(get_store),
):
    if not (1 <= month <= 12):
        raise HTTPException(400, "Month must be between 1 and 12")

    from datetime import date
    from_date = date(year, month, 1)
    import calendar
    last_day = calendar.monthrange(year, month)[1]
    to_date = date(year, month, last_day)

    txns = store.list_transactions(from_date=from_date, to_date=to_date)

    total_income = Decimal("0")
    total_expense = Decimal("0")
    by_category: dict[str, Decimal] = {}

    for txn in txns:
        cat = store.get_category(txn.category_id)
        if not cat:
            continue
        by_category[cat.name] = by_category.get(cat.name, Decimal("0")) + txn.amount
        if cat.type == CategoryType.income.value:
            total_income += txn.amount
        else:
            total_expense += txn.amount

    return MonthlySummary(
        year=year,
        month=month,
        total_income=total_income,
        total_expense=total_expense,
        net=total_income - total_expense,
        by_category=by_category,
    )
```
