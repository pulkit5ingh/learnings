## Background

### gRPC vs REST
gRPC uses Protocol Buffers (a binary format) instead of JSON. It is faster (smaller payloads, binary encoding), type-safe (schema enforced at the framework level), and supports streaming. The downside: harder to debug (not human-readable), requires tooling to work with, and is overkill for simple CRUD APIs. Use gRPC for high-throughput internal service-to-service communication where performance matters.

### Protocol Buffers
Proto files define your service API — message types and RPC methods. You run a code generator to produce Python stubs. Your server implements the stub interface. Your client uses the stub to call methods as if they were local Python functions. The framework handles serialization, connection management, and multiplexing.

### Streaming
gRPC supports four communication patterns: unary (one request, one response), server streaming (one request, many responses), client streaming (many requests, one response), and bidirectional streaming. REST can only do unary. gRPC streaming is useful for: real-time feeds, file uploads, live data.

## Tasks

### Task 1: Define the Proto File
```protobuf
// proto/user_service.proto
syntax = "proto3";

package user_service;

option python_package = "app.grpc.generated";

// Message types
message User {
  int32 id = 1;
  string email = 2;
  string full_name = 3;
  bool is_active = 4;
  string created_at = 5;
}

message GetUserRequest {
  int32 user_id = 1;
}

message GetUserResponse {
  User user = 1;
}

message ListUsersRequest {
  int32 page = 1;
  int32 page_size = 2;
}

message ListUsersResponse {
  repeated User users = 1;
  int32 total = 2;
}

message CreateUserRequest {
  string email = 1;
  string password = 2;
  string full_name = 3;
}

message UserStream {
  User user = 1;
  string event_type = 2;  // "created", "updated", "deleted"
}

// Service definition
service UserService {
  // Unary: single request, single response
  rpc GetUser (GetUserRequest) returns (GetUserResponse);
  
  // Unary: create a user
  rpc CreateUser (CreateUserRequest) returns (GetUserResponse);
  
  // Server streaming: stream all users matching criteria
  rpc ListUsers (ListUsersRequest) returns (stream UserStream);
}
```

### Task 3: gRPC Server
```python
# app/grpc/server.py
import grpc
import asyncio
from app.grpc.generated import user_service_pb2, user_service_pb2_grpc
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.database import AsyncSessionLocal
from app.models import User

class UserServiceServicer(user_service_pb2_grpc.UserServiceServicer):
    """Implements the UserService gRPC server."""

    async def GetUser(self, request, context):
        async with AsyncSessionLocal() as session:
            user = await session.get(User, request.user_id)
        
        if user is None:
            await context.abort(grpc.StatusCode.NOT_FOUND, f"User {request.user_id} not found")
            return
        
        return user_service_pb2.GetUserResponse(
            user=user_service_pb2.User(
                id=user.id,
                email=user.email,
                full_name=user.full_name,
                is_active=user.is_active,
                created_at=str(user.created_at),
            )
        )

    async def CreateUser(self, request, context):
        from app.auth.password import hash_password
        async with AsyncSessionLocal() as session:
            # Check duplicate
            from sqlalchemy import select
            result = await session.execute(select(User).where(User.email == request.email))
            if result.scalars().first():
                await context.abort(grpc.StatusCode.ALREADY_EXISTS, "Email already registered")
                return
            
            user = User(
                email=request.email,
                hashed_password=hash_password(request.password),
                full_name=request.full_name,
            )
            session.add(user)
            await session.commit()
            await session.refresh(user)
        
        return user_service_pb2.GetUserResponse(
            user=user_service_pb2.User(
                id=user.id,
                email=user.email,
                full_name=user.full_name,
                is_active=user.is_active,
            )
        )

    async def ListUsers(self, request, context):
        """Server streaming: yield users one by one."""
        offset = (request.page - 1) * request.page_size
        
        async with AsyncSessionLocal() as session:
            stmt = select(User).offset(offset).limit(request.page_size)
            result = await session.execute(stmt)
            users = result.scalars().all()
        
        for user in users:
            yield user_service_pb2.UserStream(
                user=user_service_pb2.User(
                    id=user.id,
                    email=user.email,
                    full_name=user.full_name,
                    is_active=user.is_active,
                ),
                event_type="list",
            )
            await asyncio.sleep(0)  # yield control to event loop between items

async def serve():
    server = grpc.aio.server()
    user_service_pb2_grpc.add_UserServiceServicer_to_server(
        UserServiceServicer(), server
    )
    server.add_insecure_port("[::]:50051")
    await server.start()
    print("gRPC server running on port 50051")
    await server.wait_for_termination()

if __name__ == "__main__":
    asyncio.run(serve())
```

### Task 4: gRPC Client
```python
# app/grpc/client.py
import grpc
import grpc.aio
from app.grpc.generated import user_service_pb2, user_service_pb2_grpc

USER_SERVICE_GRPC_URL = "localhost:50051"

class UserServiceGrpcClient:
    def __init__(self, address: str = USER_SERVICE_GRPC_URL):
        self.address = address
        self._channel = None
        self._stub = None

    async def _get_stub(self) -> user_service_pb2_grpc.UserServiceStub:
        if self._channel is None:
            self._channel = grpc.aio.insecure_channel(self.address)
            self._stub = user_service_pb2_grpc.UserServiceStub(self._channel)
        return self._stub

    async def get_user(self, user_id: int) -> dict | None:
        stub = await self._get_stub()
        try:
            response = await stub.GetUser(
                user_service_pb2.GetUserRequest(user_id=user_id)
            )
            u = response.user
            return {"id": u.id, "email": u.email, "full_name": u.full_name}
        except grpc.aio.AioRpcError as e:
            if e.code() == grpc.StatusCode.NOT_FOUND:
                return None
            raise

    async def list_users_stream(self, page: int = 1, page_size: int = 20):
        """Consume a server stream."""
        stub = await self._get_stub()
        async for user_stream in stub.ListUsers(
            user_service_pb2.ListUsersRequest(page=page, page_size=page_size)
        ):
            yield {
                "id": user_stream.user.id,
                "email": user_stream.user.email,
                "event_type": user_stream.event_type,
            }

    async def close(self):
        if self._channel:
            await self._channel.close()
```

### Task 5: FastAPI Route Using gRPC Client
```python
# app/users/router.py
from fastapi import APIRouter, HTTPException
from app.grpc.client import UserServiceGrpcClient

router = APIRouter(prefix="/users", tags=["Users"])

grpc_client = UserServiceGrpcClient()

@router.get("/{user_id}")
async def get_user_via_grpc(user_id: int):
    user = await grpc_client.get_user(user_id)
    if user is None:
        raise HTTPException(status_code=404, detail="User not found")
    return user

@router.get("/stream/all")
async def stream_all_users():
    from fastapi.responses import StreamingResponse
    import json
    
    async def generate():
        async for user in grpc_client.list_users_stream():
            yield json.dumps(user) + "\n"
    
    return StreamingResponse(generate(), media_type="application/x-ndjson")
```
