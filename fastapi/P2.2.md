# P2.2 — Inventory Management System

**Phase:** 2 — Database with SQLAlchemy & PostgreSQL
**Topic:** Async SQLAlchemy, relationships, aggregate queries, Alembic
**Type:** Project

---

## Goal

Build an inventory management API for tracking products, categories, suppliers, and stock levels with low-stock alerts.

## What to Build

- **Products**: CRUD, with category and supplier FKs
- **Categories**: hierarchical (parent/child) categories
- **Suppliers**: CRUD with contact info
- **Stock movements**: record incoming/outgoing stock, auto-update product quantity
- `GET /products/low-stock?threshold=10` — products below threshold
- `GET /reports/inventory-value` — total inventory value per category

## Implementation

### app/models/product.py

```python
from sqlalchemy import String, Numeric, Integer, ForeignKey, CheckConstraint
from sqlalchemy.orm import Mapped, mapped_column, relationship
from app.models.base import Base, TimestampMixin
from decimal import Decimal

class Product(Base, TimestampMixin):
    __tablename__ = "products"
    __table_args__ = (
        CheckConstraint("quantity >= 0", name="non_negative_quantity"),
        CheckConstraint("price >= 0", name="non_negative_price"),
    )

    id: Mapped[int] = mapped_column(primary_key=True)
    sku: Mapped[str] = mapped_column(String(50), unique=True, index=True)
    name: Mapped[str] = mapped_column(String(200))
    description: Mapped[str | None] = mapped_column(String(1000))
    price: Mapped[Decimal] = mapped_column(Numeric(10, 2))
    quantity: Mapped[int] = mapped_column(Integer, default=0)
    category_id: Mapped[int | None] = mapped_column(ForeignKey("categories.id"))
    supplier_id: Mapped[int | None] = mapped_column(ForeignKey("suppliers.id"))

    category: Mapped["Category"] = relationship(lazy="selectin")
    supplier: Mapped["Supplier"] = relationship(lazy="selectin")
    movements: Mapped[list["StockMovement"]] = relationship(back_populates="product")
```

### app/models/stock_movement.py

```python
from sqlalchemy import Integer, ForeignKey, String, Enum as SAEnum
from sqlalchemy.orm import Mapped, mapped_column, relationship
from app.models.base import Base, TimestampMixin
import enum

class MovementType(str, enum.Enum):
    inbound = "inbound"
    outbound = "outbound"
    adjustment = "adjustment"

class StockMovement(Base, TimestampMixin):
    __tablename__ = "stock_movements"

    id: Mapped[int] = mapped_column(primary_key=True)
    product_id: Mapped[int] = mapped_column(ForeignKey("products.id", ondelete="CASCADE"))
    quantity: Mapped[int] = mapped_column(Integer)  # positive = in, negative = out
    movement_type: Mapped[MovementType]
    reference: Mapped[str | None] = mapped_column(String(100))  # PO number, etc.
    note: Mapped[str | None] = mapped_column(String(500))

    product: Mapped["Product"] = relationship(back_populates="movements")
```

### app/repositories/stock_repo.py

```python
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update
from app.models.product import Product
from app.models.stock_movement import StockMovement, MovementType
from fastapi import HTTPException

class StockRepository:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def record_movement(
        self,
        product_id: int,
        quantity_delta: int,
        movement_type: MovementType,
        reference: str | None = None,
        note: str | None = None,
    ) -> StockMovement:
        async with self.session.begin():
            product = await self.session.get(Product, product_id, with_for_update=True)
            if not product:
                raise HTTPException(404, "Product not found")

            new_qty = product.quantity + quantity_delta
            if new_qty < 0:
                raise HTTPException(422, f"Insufficient stock: have {product.quantity}, need {abs(quantity_delta)}")

            product.quantity = new_qty
            movement = StockMovement(
                product_id=product_id,
                quantity=quantity_delta,
                movement_type=movement_type,
                reference=reference,
                note=note,
            )
            self.session.add(movement)

        await self.session.refresh(movement)
        return movement

    async def get_low_stock(self, threshold: int) -> list[Product]:
        result = await self.session.execute(
            select(Product).where(Product.quantity <= threshold).order_by(Product.quantity)
        )
        return result.scalars().all()
```

### app/routers/reports.py

```python
from fastapi import APIRouter, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, func
from app.database import get_session
from app.models.product import Product
from app.models.category import Category

router = APIRouter(prefix="/reports", tags=["reports"])

@router.get("/inventory-value")
async def inventory_value(session: AsyncSession = Depends(get_session)):
    result = await session.execute(
        select(
            Category.name,
            func.sum(Product.price * Product.quantity).label("total_value"),
            func.count(Product.id).label("product_count"),
        )
        .join(Product, Product.category_id == Category.id)
        .group_by(Category.name)
        .order_by(func.sum(Product.price * Product.quantity).desc())
    )
    return [
        {"category": row.name, "total_value": float(row.total_value), "product_count": row.product_count}
        for row in result
    ]
```
