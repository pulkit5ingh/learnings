# P6.1 — E-commerce Product Catalog API

**Phase:** 6 — API Design & Documentation
**Topic:** API versioning, comprehensive OpenAPI docs, RFC 7807 error format, CORS
**Type:** Project

---

## Goal

Build a versioned e-commerce product catalog API with rich OpenAPI documentation, custom error responses following RFC 7807, and proper CORS/security headers.

## What to Build

- `/v1/products` and `/v2/products` (v2 adds `variants` field and cursor pagination)
- Comprehensive OpenAPI docs: tags, examples, deprecated endpoints
- RFC 7807 `application/problem+json` error responses
- CORS configured for specific origins
- Security headers middleware (X-Content-Type-Options, etc.)
- Custom 404/422/500 handlers with problem+json format

## Implementation

### app/errors.py

```python
from fastapi import FastAPI, Request, HTTPException
from fastapi.exceptions import RequestValidationError
from fastapi.responses import JSONResponse
from pydantic import BaseModel

class ProblemDetail(BaseModel):
    type: str
    title: str
    status: int
    detail: str
    instance: str | None = None
    errors: list[dict] | None = None  # for validation errors

PROBLEM_CONTENT_TYPE = "application/problem+json"

def problem_response(
    status: int,
    type_: str,
    title: str,
    detail: str,
    instance: str | None = None,
    extra: dict | None = None,
) -> JSONResponse:
    body = {
        "type": f"https://catalog.example.com/errors/{type_}",
        "title": title,
        "status": status,
        "detail": detail,
    }
    if instance:
        body["instance"] = instance
    if extra:
        body.update(extra)
    return JSONResponse(status_code=status, content=body, media_type=PROBLEM_CONTENT_TYPE)

def register_exception_handlers(app: FastAPI) -> None:
    @app.exception_handler(HTTPException)
    async def http_exception_handler(request: Request, exc: HTTPException):
        type_map = {
            400: "bad-request",
            401: "unauthorized",
            403: "forbidden",
            404: "not-found",
            409: "conflict",
            422: "unprocessable-entity",
            429: "rate-limit-exceeded",
        }
        return problem_response(
            status=exc.status_code,
            type_=type_map.get(exc.status_code, "error"),
            title=exc.detail if isinstance(exc.detail, str) else "Error",
            detail=str(exc.detail),
            instance=str(request.url),
        )

    @app.exception_handler(RequestValidationError)
    async def validation_exception_handler(request: Request, exc: RequestValidationError):
        errors = [
            {"field": ".".join(str(x) for x in e["loc"]), "message": e["msg"]}
            for e in exc.errors()
        ]
        return problem_response(
            status=422,
            type_="validation-error",
            title="Request Validation Error",
            detail="One or more fields failed validation.",
            instance=str(request.url),
            extra={"errors": errors},
        )

    @app.exception_handler(Exception)
    async def generic_exception_handler(request: Request, exc: Exception):
        return problem_response(
            status=500,
            type_="internal-server-error",
            title="Internal Server Error",
            detail="An unexpected error occurred.",
            instance=str(request.url),
        )
```

### app/middleware.py

```python
from starlette.middleware.base import BaseHTTPMiddleware
from starlette.requests import Request

class SecurityHeadersMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        response = await call_next(request)
        response.headers["X-Content-Type-Options"] = "nosniff"
        response.headers["X-Frame-Options"] = "DENY"
        response.headers["X-XSS-Protection"] = "1; mode=block"
        response.headers["Referrer-Policy"] = "strict-origin-when-cross-origin"
        response.headers["Permissions-Policy"] = "geolocation=(), microphone=()"
        return response
```

### app/api/v1/products.py

```python
from fastapi import APIRouter, Query, HTTPException
from app.schemas.v1.product import ProductV1, ProductListV1

router = APIRouter()

PRODUCTS_DB: dict[int, dict] = {
    1: {"id": 1, "name": "Laptop Pro", "price": 1299.99, "category": "Electronics", "in_stock": True},
    2: {"id": 2, "name": "Wireless Mouse", "price": 29.99, "category": "Electronics", "in_stock": True},
}

@router.get(
    "",
    response_model=ProductListV1,
    summary="List products",
    description="Returns a paginated list of products. Use `offset` and `limit` for pagination.",
    response_description="A list of products with total count.",
    responses={
        200: {
            "content": {
                "application/json": {
                    "example": {"items": [{"id": 1, "name": "Laptop", "price": 999.99}], "total": 1}
                }
            }
        }
    },
    tags=["Products v1"],
)
async def list_products_v1(
    offset: int = Query(default=0, ge=0, description="Number of items to skip"),
    limit: int = Query(default=20, ge=1, le=100, description="Max items to return"),
    category: str | None = Query(default=None, description="Filter by category"),
):
    items = list(PRODUCTS_DB.values())
    if category:
        items = [p for p in items if p["category"].lower() == category.lower()]
    return {"items": items[offset:offset + limit], "total": len(items)}

@router.get(
    "/{product_id}",
    response_model=ProductV1,
    tags=["Products v1"],
    responses={404: {"description": "Product not found", "content": {"application/problem+json": {}}}},
)
async def get_product_v1(product_id: int):
    product = PRODUCTS_DB.get(product_id)
    if not product:
        raise HTTPException(404, f"Product {product_id} not found")
    return product
```

### app/schemas/v2/product.py

```python
from pydantic import BaseModel
from decimal import Decimal

class VariantV2(BaseModel):
    sku: str
    size: str | None = None
    color: str | None = None
    price_modifier: Decimal = Decimal("0")
    stock: int = 0

class ProductV2(BaseModel):
    id: int
    name: str
    price: Decimal
    category: str
    in_stock: bool
    variants: list[VariantV2] = []  # new in v2
    slug: str                        # new in v2

class CursorPageV2(BaseModel):         # v2 uses cursor pagination
    items: list[ProductV2]
    next_cursor: int | None
    has_more: bool
```

### app/main.py

```python
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from app.api.v1.router import router as v1_router
from app.api.v2.router import router as v2_router
from app.errors import register_exception_handlers
from app.middleware import SecurityHeadersMiddleware

app = FastAPI(
    title="Product Catalog API",
    version="2.0.0",
    description="E-commerce product catalog with versioned endpoints.",
    contact={"name": "API Team", "email": "api@example.com"},
    license_info={"name": "MIT"},
    openapi_tags=[
        {"name": "Products v1", "description": "Stable v1 endpoints (offset pagination)"},
        {"name": "Products v2", "description": "Current v2 endpoints (cursor pagination, variants)"},
    ],
)

app.add_middleware(SecurityHeadersMiddleware)
app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://shop.example.com", "http://localhost:3000"],
    allow_credentials=True,
    allow_methods=["GET", "POST", "PUT", "DELETE", "PATCH"],
    allow_headers=["Authorization", "Content-Type"],
)

register_exception_handlers(app)
app.include_router(v1_router, prefix="/v1")
app.include_router(v2_router, prefix="/v2")
```
