## Goal
Centralize all configuration in a type-safe `Settings` class that reads from environment variables, with validation and defaults.

## Background

### Why Centralized Settings
Scattered `os.getenv("SOME_KEY")` calls throughout the codebase are hard to audit, easy to misspell, and have no validation. A `Settings` class (using `pydantic-settings`) solves all three: all config is in one place, keys are typed (automatic coercion from string to int/bool/etc.), and missing required config fails at startup with a clear error.

### pydantic-settings
`pydantic-settings` is an extension of Pydantic that loads values from environment variables (and optionally `.env` files). It uses the field name as the env var name by default. You can customize the prefix, make fields required, and add validators.

### Secrets vs Configuration
Configuration (database URL, port, log level) changes between environments. Secrets (API keys, JWT secret) should never be hardcoded or committed to git. In production, inject secrets via the platform's secrets manager (Railway secrets, Fly.io secrets, AWS Secrets Manager). In development, use a `.env` file that is gitignored.

```python
# app/config.py
from pydantic_settings import BaseSettings, SettingsConfigDict
from pydantic import Field, field_validator, PostgresDsn, RedisDsn
from typing import Annotated
from functools import lru_cache
import secrets

class Settings(BaseSettings):
    model_config = SettingsConfigDict(
        env_file=".env",
        env_file_encoding="utf-8",
        case_sensitive=False,
        extra="ignore",  # ignore unknown env vars
    )

    # ── Application ──────────────────────────────
    app_name: str = "My FastAPI App"
    app_version: str = "1.0.0"
    environment: str = Field(default="development", alias="ENV")
    debug: bool = False
    port: int = 8000

    # ── Database ─────────────────────────────────
    database_url: str = Field(
        ...,  # required — no default
        description="PostgreSQL connection string (asyncpg format)",
    )
    db_pool_size: int = 10
    db_max_overflow: int = 20
    db_echo: bool = False

    # ── Redis ─────────────────────────────────────
    redis_url: str = "redis://localhost:6379/0"

    # ── Security ──────────────────────────────────
    secret_key: str = Field(
        default_factory=lambda: secrets.token_hex(32),
        description="JWT signing key — set in production!",
    )
    algorithm: str = "HS256"
    access_token_expire_minutes: int = 30
    refresh_token_expire_days: int = 30

    # ── External APIs ──────────────────────────────
    sendgrid_api_key: str | None = None
    twilio_account_sid: str | None = None
    twilio_auth_token: str | None = None
    sentry_dsn: str | None = None

    # ── CORS ──────────────────────────────────────
    allowed_origins: list[str] = ["http://localhost:3000", "http://localhost:5173"]

    # ── Features ──────────────────────────────────
    enable_registration: bool = True
    enable_email_verification: bool = False
    max_file_upload_mb: int = 10

    @field_validator("environment")
    @classmethod
    def validate_environment(cls, v: str) -> str:
        allowed = {"development", "testing", "staging", "production"}
        if v not in allowed:
            raise ValueError(f"environment must be one of {allowed}")
        return v

    @field_validator("secret_key")
    @classmethod
    def warn_insecure_secret(cls, v: str) -> str:
        if v == "dev-secret-key-change-in-production":
            import warnings
            warnings.warn("Using insecure default SECRET_KEY! Set a real key in production.")
        return v

    @property
    def is_production(self) -> bool:
        return self.environment == "production"

    @property
    def is_development(self) -> bool:
        return self.environment == "development"

    @property
    def is_testing(self) -> bool:
        return self.environment == "testing"

    @property
    def db_echo_enabled(self) -> bool:
        return self.db_echo or self.is_development


@lru_cache  # create once, reuse
def get_settings() -> Settings:
    return Settings()

# Shorthand for dependency injection
settings = get_settings()
```

### Task 2: Environment-Specific Config
```python
# app/config.py (continued)

# Computed settings based on environment
class DevelopmentSettings(Settings):
    debug: bool = True
    db_echo: bool = True
    # Override allowed origins to be more permissive in dev
    allowed_origins: list[str] = ["*"]

class ProductionSettings(Settings):
    debug: bool = False
    enable_email_verification: bool = True
    
    @field_validator("secret_key")
    @classmethod
    def require_strong_secret(cls, v: str) -> str:
        if len(v) < 32:
            raise ValueError("Production SECRET_KEY must be at least 32 characters")
        return v

class TestingSettings(Settings):
    database_url: str = "postgresql+asyncpg://testuser:testpass@localhost:5432/testdb"
    redis_url: str = "redis://localhost:6379/1"  # use DB 1 for tests
    secret_key: str = "test-secret-key-only-for-testing"
    enable_email_verification: bool = False

def get_settings_for_env() -> Settings:
    env = os.getenv("ENV", "development")
    settings_map = {
        "development": DevelopmentSettings,
        "testing": TestingSettings,
        "production": ProductionSettings,
    }
    cls = settings_map.get(env, Settings)
    return cls()
```

### Task 5: Validating at Startup
```python
# app/main.py — fail fast on missing/invalid config
from contextlib import asynccontextmanager
from app.config import get_settings

@asynccontextmanager
async def lifespan(app: FastAPI):
    # This runs at startup — any config errors appear immediately
    settings = get_settings()
    
    if settings.is_production:
        # Additional production checks
        if not settings.sendgrid_api_key:
            raise ValueError("SENDGRID_API_KEY required in production")
        if not settings.sentry_dsn:
            import warnings
            warnings.warn("SENTRY_DSN not set — errors will not be tracked")
    
    print(f"Starting {settings.app_name} v{settings.app_version}")
    print(f"Environment: {settings.environment}")
    print(f"Debug: {settings.debug}")
    
    yield
    
    print("Shutting down...")

app = FastAPI(lifespan=lifespan)
```
