## Goal
Implement reliable event-driven communication with domain events, transactional outbox pattern, and idempotent consumers.

## Background

### Domain Events
A domain event records something that happened in your domain: `OrderPlaced`, `PaymentProcessed`, `UserRegistered`. Events are facts — immutable records of the past. They are published AFTER the state change is committed to the database. Consumers react to events: send a notification, update a read model, trigger a workflow.

### The Dual-Write Problem
Publishing an event after a DB write creates a dual-write problem: the DB commit might succeed but the event publish might fail (network error, broker down). The consumer never gets the event. The solution is the Transactional Outbox Pattern: write the event to an outbox table IN THE SAME DATABASE TRANSACTION as the business data. A separate process reads the outbox and publishes to the broker. This guarantees at-least-once delivery.

### Idempotency
At-least-once delivery means messages can be delivered more than once (e.g., consumer acks but message gets redelivered due to network issue). Your consumers must be idempotent: processing the same event twice must produce the same result as processing it once. Use an idempotency key (usually the event ID) and check if it's been processed before acting.

# app/events/domain_events.py
from pydantic import BaseModel, Field
from datetime import datetime, timezone
from typing import Any
import uuid

class DomainEvent(BaseModel):
    event_id: str = Field(default_factory=lambda: str(uuid.uuid4()))
    event_type: str
    occurred_at: datetime = Field(default_factory=lambda: datetime.now(timezone.utc))
    aggregate_id: str
    aggregate_type: str
    version: int = 1
    payload: dict[str, Any]

class OrderPlacedEvent(DomainEvent):
    event_type: str = "order.placed"
    aggregate_type: str = "Order"
    
    @classmethod
    def create(cls, order_id: int, user_id: int, total: float, items: list[dict]) -> "OrderPlacedEvent":
        return cls(
            aggregate_id=str(order_id),
            payload={
                "order_id": order_id,
                "user_id": user_id,
                "total": total,
                "items": items,
            },
        )

class UserRegisteredEvent(DomainEvent):
    event_type: str = "user.registered"
    aggregate_type: str = "User"
    
    @classmethod
    def create(cls, user_id: int, email: str) -> "UserRegisteredEvent":
        return cls(
            aggregate_id=str(user_id),
            payload={"user_id": user_id, "email": email},
        )

class PaymentProcessedEvent(DomainEvent):
    event_type: str = "payment.processed"
    aggregate_type: str = "Payment"
    
    @classmethod
    def create(cls, order_id: int, amount: float, status: str) -> "PaymentProcessedEvent":
        return cls(
            aggregate_id=str(order_id),

        )

# app/models.py (add outbox table)
from sqlalchemy.orm import Mapped, mapped_column
from sqlalchemy import String, Boolean, JSON, Integer, DateTime
from app.database import Base
from sqlalchemy.sql import func

class OutboxEvent(Base):
    __tablename__ = "outbox_events"
    
    id: Mapped[int] = mapped_column(Integer, primary_key=True)
    event_id: Mapped[str] = mapped_column(String(50), unique=True, nullable=False)
    event_type: Mapped[str] = mapped_column(String(100), nullable=False)
    aggregate_id: Mapped[str] = mapped_column(String(50), nullable=False)
    payload: Mapped[dict] = mapped_column(JSON, nullable=False)
    published: Mapped[bool] = mapped_column(Boolean, default=False, nullable=False)
    created_at: Mapped[datetime] = mapped_column(server_default=func.now())
    published_at: Mapped[datetime | None] = mapped_column(DateTime, nullable=True)
```

# app/events/outbox.py
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update
from app.models import OutboxEvent
from app.events.domain_events import DomainEvent
from datetime import datetime, timezone

async def save_event_to_outbox(session: AsyncSession, event: DomainEvent) -> None:
    """Save event to outbox IN THE SAME TRANSACTION as business logic."""
    outbox_entry = OutboxEvent(
        event_id=event.event_id,
        event_type=event.event_type,
        aggregate_id=event.aggregate_id,
        payload=event.model_dump(),
    )
    session.add(outbox_entry)
    # No commit here — it's part of the outer transaction

async def get_unpublished_events(session: AsyncSession, limit: int = 100) -> list[OutboxEvent]:
    stmt = (
        select(OutboxEvent)
        .where(OutboxEvent.published == False)
        .order_by(OutboxEvent.created_at)
        .limit(limit)
    )
    result = await session.execute(stmt)
    return list(result.scalars().all())

async def mark_event_published(session: AsyncSession, event_id: str) -> None:
    await session.execute(
        update(OutboxEvent)
        .where(OutboxEvent.event_id == event_id)
        .values(published=True, published_at=datetime.now(timezone.utc))
    )
```

### Task 3: Outbox Publisher (Background Process)
```python
# app/events/publisher.py
import asyncio
import logging
from app.database import AsyncSessionLocal
from app.events.outbox import get_unpublished_events, mark_event_published
from app.messaging.events import publish_event

logger = logging.getLogger(__name__)

async def publish_outbox_events() -> None:
    """Polling publisher: reads unpublished events and publishes them."""
    async with AsyncSessionLocal() as session:
        events = await get_unpublished_events(session, limit=50)
        
        for event in events:
            try:
                await publish_event(
                    routing_key=event.event_type,
                    payload=event.payload,
                )
                await mark_event_published(session, event.event_id)
                logger.info(f"Published event {event.event_id}: {event.event_type}")
            except Exception as e:
                logger.error(f"Failed to publish event {event.event_id}: {e}")
                # Don't mark as published — will retry next poll
        
        await session.commit()

async def run_outbox_publisher():
    """Run the publisher loop every 5 seconds."""
    while True:
        try:
            await publish_outbox_events()
        except Exception as e:
            logger.error(f"Outbox publisher error: {e}")
        await asyncio.sleep(5)
```

### Task 4: Using Outbox in Service Layer
```python
# app/orders/service.py
from sqlalchemy.ext.asyncio import AsyncSession
from app.models import Order, OrderItem
from app.events.domain_events import OrderPlacedEvent
from app.events.outbox import save_event_to_outbox

class OrderService:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def place_order(self, user_id: int, items: list[dict]) -> Order:
        # 1. Create the order (business logic)
        total = sum(item["price"] * item["qty"] for item in items)
        order = Order(user_id=user_id, total=total, status="pending")
        self.session.add(order)
        await self.session.flush()  # get order.id
        
        for item in items:
            order_item = OrderItem(order_id=order.id, **item)
            self.session.add(order_item)
        
        # 2. Save domain event to outbox IN SAME TRANSACTION
        event = OrderPlacedEvent.create(
            order_id=order.id,
            user_id=user_id,
            total=total,
            items=items,
        )
        await save_event_to_outbox(self.session, event)
        
        # 3. Commit BOTH order AND event atomically
        # (get_db dependency commits after the route returns)
        
        return order
        # If commit fails: no order, no event (consistent)
        # If publish fails later: event is in outbox, will retry
```

### Task 5: Idempotent Consumer
```python
# app/events/consumer.py
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models import ProcessedEvent

class IdempotentConsumer:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def is_processed(self, event_id: str) -> bool:
        result = await self.session.execute(
            select(ProcessedEvent).where(ProcessedEvent.event_id == event_id)
        )
        return result.scalars().first() is not None

    async def mark_processed(self, event_id: str, event_type: str) -> None:
        processed = ProcessedEvent(
            event_id=event_id,
            event_type=event_type,
            processed_at=datetime.now(timezone.utc),
        )
        self.session.add(processed)
        await self.session.flush()


# Idempotent order event handler
async def handle_order_placed(message_data: dict) -> None:
    event_id = message_data["event_id"]
    
    async with AsyncSessionLocal() as session:
        consumer = IdempotentConsumer(session)
        
        # Check if already processed (idempotency check)
        if await consumer.is_processed(event_id):
            logger.info(f"Event {event_id} already processed, skipping")
            return
        
        # Process the event
        payload = message_data["payload"]
        order_id = payload["order_id"]
        user_id = payload["user_id"]
        
        # Send notification, update inventory, etc.
        await send_order_confirmation_email(user_id, order_id)
        
        # Mark as processed AFTER successful handling
        await consumer.mark_processed(event_id, message_data["event_type"])
        await session.commit()
```

### Task 6: Event Version Migration
```python
# Handling different event versions gracefully
def normalize_order_placed_event(raw_payload: dict) -> dict:
    """Normalize event payload across versions."""
    version = raw_payload.get("version", 1)
    
    if version == 1:
        # Old format: flat structure
        return {
            "order_id": raw_payload.get("order_id"),
            "user_id": raw_payload.get("user_id"),
            "total": raw_payload.get("total", 0),
            "items": raw_payload.get("items", []),
        }
    elif version == 2:
        # New format: nested user object
        return {
            "order_id": raw_payload.get("order_id"),
            "user_id": raw_payload["user"]["id"],
            "total": raw_payload.get("total", 0),
            "items": raw_payload.get("line_items", []),  # renamed in v2
        }
    else:
        raise ValueError(f"Unknown event version: {version}")
```
