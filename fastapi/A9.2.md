## Goal
Build a GraphQL API using Strawberry that avoids N+1 queries with DataLoaders and enforces permissions.

## Background

### GraphQL vs REST
REST exposes fixed endpoints that return fixed shapes. GraphQL exposes a single endpoint where clients specify exactly what data they need. This eliminates over-fetching (getting more data than needed) and under-fetching (needing multiple requests). The downside: caching is harder, N+1 queries are easy to introduce, and the learning curve is higher.

### Strawberry
Strawberry is a Python GraphQL library that uses Python type hints and decorators to define the schema — similar to FastAPI's approach. It integrates natively with FastAPI via `strawberry.fastapi.GraphQLRouter`. It supports Query (read), Mutation (write), and Subscription (real-time via WebSockets).

### DataLoader
The DataLoader pattern solves the N+1 problem in GraphQL. When a field resolver fetches a related object, it doesn't immediately fire a DB query. Instead, it registers the ID in a DataLoader. The DataLoader batches all IDs collected during the request and fires ONE query for all of them. This is the GraphQL equivalent of `selectinload`.

### Task 2: Query and Mutation
```python
# app/graphql/resolvers.py
import strawberry
from strawberry.types import Info
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from typing import Optional
from app.models import User, Post
from app.graphql.types import UserType, PostType, CreatePostInput

def user_to_type(u: User) -> UserType:
    return UserType(
        id=u.id,
        email=u.email,
        full_name=u.full_name,
        is_active=u.is_active,
        created_at=u.created_at,
    )

def post_to_type(p: Post) -> PostType:
    return PostType(
        id=p.id,
        title=p.title,
        body=p.body,
        published=p.published,
        created_at=p.created_at,
        author_id=p.user_id,
    )

@strawberry.type
class Query:
    @strawberry.field
    async def user(self, info: Info, id: int) -> Optional[UserType]:
        db: AsyncSession = info.context["db"]
        user = await db.get(User, id)
        return user_to_type(user) if user else None

    @strawberry.field
    async def users(self, info: Info, limit: int = 20, offset: int = 0) -> list[UserType]:
        db: AsyncSession = info.context["db"]
        result = await db.execute(select(User).limit(limit).offset(offset))
        return [user_to_type(u) for u in result.scalars().all()]

    @strawberry.field
    async def posts(
        self,
        info: Info,
        published_only: bool = True,
        limit: int = 20,
        offset: int = 0,
    ) -> list[PostType]:
        db: AsyncSession = info.context["db"]
        stmt = select(Post).limit(limit).offset(offset)
        if published_only:
            stmt = stmt.where(Post.published == True)
        result = await db.execute(stmt)
        return [post_to_type(p) for p in result.scalars().all()]

@strawberry.type
class Mutation:
    @strawberry.mutation
    async def create_post(self, info: Info, input: CreatePostInput) -> PostType:
        db: AsyncSession = info.context["db"]
        
        # Require authentication
        current_user = info.context.get("current_user")
        if not current_user:
            raise Exception("Authentication required")
        
        post = Post(
            title=input.title,
            body=input.body,
            published=input.published,
            user_id=current_user.id,
        )
        db.add(post)
        await db.flush()
        await db.refresh(post)
        
        return post_to_type(post)

    @strawberry.mutation
    async def delete_post(self, info: Info, id: int) -> bool:
        db: AsyncSession = info.context["db"]
        current_user = info.context.get("current_user")
        
        post = await db.get(Post, id)
        if not post:
            raise Exception(f"Post {id} not found")
        
        if not current_user or (post.user_id != current_user.id and current_user.role != "admin"):
            raise Exception("Forbidden")
        
        await db.delete(post)
        return True
```

### Task 3: DataLoader for N+1 Prevention
```python
# app/graphql/dataloaders.py
from strawberry.dataloader import DataLoader
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from app.models import User

async def load_users(keys: list[int], db: AsyncSession) -> list[User | None]:
    """Batch-load users by ID. Called once with all IDs collected during request."""
    result = await db.execute(select(User).where(User.id.in_(keys)))
    users_by_id = {u.id: u for u in result.scalars().all()}
    # Return in the same order as keys (None for missing)
    return [users_by_id.get(key) for key in keys]

def create_dataloaders(db: AsyncSession) -> dict:
    return {
        "user_loader": DataLoader(load_fn=lambda keys: load_users(keys, db)),
    }

# Using DataLoader in a resolver:
@strawberry.type
class PostTypeWithLoader:
    id: int
    title: str
    author_id: int
    
    @strawberry.field
    async def author(self, info: Info) -> Optional[UserType]:
        # Instead of fetching from DB directly (N+1),
        # add to DataLoader batch:
        loader = info.context["dataloaders"]["user_loader"]
        user = await loader.load(self.author_id)
        return user_to_type(user) if user else None
```

### Task 4: Context and Authentication
```python
# app/graphql/schema.py
import strawberry
from strawberry.fastapi import GraphQLRouter
from fastapi import Depends, Request
from sqlalchemy.ext.asyncio import AsyncSession
from app.deps import get_db
from app.auth.tokens import decode_access_token
from app.graphql.dataloaders import create_dataloaders

async def get_graphql_context(
    request: Request,
    db: AsyncSession = Depends(get_db),
) -> dict:
    context = {"db": db, "request": request}
    
    # Try to authenticate
    auth_header = request.headers.get("Authorization", "")
    if auth_header.startswith("Bearer "):
        token = auth_header.removeprefix("Bearer ").strip()
        try:
            payload = decode_access_token(token)
            user_id = int(payload["sub"])
            user = await db.get(User, user_id)
            context["current_user"] = user
        except (ValueError, KeyError):
            context["current_user"] = None
    
    # Add DataLoaders
    context["dataloaders"] = create_dataloaders(db)
    
    return context

schema = strawberry.Schema(query=Query, mutation=Mutation)

graphql_router = GraphQLRouter(
    schema,
    context_getter=get_graphql_context,
    graphiql=True,  # enable GraphiQL UI at /graphql
)
```
