## Goal

## Background

### SQLAlchemy 2.0 ORM Changes
SQLAlchemy 2.0 introduced a new typing-based ORM style. Instead of `Column()` you use `mapped_column()`, and instead of `relationship()` returning untyped attributes, you use `Mapped[T]` type annotations. This provides full type-checker support (mypy/pyright). The old style still works but is considered legacy.

### Sessions
The ORM `Session` is the unit of work. It tracks Python objects and their state (new, modified, deleted). When you `session.add(obj)`, SQLAlchemy tracks it as "pending." When you `session.flush()` or `session.commit()`, it generates the SQL. The `Session` also acts as a first-level cache: querying the same row twice returns the same Python object (identity map).

### Relationships and Loading
SQLAlchemy relationships define how objects relate. The crucial decision is loading strategy: lazy (default — issues a new query when you access the relationship), eager/joined (loads in the same query via JOIN), or selectin (issues a second SELECT IN query). Lazy loading is convenient but causes N+1 problems. Always choose the right loading strategy for your use case.

### Task 1: Declarative Models
```python
# models.py
from sqlalchemy import String, Integer, Float, Boolean, ForeignKey, Text, Table, Column
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, relationship
from sqlalchemy.sql import func
from datetime import datetime

class Base(DeclarativeBase):
    pass

# Many-to-many association table (not a model)
post_tags = Table(
    "post_tags",
    Base.metadata,
    Column("post_id", Integer, ForeignKey("posts.id", ondelete="CASCADE"), primary_key=True),
    Column("tag_id", Integer, ForeignKey("tags.id", ondelete="CASCADE"), primary_key=True),
)

class User(Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(Integer, primary_key=True)
    name: Mapped[str] = mapped_column(String(100), nullable=False)
    email: Mapped[str] = mapped_column(String(255), unique=True, nullable=False)
    is_active: Mapped[bool] = mapped_column(Boolean, default=True)
    created_at: Mapped[datetime] = mapped_column(server_default=func.now())

    # One-to-many: user has many posts
    posts: Mapped[list["Post"]] = relationship("Post", back_populates="author", cascade="all, delete-orphan")

    def __repr__(self) -> str:
        return f"<User id={self.id} email={self.email!r}>"


class Post(Base):
    __tablename__ = "posts"

    id: Mapped[int] = mapped_column(Integer, primary_key=True)
    title: Mapped[str] = mapped_column(String(200), nullable=False)
    body: Mapped[str] = mapped_column(Text, nullable=False)
    published: Mapped[bool] = mapped_column(Boolean, default=False)
    user_id: Mapped[int] = mapped_column(ForeignKey("users.id"), nullable=False)
    created_at: Mapped[datetime] = mapped_column(server_default=func.now())

    # Many-to-one: post belongs to user
    author: Mapped["User"] = relationship("User", back_populates="posts")
    
    # Many-to-many: post has many tags
    tags: Mapped[list["Tag"]] = relationship("Tag", secondary=post_tags, back_populates="posts")


class Tag(Base):
    __tablename__ = "tags"

    id: Mapped[int] = mapped_column(Integer, primary_key=True)
    name: Mapped[str] = mapped_column(String(50), unique=True, nullable=False)

    posts: Mapped[list["Post"]] = relationship("Post", secondary=post_tags, back_populates="tags")
```

### Task 2: Creating and Querying
```python
# orm_crud.py
from sqlalchemy import create_engine, select
from sqlalchemy.orm import Session
from models import Base, User, Post, Tag

engine = create_engine("postgresql://user:password@localhost/testdb", echo=True)
Base.metadata.create_all(engine)

def create_data():
    with Session(engine) as session:
        # Create user with posts
        alice = User(name="Alice", email="alice@example.com")
        
        tag_python = Tag(name="python")
        tag_fastapi = Tag(name="fastapi")
        session.add_all([tag_python, tag_fastapi])
        session.flush()  # get IDs without committing

        post1 = Post(
            title="Getting Started with FastAPI",
            body="FastAPI is great...",
            author=alice,  # SQLAlchemy sets user_id automatically
            tags=[tag_python, tag_fastapi],
        )
        post2 = Post(
            title="Python async/await",
            body="Let's talk about coroutines...",
            author=alice,
            tags=[tag_python],
        )
        session.add(alice)
        session.commit()
        print(f"Alice created with {len(alice.posts)} posts")


def query_examples():
    with Session(engine) as session:
        # Get by primary key
        user = session.get(User, 1)
        print(user)

        # Select with WHERE
        stmt = select(User).where(User.is_active == True)
        users = session.scalars(stmt).all()

        # Select with ORDER BY and LIMIT
        stmt = select(Post).order_by(Post.created_at.desc()).limit(10)
        posts = session.scalars(stmt).all()

        # Filter by relationship attribute
        stmt = select(Post).where(Post.author.has(User.email == "alice@example.com"))
        posts = session.scalars(stmt).all()
```

### Task 3: Lazy vs Eager Loading
```python
# loading_strategies.py
from sqlalchemy import select
from sqlalchemy.orm import Session, joinedload, selectinload, lazyload

def lazy_loading_n_plus_1_problem(session: Session):
    """BAD: triggers N+1 queries."""
    users = session.scalars(select(User)).all()
    for user in users:
        # Each access to user.posts issues a new SELECT query!
        print(f"{user.name} has {len(user.posts)} posts")

def eager_loading_joinedload(session: Session):
    """GOOD for single object: loads posts in the same query via JOIN."""
    stmt = (
        select(User)
        .options(joinedload(User.posts))
    )
    users = session.scalars(stmt).unique().all()
    for user in users:
        print(f"{user.name} has {len(user.posts)} posts")  # no extra queries

def eager_loading_selectinload(session: Session):
    """GOOD for collections: issues one extra SELECT IN query (avoids JOINs)."""
    stmt = (
        select(User)
        .options(
            selectinload(User.posts).selectinload(Post.tags)
        )
    )
    users = session.scalars(stmt).all()
    for user in users:
        for post in user.posts:
            tag_names = [t.name for t in post.tags]
            print(f"  Post: {post.title}, Tags: {tag_names}")
```

### Task 4: Updating and Deleting
```python
def update_examples(session: Session):
    # Get and update
    user = session.get(User, 1)
    if user:
        user.name = "Alice Updated"  # just change the attribute
        session.commit()  # SQLAlchemy detects the change and issues UPDATE

    # Bulk update (Core-style, no loading)
    from sqlalchemy import update
    session.execute(
        update(User).where(User.is_active == False).values(name="[Deactivated]")
    )
    session.commit()

def delete_examples(session: Session):
    # Delete one
    user = session.get(User, 1)
    if user:
        session.delete(user)  # cascade="all, delete-orphan" deletes posts too
        session.commit()

    # Bulk delete (Core-style)
    from sqlalchemy import delete
    session.execute(delete(Post).where(Post.published == False))
    session.commit()
```

### Task 5: Repository Pattern
```python
# repositories.py
from typing import Generic, TypeVar
from sqlalchemy import select
from sqlalchemy.orm import Session
from models import Base, User, Post

T = TypeVar("T", bound=Base)

class BaseRepository(Generic[T]):
    def __init__(self, session: Session, model: type[T]):
        self.session = session
        self.model = model

    def get(self, id: int) -> T | None:
        return self.session.get(self.model, id)

    def get_all(self, limit: int = 100, offset: int = 0) -> list[T]:
        stmt = select(self.model).limit(limit).offset(offset)
        return list(self.session.scalars(stmt).all())

    def create(self, **kwargs) -> T:
        obj = self.model(**kwargs)
        self.session.add(obj)
        self.session.flush()
        return obj

    def delete(self, id: int) -> bool:
        obj = self.get(id)
        if obj is None:
            return False
        self.session.delete(obj)
        return True


class UserRepository(BaseRepository[User]):
    def get_by_email(self, email: str) -> User | None:
        stmt = select(User).where(User.email == email)
        return self.session.scalars(stmt).first()

    def get_with_posts(self, user_id: int) -> User | None:
        from sqlalchemy.orm import selectinload
        stmt = (
            select(User)
            .where(User.id == user_id)
            .options(selectinload(User.posts))
        )
        return self.session.scalars(stmt).first()


class PostRepository(BaseRepository[Post]):
    def get_published(self) -> list[Post]:
        stmt = select(Post).where(Post.published == True)
        return list(self.session.scalars(stmt).all())

    def get_by_tag(self, tag_name: str) -> list[Post]:
        stmt = (
            select(Post)
            .join(Post.tags)
            .where(Tag.name == tag_name)
        )
        return list(self.session.scalars(stmt).all())
```
