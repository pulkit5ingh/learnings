# P5.2 — High-Performance Products API

**Phase:** 5 — Performance & Caching
**Topic:** Redis cache-aside, rate limiting, cursor pagination, load testing
**Type:** Project

---

## Goal

Build a high-performance products API with Redis caching, cursor-based pagination, and rate limiting that can handle thousands of requests per second.

## What to Build

- `GET /products` — cursor-based paginated product list with Redis cache
- `GET /products/{id}` — cached product detail (cache-aside, TTL 5 min)
- `POST/PUT/DELETE /products` — write-through cache invalidation
- Rate limiting: 100 req/min per IP for reads, 20 req/min for writes
- Load test with Locust showing cache hit vs miss latency difference
- Cache warming endpoint: `POST /admin/cache/warm`

## Implementation

### app/cache/product_cache.py

```python
import json
from typing import Any
from app.redis_client import redis_client

PRODUCT_TTL = 300  # 5 minutes
LIST_TTL = 60      # 1 minute

class ProductCache:
    @staticmethod
    def _product_key(product_id: int) -> str:
        return f"product:{product_id}"

    @staticmethod
    def _list_key(cursor: int | None, limit: int, category: str | None) -> str:
        return f"products:list:{cursor}:{limit}:{category or 'all'}"

    async def get_product(self, product_id: int) -> dict | None:
        raw = await redis_client.get(self._product_key(product_id))
        return json.loads(raw) if raw else None

    async def set_product(self, product_id: int, data: dict) -> None:
        await redis_client.setex(
            self._product_key(product_id), PRODUCT_TTL, json.dumps(data)
        )

    async def invalidate_product(self, product_id: int) -> None:
        await redis_client.delete(self._product_key(product_id))

    async def get_list(self, cursor: int | None, limit: int, category: str | None) -> list | None:
        raw = await redis_client.get(self._list_key(cursor, limit, category))
        return json.loads(raw) if raw else None

    async def set_list(self, cursor: int | None, limit: int, category: str | None, data: list) -> None:
        await redis_client.setex(
            self._list_key(cursor, limit, category), LIST_TTL, json.dumps(data)
        )

    async def invalidate_list_pattern(self) -> None:
        # Scan and delete all list cache keys
        async for key in redis_client.scan_iter("products:list:*"):
            await redis_client.delete(key)

product_cache = ProductCache()
```

### app/routers/products.py

```python
from fastapi import APIRouter, Depends, HTTPException, Query
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from slowapi import Limiter
from slowapi.util import get_remote_address
from app.database import get_session
from app.models.product import Product
from app.schemas.product import ProductCreate, ProductResponse, CursorPage
from app.cache.product_cache import product_cache

limiter = Limiter(key_func=get_remote_address)
router = APIRouter(prefix="/products", tags=["products"])

@router.get("", response_model=CursorPage)
@limiter.limit("100/minute")
async def list_products(
    request,
    cursor: int | None = Query(default=None),
    limit: int = Query(default=20, le=100),
    category: str | None = None,
    session: AsyncSession = Depends(get_session),
):
    cached = await product_cache.get_list(cursor, limit, category)
    if cached:
        return {"items": cached, "next_cursor": cached[-1]["id"] if cached else None, "cache": "HIT"}

    q = select(Product).order_by(Product.id).limit(limit + 1)
    if cursor:
        q = q.where(Product.id > cursor)
    if category:
        q = q.where(Product.category == category)

    result = await session.execute(q)
    items = result.scalars().all()
    has_more = len(items) > limit
    items = items[:limit]

    serialized = [ProductResponse.model_validate(p).model_dump() for p in items]
    await product_cache.set_list(cursor, limit, category, serialized)

    return {
        "items": serialized,
        "next_cursor": items[-1].id if has_more else None,
        "cache": "MISS",
    }

@router.get("/{product_id}", response_model=ProductResponse)
@limiter.limit("100/minute")
async def get_product(request, product_id: int, session: AsyncSession = Depends(get_session)):
    cached = await product_cache.get_product(product_id)
    if cached:
        return cached

    product = await session.get(Product, product_id)
    if not product:
        raise HTTPException(404)

    data = ProductResponse.model_validate(product).model_dump()
    await product_cache.set_product(product_id, data)
    return data

@router.post("", response_model=ProductResponse, status_code=201)
@limiter.limit("20/minute")
async def create_product(request, body: ProductCreate, session: AsyncSession = Depends(get_session)):
    product = Product(**body.model_dump())
    session.add(product)
    await session.commit()
    await session.refresh(product)
    await product_cache.invalidate_list_pattern()
    return product

@router.delete("/{product_id}", status_code=204)
@limiter.limit("20/minute")
async def delete_product(request, product_id: int, session: AsyncSession = Depends(get_session)):
    product = await session.get(Product, product_id)
    if not product:
        raise HTTPException(404)
    await session.delete(product)
    await session.commit()
    await product_cache.invalidate_product(product_id)
    await product_cache.invalidate_list_pattern()
```

### locustfile.py

```python
from locust import HttpUser, task, between
import random

class ProductUser(HttpUser):
    wait_time = between(0.1, 0.5)
    host = "http://localhost:8000"

    @task(10)
    def list_products(self):
        self.client.get("/products?limit=20")

    @task(5)
    def get_product(self):
        product_id = random.randint(1, 100)
        self.client.get(f"/products/{product_id}")

    @task(1)
    def create_product(self):
        self.client.post("/products", json={
            "name": f"Product {random.randint(1000, 9999)}",
            "price": round(random.uniform(1, 500), 2),
            "category": random.choice(["electronics", "clothing", "food"]),
            "stock": random.randint(0, 1000),
        })
```

### app/schemas/product.py

```python
from pydantic import BaseModel, Field
from decimal import Decimal

class ProductCreate(BaseModel):
    name: str = Field(min_length=1, max_length=200)
    price: Decimal = Field(gt=0, decimal_places=2)
    category: str
    stock: int = Field(ge=0, default=0)

class ProductResponse(ProductCreate):
    id: int

    model_config = {"from_attributes": True}

class CursorPage(BaseModel):
    items: list[ProductResponse]
    next_cursor: int | None
    cache: str = "MISS"
```
