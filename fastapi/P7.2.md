# P7.2 — Notification Service

**Phase:** 7 — Microservices & Messaging
**Topic:** Event-driven notifications, template engine, retry with exponential backoff
**Type:** Project

---

## Goal

Build a standalone notification service that consumes domain events from RabbitMQ and delivers email/SMS notifications using templates, with retry logic and delivery tracking.

## What to Build

- Consume `order.confirmed`, `order.cancelled`, `user.registered` events from RabbitMQ
- Render notifications using Jinja2 templates (email HTML + plain text)
- Mock email delivery (log to file) and SMS delivery (log to console)
- Retry failed deliveries with exponential backoff (max 3 retries)
- Track delivery status in a simple SQLite DB
- REST API: `GET /notifications/{id}` and `GET /notifications?user_id=X`

### app/models.py

```python
from sqlalchemy import String, Enum as SAEnum, Text, Integer
from sqlalchemy.orm import Mapped, mapped_column
from app.database import Base, TimestampMixin
import enum

class NotificationStatus(str, enum.Enum):
    pending = "pending"
    sent = "sent"
    failed = "failed"
    retrying = "retrying"

class NotificationChannel(str, enum.Enum):
    email = "email"
    sms = "sms"

class Notification(Base, TimestampMixin):
    __tablename__ = "notifications"

    id: Mapped[int] = mapped_column(primary_key=True)
    user_id: Mapped[int]
    channel: Mapped[NotificationChannel]
    event_type: Mapped[str] = mapped_column(String(100))
    recipient: Mapped[str] = mapped_column(String(255))  # email or phone
    subject: Mapped[str | None] = mapped_column(String(500))
    body: Mapped[str] = mapped_column(Text)
    status: Mapped[NotificationStatus] = mapped_column(default=NotificationStatus.pending)
    attempts: Mapped[int] = mapped_column(Integer, default=0)
    error: Mapped[str | None] = mapped_column(Text)
    correlation_id: Mapped[str | None] = mapped_column(String(36))
```

### app/dispatcher.py

```python
import asyncio
import structlog
from jinja2 import Environment, FileSystemLoader
from sqlalchemy.ext.asyncio import AsyncSession
from app.models import Notification, NotificationStatus, NotificationChannel
from app.channels.email import send_email
from app.channels.sms import send_sms

log = structlog.get_logger()

jinja_env = Environment(
    loader=FileSystemLoader("app/templates"),
    autoescape=True,
)

EVENT_TEMPLATES = {
    "order.confirmed": {
        "email": {"template": "order_confirmed.html", "subject": "Your order is confirmed!"},
        "sms": {"template": "order_confirmed_sms.txt"},
    },
    "order.cancelled": {
        "email": {"template": "order_cancelled.html", "subject": "Your order was cancelled"},
    },
    "user.registered": {
        "email": {"template": "welcome.html", "subject": "Welcome to our platform!"},
    },
}

async def dispatch_notification(
    session: AsyncSession,
    event_type: str,
    user_id: int,
    email: str,
    phone: str | None,
    context: dict,
    correlation_id: str,
) -> None:
    templates = EVENT_TEMPLATES.get(event_type, {})

    for channel_name, tmpl_config in templates.items():
        channel = NotificationChannel(channel_name)
        recipient = email if channel == NotificationChannel.email else phone
        if not recipient:
            continue

        # Render template
        template = jinja_env.get_template(tmpl_config["template"])
        body = template.render(**context)
        subject = tmpl_config.get("subject")

        notification = Notification(
            user_id=user_id,
            channel=channel,
            event_type=event_type,
            recipient=recipient,
            subject=subject,
            body=body,
            correlation_id=correlation_id,
        )
        session.add(notification)
        await session.flush()

        await deliver_with_retry(session, notification)

async def deliver_with_retry(session: AsyncSession, notification: Notification, max_attempts: int = 3) -> None:
    for attempt in range(1, max_attempts + 1):
        try:
            notification.attempts = attempt
            if notification.channel == NotificationChannel.email:
                await send_email(notification.recipient, notification.subject, notification.body)
            else:
                await send_sms(notification.recipient, notification.body)

            notification.status = NotificationStatus.sent
            log.info("notification_sent", id=notification.id, channel=notification.channel, attempt=attempt)
            await session.commit()
            return

        except Exception as e:
            notification.error = str(e)
            if attempt < max_attempts:
                notification.status = NotificationStatus.retrying
                await session.commit()
                backoff = 2 ** attempt
                log.warning("notification_retry", id=notification.id, attempt=attempt, backoff=backoff)
                await asyncio.sleep(backoff)
            else:
                notification.status = NotificationStatus.failed
                await session.commit()
                log.error("notification_failed", id=notification.id, error=str(e))
```

### app/consumer.py

```python
import json
import aio_pika
import structlog
from app.database import AsyncSessionLocal
from app.dispatcher import dispatch_notification

log = structlog.get_logger()

async def start_consumer(rabbitmq_url: str) -> None:
    connection = await aio_pika.connect_robust(rabbitmq_url)
    channel = await connection.channel()
    await channel.set_qos(prefetch_count=10)

    exchange = await channel.declare_exchange("orders", aio_pika.ExchangeType.TOPIC, durable=True)
    queue = await channel.declare_queue(
        "notification.events",
        durable=True,
        arguments={"x-dead-letter-exchange": "orders.dlx"},
    )
    await queue.bind(exchange, routing_key="order.*")
    await queue.bind(exchange, routing_key="user.*")

    async with queue.iterator() as q:
        async for message in q:
            async with message.process(requeue=False):
                try:
                    body = json.loads(message.body)
                    event_type = body["event_type"]
                    payload = body["payload"]
                    correlation_id = body.get("correlation_id", "")

                    async with AsyncSessionLocal() as session:
                        await dispatch_notification(
                            session=session,
                            event_type=event_type,
                            user_id=payload["user_id"],
                            email=payload.get("email", ""),
                            phone=payload.get("phone"),
                            context=payload,
                            correlation_id=correlation_id,
                        )
                    log.info("event_processed", event_type=event_type)
                except Exception as e:
                    log.error("event_processing_failed", error=str(e))
                    raise  # nack → DLQ
```
