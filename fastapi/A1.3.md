## Background

### Why Pydantic v2
Pydantic v2 was a complete rewrite in Rust, making it 5–50x faster than v1. The API changed significantly: validators use `@field_validator` instead of `@validator`, config is now `model_config = ConfigDict(...)` instead of a nested `Config` class, and `model_dump()` replaces `dict()`. FastAPI 0.100+ uses Pydantic v2 by default.

### Validation vs Serialization
Pydantic does two things: validates data coming in (input) and serializes data going out (output). You can have different behavior for each using `model_config`. For example, you might accept a datetime string as input but always serialize it as a Unix timestamp for output.

### The Role of Pydantic in FastAPI
Every FastAPI request body, response model, and query parameter uses Pydantic under the hood. When FastAPI receives a JSON body, it passes it to Pydantic for parsing. If Pydantic raises a `ValidationError`, FastAPI automatically returns a 422 with details. This means your endpoint function is only called if input is valid.

```python
# pydantic_basics.py
from pydantic import BaseModel, Field
from datetime import datetime

class Address(BaseModel):
    street: str
    city: str
    country: str = "India"
    postal_code: str | None = None

class User(BaseModel):
    id: int
    name: str = Field(min_length=2, max_length=50)
    email: str
    age: int = Field(ge=0, le=150)
    address: Address  # nested model
    created_at: datetime = Field(default_factory=datetime.utcnow)
    tags: list[str] = []

# Create from dict (e.g., from JSON)
user = User.model_validate({
    "id": 1,
    "name": "Alice",
    "email": "alice@example.com",
    "age": 30,
    "address": {"street": "123 Main St", "city": "Mumbai"},
})

print(user.model_dump())
print(user.model_dump_json(indent=2))
print(user.address.country)  # India (default)
```

### Task 2: Field Validators
```python
# validators.py
from pydantic import BaseModel, Field, field_validator, EmailStr
import re

class CreateUserRequest(BaseModel):
    username: str
    email: str
    password: str
    phone: str | None = None

    @field_validator("username")
    @classmethod
    def username_alphanumeric(cls, v: str) -> str:
        if not re.match(r"^[a-zA-Z0-9_]{3,20}$", v):
            raise ValueError(
                "Username must be 3-20 characters, alphanumeric and underscores only"
            )
        return v.lower()  # normalize to lowercase

    @field_validator("email")
    @classmethod
    def email_lowercase(cls, v: str) -> str:
        # Basic email validation
        if "@" not in v or "." not in v.split("@")[-1]:
            raise ValueError("Invalid email address")
        return v.lower()

    @field_validator("password")
    @classmethod
    def password_strength(cls, v: str) -> str:
        if len(v) < 8:
            raise ValueError("Password must be at least 8 characters")
        if not any(c.isupper() for c in v):
            raise ValueError("Password must contain at least one uppercase letter")
        if not any(c.isdigit() for c in v):
            raise ValueError("Password must contain at least one digit")
        return v

    @field_validator("phone")
    @classmethod
    def phone_format(cls, v: str | None) -> str | None:
        if v is None:
            return v
        digits = re.sub(r"\D", "", v)
        if len(digits) < 10:
            raise ValueError("Phone must have at least 10 digits")
        return digits  # normalize to digits only
```

### Task 3: model_validator (Cross-field Validation)
```python
# model_validators.py
from pydantic import BaseModel, model_validator
from datetime import date
from typing import Self

class BookingRequest(BaseModel):
    check_in: date
    check_out: date
    guests: int
    room_type: str  # "single", "double", "suite"

    @model_validator(mode="after")
    def validate_dates_and_capacity(self) -> Self:
        if self.check_out <= self.check_in:
            raise ValueError("check_out must be after check_in")
        
        max_guests = {"single": 1, "double": 2, "suite": 4}
        if self.room_type not in max_guests:
            raise ValueError(f"room_type must be one of {list(max_guests.keys())}")
        
        if self.guests > max_guests[self.room_type]:
            raise ValueError(
                f"Room type '{self.room_type}' allows max {max_guests[self.room_type]} guests, "
                f"but {self.guests} requested"
            )
        return self

    @model_validator(mode="before")
    @classmethod
    def normalize_room_type(cls, data: dict) -> dict:
        # mode="before" runs before field parsing
        if isinstance(data, dict) and "room_type" in data:
            data["room_type"] = data["room_type"].lower().strip()
        return data
```

### Task 4: Computed Fields
```python
# computed_fields.py
from pydantic import BaseModel, computed_field, Field
from datetime import date

class Product(BaseModel):
    name: str
    price: float = Field(gt=0)
    discount_percent: float = Field(ge=0, le=100, default=0)
    quantity: int = Field(ge=0)

    @computed_field
    @property
    def discounted_price(self) -> float:
        return round(self.price * (1 - self.discount_percent / 100), 2)

    @computed_field
    @property
    def total_value(self) -> float:
        return round(self.discounted_price * self.quantity, 2)

    @computed_field
    @property
    def in_stock(self) -> bool:
        return self.quantity > 0


p = Product(name="Laptop", price=1000, discount_percent=10, quantity=5)
print(p.discounted_price)   # 900.0
print(p.total_value)        # 4500.0
print(p.model_dump())       # includes computed fields
```

### Task 5: model_config
```python
# config_examples.py
from pydantic import BaseModel, ConfigDict, Field, AliasChoices

class APIResponse(BaseModel):
    model_config = ConfigDict(
        # Accept camelCase input and snake_case
        populate_by_name=True,
        # Forbid extra fields (returns error for unknown keys)
        extra="forbid",
        # Validate default values too
        validate_default=True,
        # Serialize datetime as string (ISO 8601)
        ser_json_timedelta="iso8601",
    )
    
    user_id: int = Field(validation_alias=AliasChoices("user_id", "userId", "uid"))
    full_name: str = Field(serialization_alias="fullName")


# Works with snake_case
r1 = APIResponse(user_id=1, full_name="Alice")
# Also works with camelCase (AliasChoices)
r2 = APIResponse(userId=1, full_name="Alice")

# Serializes with alias
print(r1.model_dump(by_alias=True))  # {"fullName": "Alice", "user_id": 1}
```

### Task 6: Discriminated Unions
```python
# discriminated_unions.py
from pydantic import BaseModel
from typing import Annotated, Literal
from pydantic import Field

class EmailNotification(BaseModel):
    type: Literal["email"]
    to: str
    subject: str
    body: str

class SMSNotification(BaseModel):
    type: Literal["sms"]
    to: str
    message: str

class PushNotification(BaseModel):
    type: Literal["push"]
    device_token: str
    title: str
    body: str

# Discriminated union based on the "type" field
Notification = Annotated[
    EmailNotification | SMSNotification | PushNotification,
    Field(discriminator="type"),
]

class NotificationRequest(BaseModel):
    notification: Notification
    priority: int = 1


# Pydantic uses the "type" field to know which model to parse into
req = NotificationRequest.model_validate({
    "notification": {
        "type": "email",
        "to": "user@example.com",
        "subject": "Hello",
        "body": "World",
    }
})
print(type(req.notification))  # EmailNotification
```
