## Goal

Build a real-time chat API with WebSocket rooms, Redis pub/sub for multi-instance message broadcasting, and user presence tracking.

## What to Build

- `WS /ws/{room_id}?token=...` — authenticated WebSocket connection
- Multiple chat rooms; users can join/leave
- Redis pub/sub for broadcasting across multiple server instances
- Presence: track who is online in a room
- Message history stored in PostgreSQL (last 50 messages on join)
- REST: `GET /rooms`, `POST /rooms`, `GET /rooms/{id}/history`

### app/connection_manager.py

```python
import json
import asyncio
import structlog
from fastapi import WebSocket
from redis.asyncio import Redis

log = structlog.get_logger()

class ConnectionManager:
    def __init__(self, redis: Redis):
        self.redis = redis
        # room_id -> {user_id -> WebSocket}
        self._connections: dict[str, dict[int, WebSocket]] = {}
        self._pubsub_tasks: dict[str, asyncio.Task] = {}

    async def connect(self, room_id: str, user_id: int, websocket: WebSocket) -> None:
        await websocket.accept()
        if room_id not in self._connections:
            self._connections[room_id] = {}
            # Start Redis subscriber for this room
            task = asyncio.create_task(self._subscribe_room(room_id))
            self._pubsub_tasks[room_id] = task

        self._connections[room_id][user_id] = websocket
        log.info("ws_connected", room_id=room_id, user_id=user_id)

    async def disconnect(self, room_id: str, user_id: int) -> None:
        if room_id in self._connections:
            self._connections[room_id].pop(user_id, None)
            if not self._connections[room_id]:
                del self._connections[room_id]
                task = self._pubsub_tasks.pop(room_id, None)
                if task:
                    task.cancel()
        log.info("ws_disconnected", room_id=room_id, user_id=user_id)

    async def broadcast(self, room_id: str, message: dict) -> None:
        """Publish to Redis; all instances pick it up via _subscribe_room."""
        channel = f"chat:{room_id}"
        await self.redis.publish(channel, json.dumps(message))

    async def _subscribe_room(self, room_id: str) -> None:
        """Run in background: receive from Redis, deliver to local WebSockets."""
        pubsub = self.redis.pubsub()
        await pubsub.subscribe(f"chat:{room_id}")
        try:
            async for msg in pubsub.listen():
                if msg["type"] != "message":
                    continue
                data = json.loads(msg["data"])
                await self._deliver_locally(room_id, data)
        except asyncio.CancelledError:
            await pubsub.unsubscribe(f"chat:{room_id}")
            await pubsub.aclose()

    async def _deliver_locally(self, room_id: str, message: dict) -> None:
        connections = self._connections.get(room_id, {})
        dead = []
        for user_id, ws in connections.items():
            try:
                await ws.send_json(message)
            except Exception:
                dead.append(user_id)
        for uid in dead:
            connections.pop(uid, None)
```

### app/presence.py

```python
import json
from redis.asyncio import Redis

class PresenceTracker:
    def __init__(self, redis: Redis):
        self.redis = redis

    def _key(self, room_id: str) -> str:
        return f"presence:{room_id}"

    async def join(self, room_id: str, user_id: int, username: str) -> None:
        await self.redis.hset(self._key(room_id), str(user_id), username)
        await self.redis.expire(self._key(room_id), 3600)

    async def leave(self, room_id: str, user_id: int) -> None:
        await self.redis.hdel(self._key(room_id), str(user_id))

    async def get_online_users(self, room_id: str) -> list[dict]:
        members = await self.redis.hgetall(self._key(room_id))
        return [{"user_id": int(uid), "username": uname.decode()} for uid, uname in members.items()]
```

### app/routers/ws.py

```python
from fastapi import APIRouter, WebSocket, WebSocketDisconnect, Query, HTTPException
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, desc
from datetime import datetime, UTC
from app.connection_manager import ConnectionManager
from app.presence import PresenceTracker
from app.security.jwt import decode_token
from app.models.message import Message
from app.models.room import Room
from app.schemas.chat import IncomingMessage, OutgoingMessage, MessageType
from app.dependencies import get_manager, get_presence, get_session

router = APIRouter()

@router.websocket("/ws/{room_id}")
async def websocket_endpoint(
    websocket: WebSocket,
    room_id: str,
    token: str = Query(...),
    manager: ConnectionManager = Depends(get_manager),
    presence: PresenceTracker = Depends(get_presence),
    session: AsyncSession = Depends(get_session),
):
    # Authenticate via query param token
    try:
        payload = decode_token(token)
        user_id = int(payload["sub"])
        username = payload.get("username", f"user_{user_id}")
    except ValueError:
        await websocket.close(code=4001, reason="Invalid token")
        return

    # Check room exists
    room = await session.get(Room, room_id)
    if not room:
        await websocket.close(code=4004, reason="Room not found")
        return

    await manager.connect(room_id, user_id, websocket)
    await presence.join(room_id, user_id, username)

    # Send history
    history = await session.execute(
        select(Message).where(Message.room_id == room_id)
        .order_by(desc(Message.created_at)).limit(50)
    )
    history_msgs = list(reversed(history.scalars().all()))
    await websocket.send_json({
        "type": "history",
        "messages": [{"user_id": m.user_id, "content": m.content, "timestamp": m.created_at.isoformat()} for m in history_msgs],
    })

    # Announce join
    online = await presence.get_online_users(room_id)
    await manager.broadcast(room_id, {
        "type": "presence",
        "event": "joined",
        "user_id": user_id,
        "username": username,
        "online_count": len(online),
    })

    try:
        while True:
            data = await websocket.receive_json()
            msg = IncomingMessage(**data)

            # Persist
            db_msg = Message(room_id=room_id, user_id=user_id, content=msg.content)
            session.add(db_msg)
            await session.commit()

            # Broadcast
            await manager.broadcast(room_id, {
                "type": "chat",
                "room_id": room_id,
                "user_id": user_id,
                "username": username,
                "content": msg.content,
                "timestamp": datetime.now(UTC).isoformat(),
            })

    except WebSocketDisconnect:
        await manager.disconnect(room_id, user_id)
        await presence.leave(room_id, user_id)
        online = await presence.get_online_users(room_id)
        await manager.broadcast(room_id, {
            "type": "presence",
            "event": "left",
            "user_id": user_id,
            "username": username,
            "online_count": len(online),
        })
```
