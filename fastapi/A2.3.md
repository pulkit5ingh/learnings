## Goal
Manage database schema changes safely using Alembic migrations with autogenerate, rollbacks, and data migrations.

## Background

### Why Migrations
Your database schema will change over time — adding columns, changing types, dropping tables. Migrations are versioned scripts that apply these changes incrementally. Alembic generates migration scripts based on the diff between your SQLAlchemy models and the current database state. Each migration has an `upgrade()` (apply) and `downgrade()` (revert) function.

### The Migration Chain
Alembic tracks migrations as a linked list of revisions. Each migration has a `revision` ID and a `down_revision` pointing to its parent. Running `alembic upgrade head` runs all pending migrations. Running `alembic downgrade -1` reverts the last one. This creates a safe, reversible history of schema changes.

### autogenerate vs Manual
`alembic revision --autogenerate` compares your models to the live database and generates a migration script for the differences. This works for most changes (add column, new table, new index) but does NOT detect: column type changes where the DB representation is the same, changes in server defaults, or custom constraints. Always review autogenerated migrations before running them.

### Task 2: Configure env.py
```python
# alembic/env.py — full working version
from logging.config import fileConfig
from sqlalchemy import engine_from_config, pool
from alembic import context
import os

# Import your models' Base
import sys
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))
from app.models import Base  # adjust import path

config = context.config

# Override sqlalchemy.url with env variable if set
database_url = os.getenv("DATABASE_URL")
if database_url:
    config.set_main_option("sqlalchemy.url", database_url)

if config.config_file_name is not None:
    fileConfig(config.config_file_name)

target_metadata = Base.metadata  # tells autogenerate what to compare against


def run_migrations_offline() -> None:
    """Run migrations without a live DB connection (generates SQL only)."""
    url = config.get_main_option("sqlalchemy.url")
    context.configure(
        url=url,
        target_metadata=target_metadata,
        literal_binds=True,
        dialect_opts={"paramstyle": "named"},
    )
    with context.begin_transaction():
        context.run_migrations()


def run_migrations_online() -> None:
    """Run migrations against a live DB connection."""
    connectable = engine_from_config(
        config.get_section(config.config_ini_section, {}),
        prefix="sqlalchemy.",
        poolclass=pool.NullPool,
    )
    with connectable.connect() as connection:
        context.configure(connection=connection, target_metadata=target_metadata)
        with context.begin_transaction():
            context.run_migrations()


if context.is_offline_mode():
    run_migrations_offline()
else:
    run_migrations_online()
```

### Task 3: Creating and Running Migrations
```bash
# Generate initial migration from models
alembic revision --autogenerate -m "create users and posts tables"

# Review the generated file in alembic/versions/
# It should have upgrade() and downgrade() functions

# Apply the migration
alembic upgrade head

# Check current state
alembic current

# View migration history
alembic history --verbose

# Apply up to a specific revision
alembic upgrade abc123

# Roll back one step
alembic downgrade -1

# Roll back to base (empty database)
alembic downgrade base
```

### Task 4: Example Migration Files
```python
# alembic/versions/001_create_initial_tables.py
"""create users and posts tables

Revision ID: a1b2c3d4e5f6
Revises: 
Create Date: 2024-01-15 10:00:00.000000
"""
from alembic import op
import sqlalchemy as sa

revision = "a1b2c3d4e5f6"
down_revision = None
branch_labels = None
depends_on = None


def upgrade() -> None:
    op.create_table(
        "users",
        sa.Column("id", sa.Integer(), nullable=False),
        sa.Column("name", sa.String(length=100), nullable=False),
        sa.Column("email", sa.String(length=255), nullable=False),
        sa.Column("is_active", sa.Boolean(), nullable=False, server_default="true"),
        sa.Column("created_at", sa.DateTime(), server_default=sa.text("now()")),
        sa.PrimaryKeyConstraint("id"),
        sa.UniqueConstraint("email"),
    )
    op.create_index("ix_users_email", "users", ["email"])

    op.create_table(
        "posts",
        sa.Column("id", sa.Integer(), nullable=False),
        sa.Column("title", sa.String(length=200), nullable=False),
        sa.Column("body", sa.Text(), nullable=False),
        sa.Column("user_id", sa.Integer(), nullable=False),
        sa.Column("created_at", sa.DateTime(), server_default=sa.text("now()")),
        sa.ForeignKeyConstraint(["user_id"], ["users.id"], ondelete="CASCADE"),
        sa.PrimaryKeyConstraint("id"),
    )


def downgrade() -> None:
    op.drop_table("posts")
    op.drop_index("ix_users_email")
    op.drop_table("users")
```

```python
# alembic/versions/002_add_published_to_posts.py
"""add published column to posts

Revision ID: b2c3d4e5f6a7
Revises: a1b2c3d4e5f6
Create Date: 2024-01-16 10:00:00.000000
"""
from alembic import op
import sqlalchemy as sa

revision = "b2c3d4e5f6a7"
down_revision = "a1b2c3d4e5f6"


def upgrade() -> None:
    op.add_column(
        "posts",
        sa.Column("published", sa.Boolean(), nullable=False, server_default="false"),
    )
    op.add_column(
        "posts",
        sa.Column("published_at", sa.DateTime(), nullable=True),
    )


def downgrade() -> None:
    op.drop_column("posts", "published_at")
    op.drop_column("posts", "published")
```

### Task 5: Data Migrations
```python
# alembic/versions/003_normalize_email_to_lowercase.py
"""normalize existing emails to lowercase

Revision ID: c3d4e5f6a7b8
Revises: b2c3d4e5f6a7
"""
from alembic import op
import sqlalchemy as sa
from sqlalchemy.sql import table, column

revision = "c3d4e5f6a7b8"
down_revision = "b2c3d4e5f6a7"

# Define a lightweight table reference for data migration
users_table = table("users", column("id", sa.Integer), column("email", sa.String))


def upgrade() -> None:
    # Step 1: Schema change (add temp column)
    op.add_column("users", sa.Column("email_lower", sa.String(255), nullable=True))
    
    # Step 2: Data migration using raw SQL
    op.execute(
        "UPDATE users SET email_lower = LOWER(email)"
    )
    
    # Step 3: Drop old column, rename new one
    op.drop_column("users", "email")
    op.alter_column("users", "email_lower", new_column_name="email", nullable=False)
    
    # Step 4: Recreate index on new column
    op.create_unique_constraint("uq_users_email", "users", ["email"])


def downgrade() -> None:
    # Data migration downgrades are often not possible (you can't un-lowercase)
    # Just restore the schema, accepting data loss
    pass  # or raise NotImplementedError("Cannot reverse data migration")
```

### Task 6: Multiple Heads and Branches
```bash
# Check for multiple heads (conflicts)
alembic heads

# If two developers created migrations from the same revision,
# you'll have two heads. Merge them:
alembic merge -m "merge heads" head1_id head2_id

# This creates a new migration that has both as down_revision:
# down_revision = ("head1_id", "head2_id")
```
