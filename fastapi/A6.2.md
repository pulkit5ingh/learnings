**Type:** Assignment

## Goal
Implement clean API versioning so old clients keep working while new clients get improved endpoints.

## Background

### Why Version APIs
Once clients depend on your API, changing it breaks them. Versioning lets you introduce breaking changes while keeping old versions running for a migration period. A breaking change is: removing a field, changing a field's type, changing a URL, changing behavior that clients rely on. Adding new optional fields or endpoints is NOT breaking.

### URL Versioning vs Header Versioning
URL versioning (`/v1/items`, `/v2/items`) is explicit, bookmarkable, and easy to debug. It is the most common approach and what most developers expect. Header versioning (`Accept: application/vnd.myapi.v2+json`) is "cleaner" according to REST purists but is harder to use in browsers and requires more client-side setup. URL versioning is recommended unless you have a specific reason to use headers.

### Maintaining Multiple Versions
In FastAPI, you create separate routers for each version and include them under different prefixes. For route-level changes (a new field), you can reuse most code and only override the changed parts. For large changes, you may have separate service/schema files per version.

from fastapi import APIRouter
from pydantic import BaseModel

router = APIRouter()

class ItemResponseV1(BaseModel):
    id: int
    name: str
    price: float  # v1: price as float

@router.get("/{item_id}", response_model=ItemResponseV1)
async def get_item_v1(item_id: int):
    return {"id": item_id, "name": "Widget", "price": 9.99}

@router.get("/")
async def list_items_v1():
    return [{"id": 1, "name": "Widget", "price": 9.99}]

# app/api/v2/items.py
from fastapi import APIRouter
from pydantic import BaseModel
from decimal import Decimal

router = APIRouter()

class PriceInfo(BaseModel):
    amount: Decimal
    currency: str = "USD"

class ItemResponseV2(BaseModel):
    id: int
    name: str
    price: PriceInfo       # v2: price is an object with currency
    tags: list[str] = []   # v2: added tags field
    slug: str              # v2: added slug

@router.get("/{item_id}", response_model=ItemResponseV2)
async def get_item_v2(item_id: int):
    return {
        "id": item_id,
        "name": "Widget",
        "price": {"amount": "9.99", "currency": "USD"},
        "tags": ["electronics"],
        "slug": "widget",
    }

@router.get("/")
async def list_items_v2(currency: str = "USD"):
    # v2 supports currency filtering — new feature
    return []

# app/main.py
from fastapi import FastAPI
from app.api.v1 import items as v1_items, users as v1_users
from app.api.v2 import items as v2_items, users as v2_users

app = FastAPI(
    title="My API",
    description="Supports v1 and v2",
    version="2.0.0",

app.include_router(
    v1_items.router,
    prefix="/v1/items",
    tags=["Items (v1 - deprecated)"],

app.include_router(
    v1_users.router,
    prefix="/v1/users",
    tags=["Users (v1 - deprecated)"],

app.include_router(
    v2_items.router,
    prefix="/v2/items",
    tags=["Items"],

    v2_users.router,

    tags=["Items (current)"],

### Task 3: Header Versioning (Alternative)
```python
# app/version_middleware.py
from fastapi import Request, HTTPException

SUPPORTED_VERSIONS = {"1", "2"}
DEFAULT_VERSION = "2"

async def version_middleware(request: Request, call_next):
    """
    Read API version from Accept header:
    Accept: application/vnd.myapi.v2+json
    or custom header:
    X-API-Version: 2
    """
    # Check custom header first
    version = request.headers.get("X-API-Version")
    
    # Fall back to Accept header
    if not version:
        accept = request.headers.get("Accept", "")
        import re
        match = re.search(r"application/vnd\.myapi\.v(\d+)\+json", accept)
        if match:
            version = match.group(1)
    
    # Default to latest
    if not version:
        version = DEFAULT_VERSION
    
    if version not in SUPPORTED_VERSIONS:
        raise HTTPException(
            status_code=400,
            detail=f"Unsupported API version: {version}. Supported: {SUPPORTED_VERSIONS}",
        )
    
    request.state.api_version = version
    response = await call_next(request)
    response.headers["X-API-Version"] = version
    return response
```

### Task 4: Shared Business Logic Across Versions
```python
# app/services/items_service.py — version-agnostic business logic
from sqlalchemy.ext.asyncio import AsyncSession
from app.models import Item

class ItemService:
    def __init__(self, session: AsyncSession):
        self.session = session

    async def get_item(self, item_id: int) -> Item | None:
        return await self.session.get(Item, item_id)

    async def list_items(self, limit: int = 20, offset: int = 0) -> list[Item]:
        from sqlalchemy import select
        result = await self.session.execute(
            select(Item).limit(limit).offset(offset)
        )
        return list(result.scalars().all())


# v1 route uses the service, formats response as v1
from app.api.v1.items import ItemResponseV1

async def get_item_v1(item_id: int, service: ItemService) -> ItemResponseV1:
    item = await service.get_item(item_id)
    if item is None:
        raise HTTPException(status_code=404, detail="Not found")
    # v1 format: flat price
    return ItemResponseV1(id=item.id, name=item.name, price=float(item.price))

# v2 route uses the same service, formats response as v2
from app.api.v2.items import ItemResponseV2, PriceInfo

async def get_item_v2(item_id: int, service: ItemService) -> ItemResponseV2:
    item = await service.get_item(item_id)
    if item is None:
        raise HTTPException(status_code=404, detail="Not found")
    # v2 format: price object + tags + slug
    return ItemResponseV2(
        id=item.id,
        name=item.name,
        price=PriceInfo(amount=item.price, currency=item.currency),
        tags=item.tags or [],
        slug=item.slug,
    )
```

### Task 5: Version Deprecation Strategy
```python
# Communicate deprecation via headers and documentation

async def add_deprecation_headers(request: Request, call_next):
    response = await call_next(request)
    
    if request.url.path.startswith("/v1/"):
        response.headers["Deprecation"] = "true"
        response.headers["Sunset"] = "2025-12-31T00:00:00Z"  # RFC 8594
        response.headers["Link"] = (
            '</v2' + request.url.path[3:] + '>; rel="successor-version"'
        )
    
    return response

app.middleware("http")(add_deprecation_headers)
```

# tests/test_versioning.py
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)

def test_v1_item_returns_flat_price():
    response = client.get("/v1/items/1")
    assert response.status_code == 200
    data = response.json()
    assert isinstance(data["price"], float)  # v1: flat float
    assert "currency" not in data  # v1: no currency

def test_v2_item_returns_price_object():
    response = client.get("/v2/items/1")
    assert response.status_code == 200
    data = response.json()
    assert isinstance(data["price"], dict)  # v2: price object
    assert "amount" in data["price"]
    assert "currency" in data["price"]
    assert "tags" in data  # v2: has tags

def test_v1_returns_deprecation_header():
    response = client.get("/v1/items/1")
    assert response.headers.get("Deprecation") == "true"
    assert "Sunset" in response.headers

def test_both_versions_use_same_item():
    """Same item_id returns the same item in both versions."""
    v1 = client.get("/v1/items/1").json()
    v2 = client.get("/v2/items/1").json()
    assert v1["id"] == v2["id"]
    assert v1["name"] == v2["name"]
