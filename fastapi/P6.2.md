**Topic:** OpenAPI spec, SDK generation, API key auth, client library
**Type:** Project

## Goal

Build a public-facing API and generate a typed Python client SDK from its OpenAPI spec, then publish a usage example showing the SDK in action.

## What to Build

- A weather/data API with API key authentication
- Comprehensive OpenAPI spec with all schemas, examples, and descriptions
- Generate Python SDK using `openapi-python-client` or write a thin wrapper manually
- SDK features: typed return types, retry logic, async support
- Usage example: `sdk.weather.get_forecast(city="London")`

### api/auth.py

```python
from fastapi import Security, HTTPException
from fastapi.security import APIKeyHeader
from app.config import settings

api_key_header = APIKeyHeader(name="X-API-Key", auto_error=False)

VALID_API_KEYS = {"demo-key-123", "prod-key-xyz"}  # in real app, DB-backed

async def require_api_key(api_key: str = Security(api_key_header)) -> str:
    if not api_key or api_key not in VALID_API_KEYS:
        raise HTTPException(
            status_code=401,
            detail="Invalid or missing API key. Include X-API-Key header.",
            headers={"WWW-Authenticate": "ApiKey"},
        )
    return api_key
```

### api/schemas/weather.py

```python
from pydantic import BaseModel, Field
from datetime import date

class CurrentWeather(BaseModel):
    city: str
    country: str
    temperature_c: float = Field(description="Temperature in Celsius")
    feels_like_c: float
    humidity_pct: int = Field(ge=0, le=100)
    wind_speed_kmh: float
    condition: str
    description: str
    timestamp: str

class ForecastDay(BaseModel):
    date: date
    temp_min_c: float
    temp_max_c: float
    condition: str
    precipitation_mm: float

class Forecast(BaseModel):
    city: str
    country: str
    days: list[ForecastDay] = Field(description="7-day forecast")

class ErrorResponse(BaseModel):
    type: str
    title: str
    status: int
    detail: str
```

### api/routers/weather.py

```python
from fastapi import APIRouter, Depends, Query
from api.auth import require_api_key
from api.schemas.weather import CurrentWeather, Forecast, ForecastDay
from datetime import date, timedelta
import random

router = APIRouter(prefix="/weather", tags=["Weather"])

MOCK_DATA = {
    "london": {"country": "GB", "temp": 15.2, "condition": "Cloudy"},
    "tokyo": {"country": "JP", "temp": 22.8, "condition": "Sunny"},
    "new york": {"country": "US", "temp": 18.5, "condition": "Partly Cloudy"},
}

@router.get(
    "/current",
    response_model=CurrentWeather,
    summary="Get current weather",
    description="Returns the current weather conditions for the specified city.",
    responses={
        200: {"description": "Current weather data"},
        401: {"description": "Invalid API key"},
        404: {"description": "City not found"},
    },
)
async def get_current_weather(
    city: str = Query(description="City name", example="London"),
    _: str = Depends(require_api_key),
):
    from fastapi import HTTPException
    from datetime import datetime, UTC

    city_lower = city.lower()
    if city_lower not in MOCK_DATA:
        raise HTTPException(404, f"City '{city}' not found")

    data = MOCK_DATA[city_lower]
    return CurrentWeather(
        city=city.title(),
        country=data["country"],
        temperature_c=data["temp"],
        feels_like_c=data["temp"] - 2,
        humidity_pct=random.randint(40, 90),
        wind_speed_kmh=random.uniform(5, 40),
        condition=data["condition"],
        description=f"{data['condition']} skies",
        timestamp=datetime.now(UTC).isoformat(),
    )

@router.get("/forecast", response_model=Forecast, summary="Get 7-day forecast")
async def get_forecast(
    city: str = Query(description="City name"),
    days: int = Query(default=7, ge=1, le=14, description="Number of forecast days"),
    _: str = Depends(require_api_key),
):
    from fastapi import HTTPException
    city_lower = city.lower()
    if city_lower not in MOCK_DATA:
        raise HTTPException(404, f"City '{city}' not found")
    data = MOCK_DATA[city_lower]
    forecast_days = [
        ForecastDay(
            date=date.today() + timedelta(days=i),
            temp_min_c=data["temp"] - random.uniform(2, 5),
            temp_max_c=data["temp"] + random.uniform(1, 4),
            condition=data["condition"],
            precipitation_mm=random.uniform(0, 10),
        )
        for i in range(days)
    ]
    return Forecast(city=city.title(), country=data["country"], days=forecast_days)
```

### sdk/client.py

```python
import httpx
from typing import Any
from sdk.exceptions import APIError, AuthError, NotFoundError

class WeatherClient:
    """Typed Python SDK for the Weather API."""

    def __init__(self, api_key: str, base_url: str = "https://api.example.com"):
        self.api_key = api_key
        self.base_url = base_url
        self._headers = {"X-API-Key": api_key, "Accept": "application/json"}

    def _handle_response(self, response: httpx.Response) -> dict:
        if response.status_code == 401:
            raise AuthError("Invalid API key")
        if response.status_code == 404:
            raise NotFoundError(response.json().get("detail", "Not found"))
        if response.status_code >= 400:
            raise APIError(f"API error {response.status_code}: {response.text}")
        return response.json()

    def get_current_weather(self, city: str) -> dict:
        with httpx.Client() as client:
            resp = client.get(
                f"{self.base_url}/weather/current",
                headers=self._headers,
                params={"city": city},
                timeout=10,
            )
        return self._handle_response(resp)

    def get_forecast(self, city: str, days: int = 7) -> dict:
        with httpx.Client() as client:
            resp = client.get(
                f"{self.base_url}/weather/forecast",
                headers=self._headers,
                params={"city": city, "days": days},
                timeout=10,
            )
        return self._handle_response(resp)
```

### sdk/exceptions.py

```python
class APIError(Exception):
    pass

class AuthError(APIError):
    pass

class NotFoundError(APIError):
    pass

class RateLimitError(APIError):
    def __init__(self, retry_after: int):
        self.retry_after = retry_after
        super().__init__(f"Rate limited. Retry after {retry_after} seconds.")
```

### examples/usage.py

```python
from sdk.client import WeatherClient
from sdk.exceptions import NotFoundError, AuthError

client = WeatherClient(api_key="demo-key-123", base_url="http://localhost:8000")

try:
    current = client.get_current_weather("London")
    print(f"London: {current['temperature_c']}°C, {current['condition']}")

    forecast = client.get_forecast("Tokyo", days=3)
    for day in forecast["days"]:
        print(f"  {day['date']}: {day['temp_min_c']}–{day['temp_max_c']}°C")

except NotFoundError as e:
    print(f"City not found: {e}")
except AuthError:
    print("Check your API key")
```
