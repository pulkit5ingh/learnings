**Type:** Project

## Goal

Build a full blog GraphQL API using Strawberry with queries, mutations, subscriptions, N+1-free DataLoaders, and JWT authentication.

## What to Build

- **Queries**: `posts`, `post(id)`, `me`, `users`
- **Mutations**: `createPost`, `updatePost`, `deletePost`, `login`
- **Subscriptions**: `postAdded` — real-time notifications via async generator
- **DataLoaders**: batch-load authors and tags to avoid N+1 queries
- Auth: JWT in `Authorization` header, applied via Strawberry `Context`

### app/dataloaders.py

```python
from strawberry.dataloader import DataLoader
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, and_
from app.models import User, Post, Tag, post_tag

def make_user_loader(session: AsyncSession) -> DataLoader:
    async def batch_load_users(user_ids: list[int]) -> list[User | None]:
        result = await session.execute(select(User).where(User.id.in_(user_ids)))
        users = {u.id: u for u in result.scalars().all()}
        return [users.get(uid) for uid in user_ids]
    return DataLoader(load_fn=batch_load_users)

def make_post_tags_loader(session: AsyncSession) -> DataLoader:
    async def batch_load_tags(post_ids: list[int]) -> list[list[Tag]]:
        result = await session.execute(
            select(Tag, post_tag.c.post_id)
            .join(post_tag, post_tag.c.tag_id == Tag.id)
            .where(post_tag.c.post_id.in_(post_ids))
        )
        tags_by_post: dict[int, list[Tag]] = {pid: [] for pid in post_ids}
        for tag, pid in result:
            tags_by_post[pid].append(tag)
        return [tags_by_post[pid] for pid in post_ids]
    return DataLoader(load_fn=batch_load_tags)

class Loaders:
    def __init__(self, session: AsyncSession):
        self.user_loader = make_user_loader(session)
        self.post_tags_loader = make_post_tags_loader(session)
```

### app/context.py

```python
from fastapi import Request
from sqlalchemy.ext.asyncio import AsyncSession
from app.database import AsyncSessionLocal
from app.dataloaders import Loaders
from app.security.jwt import decode_token

class GraphQLContext:
    def __init__(self, session: AsyncSession, user_id: int | None):
        self.session = session
        self.user_id = user_id
        self.loaders = Loaders(session)

async def get_context(request: Request) -> GraphQLContext:
    session = AsyncSessionLocal()
    user_id = None
    auth = request.headers.get("Authorization", "")
    if auth.startswith("Bearer "):
        try:
            payload = decode_token(auth[7:])
            user_id = int(payload["sub"])
        except ValueError:
            pass
    return GraphQLContext(session=session, user_id=user_id)
```

### app/resolvers/query.py

```python
import strawberry
from strawberry.types import Info
from sqlalchemy import select
from app.models import Post, User
from app.types.post import PostType, UserType
from fastapi import HTTPException

@strawberry.type
class Query:
    @strawberry.field
    async def posts(
        self,
        info: Info,
        published_only: bool = True,
        limit: int = 20,
        offset: int = 0,
    ) -> list[PostType]:
        session = info.context.session
        q = select(Post).order_by(Post.created_at.desc()).offset(offset).limit(limit)
        if published_only:
            q = q.where(Post.published == True)
        result = await session.execute(q)
        return result.scalars().all()

    @strawberry.field
    async def post(self, info: Info, id: int) -> PostType | None:
        return await info.context.session.get(Post, id)

    @strawberry.field
    async def me(self, info: Info) -> UserType:
        if not info.context.user_id:
            raise strawberry.exceptions.PermissionError("Not authenticated")
        user = await info.context.session.get(User, info.context.user_id)
        if not user:
            raise strawberry.exceptions.PermissionError("User not found")
        return user
```

### app/resolvers/mutation.py

```python
import strawberry
from strawberry.types import Info
from app.models import Post, Tag
from app.types.post import PostType, CreatePostInput
from sqlalchemy import select

@strawberry.type
class Mutation:
    @strawberry.mutation
    async def create_post(self, info: Info, input: CreatePostInput) -> PostType:
        if not info.context.user_id:
            raise strawberry.exceptions.PermissionError("Must be logged in")

        session = info.context.session
        post = Post(
            title=input.title,
            body=input.body,
            published=input.published,
            author_id=info.context.user_id,
        )
        if input.tag_ids:
            result = await session.execute(select(Tag).where(Tag.id.in_(input.tag_ids)))
            post.tags = result.scalars().all()

        session.add(post)
        await session.commit()
        await session.refresh(post)
        return post

    @strawberry.mutation
    async def delete_post(self, info: Info, id: int) -> bool:
        if not info.context.user_id:
            raise strawberry.exceptions.PermissionError("Must be logged in")

        session = info.context.session
        post = await session.get(Post, id)
        if not post:
            return False
        if post.author_id != info.context.user_id:
            raise strawberry.exceptions.PermissionError("Not your post")

        await session.delete(post)
        await session.commit()
        return True
```

### app/resolvers/subscription.py

```python
import strawberry
import asyncio
from typing import AsyncGenerator
from app.types.post import PostType

# In production, use Redis pub/sub for multi-instance support
_post_queues: list[asyncio.Queue] = []

async def notify_new_post(post: PostType) -> None:
    for queue in _post_queues:
        await queue.put(post)

@strawberry.type
class Subscription:
    @strawberry.subscription
    async def post_added(self) -> AsyncGenerator[PostType, None]:
        queue: asyncio.Queue = asyncio.Queue()
        _post_queues.append(queue)
        try:
            while True:
                post = await queue.get()
                yield post
        finally:
            _post_queues.remove(queue)
```
