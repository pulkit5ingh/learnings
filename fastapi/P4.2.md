**Topic:** Testing auth flows, mocking Redis, factory_boy, pytest-mock
**Type:** Project

## Goal

Write a complete test suite for the auth service from Phase 3 (P3.1), covering all auth flows including edge cases for token rotation, blacklisting, and email verification.

## What to Build

- Unit tests for JWT encode/decode, password hashing
- Integration tests for registration, login, refresh, logout
- Mocked Redis in unit/integration tests (no real Redis dependency)
- factory_boy factories for User and RefreshToken
- Test all error paths: wrong password, unverified email, reused refresh token

### tests/unit/test_jwt.py

```python
import pytest
from datetime import datetime, UTC, timedelta
from jose import jwt
from app.security.jwt import create_access_token, create_refresh_token, decode_token
from app.config import settings

class TestJWT:
    def test_access_token_has_correct_type(self):
        token = create_access_token("42")
        payload = decode_token(token)
        assert payload["sub"] == "42"
        assert payload["type"] == "access"

    def test_refresh_token_has_jti(self):
        token = create_refresh_token("42", jti="abc-123")
        payload = decode_token(token)
        assert payload["jti"] == "abc-123"
        assert payload["type"] == "refresh"

    def test_expired_token_raises(self):
        payload = {"sub": "1", "exp": datetime.now(UTC) - timedelta(seconds=1), "type": "access"}
        token = jwt.encode(payload, settings.secret_key, algorithm="HS256")
        with pytest.raises(ValueError, match="expired"):
            decode_token(token)

    def test_tampered_token_raises(self):
        token = create_access_token("99")
        tampered = token[:-5] + "XXXXX"
        with pytest.raises(ValueError):
            decode_token(tampered)
```

### tests/integration/test_refresh.py

```python
import pytest
from app.services.auth_service import AuthService
from app.security.jwt import create_refresh_token
from tests.factories import UserFactory, RefreshTokenFactory
import uuid

@pytest.mark.integration
class TestRefreshToken:
    async def test_refresh_issues_new_tokens(self, session, mock_redis):
        UserFactory._meta.sqlalchemy_session = session
        RefreshTokenFactory._meta.sqlalchemy_session = session

        user = UserFactory()
        jti = str(uuid.uuid4())
        RefreshTokenFactory(jti=jti, user_id=user.id)

        token = create_refresh_token(str(user.id), jti)
        svc = AuthService(session)
        result = await svc.refresh(token)

        assert "access_token" in result
        assert "refresh_token" in result
        assert result["refresh_token"] != token

    async def test_refresh_rejects_revoked_token(self, session, mock_redis):
        UserFactory._meta.sqlalchemy_session = session
        RefreshTokenFactory._meta.sqlalchemy_session = session

        user = UserFactory()
        jti = str(uuid.uuid4())
        RefreshTokenFactory(jti=jti, user_id=user.id, is_revoked=True)

        token = create_refresh_token(str(user.id), jti)
        svc = AuthService(session)
        with pytest.raises(Exception) as exc_info:
            await svc.refresh(token)
        assert exc_info.value.status_code == 401

    async def test_refresh_rejects_blacklisted_jti(self, session, mock_redis):
        user = UserFactory()
        jti = str(uuid.uuid4())
        mock_redis.exists.return_value = True  # simulate Redis blacklist hit

        token = create_refresh_token(str(user.id), jti)
        svc = AuthService(session)
        with pytest.raises(Exception) as exc_info:
            await svc.refresh(token)
        assert exc_info.value.status_code == 401
```
