## Goal
Write endpoint tests using FastAPI's test utilities, override dependencies for isolation, and set up a dedicated test database.

## Background

### TestClient vs AsyncClient
`TestClient` from `fastapi.testclient` is a synchronous HTTP client that internally runs the async app in a thread. It is convenient for most tests. `AsyncClient` from `httpx` is fully async and is required when your test itself needs to be async (e.g., using `pytest-asyncio` with an async test database fixture). Use `TestClient` by default; switch to `AsyncClient` when you need async fixtures.

### override_dependency
`app.dependency_overrides` is a dictionary that maps a dependency function to a replacement. FastAPI checks this dict when resolving dependencies. This is the primary mechanism for isolation in tests: replace `get_db` with a function that returns a test database session, replace `get_current_user` with a function that returns a fake user. Always clear `dependency_overrides` after each test to prevent test pollution.

### Test Database
Tests should never run against the production database. Set up a separate test PostgreSQL database (or use SQLite for fast unit tests). Create tables before tests, drop them after. Use transactions to isolate tests: start a transaction at the beginning of each test, rollback at the end — so no data from one test bleeds into another.

### Task 1: Basic TestClient Usage
```python
# tests/test_items.py
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)

def test_root():
    response = client.get("/")
    assert response.status_code == 200
    assert response.json() == {"message": "Hello World"}

def test_create_item():
    response = client.post(
        "/items/",
        json={"name": "Test Item", "price": 9.99, "in_stock": True},
    )
    assert response.status_code == 201
    data = response.json()
    assert data["name"] == "Test Item"
    assert data["price"] == 9.99
    assert "id" in data

def test_create_item_invalid():
    response = client.post(
        "/items/",
        json={"name": "", "price": -1},  # invalid
    )
    assert response.status_code == 422
    errors = response.json()["detail"]
    assert len(errors) > 0  # at least one validation error

def test_get_item_not_found():
    response = client.get("/items/99999")
    assert response.status_code == 404
    assert "not found" in response.json()["detail"].lower()
```

```python
import pytest
import pytest_asyncio
from sqlalchemy.ext.asyncio import create_async_engine, AsyncSession, async_sessionmaker
from sqlalchemy.orm import sessionmaker
from httpx import AsyncClient, ASGITransport

from app.main import app
from app.models import Base
from app.deps import get_db

TEST_DATABASE_URL = "postgresql+asyncpg://user:password@localhost/testdb"

@pytest_asyncio.fixture(scope="session")
async def test_engine():
    engine = create_async_engine(TEST_DATABASE_URL, echo=False)
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.drop_all)
        await conn.run_sync(Base.metadata.create_all)
    yield engine
    async with engine.begin() as conn:
        await conn.run_sync(Base.metadata.drop_all)
    await engine.dispose()


@pytest_asyncio.fixture
async def db_session(test_engine):
    """Each test gets a session that rolls back after the test."""
    async with test_engine.connect() as connection:
        await connection.begin()
        session = AsyncSession(bind=connection, expire_on_commit=False)
        yield session
        await session.close()
        await connection.rollback()


@pytest_asyncio.fixture
async def client(db_session: AsyncSession):
    """HTTP client with the real DB replaced by the test session."""
    async def override_get_db():
        yield db_session

    app.dependency_overrides[get_db] = override_get_db
    
    async with AsyncClient(
        transport=ASGITransport(app=app),
        base_url="http://test",
    ) as ac:
        yield ac
    
    app.dependency_overrides.clear()
```

### Task 3: Async Tests
```python
# tests/test_users_async.py
import pytest
import pytest_asyncio

pytestmark = pytest.mark.asyncio  # mark all tests in this file as async

async def test_create_user(client):
    response = await client.post(
        "/auth/register",
        json={
            "email": "test@example.com",
            "password": "TestPass123!",
            "full_name": "Test User",
        },
    )
    assert response.status_code == 201
    data = response.json()
    assert data["email"] == "test@example.com"
    assert "hashed_password" not in data

async def test_register_duplicate_email(client):
    payload = {
        "email": "dup@example.com",
        "password": "TestPass123!",
        "full_name": "User",
    }
    # First registration
    r1 = await client.post("/auth/register", json=payload)
    assert r1.status_code == 201
    
    # Duplicate
    r2 = await client.post("/auth/register", json=payload)
    assert r2.status_code == 409

async def test_full_auth_flow(client):
    # Register
    await client.post(
        "/auth/register",
        json={"email": "flow@test.com", "password": "FlowPass123!", "full_name": "Flow"},
    )
    
    # Login
    login_resp = await client.post(
        "/auth/token",
        data={"username": "flow@test.com", "password": "FlowPass123!"},
    )
    assert login_resp.status_code == 200
    token = login_resp.json()["access_token"]
    
    # Access protected route
    me_resp = await client.get(
        "/users/me",
        headers={"Authorization": f"Bearer {token}"},
    )
    assert me_resp.status_code == 200
    assert me_resp.json()["email"] == "flow@test.com"
```

### Task 4: Overriding Dependencies
```python
# tests/test_with_overrides.py
import pytest
from fastapi.testclient import TestClient
from app.main import app
from app.auth.dependencies import get_current_user
from app.models import User, UserRole

def make_fake_user(user_id: int = 1, role: UserRole = UserRole.USER) -> User:
    user = User.__new__(User)
    user.id = user_id
    user.email = f"user{user_id}@test.com"
    user.full_name = "Test User"
    user.role = role
    user.is_active = True
    return user

@pytest.fixture
def regular_user_client():
    fake_user = make_fake_user(user_id=1, role=UserRole.USER)
    app.dependency_overrides[get_current_user] = lambda: fake_user
    client = TestClient(app)
    yield client
    app.dependency_overrides.clear()

@pytest.fixture
def admin_client():
    fake_admin = make_fake_user(user_id=99, role=UserRole.ADMIN)
    app.dependency_overrides[get_current_user] = lambda: fake_admin
    client = TestClient(app)
    yield client
    app.dependency_overrides.clear()

def test_regular_user_cannot_access_admin(regular_user_client):
    response = regular_user_client.get("/admin/users")
    assert response.status_code == 403

def test_admin_can_access_admin_routes(admin_client):
    response = admin_client.get("/admin/users")
    assert response.status_code == 200

def test_unauthenticated_cannot_access_protected(regular_user_client):
    # Remove the override to test without auth
    app.dependency_overrides.clear()
    client = TestClient(app)
    response = client.get("/users/me")
    assert response.status_code == 401
```

```python
# tests/test_error_cases.py
import pytest
from fastapi.testclient import TestClient
from app.main import app

client = TestClient(app)

def test_validation_error_structure():
    """Ensure validation errors have the expected structure."""
    response = client.post("/items/", json={"price": "not_a_number"})
    assert response.status_code == 422
    body = response.json()
    assert "detail" in body
    # Each error has loc, msg, type
    error = body["detail"][0]
    assert "loc" in error
    assert "msg" in error
    assert "type" in error

def test_method_not_allowed():
    response = client.put("/")  # if PUT / is not defined
    assert response.status_code == 405

def test_headers_in_response():
    response = client.get("/")
    assert "content-type" in response.headers
    assert "application/json" in response.headers["content-type"]

@pytest.mark.parametrize("item_id", [-1, 0, "abc", 2**63])
def test_invalid_item_ids(item_id):
    response = client.get(f"/items/{item_id}")
    # Either 422 (validation) or 404 (not found) — not 500
    assert response.status_code in (404, 422)
```
