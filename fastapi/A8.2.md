## Background

### Docker Compose for Development
Docker Compose lets you define and run multi-container applications. For a FastAPI app, you typically need: the app itself, PostgreSQL, Redis, a Celery worker, and optionally Flower (Celery monitor). Docker Compose starts all of these with one command and handles networking between them.

### Volumes
Docker containers are ephemeral — data is lost when they stop. Volumes persist data. Use named volumes for databases (PostgreSQL data, Redis data). Use bind mounts for source code in development — changes to local files immediately affect the running container without rebuilding.

### Environment Files
Sensitive configuration (passwords, secrets) should not be in the docker-compose.yml file (which is committed to git). Use `.env` files which Docker Compose reads automatically. Have separate `.env.development` and `.env.production` files. The `.env` file should be in `.gitignore`.

## Tasks

### Task 1: docker-compose.yml
```yaml
# docker-compose.yml
version: "3.9"

services:
  # ─────────────────────────────────────
  # Application
  # ─────────────────────────────────────
  app:
    build:
      context: .
      target: runtime       # use the runtime stage from multi-stage build
    ports:
      - "8000:8000"
    volumes:
      - ./app:/app/app      # bind mount for hot reload in development
    environment:
      - DATABASE_URL=postgresql+asyncpg://appuser:apppass@postgres:5432/appdb
      - REDIS_URL=redis://redis:6379/0
      - SECRET_KEY=${SECRET_KEY:-dev-secret-key}
      - ENV=development
    env_file:
      - .env
    depends_on:
      postgres:
        condition: service_healthy
      redis:
        condition: service_healthy
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload
    restart: unless-stopped
    networks:
      - app_network

  # ─────────────────────────────────────
  # PostgreSQL
  # ─────────────────────────────────────
  postgres:
    image: postgres:16-alpine
    ports:
      - "5432:5432"
    environment:
      POSTGRES_USER: appuser
      POSTGRES_PASSWORD: apppass
      POSTGRES_DB: appdb
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./scripts/init_db.sql:/docker-entrypoint-initdb.d/init.sql
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U appuser -d appdb"]
      interval: 10s
      timeout: 5s
      retries: 5
      start_period: 30s
    restart: unless-stopped
    networks:
      - app_network

  # ─────────────────────────────────────
  # Redis
  # ─────────────────────────────────────
  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    command: redis-server --appendonly yes --requirepass ${REDIS_PASSWORD:-redispass}
    healthcheck:
      test: ["CMD", "redis-cli", "--no-auth-warning", "-a", "${REDIS_PASSWORD:-redispass}", "ping"]
      interval: 10s
      timeout: 5s
      retries: 5
    restart: unless-stopped
    networks:
      - app_network

  # ─────────────────────────────────────
  # Celery Worker
  # ─────────────────────────────────────
  celery_worker:
    build:
      context: .
      target: runtime
    volumes:
      - ./app:/app/app
    environment:
      - DATABASE_URL=postgresql+asyncpg://appuser:apppass@postgres:5432/appdb
      - REDIS_URL=redis://:${REDIS_PASSWORD:-redispass}@redis:6379/0
    env_file:
      - .env
    command: >
      celery -A app.celery_app worker
      --loglevel=info
      --concurrency=4
      -Q default,email,reports
    depends_on:
      redis:
        condition: service_healthy
      postgres:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - app_network

  # ─────────────────────────────────────
  # Celery Beat (Scheduler)
  # ─────────────────────────────────────
  celery_beat:
    build:
      context: .
      target: runtime
    volumes:
      - ./app:/app/app
    environment:
      - REDIS_URL=redis://:${REDIS_PASSWORD:-redispass}@redis:6379/0
    env_file:
      - .env
    command: celery -A app.celery_app beat --loglevel=info
    depends_on:
      redis:
        condition: service_healthy
    restart: unless-stopped
    networks:
      - app_network

  # ─────────────────────────────────────
  # Flower (Celery UI)
  # ─────────────────────────────────────
  flower:
    build:
      context: .
      target: runtime
    ports:
      - "5555:5555"
    environment:
      - CELERY_BROKER_URL=redis://:${REDIS_PASSWORD:-redispass}@redis:6379/0
      - FLOWER_BASIC_AUTH=admin:${FLOWER_PASSWORD:-flowerpass}
    command: >
      celery -A app.celery_app flower
      --port=5555
      --basic_auth=admin:${FLOWER_PASSWORD:-flowerpass}
    depends_on:
      - redis
      - celery_worker
    networks:
      - app_network

volumes:
  postgres_data:
  redis_data:

networks:
  app_network:
    driver: bridge
```

services:
  app:
    command: uvicorn app.main:app --host 0.0.0.0 --port 8000 --reload --log-level debug

      - "5432:5432"  # expose for local tools (pgAdmin, DBeaver)

      PGADMIN_DEFAULT_PASSWORD: admin
    depends_on:
      - postgres
    networks:
      - app_network

### Task 4: Common Docker Compose Commands
```bash
# Start all services
docker compose up -d

# Start specific services
docker compose up -d app postgres redis

# View logs
docker compose logs -f app
docker compose logs -f celery_worker

# Stop all services
docker compose down

# Stop and remove volumes (DESTROYS DATA)
docker compose down -v

# Rebuild images
docker compose build --no-cache

# Run a command in a running container
docker compose exec app python -m alembic upgrade head
docker compose exec app pytest -v
docker compose exec postgres psql -U appuser -d appdb

# Scale workers
docker compose up -d --scale celery_worker=3

# Check service health
docker compose ps

# View resource usage
docker stats
```

up:
	docker compose up -d
	docker compose exec app alembic upgrade head
	@echo "App running at http://localhost:8000"

down:
	docker compose down

logs:
	docker compose logs -f

shell:
	docker compose exec app /bin/bash

migrate:
	docker compose exec app alembic upgrade head

test:
	docker compose exec app pytest -v tests/

seed:
	docker compose exec app python -m scripts.seed_db
