## Goal
Isolate units under test by mocking external dependencies, and generate realistic test data using factory_boy.

## Background

### Mocking vs Faking vs Stubbing
A mock records calls and lets you assert on them. A stub returns a pre-defined value without recording calls. A fake is a simplified working implementation (like an in-memory database). In practice, these terms are often used interchangeably. Python's `unittest.mock` provides `MagicMock` which combines all three: it records calls, can return pre-defined values, and can simulate exceptions.

### When to Mock
Mock external dependencies: HTTP APIs, email services, SMS gateways, payment processors, time (`datetime.now`), random numbers. Do NOT mock: your own business logic, SQLAlchemy models, or Pydantic validators — those should be tested directly. Over-mocking leads to tests that pass even when the code is broken.

### pytest-mock
`pytest-mock` provides the `mocker` fixture, which wraps `unittest.mock` with automatic cleanup. Any mock created via `mocker.patch` is automatically undone after the test, avoiding the common mistake of forgetting to call `patcher.stop()`.

### Task 1: MagicMock Basics
```python
# tests/test_mock_basics.py
from unittest.mock import MagicMock, AsyncMock, call

def test_mock_return_value():
    mock_service = MagicMock()
    mock_service.get_user.return_value = {"id": 1, "name": "Alice"}
    
    result = mock_service.get_user(1)
    
    assert result == {"id": 1, "name": "Alice"}
    mock_service.get_user.assert_called_once_with(1)

def test_mock_side_effect_list():
    """Return different values on successive calls."""
    mock_fetch = MagicMock()
    mock_fetch.side_effect = [
        {"page": 1, "data": [1, 2, 3]},
        {"page": 2, "data": [4, 5, 6]},
        {"page": 3, "data": []},
    ]
    
    assert mock_fetch()["page"] == 1
    assert mock_fetch()["page"] == 2
    assert mock_fetch()["data"] == []

def test_mock_side_effect_exception():
    """Raise an exception."""
    mock_db = MagicMock()
    mock_db.query.side_effect = ConnectionError("DB connection failed")
    
    import pytest
    with pytest.raises(ConnectionError, match="DB connection failed"):
        mock_db.query("SELECT 1")

async def test_async_mock():
    """AsyncMock for async functions."""
    mock_async_fn = AsyncMock()
    mock_async_fn.return_value = {"status": "ok"}
    
    result = await mock_async_fn("arg1")
    assert result == {"status": "ok"}
    mock_async_fn.assert_awaited_once_with("arg1")
```

### Task 2: patch Decorator and Context Manager
```python
# app/services/email.py
import smtplib

def send_email(to: str, subject: str, body: str) -> bool:
    with smtplib.SMTP("smtp.example.com", 587) as server:
        server.sendmail("noreply@app.com", to, f"Subject: {subject}\n\n{body}")
    return True

def get_current_time():
    from datetime import datetime
    return datetime.now()
```

```python
# tests/test_mocking.py
from unittest.mock import patch, MagicMock
from datetime import datetime
import pytest

# Patch as decorator — mock is passed as argument
@patch("app.services.email.smtplib.SMTP")
def test_send_email_success(mock_smtp_cls):
    mock_smtp = MagicMock()
    mock_smtp_cls.return_value.__enter__.return_value = mock_smtp
    
    from app.services.email import send_email
    result = send_email("user@test.com", "Hello", "World")
    
    assert result == True
    mock_smtp.sendmail.assert_called_once()
    args = mock_smtp.sendmail.call_args[0]
    assert args[1] == "user@test.com"

# Patch as context manager
def test_current_time_mocked():
    fake_time = datetime(2024, 1, 15, 12, 0, 0)
    
    with patch("app.services.email.datetime") as mock_dt:
        mock_dt.now.return_value = fake_time
        from app.services.email import get_current_time
        result = get_current_time()
        assert result == fake_time

# Patch multiple things at once
@patch("app.services.email.smtplib.SMTP")
@patch("app.services.email.get_current_time")
def test_with_multiple_patches(mock_time, mock_smtp):  # reversed order!
    mock_time.return_value = datetime(2024, 1, 1)
    mock_smtp.return_value.__enter__.return_value = MagicMock()
    
    from app.services.email import send_email, get_current_time
    assert send_email("a@b.com", "S", "B") == True
    assert get_current_time() == datetime(2024, 1, 1)
```

### Task 3: pytest-mock
```python
# tests/test_pytest_mock.py
# Install: uv add pytest-mock
import pytest
from app.services.email import send_email

def test_email_with_mocker(mocker):
    """mocker.patch automatically undoes the patch after the test."""
    mock_smtp = mocker.patch("app.services.email.smtplib.SMTP")
    mock_server = mocker.MagicMock()
    mock_smtp.return_value.__enter__.return_value = mock_server
    
    result = send_email("test@example.com", "Test", "Body")
    
    assert result == True
    assert mock_server.sendmail.call_count == 1

def test_spy_on_method(mocker):
    """Spy: observe real calls without replacing them."""
    import app.services.email as email_module
    spy = mocker.spy(email_module, "get_current_time")
    
    email_module.get_current_time()
    
    assert spy.call_count == 1
    assert spy.return_value is not None  # real function returned something

def test_mock_async_with_mocker(mocker):
    """mocker.AsyncMock for async functions."""
    mock_send = mocker.AsyncMock(return_value={"message_id": "abc123"})
    mocker.patch("app.services.notification.send_notification", mock_send)
    
    # Use the patched function
    import asyncio
    result = asyncio.run(mock_send("user@test.com", "Hello"))
    assert result["message_id"] == "abc123"
```

### Task 4: factory_boy for Test Data
```python
# tests/factories.py
# Install: uv add factory_boy

import factory
from factory import Faker, LazyAttribute, SubFactory
from factory.alchemy import SQLAlchemyModelFactory
from datetime import datetime, timezone
from app.models import User, Post, Tag, UserRole

class UserFactory(SQLAlchemyModelFactory):
    class Meta:
        model = User
        sqlalchemy_session_persistence = "flush"  # flush after create
    
    id = factory.Sequence(lambda n: n + 1)
    email = Faker("email")
    full_name = Faker("name")
    hashed_password = "$2b$12$fakehashedfakehashed"  # don't hash in tests
    role = UserRole.USER
    is_active = True
    created_at = factory.LazyFunction(lambda: datetime.now(timezone.utc))


class PostFactory(SQLAlchemyModelFactory):
    class Meta:
        model = Post
        sqlalchemy_session_persistence = "flush"
    
    id = factory.Sequence(lambda n: n + 1)
    title = Faker("sentence", nb_words=6)
    body = Faker("paragraphs", nb=3, as_text=True)
    published = False
    author = SubFactory(UserFactory)  # creates a user automatically


class AdminUserFactory(UserFactory):
    """Subclass factory for admin users."""
    role = UserRole.ADMIN
    email = LazyAttribute(lambda o: f"admin_{o.id}@example.com")
```

```python
# tests/test_with_factories.py
from tests.factories import UserFactory, PostFactory, AdminUserFactory

def test_user_factory_creates_valid_user():
    user = UserFactory.build()  # build() doesn't hit DB
    assert user.email
    assert "@" in user.email
    assert user.role == UserRole.USER

def test_post_factory_creates_author():
    post = PostFactory.build()
    assert post.author is not None
    assert post.author.email

def test_batch_creation():
    users = UserFactory.build_batch(10)
    assert len(users) == 10
    # All emails should be unique (Faker generates unique emails by default)
    emails = [u.email for u in users]
    assert len(set(emails)) == 10

def test_factory_with_overrides():
    user = UserFactory.build(email="custom@test.com", role=UserRole.ADMIN)
    assert user.email == "custom@test.com"
    assert user.role == UserRole.ADMIN
```

### Task 5: Mocking in FastAPI Services
```python
# app/services/notification.py
import httpx

async def send_push_notification(device_token: str, message: str) -> dict:
    async with httpx.AsyncClient() as client:
        response = await client.post(
            "https://fcm.googleapis.com/v1/messages:send",
            json={"token": device_token, "notification": {"body": message}},
            headers={"Authorization": "Bearer FCM_KEY"},
        )
        response.raise_for_status()
        return response.json()
```

```python
# tests/test_notification_service.py
import pytest
import httpx
from unittest.mock import AsyncMock, MagicMock

async def test_push_notification_success(mocker):
    mock_response = MagicMock()
    mock_response.json.return_value = {"name": "projects/myapp/messages/abc"}
    mock_response.raise_for_status = MagicMock()
    
    mock_client = AsyncMock()
    mock_client.post = AsyncMock(return_value=mock_response)
    mock_client.__aenter__ = AsyncMock(return_value=mock_client)
    mock_client.__aexit__ = AsyncMock(return_value=None)
    
    mocker.patch("app.services.notification.httpx.AsyncClient", return_value=mock_client)
    
    from app.services.notification import send_push_notification
    result = await send_push_notification("device_token_123", "Hello!")
    
    assert "name" in result
    mock_client.post.assert_awaited_once()
```
