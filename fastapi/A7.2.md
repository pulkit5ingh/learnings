## Background

### RabbitMQ Concepts
RabbitMQ is a message broker implementing AMQP. Producers send messages to exchanges. Exchanges route messages to queues based on routing rules. Consumers read from queues. This decouples producers from consumers: the producer doesn't know which service will handle the message.

### Exchange Types
- **Direct**: routes messages to queues where the binding key exactly matches the routing key. Use for task queues (send to specific service).
- **Topic**: routes using wildcard patterns (`order.created`, `order.*`, `#`). Use for flexible routing.
- **Fanout**: broadcasts to ALL bound queues regardless of routing key. Use for notifications where every subscriber needs the message.
- **Headers**: routes based on message headers (rarely used).

### Message Acknowledgment
By default, a message is removed from the queue when delivered to a consumer. With explicit acknowledgment (`ack`), the message stays in the queue until the consumer calls `message.ack()`. If the consumer crashes without acking, the message goes back to the queue and another consumer picks it up. Always use explicit ack for important tasks.

```python
# app/messaging/connection.py
import aio_pika
import os
from aio_pika.abc import AbstractRobustConnection, AbstractChannel

RABBITMQ_URL = os.getenv("RABBITMQ_URL", "amqp://guest:guest@localhost:5672/")

_connection: AbstractRobustConnection | None = None

async def get_connection() -> AbstractRobustConnection:
    global _connection
    if _connection is None or _connection.is_closed:
        _connection = await aio_pika.connect_robust(
            RABBITMQ_URL,
            reconnect_interval=5,  # retry connection every 5 seconds if lost
        )
    return _connection

async def get_channel() -> AbstractChannel:
    conn = await get_connection()
    # Create a new channel per use (channels are lightweight)
    return await conn.channel()

async def close_connection():
    global _connection
    if _connection and not _connection.is_closed:
        await _connection.close()
    _connection = None
```

### Task 2: Direct Exchange — Task Queue
```python
# app/messaging/producer.py
import aio_pika
import json
from app.messaging.connection import get_channel

TASK_EXCHANGE = "tasks"
TASK_QUEUE = "email_tasks"

async def publish_email_task(task_data: dict) -> None:
    channel = await get_channel()
    
    # Declare exchange (idempotent — safe to call multiple times)
    exchange = await channel.declare_exchange(
        TASK_EXCHANGE,
        aio_pika.ExchangeType.DIRECT,
        durable=True,  # survive broker restart
    )
    
    # Declare queue
    queue = await channel.declare_queue(
        TASK_QUEUE,
        durable=True,
        arguments={
            # Dead letter exchange for failed messages
            "x-dead-letter-exchange": "dead_letters",
            "x-dead-letter-routing-key": "dead.email",
            # Max retries via message TTL
            "x-message-ttl": 60000,  # 60 seconds TTL
        },
    )
    
    # Bind queue to exchange with routing key
    await queue.bind(exchange, routing_key="email")
    
    # Publish message
    await exchange.publish(
        aio_pika.Message(
            body=json.dumps(task_data).encode(),
            delivery_mode=aio_pika.DeliveryMode.PERSISTENT,  # survive broker restart
            content_type="application/json",
        ),
        routing_key="email",
    )
    
    await channel.close()
```

### Task 3: Consumer with Acknowledgment
```python
# app/messaging/consumer.py
import aio_pika
import json
import asyncio
import logging
from app.messaging.connection import get_channel

logger = logging.getLogger(__name__)

async def process_email_task(message: aio_pika.abc.AbstractIncomingMessage) -> None:
    async with message.process(requeue=False):  # auto ack/nack with context manager
        try:
            data = json.loads(message.body.decode())
            logger.info(f"Processing email task: {data}")
            
            # Simulate email sending
            await asyncio.sleep(0.5)
            
            if data.get("force_fail"):
                raise ValueError("Simulated email failure")
            
            logger.info(f"Email sent to {data.get('to')}")
            # Message is automatically acked when context exits without exception
            
        except ValueError as e:
            logger.error(f"Email task failed: {e}")
            # Context manager will nack the message (send to DLQ if configured)
            raise

async def start_email_consumer():
    channel = await get_channel()
    await channel.set_qos(prefetch_count=10)  # process max 10 messages at once
    
    queue = await channel.declare_queue("email_tasks", durable=True)
    
    await queue.consume(process_email_task)
    logger.info("Email consumer started. Waiting for messages...")
    
    # Keep running
    await asyncio.Future()

# Run consumer as a separate process or in FastAPI lifespan
if __name__ == "__main__":
    asyncio.run(start_email_consumer())
```

### Task 4: Topic Exchange — Event Bus
```python
# app/messaging/events.py
import aio_pika
import json
from app.messaging.connection import get_channel

EVENT_EXCHANGE = "domain_events"

async def publish_event(routing_key: str, payload: dict) -> None:
    """
    Publish a domain event.
    routing_key examples: "order.created", "order.updated", "user.registered"
    """
    channel = await get_channel()
    
    exchange = await channel.declare_exchange(
        EVENT_EXCHANGE,
        aio_pika.ExchangeType.TOPIC,
        durable=True,
    )
    
    await exchange.publish(
        aio_pika.Message(
            body=json.dumps(payload).encode(),
            content_type="application/json",
            headers={"event_type": routing_key},
        ),
        routing_key=routing_key,
    )
    await channel.close()

async def subscribe_to_events(
    queue_name: str,
    binding_patterns: list[str],
    handler,
) -> None:
    """
    Subscribe to events matching patterns.
    binding_patterns: ["order.*", "user.registered"]
    """
    channel = await get_channel()
    await channel.set_qos(prefetch_count=1)
    
    exchange = await channel.declare_exchange(
        EVENT_EXCHANGE,
        aio_pika.ExchangeType.TOPIC,
        durable=True,
    )
    
    queue = await channel.declare_queue(queue_name, durable=True)
    
    for pattern in binding_patterns:
        await queue.bind(exchange, routing_key=pattern)
    
    await queue.consume(handler)
    print(f"Subscribed {queue_name} to patterns: {binding_patterns}")
    await asyncio.Future()
```

```python
import aio_pika

        "dead_letter_queue",
```

### Task 6: Fanout for Broadcasting
```python
# app/messaging/notifications.py

async def broadcast_notification(message: str) -> None:
    """Send to ALL subscribers (fanout)."""
    channel = await get_channel()
    
    exchange = await channel.declare_exchange(
        "notifications",
        aio_pika.ExchangeType.FANOUT,
        durable=True,
    )
    
    # Fanout ignores routing_key — delivers to all bound queues
    await exchange.publish(
        aio_pika.Message(
            body=json.dumps({"message": message}).encode(),
        ),
        routing_key="",  # ignored for fanout
    )
    await channel.close()

async def setup_notification_subscriber(service_name: str, handler) -> None:
    """Each service creates its own exclusive queue."""
    channel = await get_channel()
    
    exchange = await channel.declare_exchange("notifications", aio_pika.ExchangeType.FANOUT)
    
    # Exclusive queue: auto-deleted when consumer disconnects
    queue = await channel.declare_queue(
        f"notifications.{service_name}",
        exclusive=True,  # only one consumer, auto-deleted on disconnect
    )
    
    await queue.bind(exchange)
    await queue.consume(handler)
```
