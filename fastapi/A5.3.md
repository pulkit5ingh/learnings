## Goal
Identify and fix common async performance problems: N+1 queries, missing connection pool configuration, and inefficient data access patterns.

## Background

### N+1 Query Problem
The N+1 problem occurs when you fetch N objects and then issue one more query per object to fetch related data — for N=100 objects, you issue 101 queries. This is catastrophic for performance. The fix is eager loading: fetch the main objects AND their relations in one or two queries using `joinedload` or `selectinload`. Always profile your queries in development with `echo=True` and count how many SQL statements are issued.

### Connection Pooling
Every database query needs a connection. Creating a new connection is expensive (TCP handshake, authentication, SSL). Connection pooling maintains a pool of open connections that are reused. SQLAlchemy's pool defaults are conservative. For a production FastAPI app, configure pool size based on your expected concurrency. With async, each request uses a connection only while waiting for I/O, so you need fewer connections than you might think.

### Batch vs Individual Operations
Inserting 1000 records one by one is slow because each insert is a separate roundtrip. Use bulk operations: `session.bulk_save_objects()`, `insert().values([...])`, or `COPY` for very large datasets. The same applies to updates and deletes.

### Task 1: Detecting N+1 Queries
```python
# Enable SQL logging to spot N+1 issues
import logging
logging.basicConfig()
logging.getLogger("sqlalchemy.engine").setLevel(logging.INFO)

# BAD — N+1 pattern
async def get_posts_with_authors_bad(session: AsyncSession) -> list[dict]:
    stmt = select(Post)
    result = await session.execute(stmt)
    posts = result.scalars().all()
    
    output = []
    for post in posts:
        # DANGER: each of these triggers a new SELECT query!
        # With 100 posts, this is 101 queries.
        author = await session.get(User, post.user_id)
        output.append({
            "title": post.title,
            "author": author.name if author else None,
        })
    return output

# GOOD — single extra query with selectinload
async def get_posts_with_authors_good(session: AsyncSession) -> list[dict]:
    stmt = (
        select(Post)
        .options(selectinload(Post.author))  # one extra SELECT IN (...) query
    )
    result = await session.execute(stmt)
    posts = result.scalars().all()
    
    return [
        {"title": post.title, "author": post.author.name if post.author else None}
        for post in posts
    ]
    # Total queries: 2 (one for posts, one for all authors in one go)
```

### Task 3: Batch Operations
```python
# app/batch_ops.py
from sqlalchemy import insert, update, delete
from sqlalchemy.ext.asyncio import AsyncSession
from app.models import Item
import time

async def bulk_insert_items(session: AsyncSession, items: list[dict]) -> int:
    """Insert many items in a single statement."""
    if not items:
        return 0
    
    stmt = insert(Item).values(items)
    result = await session.execute(stmt)
    return result.rowcount

async def bulk_update_prices(session: AsyncSession, price_map: dict[int, float]) -> int:
    """Update prices for multiple items. Uses CASE WHEN for a single query."""
    from sqlalchemy import case, literal
    
    if not price_map:
        return 0
    
    # Build CASE WHEN id = 1 THEN 9.99 WHEN id = 2 THEN 19.99 ...
    when_clauses = [(Item.id == id_, literal(price)) for id_, price in price_map.items()]
    stmt = (
        update(Item)
        .where(Item.id.in_(price_map.keys()))
        .values(price=case(*when_clauses))
    )
    result = await session.execute(stmt)
    return result.rowcount

# Benchmark: individual inserts vs bulk insert
async def benchmark_inserts(session: AsyncSession, n: int = 1000):
    # Individual inserts
    start = time.perf_counter()
    for i in range(n):
        session.add(Item(name=f"item_{i}", price=9.99))
    await session.flush()
    individual_time = time.perf_counter() - start
    await session.rollback()
    
    # Bulk insert
    start = time.perf_counter()
    items = [{"name": f"item_{i}", "price": 9.99} for i in range(n)]
    await session.execute(insert(Item).values(items))
    bulk_time = time.perf_counter() - start
    await session.rollback()
    
    print(f"Individual: {individual_time:.3f}s")
    print(f"Bulk: {bulk_time:.3f}s")
    print(f"Speedup: {individual_time / bulk_time:.1f}x")
```

### Task 4: Query Optimization Patterns
```python
# app/optimized_queries.py

async def count_efficiently(session: AsyncSession, model) -> int:
    """Count rows without loading all data."""
    from sqlalchemy import func, select
    result = await session.execute(select(func.count()).select_from(model))
    return result.scalar() or 0

async def exists_check(session: AsyncSession, model, **filters) -> bool:
    """Check existence without loading the object."""
    from sqlalchemy import select, exists as sa_exists
    stmt = select(sa_exists().where(*[
        getattr(model, k) == v for k, v in filters.items()
    ]))
    result = await session.execute(stmt)
    return bool(result.scalar())

async def paginated_query_with_total(
    session: AsyncSession,
    model,
    page: int = 1,
    page_size: int = 20,
) -> tuple[list, int]:
    """Fetch page AND total count in parallel."""
    import asyncio
    from sqlalchemy import select, func
    
    # Run both queries concurrently
    items_stmt = select(model).offset((page - 1) * page_size).limit(page_size)
    count_stmt = select(func.count()).select_from(model)
    
    items_result, count_result = await asyncio.gather(
        session.execute(items_stmt),
        session.execute(count_stmt),
    )
    
    return list(items_result.scalars().all()), count_result.scalar() or 0

async def chunked_processing(session: AsyncSession, model, chunk_size: int = 500):
    """Process large datasets in chunks to avoid loading everything into memory."""
    from sqlalchemy import select
    offset = 0
    while True:
        stmt = select(model).offset(offset).limit(chunk_size)
        result = await session.execute(stmt)
        chunk = result.scalars().all()
        
        if not chunk:
            break
        
        for item in chunk:
            # Process item
            pass
        
        offset += chunk_size
        # Expunge processed objects to free memory
        session.expunge_all()
```

### Task 6: Async Concurrency Pattern
```python
# Running independent DB queries concurrently
async def get_dashboard_data(session: AsyncSession, user_id: int) -> dict:
    """Fetch multiple independent queries concurrently."""
    import asyncio
    
    # These queries are independent — run them at the same time
    user_task = session.get(User, user_id)
    
    posts_stmt = select(func.count()).select_from(Post).where(Post.user_id == user_id)
    posts_task = session.execute(posts_stmt)
    
    comments_stmt = select(func.count()).select_from(Comment).where(Comment.user_id == user_id)
    comments_task = session.execute(comments_stmt)
    
    # Wait for all three concurrently
    user, posts_result, comments_result = await asyncio.gather(
        user_task,
        posts_task,
        comments_task,
    )
    
    return {
        "user": {"id": user.id, "name": user.full_name},
        "post_count": posts_result.scalar() or 0,
        "comment_count": comments_result.scalar() or 0,
    }
    # Total time = max(user query, posts query, comments query) — not sum!
```
