## Goal
Implement API key authentication alongside JWT, add OAuth2 scopes for fine-grained permissions, and configure multiple security schemes.

## Background

### API Keys
API keys are long random strings passed in a header (typically `X-API-Key`) or query parameter. They are simpler than JWT: no expiry, no signing — just a secret string stored in the database and checked on every request. They are ideal for server-to-server communication, CI/CD pipelines, and third-party integrations where storing and refreshing JWTs is impractical.

### OAuth2 Scopes
Scopes add fine-grained authorization to OAuth2. Instead of just "is this user authenticated?", scopes answer "is this token allowed to do this specific thing?" Examples: `items:read`, `items:write`, `admin:users`. The token endpoint returns tokens with a list of scopes, and protected routes check for the required scope.

### Multiple Security Schemes
A FastAPI app can support multiple security schemes simultaneously. Some routes might accept both JWT and API keys. FastAPI's OpenAPI spec supports this via `SecurityRequirement` — you can declare that a route requires scheme A OR scheme B.

```python
# app/models.py (additions)
import secrets
from sqlalchemy import String, Integer, Boolean, ForeignKey, DateTime
from sqlalchemy.orm import Mapped, mapped_column
from sqlalchemy.sql import func
from datetime import datetime

class APIKey(Base):
    __tablename__ = "api_keys"
    id: Mapped[int] = mapped_column(Integer, primary_key=True)
    key_hash: Mapped[str] = mapped_column(String(255), unique=True, nullable=False)
    name: Mapped[str] = mapped_column(String(100))  # e.g. "My CI Pipeline"
    user_id: Mapped[int] = mapped_column(ForeignKey("users.id", ondelete="CASCADE"))
    scopes: Mapped[str] = mapped_column(String(500), default="")  # space-separated
    is_active: Mapped[bool] = mapped_column(Boolean, default=True)
    last_used_at: Mapped[datetime | None] = mapped_column(DateTime, nullable=True)
    created_at: Mapped[datetime] = mapped_column(server_default=func.now())
    expires_at: Mapped[datetime | None] = mapped_column(DateTime, nullable=True)

def generate_api_key() -> tuple[str, str]:
    """Returns (raw_key, hashed_key). Store hash, give raw to user."""
    raw_key = f"sk_{secrets.token_urlsafe(32)}"
    import hashlib
    key_hash = hashlib.sha256(raw_key.encode()).hexdigest()
    return raw_key, key_hash
```

### Task 2: API Key Authentication Dependency
```python
# app/auth/api_key.py
import hashlib
from fastapi import Security, HTTPException, status
from fastapi.security import APIKeyHeader
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select, update
from datetime import datetime, timezone
from typing import Annotated

from app.models import APIKey, User
from app.deps import get_db

api_key_header = APIKeyHeader(name="X-API-Key", auto_error=False)

async def get_api_key_user(
    raw_key: Annotated[str | None, Security(api_key_header)],
    db: Annotated[AsyncSession, Depends(get_db)],
) -> User | None:
    if raw_key is None:
        return None
    
    key_hash = hashlib.sha256(raw_key.encode()).hexdigest()
    
    stmt = (
        select(APIKey)
        .where(APIKey.key_hash == key_hash)
        .where(APIKey.is_active == True)
    )
    result = await db.execute(stmt)
    api_key = result.scalars().first()
    
    if api_key is None:
        return None
    
    # Check expiry
    if api_key.expires_at and api_key.expires_at < datetime.now(timezone.utc):
        return None
    
    # Update last_used_at
    await db.execute(
        update(APIKey)
        .where(APIKey.id == api_key.id)
        .values(last_used_at=datetime.now(timezone.utc))
    )
    
    # Fetch the user
    user = await db.get(User, api_key.user_id)
    return user
```

### Task 3: Flexible Auth — JWT or API Key
```python
# app/auth/flexible.py
from fastapi import Depends, HTTPException, status
from fastapi.security import OAuth2PasswordBearer, APIKeyHeader
from typing import Annotated

from app.models import User

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/auth/token", auto_error=False)
api_key_header = APIKeyHeader(name="X-API-Key", auto_error=False)

async def get_current_user_flexible(
    token: Annotated[str | None, Depends(oauth2_scheme)],
    api_key: Annotated[str | None, Security(api_key_header)],
    db: Annotated[AsyncSession, Depends(get_db)],
) -> User:
    """Accept either a JWT Bearer token OR an X-API-Key header."""
    user = None
    
    if token:
        try:
            from app.auth.tokens import decode_access_token
            payload = decode_access_token(token)
            user_id = int(payload["sub"])
            user = await db.get(User, user_id)
        except (ValueError, KeyError):
            pass
    
    if user is None and api_key:
        import hashlib
        key_hash = hashlib.sha256(api_key.encode()).hexdigest()
        from sqlalchemy import select
        result = await db.execute(
            select(APIKey).where(APIKey.key_hash == key_hash, APIKey.is_active == True)
        )
        key_obj = result.scalars().first()
        if key_obj:
            user = await db.get(User, key_obj.user_id)
    
    if user is None:
        raise HTTPException(
            status_code=status.HTTP_401_UNAUTHORIZED,
            detail="Authentication required (Bearer token or X-API-Key)",
        )
    return user
```

### Task 4: Scopes
```python
# app/auth/scopes.py
from fastapi import Depends, HTTPException, status, Security
from fastapi.security import SecurityScopes
from typing import Annotated
from app.models import User

# Scopes the app supports
SCOPES = {
    "items:read": "Read items",
    "items:write": "Create and update items",
    "items:delete": "Delete items",
    "admin:users": "Manage users (admin only)",
    "profile:read": "Read own profile",
    "profile:write": "Update own profile",
}

async def get_current_user_with_scopes(
    security_scopes: SecurityScopes,  # injected by FastAPI — contains required scopes
    token: Annotated[str, Depends(OAuth2PasswordBearer(tokenUrl="/auth/token"))],
) -> dict:
    if security_scopes.scopes:
        authenticate_value = f'Bearer scope="{security_scopes.scope_str}"'
    else:
        authenticate_value = "Bearer"
    
    credentials_exception = HTTPException(
        status_code=status.HTTP_401_UNAUTHORIZED,
        detail="Could not validate credentials",
        headers={"WWW-Authenticate": authenticate_value},
    )
    
    try:
        from app.auth.tokens import decode_access_token
        payload = decode_access_token(token)
        token_scopes = payload.get("scopes", [])
    except ValueError:
        raise credentials_exception
    
    # Check that token has all required scopes
    for scope in security_scopes.scopes:
        if scope not in token_scopes:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"Missing required scope: {scope}",
                headers={"WWW-Authenticate": authenticate_value},
            )
    
    return {"user_id": int(payload["sub"]), "scopes": token_scopes}

# Usage in routes — requires specific scopes:
from fastapi import Security

@router.get("/items")
async def list_items(
    user: Annotated[dict, Security(get_current_user_with_scopes, scopes=["items:read"])]
):
    return []

@router.post("/items")
async def create_item(
    user: Annotated[dict, Security(get_current_user_with_scopes, scopes=["items:write"])]
):
    return {}

@router.delete("/items/{item_id}")
async def delete_item(
    item_id: int,
    user: Annotated[dict, Security(get_current_user_with_scopes, scopes=["items:delete"])]
):
    pass
```

### Task 5: Issuing Tokens with Scopes
```python
# Modified token endpoint with scope support
from fastapi.security import OAuth2PasswordRequestForm

@router.post("/token")
async def login_with_scopes(
    form_data: Annotated[OAuth2PasswordRequestForm, Depends()],
    db: DB,
):
    # OAuth2PasswordRequestForm has a .scopes attribute
    requested_scopes = form_data.scopes
    
    service = AuthService(db)
    user = await service.authenticate(form_data.username, form_data.password)
    
    # Validate that user is allowed to have these scopes
    allowed_scopes = get_allowed_scopes_for_user(user)
    granted_scopes = [s for s in requested_scopes if s in allowed_scopes]
    
    from app.auth.tokens import create_access_token
    access_token = create_access_token(
        subject=user.id,
        additional_claims={"scopes": granted_scopes},
    )
    
    return {"access_token": access_token, "token_type": "bearer", "scope": " ".join(granted_scopes)}

def get_allowed_scopes_for_user(user: User) -> list[str]:
    from app.models import UserRole
    if user.role == UserRole.ADMIN:
        return list(SCOPES.keys())
    elif user.role == UserRole.USER:
        return ["items:read", "items:write", "profile:read", "profile:write"]
    else:
        return ["items:read", "profile:read"]
```

### Task 6: API Key Management Routes
```python
# app/api_keys/router.py
from fastapi import APIRouter, Depends
from pydantic import BaseModel
from app.auth.dependencies import CurrentUser
from app.models import APIKey, generate_api_key

router = APIRouter(prefix="/api-keys", tags=["API Keys"])

class APIKeyCreate(BaseModel):
    name: str
    scopes: list[str] = []

class APIKeyResponse(BaseModel):
    id: int
    name: str
    scopes: str
    created_at: str
    # NOTE: raw key is only returned once at creation time

class APIKeyCreatedResponse(APIKeyResponse):
    key: str  # only shown once!

@router.post("/", response_model=APIKeyCreatedResponse, status_code=201)
async def create_api_key(data: APIKeyCreate, user: CurrentUser, db: DB):
    raw_key, key_hash = generate_api_key()
    api_key = APIKey(
        key_hash=key_hash,
        name=data.name,
        user_id=user.id,
        scopes=" ".join(data.scopes),
    )
    db.add(api_key)
    await db.flush()
    return {**api_key.__dict__, "key": raw_key}

@router.delete("/{key_id}", status_code=204)
async def revoke_api_key(key_id: int, user: CurrentUser, db: DB):
    api_key = await db.get(APIKey, key_id)
    if api_key and api_key.user_id == user.id:
        api_key.is_active = False
```
