# P7.1 — Order Processing System

**Phase:** 7 — Microservices & Messaging
**Topic:** RabbitMQ, aio-pika, saga pattern, microservice communication
**Type:** Project

---

## Goal

Build an order processing system using three microservices communicating via RabbitMQ, implementing the saga pattern for distributed transaction coordination.

## What to Build

- **order-service**: accepts orders, publishes `order.placed` events
- **inventory-service**: consumes `order.placed`, reserves stock, publishes `stock.reserved` or `stock.failed`
- **notification-service**: consumes both outcomes, sends confirmation/failure notification
- Saga: if stock reservation fails, order-service rolls back order to `cancelled`
- Dead letter queue for failed messages

### shared/events.py

```python
from pydantic import BaseModel
from datetime import datetime, UTC
from typing import Any
import uuid

class Event(BaseModel):
    event_id: str = str(uuid.uuid4())
    event_type: str
    timestamp: str = datetime.now(UTC).isoformat()
    correlation_id: str  # ties saga steps together
    payload: dict[str, Any]

class OrderPlaced(Event):
    event_type: str = "order.placed"

class StockReserved(Event):
    event_type: str = "stock.reserved"

class StockFailed(Event):
    event_type: str = "stock.failed"

class OrderCancelled(Event):
    event_type: str = "order.cancelled"
```

### shared/rabbitmq.py

```python
import aio_pika
import json
from shared.events import Event

EXCHANGE_NAME = "orders"
DLX_NAME = "orders.dlx"

async def get_connection(url: str) -> aio_pika.Connection:
    return await aio_pika.connect_robust(url)

async def setup_topology(channel: aio_pika.Channel) -> None:
    # Main exchange
    exchange = await channel.declare_exchange(
        EXCHANGE_NAME, aio_pika.ExchangeType.TOPIC, durable=True
    )
    # DLX
    dlx = await channel.declare_exchange(
        DLX_NAME, aio_pika.ExchangeType.FANOUT, durable=True
    )
    dlq = await channel.declare_queue("orders.dlq", durable=True)
    await dlq.bind(dlx)

    # Queues with DLX
    dlx_args = {
        "x-dead-letter-exchange": DLX_NAME,
        "x-message-ttl": 60000,  # 60s before DLQ
    }
    inventory_q = await channel.declare_queue("inventory.orders", durable=True, arguments=dlx_args)
    await inventory_q.bind(exchange, routing_key="order.*")

    notification_q = await channel.declare_queue("notification.events", durable=True, arguments=dlx_args)
    await notification_q.bind(exchange, routing_key="stock.*")
    await notification_q.bind(exchange, routing_key="order.cancelled")

    return exchange

async def publish_event(exchange: aio_pika.Exchange, event: Event, routing_key: str) -> None:
    message = aio_pika.Message(
        body=event.model_dump_json().encode(),
        content_type="application/json",
        delivery_mode=aio_pika.DeliveryMode.PERSISTENT,
        correlation_id=event.correlation_id,
        message_id=event.event_id,
    )
    await exchange.publish(message, routing_key=routing_key)
```

### order_service/consumer.py (saga reply handler)

```python
import json
import aio_pika
import structlog
from shared.rabbitmq import get_connection
from order_service.models import Order, OrderStatus

log = structlog.get_logger()

async def start_saga_reply_consumer(session_factory, rabbitmq_url: str) -> None:
    connection = await get_connection(rabbitmq_url)
    channel = await connection.channel()

    reply_queue = await channel.declare_queue("order.saga.replies", durable=True)
    exchange = await channel.get_exchange("orders")
    await reply_queue.bind(exchange, routing_key="stock.#")

    async with reply_queue.iterator() as queue_iter:
        async for message in queue_iter:
            async with message.process(requeue=False):
                try:
                    body = json.loads(message.body)
                    event_type = body["event_type"]
                    correlation_id = body["correlation_id"]

                    async with session_factory() as session:
                        order = await session.execute(
                            select(Order).where(Order.correlation_id == correlation_id)
                        )
                        order = order.scalar_one_or_none()
                        if not order:
                            log.warning("saga_reply_for_unknown_order", correlation_id=correlation_id)
                            continue

                        if event_type == "stock.reserved":
                            order.status = OrderStatus.confirmed
                        elif event_type == "stock.failed":
                            order.status = OrderStatus.cancelled
                            log.warning("order_cancelled_no_stock", order_id=order.id)

                        await session.commit()
                        log.info("saga_step_completed", event_type=event_type, order_id=order.id)

                except Exception as e:
                    log.error("saga_reply_processing_failed", error=str(e))
                    raise  # message goes to DLQ
```

### inventory_service/consumer.py

```python
import json
import aio_pika
import structlog
from shared.events import StockReserved, StockFailed
from shared.rabbitmq import publish_event

log = structlog.get_logger()

async def handle_order_placed(message: aio_pika.IncomingMessage, exchange, session_factory) -> None:
    async with message.process(requeue=True):
        body = json.loads(message.body)
        order_id = body["payload"]["order_id"]
        items = body["payload"]["items"]
        correlation_id = body["correlation_id"]

        log.info("reserving_stock", order_id=order_id, items=items)

        async with session_factory() as session:
            # Check and reserve stock atomically
            can_reserve = await check_and_reserve(session, items)

        if can_reserve:
            event = StockReserved(
                correlation_id=correlation_id,
                payload={"order_id": order_id, "reserved": True},
            )
            await publish_event(exchange, event, routing_key="stock.reserved")
            log.info("stock_reserved", order_id=order_id)
        else:
            event = StockFailed(
                correlation_id=correlation_id,
                payload={"order_id": order_id, "reason": "Insufficient stock"},
            )
            await publish_event(exchange, event, routing_key="stock.failed")
            log.warning("stock_reservation_failed", order_id=order_id)

async def check_and_reserve(session, items: list[dict]) -> bool:
    for item in items:
        from sqlalchemy import select, update
        from inventory_service.models import Product
        result = await session.execute(
            select(Product).where(
                Product.id == item["product_id"],
                Product.stock >= item["quantity"],
            ).with_for_update()
        )
        product = result.scalar_one_or_none()
        if not product:
            return False
        product.stock -= item["quantity"]
    await session.commit()
    return True
```
