## Goal
Handle file uploads securely, store files in S3, and stream large responses without loading everything into memory.

## Background

### UploadFile
FastAPI's `UploadFile` provides access to the uploaded file as a stream. It has a `file` attribute (a SpooledTemporaryFile) that you can read in chunks. Always validate file type (by content, not just extension) and size before processing. Never trust the client-provided filename — sanitize it.

### S3 Object Storage
For production, store files in S3 (or compatible: MinIO, Cloudflare R2) instead of the local filesystem. Local storage doesn't work with horizontal scaling (multiple instances each have a different filesystem). S3 is durable, scalable, and cheap. Use `boto3` for AWS S3. For uploads, you can either: (1) upload to your server then forward to S3, or (2) generate a presigned URL and let the client upload directly to S3 (faster, less server load).

### Streaming Responses
`StreamingResponse` sends data to the client in chunks. This is critical for large files: instead of loading a 1GB file into memory and then sending it, you read it in 64KB chunks. This keeps memory usage low regardless of file size.

### Task 1: Basic File Upload
```python
# app/files/router.py
from fastapi import APIRouter, UploadFile, File, HTTPException
from fastapi.responses import FileResponse
import aiofiles
import os
import uuid
import mimetypes

router = APIRouter(prefix="/files", tags=["Files"])

UPLOAD_DIR = "/tmp/uploads"
MAX_FILE_SIZE = 10 * 1024 * 1024  # 10 MB
ALLOWED_MIME_TYPES = {"image/jpeg", "image/png", "image/webp", "image/gif", "application/pdf"}

os.makedirs(UPLOAD_DIR, exist_ok=True)

@router.post("/upload", status_code=201)
async def upload_file(
    file: UploadFile = File(...),
):
    # Validate file size
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(
            status_code=413,
            detail=f"File too large. Max size: {MAX_FILE_SIZE / 1024 / 1024:.0f}MB",
        )
    
    # Validate MIME type by reading magic bytes (not trusting content-type header)
    import magic  # pip install python-magic
    detected_type = magic.from_buffer(content, mime=True)
    if detected_type not in ALLOWED_MIME_TYPES:
        raise HTTPException(
            status_code=400,
            detail=f"File type '{detected_type}' not allowed",
        )
    
    # Sanitize filename
    ext = detected_type.split("/")[-1]
    filename = f"{uuid.uuid4()}.{ext}"
    file_path = os.path.join(UPLOAD_DIR, filename)
    
    # Save to disk
    async with aiofiles.open(file_path, "wb") as f:
        await f.write(content)
    
    return {"filename": filename, "size": len(content), "content_type": detected_type}
```

### Task 2: Chunked Upload
```python
# Large file upload without loading entire file into memory
@router.post("/upload/chunked")
async def upload_chunked(file: UploadFile = File(...)):
    filename = f"{uuid.uuid4()}.bin"
    file_path = os.path.join(UPLOAD_DIR, filename)
    total_size = 0
    CHUNK_SIZE = 1024 * 1024  # 1MB chunks
    
    async with aiofiles.open(file_path, "wb") as out_file:
        while chunk := await file.read(CHUNK_SIZE):
            total_size += len(chunk)
            if total_size > MAX_FILE_SIZE:
                # Delete partial file
                await out_file.close()
                os.unlink(file_path)
                raise HTTPException(status_code=413, detail="File too large")
            await out_file.write(chunk)
    
    return {"filename": filename, "size": total_size}
```

```python
# app/storage/s3.py
import aioboto3
import os
import uuid
from typing import AsyncIterator
import hashlib

S3_BUCKET = os.getenv("S3_BUCKET", "my-app-uploads")
S3_REGION = os.getenv("AWS_REGION", "us-east-1")
CDN_URL = os.getenv("CDN_URL", f"https://{S3_BUCKET}.s3.{S3_REGION}.amazonaws.com")

session = aioboto3.Session()

async def upload_to_s3(
    content: bytes,
    filename: str,
    content_type: str,
    folder: str = "uploads",
) -> str:
    """Upload file to S3 and return the public URL."""
    key = f"{folder}/{uuid.uuid4()}/{filename}"
    
    async with session.client("s3", region_name=S3_REGION) as s3:
        await s3.put_object(
            Bucket=S3_BUCKET,
            Key=key,
            Body=content,
            ContentType=content_type,
            # Cache for 1 year (since we use UUID in the key)
            CacheControl="max-age=31536000",
        )
    
    return f"{CDN_URL}/{key}"

async def generate_presigned_upload_url(
    filename: str,
    content_type: str,
    max_size_bytes: int = 10 * 1024 * 1024,
) -> dict:
    """Generate a presigned URL for direct client-to-S3 upload."""
    key = f"uploads/{uuid.uuid4()}/{filename}"
    
    async with session.client("s3", region_name=S3_REGION) as s3:
        url = await s3.generate_presigned_post(
            Bucket=S3_BUCKET,
            Key=key,
            Fields={"Content-Type": content_type},
            Conditions=[
                {"Content-Type": content_type},
                ["content-length-range", 1, max_size_bytes],
            ],
            ExpiresIn=3600,  # URL valid for 1 hour
        )
    
    return {
        "upload_url": url["url"],
        "fields": url["fields"],
        "file_url": f"{CDN_URL}/{key}",
    }

async def delete_from_s3(key: str) -> None:
    async with session.client("s3") as s3:
        await s3.delete_object(Bucket=S3_BUCKET, Key=key)
```

### Task 4: Streaming File Download
```python
# app/files/router.py (continued)
from fastapi.responses import StreamingResponse
import aiofiles
import os

@router.get("/download/{filename}")
async def download_file(filename: str):
    file_path = os.path.join(UPLOAD_DIR, filename)
    
    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="File not found")
    
    # Detect content type
    content_type, _ = mimetypes.guess_type(file_path)
    content_type = content_type or "application/octet-stream"
    
    file_size = os.path.getsize(file_path)
    
    async def file_generator():
        async with aiofiles.open(file_path, "rb") as f:
            while chunk := await f.read(65536):  # 64KB chunks
                yield chunk
    
    return StreamingResponse(
        file_generator(),
        media_type=content_type,
        headers={
            "Content-Disposition": f'attachment; filename="{filename}"',
            "Content-Length": str(file_size),
            "Accept-Ranges": "bytes",
        },
    )
```

### Task 5: Video Streaming with Range Requests
```python
from fastapi import Request
from fastapi.responses import Response, StreamingResponse

@router.get("/video/{filename}")
async def stream_video(filename: str, request: Request):
    file_path = os.path.join(UPLOAD_DIR, filename)
    
    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="Not found")
    
    file_size = os.path.getsize(file_path)
    range_header = request.headers.get("Range")
    
    if range_header:
        # Parse Range: bytes=0-1023
        byte_range = range_header.replace("bytes=", "")
        start_str, end_str = byte_range.split("-")
        start = int(start_str) if start_str else 0
        end = int(end_str) if end_str else file_size - 1
        end = min(end, file_size - 1)
        length = end - start + 1
        
        async def range_generator():
            async with aiofiles.open(file_path, "rb") as f:
                await f.seek(start)
                remaining = length
                while remaining > 0:
                    chunk = await f.read(min(65536, remaining))
                    if not chunk:
                        break
                    remaining -= len(chunk)
                    yield chunk
        
        return StreamingResponse(
            range_generator(),
            status_code=206,
            media_type="video/mp4",
            headers={
                "Content-Range": f"bytes {start}-{end}/{file_size}",
                "Accept-Ranges": "bytes",
                "Content-Length": str(length),
            },
        )
    
    # Full file (no Range header)
    async def full_generator():
        async with aiofiles.open(file_path, "rb") as f:
            while chunk := await f.read(65536):
                yield chunk
    
    return StreamingResponse(
        full_generator(),
        media_type="video/mp4",
        headers={"Content-Length": str(file_size), "Accept-Ranges": "bytes"},
    )
```

### Task 6: Multiple File Upload
```python
from typing import List

@router.post("/upload/multiple")
async def upload_multiple(
    files: List[UploadFile] = File(...),
    max_files: int = 10,
):
    if len(files) > max_files:
        raise HTTPException(status_code=400, detail=f"Max {max_files} files allowed")
    
    results = []
    for file in files:
        content = await file.read()
        filename = f"{uuid.uuid4()}_{file.filename}"
        # Upload to S3 or save locally
        url = await upload_to_s3(content, filename, file.content_type or "application/octet-stream")
        results.append({"filename": filename, "url": url, "size": len(content)})
    
    return {"uploaded": results, "count": len(results)}
```
