## Goal
Cache expensive database queries and API responses in Redis using the cache-aside pattern with proper TTL and invalidation.

## Background

### Cache-Aside Pattern
In cache-aside (lazy loading), the application manages the cache: check cache first, if miss, fetch from DB and populate cache, if hit, return cached value. This is the most common pattern. On writes, you can either invalidate the cache (delete the key) or update it. Invalidation is simpler and safer (no stale writes), but means the next read after any write will be a cache miss.

### Redis Data Structures
Redis supports strings (most common for caching — store JSON), hashes (for objects with multiple fields you update individually), sets (for unique collections), sorted sets (for leaderboards/pagination), and lists (for queues). For caching API responses, storing serialized JSON as a string is almost always the right choice.

### TTL (Time-To-Live)
Every cached value should have a TTL — the duration after which it expires and is automatically deleted. Choose TTL based on how stale data is acceptable: user profiles (minutes to hours), product catalogs (minutes), real-time data (seconds), static configuration (hours to days). Too short = lots of cache misses. Too long = stale data.

### Task 2: Cache-Aside Helper
```python
# app/cache.py (continued)
from datetime import timedelta
import logging

logger = logging.getLogger(__name__)

class CacheService:
    def __init__(self, redis: aioredis.Redis, prefix: str = ""):
        self.redis = redis
        self.prefix = prefix

    def _key(self, key: str) -> str:
        return f"{self.prefix}:{key}" if self.prefix else key

    async def get(self, key: str) -> Any | None:
        raw = await self.redis.get(self._key(key))
        if raw is None:
            return None
        return json.loads(raw)

    async def set(self, key: str, value: Any, ttl: int | timedelta = 300) -> None:
        if isinstance(ttl, timedelta):
            ttl = int(ttl.total_seconds())
        await self.redis.setex(self._key(key), ttl, json.dumps(value))

    async def delete(self, key: str) -> int:
        return await self.redis.delete(self._key(key))

    async def delete_pattern(self, pattern: str) -> int:
        """Delete all keys matching a pattern. Use carefully in production."""
        keys = await self.redis.keys(f"{self.prefix}:{pattern}")
        if keys:
            return await self.redis.delete(*keys)
        return 0

    async def exists(self, key: str) -> bool:
        return bool(await self.redis.exists(self._key(key)))

    async def ttl(self, key: str) -> int:
        """Returns remaining TTL in seconds. -1 = no TTL, -2 = key doesn't exist."""
        return await self.redis.ttl(self._key(key))

    async def cache_aside(
        self,
        key: str,
        fetch_fn,
        ttl: int = 300,
    ) -> Any:
        """Cache-aside pattern: try cache, fall back to fetch_fn, store result."""
        cached = await self.get(key)
        if cached is not None:
            logger.debug(f"Cache HIT: {key}")
            return cached
        
        logger.debug(f"Cache MISS: {key}")
        result = await fetch_fn()
        
        if result is not None:
            await self.set(key, result, ttl)
        
        return result
```

### Task 3: Using Cache in FastAPI Routes
```python
# app/items/router.py
from fastapi import APIRouter, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from app.deps import get_db, get_cache
from app.cache import CacheService
from typing import Annotated

router = APIRouter(prefix="/items", tags=["items"])
DB = Annotated[AsyncSession, Depends(get_db)]
Cache = Annotated[CacheService, Depends(get_cache)]

@router.get("/{item_id}")
async def get_item(item_id: int, db: DB, cache: Cache):
    cache_key = f"item:{item_id}"
    
    # Cache-aside
    cached = await cache.get(cache_key)
    if cached:
        return {**cached, "from_cache": True}
    
    # DB fetch
    from sqlalchemy import select
    from app.models import Item
    result = await db.execute(select(Item).where(Item.id == item_id))
    item = result.scalars().first()
    if item is None:
        from fastapi import HTTPException
        raise HTTPException(status_code=404, detail="Not found")
    
    item_dict = {"id": item.id, "name": item.name, "price": item.price}
    await cache.set(cache_key, item_dict, ttl=300)  # 5 minutes
    
    return {**item_dict, "from_cache": False}

@router.patch("/{item_id}")
async def update_item(item_id: int, data: dict, db: DB, cache: Cache):
    # Update in DB
    from app.models import Item
    item = await db.get(Item, item_id)
    if item is None:
        from fastapi import HTTPException
        raise HTTPException(status_code=404, detail="Not found")
    
    for key, value in data.items():
        setattr(item, key, value)
    await db.flush()
    
    # Invalidate cache
    await cache.delete(f"item:{item_id}")
    
    return {"id": item.id, "name": item.name, "price": item.price}
```

### Task 4: Caching Decorator
```python
# app/cache.py (continued)
import functools
import hashlib
from typing import Callable

def cached(
    ttl: int = 300,
    key_prefix: str = "",
    key_builder: Callable | None = None,
):
    """
    Decorator for caching async function return values.
    
    Usage:
        @cached(ttl=60, key_prefix="users")
        async def get_user(user_id: int, db: AsyncSession) -> dict:
            ...
    """
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            from fastapi import Request
            
            # Build cache key from function name + args
            if key_builder:
                cache_key = key_builder(*args, **kwargs)
            else:
                key_parts = [key_prefix or func.__name__]
                # Serialize args to build key (skip db sessions, requests)
                for arg in args:
                    if not isinstance(arg, (AsyncSession, Request)):
                        key_parts.append(str(arg))
                for k, v in kwargs.items():
                    if not isinstance(v, (AsyncSession, Request)):
                        key_parts.append(f"{k}={v}")
                cache_key = ":".join(key_parts)
            
            # Get redis from somewhere (e.g., global or from context)
            redis = await get_redis()
            cache = CacheService(redis)
            
            cached_val = await cache.get(cache_key)
            if cached_val is not None:
                return cached_val
            
            result = await func(*args, **kwargs)
            if result is not None:
                await cache.set(cache_key, result, ttl)
            return result
        
        return wrapper
    return decorator
```

### Task 5: Batch/Multi-Key Operations
```python
# Efficiently get multiple keys at once
async def get_items_batch(item_ids: list[int], cache: CacheService, db: AsyncSession) -> list[dict]:
    # Try to get all from cache at once
    keys = [f"item:{id}" for id in item_ids]
    redis = await get_redis()
    raw_values = await redis.mget(*[f"items:{id}" for id in item_ids])
    
    result = {}
    missing_ids = []
    
    for item_id, raw in zip(item_ids, raw_values):
        if raw is not None:
            result[item_id] = json.loads(raw)
        else:
            missing_ids.append(item_id)
    
    # Fetch missing from DB in one query
    if missing_ids:
        from sqlalchemy import select
        from app.models import Item
        stmt = select(Item).where(Item.id.in_(missing_ids))
        db_result = await db.execute(stmt)
        items = db_result.scalars().all()
        
        # Store in cache and result
        pipe = redis.pipeline()
        for item in items:
            item_dict = {"id": item.id, "name": item.name, "price": item.price}
            result[item.id] = item_dict
            pipe.setex(f"items:{item.id}", 300, json.dumps(item_dict))
        await pipe.execute()
    
    return [result[id] for id in item_ids if id in result]
```

```python
# Cache warming — populate cache before first requests hit
async def warm_product_cache(db: AsyncSession, cache: CacheService):
    """Call this at startup for frequently accessed items."""
    from sqlalchemy import select
    from app.models import Item
    result = await db.execute(select(Item).order_by(Item.id).limit(100))
    items = result.scalars().all()
    
    redis = await get_redis()
    pipe = redis.pipeline()
    for item in items:
        key = f"items:{item.id}"
        value = json.dumps({"id": item.id, "name": item.name})
        pipe.setex(key, 600, value)
    await pipe.execute()
    
    print(f"Warmed cache with {len(items)} items")

# Simple cache hit/miss tracking
async def get_cache_stats(redis: aioredis.Redis) -> dict:
    info = await redis.info("stats")
    return {
        "hits": info.get("keyspace_hits", 0),
        "misses": info.get("keyspace_misses", 0),
        "hit_rate": (
            info["keyspace_hits"] / (info["keyspace_hits"] + info["keyspace_misses"])
            if (info["keyspace_hits"] + info["keyspace_misses"]) > 0
            else 0
        ),
    }
```
