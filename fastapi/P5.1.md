# P5.1 — Image Processing Service

**Phase:** 5 — Performance & Caching
**Topic:** Celery + Redis, background tasks, job status, webhooks
**Type:** Project

---

## Goal

Build an image upload and processing service where uploads trigger Celery tasks that resize/convert images asynchronously, with Redis-backed job status and optional webhook callbacks.

## What to Build

- `POST /images/upload` — accept image file, enqueue Celery task, return job ID
- Celery worker resizes image to multiple sizes (thumbnail, medium, large)
- `GET /jobs/{job_id}` — return job status (pending / processing / done / failed)
- Webhook: on completion, POST results to a caller-provided URL
- `GET /images/{image_id}` — return download URLs for processed sizes

### worker/tasks.py

```python
import json
import httpx
from pathlib import Path
from PIL import Image
from worker.celery_app import celery_app
from app.redis_client import get_sync_redis

SIZES = {
    "thumbnail": (150, 150),
    "medium": (800, 600),
    "large": (1920, 1080),
}
STORAGE_DIR = Path("storage/processed")
STORAGE_DIR.mkdir(parents=True, exist_ok=True)

def update_job_status(redis, job_id: str, status: str, result: dict | None = None):
    data = {"status": status}
    if result:
        data["result"] = result
    redis.setex(f"job:{job_id}", 3600, json.dumps(data))

@celery_app.task(bind=True, max_retries=3)
def process_image(self, job_id: str, source_path: str, webhook_url: str | None = None):
    redis = get_sync_redis()
    update_job_status(redis, job_id, "processing")

    try:
        img = Image.open(source_path)
        img = img.convert("RGB")
        output_paths = {}

        for size_name, dimensions in SIZES.items():
            resized = img.copy()
            resized.thumbnail(dimensions, Image.LANCZOS)
            out_path = STORAGE_DIR / f"{job_id}_{size_name}.jpg"
            resized.save(out_path, "JPEG", quality=85)
            output_paths[size_name] = str(out_path)

        result = {"job_id": job_id, "sizes": output_paths}
        update_job_status(redis, job_id, "done", result)

        if webhook_url:
            with httpx.Client(timeout=10) as client:
                client.post(webhook_url, json=result)

        return result

    except Exception as exc:
        update_job_status(redis, job_id, "failed", {"error": str(exc)})
        raise self.retry(exc=exc, countdown=2 ** self.request.retries)
```

### app/routers/images.py

```python
import uuid
import aiofiles
from pathlib import Path
from fastapi import APIRouter, UploadFile, File, HTTPException, Form
from app.redis_client import redis_client
from worker.tasks import process_image

UPLOAD_DIR = Path("storage/uploads")
UPLOAD_DIR.mkdir(parents=True, exist_ok=True)
ALLOWED_TYPES = {"image/jpeg", "image/png", "image/webp"}

router = APIRouter(prefix="/images", tags=["images"])

@router.post("/upload", status_code=202)
async def upload_image(
    file: UploadFile = File(...),
    webhook_url: str | None = Form(default=None),
):
    if file.content_type not in ALLOWED_TYPES:
        raise HTTPException(415, f"Unsupported file type: {file.content_type}")

    max_size = 10 * 1024 * 1024  # 10 MB
    contents = await file.read()
    if len(contents) > max_size:
        raise HTTPException(413, "File too large (max 10 MB)")

    job_id = str(uuid.uuid4())
    save_path = UPLOAD_DIR / f"{job_id}_{file.filename}"

    async with aiofiles.open(save_path, "wb") as f:
        await f.write(contents)

    # Enqueue Celery task
    process_image.apply_async(
        args=[job_id, str(save_path), webhook_url],
        task_id=job_id,
    )

    import json
    await redis_client.setex(f"job:{job_id}", 3600, json.dumps({"status": "pending"}))

    return {"job_id": job_id, "status": "pending"}
```

### app/routers/jobs.py

```python
import json
from fastapi import APIRouter, HTTPException
from app.redis_client import redis_client

router = APIRouter(prefix="/jobs", tags=["jobs"])

@router.get("/{job_id}")
async def get_job_status(job_id: str):
    raw = await redis_client.get(f"job:{job_id}")
    if not raw:
        raise HTTPException(404, "Job not found")
    return json.loads(raw)
```

- [ ] Webhook POST fires on completion if `webhook_url` provided

- [ ] Files over 10 MB rejected with 413
