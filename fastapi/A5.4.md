## Goal
Protect APIs from abuse with rate limiting and implement efficient pagination for large datasets.

## Background

### Rate Limiting
Rate limiting controls how many requests a client can make in a time window. Without it, a single client can overwhelm your server or exhaust your database. You can limit by IP address (for public APIs), by user ID (for authenticated APIs), or by API key. Redis is the standard backend for rate limit counters because it is fast, atomic (using INCR + EXPIRE), and can be shared across multiple server instances.

### Offset vs Cursor Pagination
Offset pagination (`LIMIT 20 OFFSET 100`) is simple but has problems at scale: the database must skip 100 rows before returning the next 20. For page 500 of a large dataset, this is slow. Cursor pagination uses a pointer (usually the ID or timestamp of the last item seen) and fetches rows AFTER that cursor. It is efficient even for large offsets, but you cannot jump to an arbitrary page. Use offset for small datasets or admin UIs; use cursor for infinite scroll and high-performance APIs.

### Link Headers
The RFC 5988 `Link` header is the standard way to communicate pagination links. It looks like: `Link: <https://api.example.com/items?cursor=xyz>; rel="next"`. Many API clients and HTTP libraries parse this automatically. Include `rel="next"`, `rel="prev"`, `rel="first"`, and `rel="last"` where applicable.

```python
# app/rate_limit.py
from slowapi import Limiter
from fastapi import Request

def get_user_or_ip(request: Request) -> str:
    """Rate limit by user ID if authenticated, otherwise by IP."""
    # If there's a user from JWT, use their ID
    user = getattr(request.state, "user", None)
    if user:
        return f"user:{user.id}"
    # Fall back to IP
    forwarded_for = request.headers.get("X-Forwarded-For")
    if forwarded_for:
        return forwarded_for.split(",")[0].strip()
    return request.client.host if request.client else "unknown"

limiter = Limiter(key_func=get_user_or_ip)

# Per-user limits: free users get 100/hour, pro users get 1000/hour
def get_user_limit(request: Request) -> str:
    user = getattr(request.state, "user", None)
    if user and user.plan == "pro":
        return "1000/hour"
    return "100/hour"
```

### Task 3: Manual Rate Limiting with Redis
```python
# app/rate_limiter.py
import redis.asyncio as aioredis
import time
from fastapi import HTTPException, status

class SlidingWindowRateLimiter:
    def __init__(self, redis: aioredis.Redis):
        self.redis = redis

    async def check(
        self,
        key: str,
        limit: int,
        window_seconds: int,
    ) -> tuple[bool, dict]:
        """
        Sliding window rate limiter using Redis sorted sets.
        Returns (allowed: bool, info: dict)
        """
        now = time.time()
        window_start = now - window_seconds
        
        pipe = self.redis.pipeline()
        # Remove entries older than the window
        pipe.zremrangebyscore(key, 0, window_start)
        # Count remaining entries (current window)
        pipe.zcard(key)
        # Add current request
        pipe.zadd(key, {str(now): now})
        # Set expiry to window duration
        pipe.expire(key, window_seconds)
        
        results = await pipe.execute()
        count = results[1]
        
        remaining = max(0, limit - count - 1)
        reset_time = int(now + window_seconds)
        
        if count >= limit:
            return False, {
                "limit": limit,
                "remaining": 0,
                "reset": reset_time,
            }
        
        return True, {
            "limit": limit,
            "remaining": remaining,
            "reset": reset_time,
        }

async def rate_limit_middleware(request: Request, call_next):
    """Middleware that adds rate limit headers to all responses."""
    redis = await get_redis()
    limiter = SlidingWindowRateLimiter(redis)
    
    key = f"ratelimit:{request.client.host}:{request.url.path}"
    allowed, info = await limiter.check(key, limit=100, window_seconds=60)
    
    if not allowed:
        raise HTTPException(
            status_code=status.HTTP_429_TOO_MANY_REQUESTS,
            detail="Rate limit exceeded",
            headers={
                "X-RateLimit-Limit": str(info["limit"]),
                "X-RateLimit-Remaining": str(info["remaining"]),
                "X-RateLimit-Reset": str(info["reset"]),
                "Retry-After": str(info["reset"] - int(time.time())),
            },
        )
    
    response = await call_next(request)
    response.headers["X-RateLimit-Limit"] = str(info["limit"])
    response.headers["X-RateLimit-Remaining"] = str(info["remaining"])
    response.headers["X-RateLimit-Reset"] = str(info["reset"])
    return response
```

```python
# app/pagination.py
from pydantic import BaseModel, Field
from typing import TypeVar, Generic
from fastapi import Query
from typing import Annotated
import math

T = TypeVar("T")

class PaginatedResponse(BaseModel, Generic[T]):
    items: list[T]
    total: int
    page: int
    page_size: int
    pages: int
    has_next: bool
    has_prev: bool

class OffsetPagination(BaseModel):
    page: int = Field(ge=1, default=1)
    page_size: int = Field(ge=1, le=100, default=20)
    
    @property
    def offset(self) -> int:
        return (self.page - 1) * self.page_size
    
    def paginate(self, items: list[T], total: int) -> PaginatedResponse[T]:
        pages = math.ceil(total / self.page_size) if total > 0 else 1
        return PaginatedResponse(
            items=items,
            total=total,
            page=self.page,
            page_size=self.page_size,
            pages=pages,
            has_next=self.page < pages,
            has_prev=self.page > 1,
        )

# Usage in route:
@router.get("/items", response_model=PaginatedResponse[ItemResponse])
async def list_items(
    page: Annotated[int, Query(ge=1)] = 1,
    page_size: Annotated[int, Query(ge=1, le=100)] = 20,
    db: DB = Depends(get_db),
):
    pagination = OffsetPagination(page=page, page_size=page_size)
    items, total = await get_items_page(db, pagination.offset, pagination.page_size)
    return pagination.paginate(items, total)
```

### Task 5: Cursor-Based Pagination
```python
# app/cursor_pagination.py
from pydantic import BaseModel
from fastapi import Query
from typing import Annotated
import base64
import json

class CursorPage(BaseModel):
    items: list
    next_cursor: str | None
    prev_cursor: str | None
    has_more: bool

def encode_cursor(value: dict) -> str:
    return base64.b64encode(json.dumps(value).encode()).decode()

def decode_cursor(cursor: str) -> dict:
    return json.loads(base64.b64decode(cursor.encode()).decode())

@router.get("/items/cursor")
async def list_items_cursor(
    cursor: Annotated[str | None, Query()] = None,
    limit: Annotated[int, Query(ge=1, le=100)] = 20,
    db: DB = Depends(get_db),
):
    from sqlalchemy import select
    from app.models import Item
    
    stmt = select(Item).order_by(Item.id.asc()).limit(limit + 1)  # fetch +1 to detect next
    
    if cursor:
        cursor_data = decode_cursor(cursor)
        stmt = stmt.where(Item.id > cursor_data["last_id"])
    
    result = await db.execute(stmt)
    items = result.scalars().all()
    
    has_more = len(items) > limit
    if has_more:
        items = items[:limit]  # remove the extra item
    
    next_cursor = encode_cursor({"last_id": items[-1].id}) if has_more and items else None
    
    return CursorPage(
        items=[{"id": i.id, "name": i.name} for i in items],
        next_cursor=next_cursor,
        prev_cursor=None,  # cursor pagination typically doesn't support prev
        has_more=has_more,
    )
```

### Task 6: Link Headers
```python
# app/link_headers.py
from fastapi import Request, Response

def add_pagination_link_headers(
    response: Response,
    request: Request,
    page: int,
    total_pages: int,
) -> None:
    """Add RFC 5988 Link headers for pagination."""
    base_url = str(request.url).split("?")[0]
    query_params = dict(request.query_params)
    
    links = []
    
    if page > 1:
        query_params["page"] = str(page - 1)
        links.append(f'<{base_url}?{"&".join(f"{k}={v}" for k, v in query_params.items())}>; rel="prev"')
    
    if page < total_pages:
        query_params["page"] = str(page + 1)
        links.append(f'<{base_url}?{"&".join(f"{k}={v}" for k, v in query_params.items())}>; rel="next"')
    
    query_params["page"] = "1"
    links.append(f'<{base_url}?{"&".join(f"{k}={v}" for k, v in query_params.items())}>; rel="first"')
    
    query_params["page"] = str(total_pages)
    links.append(f'<{base_url}?{"&".join(f"{k}={v}" for k, v in query_params.items())}>; rel="last"')
    
    if links:
        response.headers["Link"] = ", ".join(links)

# Usage:
@router.get("/items")
async def list_items(
    request: Request,
    response: Response,
    page: int = 1,
    db: DB = Depends(get_db),
):
    items, total = await get_items_page(db, page)
    total_pages = math.ceil(total / 20)
    add_pagination_link_headers(response, request, page, total_pages)
    return {"items": items, "total": total}
```
