```typescript
  set: jest.fn(),

describe('CartService (TDD)', () => {
  let service: CartService;

  const sessionId = 'test-session-123';
  const item1 = { productId: 'prod-1', name: 'Widget', price: 9.99, quantity: 1 };
  const item2 = { productId: 'prod-2', name: 'Gadget', price: 24.99, quantity: 2 };

  beforeEach(async () => {
    const module: TestingModule = await Test.createTestingModule({
      providers: [
        CartService,
        { provide: CACHE_MANAGER, useValue: mockCacheManager },

    service = module.get<CartService>(CartService);
    jest.clearAllMocks();
    mockCacheManager.get.mockResolvedValue(null); // empty cart by default

    it('should return empty cart for new session', async () => {
      const cart = await service.getCart(sessionId);
      expect(cart.items).toEqual([]);
      expect(cart.total).toBe(0);
      expect(cart.sessionId).toBe(sessionId);
    });

    it('should return existing cart from cache', async () => {
      const cached = { sessionId, items: [item1], total: 9.99, discount: null };
      mockCacheManager.get.mockResolvedValue(cached);
      const cart = await service.getCart(sessionId);
      expect(cart.items).toHaveLength(1);

  });

  describe('addItem', () => {

      const cart = await service.addItem(sessionId, item1);
      expect(cart.items).toHaveLength(1);
      expect(cart.items[0].productId).toBe('prod-1');
    });

    it('should calculate total after adding item', async () => {
      const cart = await service.addItem(sessionId, item1);
      expect(cart.total).toBeCloseTo(9.99);
    });

    it('should increment quantity when adding duplicate item', async () => {
      mockCacheManager.get
        .mockResolvedValueOnce(null) // first getCart
        .mockResolvedValueOnce({     // second getCart (inside addItem)
          sessionId, items: [{ ...item1, quantity: 1 }], total: 9.99, discount: null,
        });

      const cart = await service.addItem(sessionId, { ...item1, quantity: 2 });
      expect(cart.items[0].quantity).toBe(3);
      expect(cart.total).toBeCloseTo(29.97);

      await service.addItem(sessionId, item1);
      expect(mockCacheManager.set).toHaveBeenCalledWith(
        `cart:${sessionId}`,
        expect.objectContaining({ items: expect.any(Array) }),

    });
  });

  describe('removeItem', () => {

      mockCacheManager.get.mockResolvedValue({
        sessionId, items: [item1, item2], total: 59.97, discount: null,
      });
    });

    it('should remove item from cart', async () => {
      const cart = await service.removeItem(sessionId, 'prod-1');
      expect(cart.items.find((i) => i.productId === 'prod-1')).toBeUndefined();

    it('should recalculate total after removing item', async () => {

    it('should return cart unchanged if item not found', async () => {
      const cart = await service.removeItem(sessionId, 'non-existent');
      expect(cart.items).toHaveLength(2);

        sessionId, items: [{ ...item1, quantity: 1 }], total: 9.99, discount: null,
      });

    it('should update item quantity', async () => {
      const cart = await service.updateQuantity(sessionId, 'prod-1', 5);
      expect(cart.items[0].quantity).toBe(5);
      expect(cart.total).toBeCloseTo(49.95);

      expect(cart.items).toHaveLength(0);
    });

    it('should throw when quantity is negative', async () => {

    it('should delete cart from cache', async () => {
      await service.clearCart(sessionId);
      expect(mockCacheManager.del).toHaveBeenCalledWith(`cart:${sessionId}`);
```

### `src/cart/cart.service.ts`
```typescript
import { Injectable, Inject, BadRequestException } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';
import { Cart, CartItem } from './cart.entity';

const CART_TTL = 86_400_000; // 24 hours in ms

@Injectable()
export class CartService {
  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  async getCart(sessionId: string): Promise<Cart> {
    const cached = await this.cacheManager.get<Cart>(`cart:${sessionId}`);
    if (cached) return cached;
    return { sessionId, items: [], total: 0, discount: null };
  }

  async addItem(sessionId: string, item: CartItem): Promise<Cart> {
    const cart = await this.getCart(sessionId);
    const existingIndex = cart.items.findIndex((i) => i.productId === item.productId);

    if (existingIndex >= 0) {
      cart.items[existingIndex].quantity += item.quantity;
    } else {
      cart.items.push({ ...item });
    }

    cart.total = this.calculateTotal(cart.items);
    await this.persist(sessionId, cart);
    return cart;
  }

  async removeItem(sessionId: string, productId: string): Promise<Cart> {
    const cart = await this.getCart(sessionId);
    cart.items = cart.items.filter((i) => i.productId !== productId);
    cart.total = this.calculateTotal(cart.items);
    await this.persist(sessionId, cart);
    return cart;
  }

  async updateQuantity(sessionId: string, productId: string, quantity: number): Promise<Cart> {
    if (quantity < 0) throw new BadRequestException('Quantity cannot be negative');
    if (quantity === 0) return this.removeItem(sessionId, productId);

    const cart = await this.getCart(sessionId);
    const item = cart.items.find((i) => i.productId === productId);
    if (item) {
      item.quantity = quantity;
      cart.total = this.calculateTotal(cart.items);
      await this.persist(sessionId, cart);
    }
    return cart;
  }

  async clearCart(sessionId: string): Promise<void> {
    await this.cacheManager.del(`cart:${sessionId}`);
  }

  private calculateTotal(items: CartItem[]): number {
    return parseFloat(
      items.reduce((sum, item) => sum + item.price * item.quantity, 0).toFixed(2),
    );
  }

  private async persist(sessionId: string, cart: Cart): Promise<void> {
    await this.cacheManager.set(`cart:${sessionId}`, cart, CART_TTL);
  }
}
```

### `src/cart/discount.service.ts`
```typescript
import { Injectable, BadRequestException } from '@nestjs/common';

interface DiscountDefinition {
  type: 'percentage' | 'fixed';
  value: number;
  expiresAt: Date | null;
}

const DISCOUNT_CODES: Record<string, DiscountDefinition> = {
  SAVE10:    { type: 'percentage', value: 10, expiresAt: new Date('2099-12-31') },
  FLAT5:     { type: 'fixed',      value: 5,  expiresAt: new Date('2099-12-31') },
  EXPIRED20: { type: 'percentage', value: 20, expiresAt: new Date('2020-01-01') },
};

@Injectable()
export class DiscountService {
  validate(code: string): Pick<DiscountDefinition, 'type' | 'value'> {
    const discount = DISCOUNT_CODES[code.toUpperCase()];
    if (!discount) throw new BadRequestException('Invalid discount code');
    if (discount.expiresAt && discount.expiresAt < new Date()) {
      throw new BadRequestException('Discount code has expired');
    }
    return { type: discount.type, value: discount.value };
  }

  applyDiscount(total: number, discount: Pick<DiscountDefinition, 'type' | 'value'>): number {
    const reduced =
      discount.type === 'percentage'
        ? total * (1 - discount.value / 100)
        : total - discount.value;
    return Math.max(0, parseFloat(reduced.toFixed(2)));
  }
}
```

### `src/cart/cart.controller.ts`
```typescript
import { Controller, Get, Post, Delete, Patch, Param, Body, HttpCode, HttpStatus } from '@nestjs/common';
import { CartService } from './cart.service';
import { DiscountService } from './discount.service';
import { AddItemDto } from './dto/add-item.dto';

@Controller('cart')
export class CartController {
  constructor(
    private readonly cartService: CartService,
    private readonly discountService: DiscountService,
  ) {}

  @Get(':sessionId')
  getCart(@Param('sessionId') sessionId: string) {
    return this.cartService.getCart(sessionId);
  }

  @Post(':sessionId/items')
  @HttpCode(HttpStatus.OK)
  addItem(@Param('sessionId') sessionId: string, @Body() dto: AddItemDto) {
    return this.cartService.addItem(sessionId, dto);
  }

  @Delete(':sessionId/items/:productId')
  removeItem(@Param('sessionId') sessionId: string, @Param('productId') productId: string) {
    return this.cartService.removeItem(sessionId, productId);
  }

  @Patch(':sessionId/items/:productId/quantity')
  updateQuantity(
    @Param('sessionId') sessionId: string,
    @Param('productId') productId: string,
    @Body('quantity') quantity: number,
  ) {
    return this.cartService.updateQuantity(sessionId, productId, quantity);
  }

  @Delete(':sessionId')
  @HttpCode(HttpStatus.NO_CONTENT)
  clearCart(@Param('sessionId') sessionId: string) {
    return this.cartService.clearCart(sessionId);
  }

  @Post(':sessionId/discount')
  @HttpCode(HttpStatus.OK)
  applyDiscount(@Param('sessionId') sessionId: string, @Body('code') code: string) {
    const discount = this.discountService.validate(code);
    return { discount, message: `Code ${code} validated successfully` };
  }
}
```
