# A4.2 — Interceptors

**Phase:** 4 — Pipes, Interceptors & Filters  
**Topic:** NestInterceptor, logging interceptor, response transform interceptor, caching interceptor, timeout  
**Type:** Assignment

---

## Goal
Write interceptors that wrap the request/response pipeline to add logging, standardise response shapes, implement simple caching, and enforce timeouts.

## Background

Interceptors implement `NestInterceptor` with an `intercept(context, next)` method. `next.handle()` returns an `Observable<any>` — the eventual response. Because interceptors use RxJS, you can:
- Run code **before** the handler: before calling `next.handle()`.
- Run code **after** the handler: by piping operators (`map`, `tap`, `catchError`, `timeout`) onto the returned observable.
- Short-circuit the handler: return an observable without calling `next.handle()` (used for caching).

The key insight is that interceptors see the *response*, not just the request. This makes them the right place for response transformation, logging elapsed time, caching, and adding tracing headers.

**`CallHandler.handle()`** can only be called once. If you cache responses, you call `handle()` the first time and return a cached value directly on subsequent requests.

## Tasks

### 1. Logging interceptor

**`src/common/interceptors/logging.interceptor.ts`**
```typescript
import {
  CallHandler,
  ExecutionContext,
  Injectable,
  Logger,
  NestInterceptor,
} from '@nestjs/common';
import { Observable, tap } from 'rxjs';
import { Request } from 'express';

@Injectable()
export class LoggingInterceptor implements NestInterceptor {
  private readonly logger = new Logger(LoggingInterceptor.name);

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest<Request>();
    const { method, url, ip } = request;
    const start = Date.now();

    this.logger.log(`→ ${method} ${url} [${ip}]`);

    return next.handle().pipe(
      tap({
        next: () => {
          const elapsed = Date.now() - start;
          const response = context.switchToHttp().getResponse();
          this.logger.log(
            `← ${method} ${url} ${response.statusCode} [${elapsed}ms]`,
          );
        },
        error: (err) => {
          const elapsed = Date.now() - start;
          this.logger.error(
            `✗ ${method} ${url} ERROR [${elapsed}ms]: ${err.message}`,
          );
        },
      }),
    );
  }
}
```

Register globally:
```typescript
// In AppModule providers:
{ provide: APP_INTERCEPTOR, useClass: LoggingInterceptor }
```

### 2. Response transform interceptor

Standardise all responses to `{ data: T, meta: { timestamp, path } }`:

**`src/common/interceptors/transform.interceptor.ts`**
```typescript
import {
  CallHandler,
  ExecutionContext,
  Injectable,
  NestInterceptor,
} from '@nestjs/common';
import { Observable, map } from 'rxjs';
import { Request } from 'express';

export interface StandardResponse<T> {
  data: T;
  meta: {
    timestamp: string;
    path: string;
  };
}

@Injectable()
export class TransformInterceptor<T>
  implements NestInterceptor<T, StandardResponse<T>>
{
  intercept(
    context: ExecutionContext,
    next: CallHandler<T>,
  ): Observable<StandardResponse<T>> {
    const request = context.switchToHttp().getRequest<Request>();

    return next.handle().pipe(
      map(data => ({
        data,
        meta: {
          timestamp: new Date().toISOString(),
          path: request.url,
        },
      })),
    );
  }
}
```

Response example:
```json
{
  "data": { "id": 1, "name": "Alice" },
  "meta": {
    "timestamp": "2024-01-15T10:30:00.000Z",
    "path": "/users/1"
  }
}
```

### 3. Timeout interceptor

**`src/common/interceptors/timeout.interceptor.ts`**
```typescript
import {
  CallHandler,
  ExecutionContext,
  Injectable,
  NestInterceptor,
  RequestTimeoutException,
} from '@nestjs/common';
import { Observable, throwError, TimeoutError } from 'rxjs';
import { catchError, timeout } from 'rxjs/operators';

@Injectable()
export class TimeoutInterceptor implements NestInterceptor {
  constructor(private readonly milliseconds: number = 5000) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    return next.handle().pipe(
      timeout(this.milliseconds),
      catchError(err => {
        if (err instanceof TimeoutError) {
          return throwError(() => new RequestTimeoutException(
            `Request exceeded ${this.milliseconds}ms timeout`,
          ));
        }
        return throwError(() => err);
      }),
    );
  }
}
```

Usage per route:
```typescript
@Get('slow')
@UseInterceptors(new TimeoutInterceptor(2000))  // 2 second timeout for this route
async slowEndpoint() {
  await new Promise(r => setTimeout(r, 3000));  // will timeout
}
```

### 4. Caching interceptor (manual)

**`src/common/interceptors/cache.interceptor.ts`**
```typescript
import {
  CallHandler,
  ExecutionContext,
  Injectable,
  NestInterceptor,
} from '@nestjs/common';
import { Observable, of, tap } from 'rxjs';
import { Request } from 'express';

@Injectable()
export class SimpleCacheInterceptor implements NestInterceptor {
  private readonly store = new Map<string, { value: any; expiresAt: number }>();
  private readonly ttlMs: number;

  constructor(ttlMs = 60_000) {
    this.ttlMs = ttlMs;
  }

  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    const request = context.switchToHttp().getRequest<Request>();

    // Only cache GET requests
    if (request.method !== 'GET') return next.handle();

    const key = request.url;
    const cached = this.store.get(key);

    if (cached && cached.expiresAt > Date.now()) {
      return of(cached.value);  // short-circuit — no database hit
    }

    return next.handle().pipe(
      tap(data => {
        this.store.set(key, { value: data, expiresAt: Date.now() + this.ttlMs });
      }),
    );
  }
}
```

### 5. Exclude transform for certain routes

Sometimes you want a route to bypass the transform interceptor. Use `@SkipInterceptors` or a custom decorator:

```typescript
// src/common/decorators/raw-response.decorator.ts
import { SetMetadata } from '@nestjs/common';

export const RAW_RESPONSE = 'rawResponse';
export const RawResponse = () => SetMetadata(RAW_RESPONSE, true);
```

Update `TransformInterceptor` to check:
```typescript
intercept(context: ExecutionContext, next: CallHandler) {
  const isRaw = this.reflector.getAllAndOverride<boolean>(RAW_RESPONSE, [
    context.getHandler(),
    context.getClass(),
  ]);

  if (isRaw) return next.handle();  // skip wrapping

  // ... rest of transform logic
}
```

Then on any route:
```typescript
@Get('health')
@RawResponse()
healthCheck() {
  return { status: 'ok' };  // returned as-is, not wrapped
}
```

### 6. Track request ID

```typescript
// Add to LoggingInterceptor
import { v4 as uuid } from 'uuid';

intercept(context: ExecutionContext, next: CallHandler) {
  const request = context.switchToHttp().getRequest<Request>();
  const requestId = (request.headers['x-request-id'] as string) ?? uuid();
  (request as any).requestId = requestId;

  const response = context.switchToHttp().getResponse();
  response.setHeader('x-request-id', requestId);

  // ... logging logic
}
```
