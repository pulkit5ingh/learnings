# P1.2 — Library Management API

**Phase:** 1 — NestJS Fundamentals
**Topic:** Nested modules, multiple resources, cross-module DI
**Type:** Project

---

## Goal

Build a library management REST API with books, authors, members, and borrow records demonstrating multi-feature module organisation and cross-module dependency injection.

## What to Build

- Books resource: CRUD, search by title/genre, link to author
- Authors resource: CRUD, list books by author
- Members resource: CRUD, member registration, active/inactive status
- Borrow records: borrow a book, return a book, member borrow history
- In-memory repositories (no database)
- A `SharedModule` exporting a `DateHelperService` used in multiple features
- A `LibraryModule` grouping all feature modules

```typescript
import { Injectable } from '@nestjs/common';

@Injectable()
export class DateHelperService {
  daysUntil(date: Date): number {
    return Math.ceil((date.getTime() - Date.now()) / 86_400_000);
  }

  isOverdue(dueDate: Date): boolean {
    return dueDate < new Date();
  }

  addDays(date: Date, days: number): Date {
    const d = new Date(date);
    d.setDate(d.getDate() + days);
    return d;
  }
}
```

```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { randomUUID } from 'crypto';
import { Author } from './author.entity';
import { CreateAuthorDto } from './dto/create-author.dto';

@Injectable()
export class AuthorsService {
  private store = new Map<string, Author>();

  create(dto: CreateAuthorDto): Author {
    const author: Author = { id: randomUUID(), name: dto.name, bio: dto.bio ?? '', createdAt: new Date() };
    this.store.set(author.id, author);
    return author;
  }

  findAll(): Author[] {
    return Array.from(this.store.values());
  }

  findOne(id: string): Author {
    const a = this.store.get(id);
    if (!a) throw new NotFoundException(`Author ${id} not found`);
    return a;
  }

  remove(id: string): void {
    this.findOne(id);
    this.store.delete(id);
  }
}
```

### `src/library/books/books.service.ts`
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { randomUUID } from 'crypto';
import { Book } from './book.entity';
import { CreateBookDto } from './dto/create-book.dto';
import { AuthorsService } from '../authors/authors.service';

@Injectable()
export class BooksService {
  private store = new Map<string, Book>();

  constructor(private readonly authorsService: AuthorsService) {}

  create(dto: CreateBookDto): Book {
    this.authorsService.findOne(dto.authorId); // validate author exists
    const book: Book = {
      id: randomUUID(),
      title: dto.title,
      isbn: dto.isbn,
      genre: dto.genre ?? 'General',
      authorId: dto.authorId,
      available: true,
      createdAt: new Date(),
    };
    this.store.set(book.id, book);
    return book;
  }

  findAll(search?: string): Book[] {
    let books = Array.from(this.store.values());
    if (search) {
      const q = search.toLowerCase();
      books = books.filter((b) => b.title.toLowerCase().includes(q) || b.genre.toLowerCase().includes(q));
    }
    return books;
  }

  findOne(id: string): Book {
    const b = this.store.get(id);
    if (!b) throw new NotFoundException(`Book ${id} not found`);
    return b;
  }

  findByAuthor(authorId: string): Book[] {
    return Array.from(this.store.values()).filter((b) => b.authorId === authorId);
  }

  setAvailability(id: string, available: boolean): Book {
    const book = this.findOne(id);
    const updated = { ...book, available };
    this.store.set(id, updated);
    return updated;
  }
}
```

```typescript
import { Injectable, NotFoundException, ConflictException } from '@nestjs/common';
import { randomUUID } from 'crypto';
import { Member } from './member.entity';
import { CreateMemberDto } from './dto/create-member.dto';

@Injectable()
export class MembersService {
  private store = new Map<string, Member>();

  create(dto: CreateMemberDto): Member {
    const exists = Array.from(this.store.values()).find((m) => m.email === dto.email);
    if (exists) throw new ConflictException('Email already registered');
    const member: Member = { id: randomUUID(), name: dto.name, email: dto.email, active: true, joinedAt: new Date() };
    this.store.set(member.id, member);
    return member;
  }

  findAll(): Member[] { return Array.from(this.store.values()); }

  findOne(id: string): Member {
    const m = this.store.get(id);
    if (!m) throw new NotFoundException(`Member ${id} not found`);
    return m;
  }

  deactivate(id: string): Member {
    const m = this.findOne(id);
    const updated = { ...m, active: false };
    this.store.set(id, updated);
    return updated;
  }
}
```

### `src/library/borrows/borrows.service.ts`
```typescript
import { Injectable, BadRequestException, NotFoundException } from '@nestjs/common';
import { randomUUID } from 'crypto';
import { BorrowRecord } from './borrow.entity';
import { CreateBorrowDto } from './dto/create-borrow.dto';
import { BooksService } from '../books/books.service';
import { MembersService } from '../members/members.service';
import { DateHelperService } from '../../shared/date-helper.service';

@Injectable()
export class BorrowsService {
  private store = new Map<string, BorrowRecord>();

  constructor(
    private readonly booksService: BooksService,
    private readonly membersService: MembersService,
    private readonly dateHelper: DateHelperService,
  ) {}

  borrow(dto: CreateBorrowDto): BorrowRecord {
    const book = this.booksService.findOne(dto.bookId);
    if (!book.available) throw new BadRequestException('Book is not available');
    this.membersService.findOne(dto.memberId); // validate member

    const now = new Date();
    const record: BorrowRecord = {
      id: randomUUID(),
      bookId: dto.bookId,
      memberId: dto.memberId,
      borrowedAt: now,
      dueDate: this.dateHelper.addDays(now, dto.durationDays ?? 14),
      returnedAt: null,
    };
    this.store.set(record.id, record);
    this.booksService.setAvailability(dto.bookId, false);
    return record;
  }

  returnBook(recordId: string): BorrowRecord {
    const record = this.store.get(recordId);
    if (!record) throw new NotFoundException(`Borrow record ${recordId} not found`);
    if (record.returnedAt) throw new BadRequestException('Book already returned');

    const updated: BorrowRecord = { ...record, returnedAt: new Date() };
    this.store.set(recordId, updated);
    this.booksService.setAvailability(record.bookId, true);
    return updated;
  }

  memberHistory(memberId: string): BorrowRecord[] {
    this.membersService.findOne(memberId);
    return Array.from(this.store.values()).filter((r) => r.memberId === memberId);
  }

  overdueRecords(): BorrowRecord[] {
    return Array.from(this.store.values()).filter(
      (r) => !r.returnedAt && this.dateHelper.isOverdue(r.dueDate),
    );
  }
}
```

## Deliverables

- [ ] `SharedModule` is `@Global()` and `DateHelperService` is injectable without importing it per module
- [ ] `BooksService` injects `AuthorsService` from a different module
- [ ] `BorrowsService` injects both `BooksService` and `MembersService`
- [ ] Borrowing an unavailable book returns 400
- [ ] Borrowing with invalid `memberId`/`bookId` returns 404
- [ ] `GET /borrows/overdue` lists all overdue unreturned records
- [ ] `GET /books/author/:authorId` returns all books for an author
- [ ] `PATCH /borrows/:id/return` marks book available again
- [ ] All DTO validations enforced via global `ValidationPipe`
- [ ] No circular imports between modules
