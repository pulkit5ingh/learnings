# P5.1 — Multi-Environment API

**Phase:** 5 — Configuration & Environment
**Topic:** ConfigModule, typed ConfigService, Joi validation, environment-specific config
**Type:** Project

---

## Goal

Build an API that loads typed, validated configuration for development, staging, and production environments, with zero hard-coded values and full env-var validation at startup.

## What to Build

- `.env.development`, `.env.staging`, `.env.production` environment files
- Joi schema validating all required environment variables at startup
- Namespaced config (database config, app config, auth config, redis config)
- Typed `ConfigService` wrappers — never use string keys in business code
- `TypeOrmModule.forRootAsync` and `JwtModule.registerAsync` using typed config
- `APP_ENV` variable that gates behaviour (e.g., Swagger only in dev/staging)
- Config factory functions exposing typed interfaces

### `src/config/validation.schema.ts`
```typescript
import * as Joi from 'joi';

export const validationSchema = Joi.object({
  NODE_ENV: Joi.string().valid('development', 'staging', 'production', 'test').default('development'),
  APP_PORT: Joi.number().port().default(3000),
  APP_NAME: Joi.string().required(),

  DATABASE_HOST: Joi.string().required(),
  DATABASE_PORT: Joi.number().port().default(5432),
  DATABASE_NAME: Joi.string().required(),
  DATABASE_USER: Joi.string().required(),
  DATABASE_PASSWORD: Joi.string().required(),
  DATABASE_SSL: Joi.boolean().default(false),

  REDIS_HOST: Joi.string().required(),
  REDIS_PORT: Joi.number().port().default(6379),

  JWT_ACCESS_SECRET: Joi.string().min(32).required(),
  JWT_REFRESH_SECRET: Joi.string().min(32).required(),
  JWT_ACCESS_EXPIRY: Joi.string().default('15m'),
  JWT_REFRESH_EXPIRY: Joi.string().default('7d'),

  SWAGGER_ENABLED: Joi.boolean().default(false),
});
```

```typescript
import { registerAs } from '@nestjs/config';
import { TypeOrmModuleOptions } from '@nestjs/typeorm';

export interface DatabaseConfig {
  host: string;
  port: number;
  name: string;
  user: string;
  password: string;
  ssl: boolean;
}

export const databaseConfig = registerAs('database', (): DatabaseConfig => ({
  host: process.env.DATABASE_HOST!,
  port: parseInt(process.env.DATABASE_PORT ?? '5432', 10),
  name: process.env.DATABASE_NAME!,
  user: process.env.DATABASE_USER!,
  password: process.env.DATABASE_PASSWORD!,
  ssl: process.env.DATABASE_SSL === 'true',
}));

export function buildTypeOrmConfig(db: DatabaseConfig): TypeOrmModuleOptions {
  return {
    type: 'postgres',
    host: db.host,
    port: db.port,
    database: db.name,
    username: db.user,
    password: db.password,
    ssl: db.ssl ? { rejectUnauthorized: false } : false,
    entities: [__dirname + '/../**/*.entity{.ts,.js}'],
    migrations: [__dirname + '/../../migrations/*{.ts,.js}'],
    synchronize: process.env.NODE_ENV === 'development',
    logging: process.env.NODE_ENV === 'development',
  };
}
```

### `src/common/services/typed-config.service.ts`
```typescript
import { Injectable } from '@nestjs/common';
import { ConfigService } from '@nestjs/config';
import { AppConfig } from '../../config/app.config';
import { DatabaseConfig } from '../../config/database.config';
import { AuthConfig } from '../../config/auth.config';
import { RedisConfig } from '../../config/redis.config';

@Injectable()
export class TypedConfigService {
  constructor(private readonly config: ConfigService) {}

  get app(): AppConfig {
    return this.config.getOrThrow<AppConfig>('app');
  }

  get database(): DatabaseConfig {
    return this.config.getOrThrow<DatabaseConfig>('database');
  }

  get auth(): AuthConfig {
    return this.config.getOrThrow<AuthConfig>('auth');
  }

  get redis(): RedisConfig {
    return this.config.getOrThrow<RedisConfig>('redis');
  }

  get isProduction(): boolean { return this.app.env === 'production'; }
  get isDevelopment(): boolean { return this.app.env === 'development'; }
  get isStaging(): boolean { return this.app.env === 'staging'; }
}
```

### `src/app.module.ts`
```typescript
import { Module } from '@nestjs/common';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { TypeOrmModule } from '@nestjs/typeorm';
import { JwtModule } from '@nestjs/jwt';
import { validationSchema } from './config/validation.schema';
import { appConfig } from './config/app.config';
import { databaseConfig, buildTypeOrmConfig, DatabaseConfig } from './config/database.config';
import { authConfig } from './config/auth.config';
import { redisConfig } from './config/redis.config';
import { TypedConfigService } from './common/services/typed-config.service';

@Module({
  imports: [
    ConfigModule.forRoot({
      isGlobal: true,
      envFilePath: `.env.${process.env.NODE_ENV ?? 'development'}`,
      load: [appConfig, databaseConfig, authConfig, redisConfig],
      validationSchema,
      validationOptions: { allowUnknown: false, abortEarly: false },
    }),
    TypeOrmModule.forRootAsync({
      inject: [ConfigService],
      useFactory: (config: ConfigService) => {
        const db = config.getOrThrow<DatabaseConfig>('database');
        return buildTypeOrmConfig(db);
      },
    }),
    JwtModule.registerAsync({
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        secret: config.getOrThrow('auth.accessSecret'),
        signOptions: { expiresIn: config.getOrThrow('auth.accessExpiry') },
      }),
    }),
  ],
  providers: [TypedConfigService],
  exports: [TypedConfigService],
})
export class AppModule {}
```

### `src/main.ts`
```typescript
import { NestFactory } from '@nestjs/core';
import { ValidationPipe } from '@nestjs/common';
import { SwaggerModule, DocumentBuilder } from '@nestjs/swagger';
import { AppModule } from './app.module';
import { TypedConfigService } from './common/services/typed-config.service';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);
  const typedConfig = app.get(TypedConfigService);

  app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));

  if (typedConfig.app.swaggerEnabled) {
    const doc = new DocumentBuilder()
      .setTitle(typedConfig.app.name)
      .setVersion('1.0')
      .addBearerAuth()
      .build();
    SwaggerModule.setup('api/docs', app, SwaggerModule.createDocument(app, doc));
    console.log(`Swagger available at /api/docs`);
  }

  await app.listen(typedConfig.app.port);
  console.log(`[${typedConfig.app.env}] ${typedConfig.app.name} running on port ${typedConfig.app.port}`);
}
bootstrap();
```

## Deliverables

- [ ] `npm run start:dev` loads `.env.development`, `npm run start:prod` loads `.env.production`
- [ ] Missing required env var at startup crashes with a descriptive Joi error message
- [ ] `JWT_ACCESS_SECRET` shorter than 32 characters fails Joi validation at startup
- [ ] `TypedConfigService.database` returns a fully-typed `DatabaseConfig` object
- [ ] `TypeOrmModule.forRootAsync` reads database config via `ConfigService`, no hard-coded values
- [ ] `JwtModule.registerAsync` reads auth config via `ConfigService`
- [ ] Swagger is only mounted when `SWAGGER_ENABLED=true`
- [ ] No raw `process.env.X` calls outside of `config/*.config.ts` files
- [ ] `synchronize: true` only in development, `false` in staging/production
- [ ] `TypedConfigService` is globally available (exported from `AppModule`)
