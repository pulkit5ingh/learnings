## Goal
Process work asynchronously with Bull queues: add jobs from services, process them in workers, handle retries on failure, and monitor queue health.

## Background

Bull is a Redis-backed job queue. Jobs are added to a queue (producer side) and picked up by workers (consumer side). Because jobs are persisted in Redis, they survive server restarts and can be processed by multiple worker instances horizontally.

**Key Bull concepts:**
- **Queue** — a named FIFO channel in Redis. Add jobs with `queue.add(name, data, options)`.
- **Processor** — a class decorated with `@Processor('queue-name')` that handles jobs via methods decorated with `@Process('job-name')`.
- **Job options** — `attempts` (max retries), `backoff` (retry delay strategy), `priority` (higher number = higher priority), `delay` (process after N ms), `removeOnComplete`, `removeOnFail`.
- **Progress** — jobs can call `job.progress(percent)` to report progress; useful for long-running jobs.
- **Events** — `@OnQueueActive`, `@OnQueueCompleted`, `@OnQueueFailed` decorators listen to queue lifecycle events.
- **Concurrency** — `@Process({ concurrency: 5 })` runs up to 5 jobs simultaneously.

```typescript
import { ConfigModule, ConfigService } from '@nestjs/config';

@Module({
  imports: [
    BullModule.forRootAsync({
      imports: [ConfigModule],
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        redis: {
          host: config.get('REDIS_HOST', 'localhost'),
          port: config.get<number>('REDIS_PORT', 6379),
          password: config.get('REDIS_PASSWORD'),
        },
        defaultJobOptions: {
          removeOnComplete: true,  // clean up completed jobs
          removeOnFail: false,     // keep failed jobs for debugging
          attempts: 3,
          backoff: { type: 'exponential', delay: 1000 },
        },
      }),
    }),
  ],
})
```

### 4. Email processor (consumer)

**`src/email/email.processor.ts`**
```typescript
import {
  Processor, Process,
  OnQueueActive, OnQueueCompleted, OnQueueFailed,
  OnQueueStalled,
} from '@nestjs/bull';
import { Job } from 'bull';
import { Logger } from '@nestjs/common';
import { EMAIL_QUEUE } from './email.module';

export interface WelcomeEmailData {
  to: string;
  name: string;
  verificationToken: string;
}

export interface PasswordResetData {
  to: string;
  resetToken: string;
  expiresIn: number;
}

@Processor(EMAIL_QUEUE)
export class EmailProcessor {
  private readonly logger = new Logger(EmailProcessor.name);

  @Process({ name: 'welcome', concurrency: 3 })
  async sendWelcomeEmail(job: Job<WelcomeEmailData>): Promise<void> {
    const { to, name, verificationToken } = job.data;

    this.logger.log(`Sending welcome email to ${to}`);
    await job.progress(10);

    // Simulate email sending steps
    await this.delay(200);  // prepare template
    await job.progress(50);

    await this.sendSmtp(to, 'Welcome!', `Hi ${name}, verify: ${verificationToken}`);
    await job.progress(100);

    this.logger.log(`Welcome email sent to ${to}`);
  }

  @Process({ name: 'password-reset', concurrency: 5 })
  async sendPasswordReset(job: Job<PasswordResetData>): Promise<void> {
    const { to, resetToken, expiresIn } = job.data;
    await this.sendSmtp(
      to,
      'Password Reset',
      `Reset your password: ${resetToken} (expires in ${expiresIn} minutes)`,
    );
  }

  @Process('bulk-notification')
  async processBulk(job: Job<{ recipients: string[]; message: string }>): Promise<void> {
    const { recipients, message } = job.data;
    const total = recipients.length;

    for (let i = 0; i < total; i++) {
      await this.sendSmtp(recipients[i], 'Notification', message);
      await job.progress(Math.round(((i + 1) / total) * 100));
    }
  }

  @OnQueueActive()
  onActive(job: Job): void {
    this.logger.debug(`Processing job #${job.id} (${job.name})`);
  }

  @OnQueueCompleted()
  onCompleted(job: Job, result: any): void {
    this.logger.log(`Job #${job.id} (${job.name}) completed`);
  }

  @OnQueueFailed()
  onFailed(job: Job, err: Error): void {
    this.logger.error(
      `Job #${job.id} (${job.name}) failed: ${err.message}. Attempts: ${job.attemptsMade}/${job.opts.attempts}`,
    );
  }

  @OnQueueStalled()
  onStalled(job: Job): void {
    this.logger.warn(`Job #${job.id} (${job.name}) stalled and will be retried`);
  }

  private async sendSmtp(to: string, subject: string, body: string): Promise<void> {
    // In real app: use nodemailer or @nestjs-modules/mailer
    console.log(`[SMTP] To: ${to} | Subject: ${subject} | Body: ${body}`);
    await this.delay(100);
  }

  private delay(ms: number): Promise<void> {
    return new Promise(resolve => setTimeout(resolve, ms));
  }
}
```

### 5. Email service (producer)

**`src/email/email.service.ts`**
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { InjectQueue } from '@nestjs/bull';
import { Queue, Job, JobOptions } from 'bull';
import { EMAIL_QUEUE, WelcomeEmailData, PasswordResetData } from './email.module';

@Injectable()
export class EmailService {
  private readonly logger = new Logger(EmailService.name);

  constructor(@InjectQueue(EMAIL_QUEUE) private readonly emailQueue: Queue) {}

  async sendWelcomeEmail(data: WelcomeEmailData): Promise<Job> {
    return this.emailQueue.add('welcome', data, {
      priority: 1,    // high priority
      attempts: 5,
      backoff: { type: 'fixed', delay: 2000 },
    });
  }

  async sendPasswordReset(data: PasswordResetData): Promise<Job> {
    return this.emailQueue.add('password-reset', data, {
      priority: 1,
      attempts: 3,
      delay: 0,
    });
  }

  async scheduleDailyDigest(recipients: string[], message: string): Promise<Job> {
    return this.emailQueue.add(
      'bulk-notification',
      { recipients, message },
      {
        delay: 60_000,  // process after 1 minute
        attempts: 2,
        removeOnComplete: true,
      },
    );
  }

  async getQueueStats() {
    const [waiting, active, completed, failed, delayed] = await Promise.all([
      this.emailQueue.getWaitingCount(),
      this.emailQueue.getActiveCount(),
      this.emailQueue.getCompletedCount(),
      this.emailQueue.getFailedCount(),
      this.emailQueue.getDelayedCount(),
    ]);
    return { waiting, active, completed, failed, delayed };
  }

  async getFailedJobs(limit = 10): Promise<Job[]> {
    return this.emailQueue.getFailed(0, limit - 1);
  }

  async retryFailedJob(jobId: string): Promise<void> {
    const job = await this.emailQueue.getJob(jobId);
    if (!job) throw new Error(`Job ${jobId} not found`);
    await job.retry();
  }

  async clearQueue(): Promise<void> {
    await this.emailQueue.empty();
    this.logger.warn('Email queue cleared');
  }
}
```

### 6. Monitor queue health

```typescript
@Controller('admin/queues')
@UseGuards(JwtAuthGuard, RolesGuard)
@Roles(Role.ADMIN)
export class QueueAdminController {
  constructor(private readonly emailService: EmailService) {}

  @Get('email/stats')
  getEmailQueueStats() {
    return this.emailService.getQueueStats();
  }

  @Get('email/failed')
  getFailedJobs() {
    return this.emailService.getFailedJobs();
  }

  @Post('email/failed/:id/retry')
  retryJob(@Param('id') id: string) {
    return this.emailService.retryFailedJob(id);
  }
}
```
