# A7.1 — Unit Testing

**Phase:** 7 — Testing  
**Topic:** jest, Test.createTestingModule, mock providers, jest.fn(), test service in isolation  
**Type:** Assignment

---

## Goal
Write isolated unit tests for NestJS services using `Test.createTestingModule`, mock all dependencies with `jest.fn()`, and verify behaviour without hitting a database or external service.

## Background

Unit tests test a single unit of code (a function, method, or class) in complete isolation. All external dependencies are replaced with mocks. In NestJS, the unit under test is typically a service, and its dependencies (repositories, other services, external clients) are mocked.

`Test.createTestingModule` creates a sandboxed NestJS module for testing. You provide the real class under test and mock replacements for all its dependencies. This way, tests run in milliseconds (no DB, no HTTP) and failures point to the exact logic, not to infrastructure.

**jest.fn()** creates a mock function that tracks calls, arguments, and return values. **jest.spyOn()** replaces a method on an existing object with a mock. Both support `.mockResolvedValue()`, `.mockRejectedValue()`, `.mockReturnValue()`, and `.mockImplementation()`.

Good unit tests follow the **AAA pattern**: Arrange (set up mocks and inputs), Act (call the method under test), Assert (verify the output and mock calls).

### 2. Unit test for UsersService

**`src/users/users.service.spec.ts`**
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { getRepositoryToken } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { NotFoundException } from '@nestjs/common';
import { UsersService } from './users.service';
import { User } from './entities/user.entity';

// Type helper for mocking a Repository
type MockRepository<T = any> = Partial<Record<keyof Repository<T>, jest.Mock>>;

const createMockRepository = <T = any>(): MockRepository<T> => ({
  find: jest.fn(),
  findOne: jest.fn(),
  findOneOrFail: jest.fn(),
  create: jest.fn(),
  save: jest.fn(),
  update: jest.fn(),
  delete: jest.fn(),
  softDelete: jest.fn(),
  count: jest.fn(),
});

describe('UsersService', () => {
  let service: UsersService;
  let mockRepo: MockRepository<User>;

  const mockUser: User = {
    id: 1,
    name: 'Alice',
    email: 'alice@example.com',
    passwordHash: 'hashedpw',
    isActive: true,
    roles: ['user'],
    createdAt: new Date('2024-01-01'),
    updatedAt: new Date('2024-01-01'),
  } as User;

  beforeEach(async () => {
    mockRepo = createMockRepository<User>();

    const module: TestingModule = await Test.createTestingModule({
      providers: [
        UsersService,
        {
          provide: getRepositoryToken(User),
          useValue: mockRepo,
        },
      ],
    }).compile();

    service = module.get<UsersService>(UsersService);
  });

  it('should be defined', () => {
    expect(service).toBeDefined();
  });

  describe('findOne', () => {
    it('should return a user when found', async () => {
      // Arrange
      mockRepo.findOne!.mockResolvedValue(mockUser);

      // Act
      const result = await service.findOne(1);

      // Assert
      expect(result).toEqual(mockUser);
      expect(mockRepo.findOne).toHaveBeenCalledWith({ where: { id: 1 } });
      expect(mockRepo.findOne).toHaveBeenCalledTimes(1);
    });

    it('should throw NotFoundException when user not found', async () => {
      // Arrange
      mockRepo.findOne!.mockResolvedValue(null);

      // Act & Assert
      await expect(service.findOne(999)).rejects.toThrow(NotFoundException);
      await expect(service.findOne(999)).rejects.toThrow('User 999 not found');
    });
  });

  describe('create', () => {
    it('should create and return a new user', async () => {
      // Arrange
      const createDto = { name: 'Bob', email: 'bob@example.com', password: 'Password1!' };
      const createdUser = { ...mockUser, id: 2, name: 'Bob', email: 'bob@example.com' };
      mockRepo.create!.mockReturnValue(createdUser);
      mockRepo.save!.mockResolvedValue(createdUser);

      // Act
      const result = await service.create(createDto);

      // Assert
      expect(mockRepo.create).toHaveBeenCalledWith(expect.objectContaining({ email: 'bob@example.com' }));
      expect(mockRepo.save).toHaveBeenCalledWith(createdUser);
      expect(result).toEqual(createdUser);
    });
  });

  describe('update', () => {
    it('should update and return the modified user', async () => {
      // Arrange
      const updatedUser = { ...mockUser, name: 'Alice Updated' };
      mockRepo.findOne!.mockResolvedValue(mockUser);
      mockRepo.save!.mockResolvedValue(updatedUser);

      // Act
      const result = await service.update(1, { name: 'Alice Updated' });

      // Assert
      expect(result.name).toBe('Alice Updated');
      expect(mockRepo.save).toHaveBeenCalledTimes(1);
    });

    it('should throw NotFoundException when updating non-existent user', async () => {
      mockRepo.findOne!.mockResolvedValue(null);
      await expect(service.update(999, { name: 'x' })).rejects.toThrow(NotFoundException);
    });
  });

  describe('softDelete', () => {
    it('should soft delete a user', async () => {
      mockRepo.softDelete!.mockResolvedValue({ affected: 1 });
      await expect(service.softDelete(1)).resolves.not.toThrow();
      expect(mockRepo.softDelete).toHaveBeenCalledWith(1);
    });

    it('should throw NotFoundException when deleting non-existent user', async () => {
      mockRepo.softDelete!.mockResolvedValue({ affected: 0 });
      await expect(service.softDelete(999)).rejects.toThrow(NotFoundException);
    });
  });
});
```

### 3. Unit test with multiple mocked dependencies

**`src/auth/auth.service.spec.ts`**
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import { UnauthorizedException } from '@nestjs/common';
import * as bcrypt from 'bcrypt';
import { AuthService } from './auth.service';
import { UsersService } from '../users/users.service';

jest.mock('bcrypt');

describe('AuthService', () => {
  let service: AuthService;
  let mockUsersService: jest.Mocked<Partial<UsersService>>;
  let mockJwtService: jest.Mocked<Partial<JwtService>>;

  beforeEach(async () => {
    mockUsersService = {
      findByEmail: jest.fn(),
      create: jest.fn(),
    };

    mockJwtService = {
      sign: jest.fn().mockReturnValue('mock.jwt.token'),
    };

    const module: TestingModule = await Test.createTestingModule({
      providers: [
        AuthService,
        { provide: UsersService, useValue: mockUsersService },
        { provide: JwtService, useValue: mockJwtService },
        {
          provide: ConfigService,
          useValue: { get: jest.fn().mockReturnValue('test-secret') },
        },
        // Mock the RefreshToken repository
        {
          provide: 'RefreshTokenRepository',
          useValue: { find: jest.fn(), save: jest.fn(), create: jest.fn() },
        },
      ],
    }).compile();

    service = module.get<AuthService>(AuthService);
  });

  describe('login', () => {
    it('should return tokens for valid credentials', async () => {
      const mockUser = { id: 1, email: 'test@example.com', passwordHash: 'hash', roles: ['user'] };
      mockUsersService.findByEmail!.mockResolvedValue(mockUser as any);
      (bcrypt.compare as jest.Mock).mockResolvedValue(true);

      const result = await service.login('test@example.com', 'password123');

      expect(result).toHaveProperty('accessToken');
      expect(result).toHaveProperty('refreshToken');
      expect(mockUsersService.findByEmail).toHaveBeenCalledWith('test@example.com');
      expect(bcrypt.compare).toHaveBeenCalledWith('password123', 'hash');
    });

    it('should throw UnauthorizedException for invalid password', async () => {
      const mockUser = { id: 1, email: 'test@example.com', passwordHash: 'hash' };
      mockUsersService.findByEmail!.mockResolvedValue(mockUser as any);
      (bcrypt.compare as jest.Mock).mockResolvedValue(false);

      await expect(service.login('test@example.com', 'wrong')).rejects.toThrow(UnauthorizedException);
    });

    it('should throw UnauthorizedException when user not found', async () => {
      mockUsersService.findByEmail!.mockResolvedValue(null);
      await expect(service.login('nobody@example.com', 'pw')).rejects.toThrow(UnauthorizedException);
    });
  });
});
```

### 4. Testing edge cases and error paths

```typescript
describe('findAll with search', () => {
  it('should call repository with ILike filter when search is provided', async () => {
    mockRepo.find!.mockResolvedValue([mockUser]);

    const result = await service.findAll('alice');

    expect(mockRepo.find).toHaveBeenCalledWith(expect.objectContaining({
      where: expect.arrayContaining([
        expect.objectContaining({ name: expect.anything() }),
      ]),
    }));
    expect(result).toHaveLength(1);
  });

  it('should return all users when no search is provided', async () => {
    mockRepo.find!.mockResolvedValue([mockUser]);

    await service.findAll();

    expect(mockRepo.find).toHaveBeenCalledWith(expect.objectContaining({
      order: { createdAt: 'DESC' },
    }));
  });
});
```
