# A4.3 — Exception Filters

**Phase:** 4 — Pipes, Interceptors & Filters  
**Topic:** ExceptionFilter, @Catch, HttpException, custom exceptions, global filter  
**Type:** Assignment

---

## Goal
Catch and format all errors consistently using exception filters, create domain-specific exception classes, and handle unexpected errors gracefully.

## Background

Without an exception filter, NestJS has a built-in exception layer that converts `HttpException` subclasses to appropriate HTTP responses and catches anything else as a 500 Internal Server Error. Exception filters intercept thrown exceptions and transform them into HTTP responses with full control over the status code, headers, and body.

An exception filter implements `ExceptionFilter` and is decorated with `@Catch(ExceptionType)`. Using `@Catch()` with no argument catches *everything*. The `catch(exception, host: ArgumentsHost)` method receives the thrown exception and the `ArgumentsHost` which gives access to the underlying request/response.

**Custom exception classes** should extend `HttpException` and pass a status code. They can carry additional structured data (error code, details) beyond the message.

**The RFC 7807 "Problem Details" format** is a standard for HTTP error responses:
```json
{
  "type": "https://example.com/errors/validation-failed",
  "title": "Validation Failed",
  "status": 400,
  "detail": "name: must not be blank",
  "instance": "/users"
}
```

## Tasks

### 1. Custom exception classes

**`src/common/exceptions/app.exception.ts`**
```typescript
import { HttpException, HttpStatus } from '@nestjs/common';

export interface ErrorDetails {
  code: string;
  message: string;
  details?: Record<string, any>;
}

export class AppException extends HttpException {
  public readonly code: string;
  public readonly details?: Record<string, any>;

  constructor(error: ErrorDetails, status: HttpStatus) {
    super(error, status);
    this.code = error.code;
    this.details = error.details;
  }
}
```

**`src/common/exceptions/business.exceptions.ts`**
```typescript
import { HttpStatus } from '@nestjs/common';
import { AppException } from './app.exception';

export class ResourceNotFoundException extends AppException {
  constructor(resource: string, id: number | string) {
    super(
      {
        code: 'RESOURCE_NOT_FOUND',
        message: `${resource} with id ${id} was not found`,
        details: { resource, id },
      },
      HttpStatus.NOT_FOUND,
    );
  }
}

export class DuplicateResourceException extends AppException {
  constructor(resource: string, field: string, value: string) {
    super(
      {
        code: 'DUPLICATE_RESOURCE',
        message: `${resource} with ${field}=${value} already exists`,
        details: { resource, field, value },
      },
      HttpStatus.CONFLICT,
    );
  }
}

export class BusinessRuleViolationException extends AppException {
  constructor(rule: string, details?: Record<string, any>) {
    super(
      {
        code: 'BUSINESS_RULE_VIOLATION',
        message: rule,
        details,
      },
      HttpStatus.UNPROCESSABLE_ENTITY,
    );
  }
}
```

### 2. Global HTTP exception filter

**`src/common/filters/http-exception.filter.ts`**
```typescript
import {
  ArgumentsHost,
  Catch,
  ExceptionFilter,
  HttpException,
  HttpStatus,
  Logger,
} from '@nestjs/common';
import { Request, Response } from 'express';

@Catch(HttpException)
export class HttpExceptionFilter implements ExceptionFilter {
  private readonly logger = new Logger(HttpExceptionFilter.name);

  catch(exception: HttpException, host: ArgumentsHost): void {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const request = ctx.getRequest<Request>();
    const status = exception.getStatus();
    const exceptionResponse = exception.getResponse();

    const errorBody = this.buildErrorBody(exception, exceptionResponse, request, status);

    this.logger.warn(
      `${request.method} ${request.url} → ${status}: ${errorBody.title}`,
    );

    response.status(status).json(errorBody);
  }

  private buildErrorBody(
    exception: HttpException,
    exceptionResponse: any,
    request: Request,
    status: HttpStatus,
  ) {
    // RFC 7807 Problem Details
    const base = {
      status,
      instance: request.url,
      timestamp: new Date().toISOString(),
    };

    // ValidationPipe errors (array of messages)
    if (typeof exceptionResponse === 'object' && Array.isArray(exceptionResponse.message)) {
      return {
        ...base,
        type: 'https://nestjs.example.com/errors/validation-failed',
        title: 'Validation Failed',
        errors: exceptionResponse.message,
      };
    }

    // Custom AppException
    if (typeof exceptionResponse === 'object' && exceptionResponse.code) {
      return {
        ...base,
        type: `https://nestjs.example.com/errors/${exceptionResponse.code.toLowerCase().replace(/_/g, '-')}`,
        title: exceptionResponse.code,
        detail: exceptionResponse.message,
        ...(exceptionResponse.details && { context: exceptionResponse.details }),
      };
    }

    // Standard HttpException
    return {
      ...base,
      type: `https://nestjs.example.com/errors/${status}`,
      title: exception.message,
      detail: typeof exceptionResponse === 'string'
        ? exceptionResponse
        : exceptionResponse?.message,
    };
  }
}
```

### 3. All-errors catch filter (500 handler)

**`src/common/filters/all-exceptions.filter.ts`**
```typescript
import {
  ArgumentsHost,
  Catch,
  ExceptionFilter,
  HttpException,
  HttpStatus,
  Logger,
} from '@nestjs/common';
import { Request, Response } from 'express';
import { QueryFailedError } from 'typeorm';

@Catch()
export class AllExceptionsFilter implements ExceptionFilter {
  private readonly logger = new Logger(AllExceptionsFilter.name);

  catch(exception: unknown, host: ArgumentsHost): void {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const request = ctx.getRequest<Request>();

    let status = HttpStatus.INTERNAL_SERVER_ERROR;
    let message = 'An unexpected error occurred';

    if (exception instanceof HttpException) {
      status = exception.getStatus();
      message = exception.message;
    } else if (exception instanceof QueryFailedError) {
      // PostgreSQL unique constraint violation
      if ((exception as any).code === '23505') {
        status = HttpStatus.CONFLICT;
        message = 'Duplicate entry: ' + (exception as any).detail;
      } else {
        this.logger.error('Database error:', exception.message);
      }
    } else if (exception instanceof Error) {
      this.logger.error('Unhandled error:', exception.stack);
    }

    response.status(status).json({
      status,
      timestamp: new Date().toISOString(),
      path: request.url,
      message,
    });
  }
}
```

### 4. Register filters globally

**Via APP_FILTER (preferred — DI supported):**
```typescript
// app.module.ts providers:
import { APP_FILTER } from '@nestjs/core';

providers: [
  { provide: APP_FILTER, useClass: AllExceptionsFilter },
  { provide: APP_FILTER, useClass: HttpExceptionFilter },
  // Registration order: filters run in reverse order of registration
  // HttpExceptionFilter runs before AllExceptionsFilter
]
```

**Via main.ts (no DI — use for non-NestJS exceptions):**
```typescript
app.useGlobalFilters(new AllExceptionsFilter(), new HttpExceptionFilter());
```

### 5. TypeORM-specific filter

```typescript
@Catch(QueryFailedError)
export class TypeOrmExceptionFilter implements ExceptionFilter {
  catch(exception: QueryFailedError, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const pgCode = (exception as any).code;

    const statusMap: Record<string, number> = {
      '23505': 409,  // unique_violation
      '23503': 400,  // foreign_key_violation
      '23502': 400,  // not_null_violation
    };

    const status = statusMap[pgCode] ?? 500;
    const messageMap: Record<string, string> = {
      '23505': 'Duplicate entry',
      '23503': 'Referenced record does not exist',
      '23502': 'Required field is missing',
    };

    response.status(status).json({
      status,
      message: messageMap[pgCode] ?? 'Database error',
      detail: (exception as any).detail,
    });
  }
}
```
