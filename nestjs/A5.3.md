# A5.3 — Secrets Management

**Phase:** 5 — Configuration & Environment  
**Topic:** Env validation, Vault integration basics, secret rotation  
**Type:** Assignment

---

## Goal
Secure sensitive configuration using strict Joi validation, understand secret injection patterns, and integrate with HashiCorp Vault for dynamic secrets.

## Background

Environment variables are the standard mechanism for injecting secrets into 12-factor applications. The risks are: accidentally committing `.env` files, logging secrets, returning them in API responses, and not rotating them when compromised.

**Secrets management best practices:**
1. Never commit secrets to source control — use `.env.example` with placeholder values.
2. Validate all secrets at startup — a missing/malformed secret should prevent the app from starting.
3. Never log secrets — use a redaction list in your logger.
4. Never return secrets in API responses — use `@Exclude()` on sensitive entity fields.
5. Rotate secrets regularly — JWT secrets, database passwords, API keys.

**HashiCorp Vault** is the industry standard for secrets management. It stores secrets encrypted, provides audit logs of every access, and can generate dynamic credentials (e.g., short-lived database passwords). NestJS can integrate with Vault by loading secrets at startup using the Vault SDK before the ConfigModule is fully initialised.

### 1. Strict Joi validation schema

**`src/config/validation.schema.ts`**
```typescript
import * as Joi from 'joi';

export const validationSchema = Joi.object({
  // App
  NODE_ENV: Joi.string().valid('development', 'staging', 'production', 'test').required(),
  PORT: Joi.number().integer().min(1024).max(65535).default(3000),
  APP_NAME: Joi.string().alphanum().min(2).max(50).required(),

  // Database
  DB_HOST: Joi.string().hostname().required(),
  DB_PORT: Joi.number().integer().min(1).max(65535).default(5432),
  DB_USER: Joi.string().required(),
  DB_PASS: Joi.string().min(8).required()
    .messages({ 'string.min': 'DB_PASS must be at least 8 characters — use a strong password' }),
  DB_NAME: Joi.string().required(),
  DB_POOL_MAX: Joi.number().integer().min(1).max(100).default(10),
  DB_SSL: Joi.boolean().default(false),

  // JWT
  JWT_SECRET: Joi.string().min(32).required()
    .messages({ 'string.min': 'JWT_SECRET must be at least 32 characters for HS256 security' }),
  JWT_EXPIRES_IN: Joi.string()
    .pattern(/^\d+(s|m|h|d)$/)
    .default('15m')
    .messages({ 'string.pattern.base': 'JWT_EXPIRES_IN must be a duration like 15m, 1h, 7d' }),
  JWT_REFRESH_SECRET: Joi.string().min(32).required(),
  JWT_REFRESH_EXPIRES_IN: Joi.string().default('7d'),

  // Redis
  REDIS_HOST: Joi.string().hostname().default('localhost'),
  REDIS_PORT: Joi.number().integer().default(6379),
  REDIS_PASSWORD: Joi.string().optional().allow(''),

  // External APIs
  SMTP_HOST: Joi.when('NODE_ENV', {
    is: 'production',
    then: Joi.string().hostname().required(),
    otherwise: Joi.string().optional(),
  }),
  SMTP_PORT: Joi.number().integer().default(587),
  SMTP_USER: Joi.string().email().optional(),
  SMTP_PASS: Joi.string().optional(),
});
```

### 2. Redacting secrets from logs

**`src/common/utils/secret-redactor.ts`**
```typescript
const SENSITIVE_KEYS = new Set([
  'password', 'pass', 'secret', 'token', 'key', 'auth',
  'authorization', 'cookie', 'credential', 'private',
]);

export function redactSecrets(obj: Record<string, any>, depth = 0): Record<string, any> {
  if (depth > 5) return { '[deep]': true };

  return Object.fromEntries(
    Object.entries(obj).map(([key, value]) => {
      const isSecret = SENSITIVE_KEYS.has(key.toLowerCase()) ||
        SENSITIVE_KEYS.has(key.toLowerCase().replace(/[_-]/g, ''));

      if (isSecret) return [key, '[REDACTED]'];
      if (typeof value === 'object' && value !== null) {
        return [key, redactSecrets(value as Record<string, any>, depth + 1)];
      }
      return [key, value];
    }),
  );
}
```

**Usage in logger:**
```typescript
this.logger.debug('Config loaded', redactSecrets({
  host: 'localhost',
  password: 'mysecret',    // → '[REDACTED]'
  jwt_secret: 'abc123',   // → '[REDACTED]'
  port: 5432,              // → 5432 (not redacted)
}));
```

### 3. Vault integration (basic pattern)

Install the official Vault SDK:
```bash
npm install node-vault
```

**`src/config/vault.loader.ts`**
```typescript
import * as vault from 'node-vault';

interface VaultSecrets {
  DB_PASS: string;
  JWT_SECRET: string;
  JWT_REFRESH_SECRET: string;
}

export async function loadVaultSecrets(): Promise<Partial<VaultSecrets>> {
  if (process.env.VAULT_ENABLED !== 'true') return {};

  const client = vault({
    apiVersion: 'v1',
    endpoint: process.env.VAULT_ADDR ?? 'http://localhost:8200',
    token: process.env.VAULT_TOKEN,
  });

  try {
    const result = await client.read(`secret/data/${process.env.APP_NAME}/config`);
    const secrets = result.data?.data as VaultSecrets;
    console.log('[Vault] Secrets loaded successfully');
    return secrets;
  } catch (err) {
    console.error('[Vault] Failed to load secrets:', (err as Error).message);
    if (process.env.NODE_ENV === 'production') {
      throw new Error('Cannot start: Vault secrets unavailable in production');
    }
    return {};
  }
}
```

**`src/main.ts`** — load Vault before bootstrapping:
```typescript
import { loadVaultSecrets } from './config/vault.loader';

async function bootstrap() {
  // Load Vault secrets into process.env BEFORE NestJS reads them
  const vaultSecrets = await loadVaultSecrets();
  Object.assign(process.env, vaultSecrets);

  // NOW create the NestJS app — ConfigModule will find secrets in process.env
  const app = await NestFactory.create(AppModule);
  // ...
}
bootstrap();
```

### 4. Secret rotation strategy

For JWT secrets, support a "transitional" period where both old and new secrets are valid:

```typescript
@Injectable()
export class JwtRotationStrategy extends PassportStrategy(Strategy, 'jwt') {
  constructor(private readonly config: ConfigService) {
    super({
      jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
      secretOrKeyProvider: async (request: Request, rawJwtToken: string, done: Function) => {
        // Try new secret first, fall back to old secret
        const newSecret = config.get<string>('JWT_SECRET');
        const oldSecret = config.get<string>('JWT_SECRET_OLD');

        try {
          // Verify with new secret
          require('jsonwebtoken').verify(rawJwtToken, newSecret);
          done(null, newSecret);
        } catch {
          if (oldSecret) {
            done(null, oldSecret);  // fall back to old secret
          } else {
            done(new Error('Invalid token'));
          }
        }
      },
    });
  }

  async validate(payload: JwtPayload) {
    return payload;
  }
}
```

### 5. Config audit endpoint (admin only)

```typescript
@Controller('admin/config')
@UseGuards(JwtAuthGuard, RolesGuard)
@Roles(Role.ADMIN)
export class ConfigAuditController {
  constructor(private readonly config: ConfigService) {}

  @Get('check')
  checkConfig() {
    // Return ONLY non-sensitive config — never secrets
    return {
      nodeEnv: this.config.get('NODE_ENV'),
      appName: this.config.get('APP_NAME'),
      dbHost: this.config.get('DB_HOST'),
      dbName: this.config.get('DB_NAME'),
      jwtExpiresIn: this.config.get('JWT_EXPIRES_IN'),
      // NEVER include: DB_PASS, JWT_SECRET, etc.
    };
  }
}
```
