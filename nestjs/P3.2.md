# P3.2 — Permission System (RBAC + ABAC)

**Phase:** 3 — Authentication & Guards
**Topic:** Role-Based + Attribute-Based Access Control, custom guards, Reflector
**Type:** Project

---

## Goal

Implement a hybrid RBAC + ABAC permission system where roles grant broad access and ownership checks provide fine-grained resource-level control.

## What to Build

- `Role` entity: ADMIN, EDITOR, VIEWER stored in DB
- `Permission` entity: action + resource pairs (e.g., `posts:update`, `users:delete`)
- `RolePermission` join table: which permissions each role has
- `@Roles()` decorator + `RolesGuard` for coarse-grained route protection
- `@RequirePermission()` decorator for specific permission checks
- `OwnershipGuard`: ensures the authenticated user owns the resource (ABAC)
- `@Public()` bypass for unauthenticated routes
- Admin panel endpoint to assign roles to users and manage role permissions
- Seed script that creates default roles and permissions

### `src/roles/entities/role.entity.ts`
```typescript
import { Entity, PrimaryGeneratedColumn, Column, ManyToMany, JoinTable } from 'typeorm';
import { Permission } from './permission.entity';
import { User } from '../../users/entities/user.entity';

export enum RoleName {
  ADMIN = 'ADMIN',
  EDITOR = 'EDITOR',
  VIEWER = 'VIEWER',
}

@Entity('roles')
export class Role {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ type: 'enum', enum: RoleName, unique: true })
  name: RoleName;

  @ManyToMany(() => Permission, { eager: true, cascade: true })
  @JoinTable({ name: 'role_permissions' })
  permissions: Permission[];

  @ManyToMany(() => User, (user) => user.roles)
  users: User[];
}
```

### `src/users/entities/user.entity.ts` (updated)
```typescript
import { Entity, PrimaryGeneratedColumn, Column, CreateDateColumn, ManyToMany, JoinTable } from 'typeorm';
import { Role } from '../../roles/entities/role.entity';

@Entity('users')
export class User {
  @PrimaryGeneratedColumn('uuid')
  id: string;

  @Column({ unique: true })
  email: string;

  @Column()
  name: string;

  @Column({ select: false })
  passwordHash: string;

  @Column({ default: false })
  emailVerified: boolean;

  @CreateDateColumn()
  createdAt: Date;

  @ManyToMany(() => Role, (role) => role.users, { eager: true })
  @JoinTable({ name: 'user_roles' })
  roles: Role[];

  hasRole(roleName: string): boolean {
    return this.roles?.some((r) => r.name === roleName) ?? false;
  }

  hasPermission(permissionKey: string): boolean {
    return this.roles?.some((r) =>
      r.permissions?.some((p) => p.key === permissionKey)
    ) ?? false;
  }
}
```

### `src/auth/guards/roles.guard.ts`
```typescript
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { ROLES_KEY } from '../decorators/roles.decorator';
import { RoleName } from '../../roles/entities/role.entity';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<RoleName[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (!requiredRoles || requiredRoles.length === 0) return true;

    const { user } = context.switchToHttp().getRequest();
    const hasRole = requiredRoles.some((role) => user?.hasRole(role));
    if (!hasRole) throw new ForbiddenException('Insufficient role');
    return true;
  }
}
```

```typescript
import { Injectable, CanActivate, ExecutionContext, ForbiddenException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { PERMISSION_KEY } from '../decorators/require-permission.decorator';

@Injectable()
export class PermissionsGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredPermission = this.reflector.getAllAndOverride<string>(PERMISSION_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (!requiredPermission) return true;

    const { user } = context.switchToHttp().getRequest();
    if (!user?.hasPermission(requiredPermission)) {
      throw new ForbiddenException(`Missing permission: ${requiredPermission}`);
    }
    return true;
  }
}
```

### `src/auth/guards/ownership.guard.ts`
```typescript
import { Injectable, CanActivate, ExecutionContext, ForbiddenException, NotFoundException } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { ModuleRef } from '@nestjs/core';
import { OWNERSHIP_KEY, OwnershipOptions } from '../decorators/check-ownership.decorator';
import { RoleName } from '../../roles/entities/role.entity';

@Injectable()
export class OwnershipGuard implements CanActivate {
  constructor(private reflector: Reflector, private moduleRef: ModuleRef) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const options = this.reflector.get<OwnershipOptions>(OWNERSHIP_KEY, context.getHandler());
    if (!options) return true;

    const request = context.switchToHttp().getRequest();
    const user = request.user;

    // Admins bypass ownership checks
    if (user?.hasRole(RoleName.ADMIN)) return true;

    const resourceId = request.params[options.idParam];
    const service = this.moduleRef.get(options.service, { strict: false });
    const resource = await service.findOne(resourceId);

    if (!resource) throw new NotFoundException('Resource not found');
    if (resource[options.ownerField] !== user.id) {
      throw new ForbiddenException('You do not own this resource');
    }
    return true;
  }
}
```

### `src/roles/roles.service.ts`
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Role, RoleName } from './entities/role.entity';
import { Permission } from './entities/permission.entity';
import { User } from '../users/entities/user.entity';

@Injectable()
export class RolesService {
  constructor(
    @InjectRepository(Role) private rolesRepo: Repository<Role>,
    @InjectRepository(Permission) private permRepo: Repository<Permission>,
    @InjectRepository(User) private usersRepo: Repository<User>,
  ) {}

  findAll() { return this.rolesRepo.find({ relations: ['permissions'] }); }

  findPermissions() { return this.permRepo.find(); }

  async assignRoleToUser(userId: string, roleName: RoleName): Promise<User> {
    const user = await this.usersRepo.findOne({ where: { id: userId }, relations: ['roles'] });
    if (!user) throw new NotFoundException(`User ${userId} not found`);

    const role = await this.rolesRepo.findOne({ where: { name: roleName } });
    if (!role) throw new NotFoundException(`Role ${roleName} not found`);

    const alreadyAssigned = user.roles.some((r) => r.id === role.id);
    if (!alreadyAssigned) {
      user.roles = [...user.roles, role];
      await this.usersRepo.save(user);
    }
    return user;
  }

  async addPermissionToRole(roleName: RoleName, permissionKey: string): Promise<Role> {
    const role = await this.rolesRepo.findOne({ where: { name: roleName }, relations: ['permissions'] });
    if (!role) throw new NotFoundException(`Role ${roleName} not found`);

    const permission = await this.permRepo.findOne({ where: { key: permissionKey } });
    if (!permission) throw new NotFoundException(`Permission ${permissionKey} not found`);

    const already = role.permissions.some((p) => p.key === permissionKey);
    if (!already) {
      role.permissions = [...role.permissions, permission];
      await this.rolesRepo.save(role);
    }
    return role;
  }
}
```

### `src/roles/roles.controller.ts`
```typescript
import { Controller, Get, Post, Param, Body, UseGuards } from '@nestjs/common';
import { RolesService } from './roles.service';
import { RoleName } from './entities/role.entity';
import { JwtAuthGuard } from '../auth/guards/jwt-auth.guard';
import { RolesGuard } from '../auth/guards/roles.guard';
import { Roles } from '../auth/decorators/roles.decorator';

@Controller('admin/roles')
@UseGuards(JwtAuthGuard, RolesGuard)
@Roles(RoleName.ADMIN)
export class RolesController {
  constructor(private readonly rolesService: RolesService) {}

  @Get()
  findAll() { return this.rolesService.findAll(); }

  @Get('permissions')
  findPermissions() { return this.rolesService.findPermissions(); }

  @Post('assign')
  assignRole(@Body('userId') userId: string, @Body('role') role: RoleName) {
    return this.rolesService.assignRoleToUser(userId, role);
  }

  @Post(':roleName/permissions')
  addPermission(@Param('roleName') roleName: RoleName, @Body('permissionKey') key: string) {
    return this.rolesService.addPermissionToRole(roleName, key);
  }
}
```

### Example protected resource controller
```typescript
// src/resources/posts/posts.controller.ts
import { Controller, Get, Post, Patch, Delete, Param, Body, UseGuards } from '@nestjs/common';
import { JwtAuthGuard } from '../../auth/guards/jwt-auth.guard';
import { RolesGuard } from '../../auth/guards/roles.guard';
import { PermissionsGuard } from '../../auth/guards/permissions.guard';
import { OwnershipGuard } from '../../auth/guards/ownership.guard';
import { Roles } from '../../auth/decorators/roles.decorator';
import { RequirePermission } from '../../auth/decorators/require-permission.decorator';
import { CheckOwnership } from '../../auth/decorators/check-ownership.decorator';
import { RoleName } from '../../roles/entities/role.entity';
import { PostsService } from './posts.service';

@Controller('posts')
@UseGuards(JwtAuthGuard, RolesGuard, PermissionsGuard)
export class PostsController {
  constructor(private readonly postsService: PostsService) {}

  @Get()
  @Roles(RoleName.VIEWER, RoleName.EDITOR, RoleName.ADMIN)
  findAll() { return this.postsService.findAll(); }

  @Post()
  @RequirePermission('posts:create')
  create(@Body() body: any) { return this.postsService.create(body); }

  @Patch(':id')
  @RequirePermission('posts:update')
  @UseGuards(OwnershipGuard)
  @CheckOwnership({ service: 'PostsService', idParam: 'id', ownerField: 'authorId' })
  update(@Param('id') id: string, @Body() body: any) { return this.postsService.update(id, body); }

  @Delete(':id')
  @Roles(RoleName.ADMIN)
  remove(@Param('id') id: string) { return this.postsService.remove(id); }
}
```

### Seed script: `src/roles/seeds/roles.seed.ts`
```typescript
import { DataSource } from 'typeorm';
import { Role, RoleName } from '../entities/role.entity';
import { Permission } from '../entities/permission.entity';

const permissions = [
  { action: 'create', resource: 'posts', key: 'posts:create' },
  { action: 'read',   resource: 'posts', key: 'posts:read' },
  { action: 'update', resource: 'posts', key: 'posts:update' },
  { action: 'delete', resource: 'posts', key: 'posts:delete' },
  { action: 'delete', resource: 'users', key: 'users:delete' },
];

const rolePermissions: Record<RoleName, string[]> = {
  [RoleName.VIEWER]: ['posts:read'],
  [RoleName.EDITOR]: ['posts:create', 'posts:read', 'posts:update'],
  [RoleName.ADMIN]:  ['posts:create', 'posts:read', 'posts:update', 'posts:delete', 'users:delete'],
};

export async function seedRoles(dataSource: DataSource) {
  const permRepo = dataSource.getRepository(Permission);
  const roleRepo = dataSource.getRepository(Role);

  const savedPerms = await Promise.all(
    permissions.map((p) => permRepo.save(permRepo.create(p)))
  );

  for (const [roleName, keys] of Object.entries(rolePermissions)) {
    const rolePerms = savedPerms.filter((p) => keys.includes(p.key));
    await roleRepo.save(roleRepo.create({ name: roleName as RoleName, permissions: rolePerms }));
  }
  console.log('Roles and permissions seeded');
}
```

## Deliverables

- [ ] `Role`, `Permission` entities created with join table `role_permissions`
- [ ] `User.hasRole()` and `User.hasPermission()` helper methods work correctly
- [ ] `RolesGuard` returns 403 when user lacks the required role
- [ ] `PermissionsGuard` returns 403 when user lacks the required permission
- [ ] `OwnershipGuard` allows ADMIN to bypass ownership, blocks others if not owner
- [ ] `GET /admin/roles` only accessible by ADMIN (returns 403 for EDITOR/VIEWER)
- [ ] `PATCH /posts/:id` blocked for non-owner non-admin users
- [ ] Seed script creates three default roles with correct permissions
- [ ] `@Roles()` and `@RequirePermission()` can be combined on same route
