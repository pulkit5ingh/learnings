# P8.2 — Event-Driven Notification System

**Phase:** 8 — Microservices
**Topic:** RabbitMQ transport, domain events, event-driven microservices
**Type:** Project

---

## Goal

Build an event-driven notification system where a core domain service emits business events over RabbitMQ and a notification microservice consumes them to send email and SMS notifications.

## What to Build

- **core-service**: HTTP API that emits domain events (user.registered, order.placed, payment.failed) to RabbitMQ
- **notification-ms**: subscribes to domain events, dispatches email (Nodemailer) and SMS (Twilio stub)
- Durable queues with Dead Letter Exchange for failed notifications
- `@EventPattern` handlers in notification-ms
- Event schemas as shared TypeScript interfaces in `libs/events`
- Retry logic with backoff via RabbitMQ's `x-death` header
- Manual acknowledgement (`ack`/`nack`) for reliable delivery

### `apps/core-service/src/events/events.service.ts`
```typescript
import { Injectable, Inject, Logger } from '@nestjs/common';
import { ClientProxy } from '@nestjs/microservices';
import { UserRegisteredEvent, OrderPlacedEvent, PaymentFailedEvent, EVENTS } from '@app/events';

@Injectable()
export class EventsService {
  private readonly logger = new Logger(EventsService.name);

  constructor(
    @Inject('NOTIFICATION_SERVICE') private readonly client: ClientProxy,
  ) {}

  emitUserRegistered(data: UserRegisteredEvent): void {
    this.logger.log(`Emitting ${EVENTS.USER_REGISTERED} for user ${data.userId}`);
    this.client.emit(EVENTS.USER_REGISTERED, data);
  }

  emitOrderPlaced(data: OrderPlacedEvent): void {
    this.logger.log(`Emitting ${EVENTS.ORDER_PLACED} for order ${data.orderId}`);
    this.client.emit(EVENTS.ORDER_PLACED, data);
  }

  emitPaymentFailed(data: PaymentFailedEvent): void {
    this.logger.log(`Emitting ${EVENTS.PAYMENT_FAILED} for order ${data.orderId}`);
    this.client.emit(EVENTS.PAYMENT_FAILED, data);
  }
}
```

### `apps/core-service/src/resources/users/users.controller.ts`
```typescript
import { Controller, Post, Body, HttpCode, HttpStatus } from '@nestjs/common';
import { EventsService } from '../../events/events.service';
import { randomUUID } from 'crypto';

@Controller('users')
export class UsersController {
  constructor(private readonly eventsService: EventsService) {}

  @Post()
  @HttpCode(HttpStatus.CREATED)
  register(@Body() body: { name: string; email: string }) {
    const userId = randomUUID();

    // Emit domain event (fire and forget)
    this.eventsService.emitUserRegistered({
      userId,
      email: body.email,
      name: body.name,
      verifyUrl: `https://example.com/verify?token=${randomUUID()}`,
      registeredAt: new Date().toISOString(),
    });

    return { id: userId, name: body.name, email: body.email, message: 'Registration queued' };
  }
}
```

### `apps/notification-ms/src/handlers/user-registered.handler.ts`
```typescript
import { Controller, Logger } from '@nestjs/common';
import { EventPattern, Payload, Ctx, RmqContext } from '@nestjs/microservices';
import { UserRegisteredEvent, EVENTS } from '@app/events';
import { NotificationService } from '../notification.service';

@Controller()
export class UserRegisteredHandler {
  private readonly logger = new Logger(UserRegisteredHandler.name);

  constructor(private readonly notificationService: NotificationService) {}

  @EventPattern(EVENTS.USER_REGISTERED)
  async handle(@Payload() data: UserRegisteredEvent, @Ctx() context: RmqContext): Promise<void> {
    const channel = context.getChannelRef();
    const originalMsg = context.getMessage();

    try {
      this.logger.log(`Processing user.registered for ${data.email}`);

      await this.notificationService.sendEmail({
        to: data.email,
        subject: 'Welcome! Verify your email',
        template: 'welcome',
        context: { name: data.name, verifyUrl: data.verifyUrl },
      });

      channel.ack(originalMsg); // acknowledge success
      this.logger.log(`Acked user.registered for ${data.email}`);
    } catch (error) {
      this.logger.error(`Failed to process user.registered: ${error.message}`);

      const deathCount = originalMsg.properties?.headers?.['x-death']?.[0]?.count ?? 0;
      if (deathCount >= 3) {
        channel.ack(originalMsg); // give up after 3 retries — DLX will handle
        this.logger.warn(`Giving up on user.registered for ${data.email} after ${deathCount} attempts`);
      } else {
        channel.nack(originalMsg, false, false); // nack, don't requeue (let DLX handle)
      }
    }
  }
}
```

### `apps/notification-ms/src/handlers/order-placed.handler.ts`
```typescript
import { Controller, Logger } from '@nestjs/common';
import { EventPattern, Payload, Ctx, RmqContext } from '@nestjs/microservices';
import { OrderPlacedEvent, EVENTS } from '@app/events';
import { NotificationService } from '../notification.service';

@Controller()
export class OrderPlacedHandler {
  private readonly logger = new Logger(OrderPlacedHandler.name);

  constructor(private readonly notificationService: NotificationService) {}

  @EventPattern(EVENTS.ORDER_PLACED)
  async handle(@Payload() data: OrderPlacedEvent, @Ctx() context: RmqContext): Promise<void> {
    const channel = context.getChannelRef();
    const originalMsg = context.getMessage();

    try {
      await this.notificationService.sendEmail({
        to: data.userEmail,
        subject: `Order Confirmed #${data.orderId}`,
        template: 'order-confirmation',
        context: { name: data.userName, orderId: data.orderId, items: data.items, total: data.total },
      });

      // Also send SMS for high-value orders
      if (data.total > 100) {
        await this.notificationService.sendSms(
          data.userEmail, // in real app use phone number
          `Your order #${data.orderId} for $${data.total.toFixed(2)} has been confirmed!`,
        );
      }

      channel.ack(originalMsg);
    } catch (error) {
      this.logger.error(`order.placed handler error: ${error.message}`);
      channel.nack(originalMsg, false, false);
    }
  }
}
```

### `apps/notification-ms/src/notification.service.ts`
```typescript
import { Injectable, Logger } from '@nestjs/common';
import * as nodemailer from 'nodemailer';
import { ConfigService } from '@nestjs/config';

interface SendEmailOptions {
  to: string;
  subject: string;
  template: string;
  context: Record<string, unknown>;
}

@Injectable()
export class NotificationService {
  private readonly logger = new Logger(NotificationService.name);
  private transporter: nodemailer.Transporter;

  constructor(private readonly config: ConfigService) {
    this.transporter = nodemailer.createTransport({
      host: config.get('SMTP_HOST', 'smtp.ethereal.email'),
      port: config.get('SMTP_PORT', 587),
      auth: { user: config.get('SMTP_USER'), pass: config.get('SMTP_PASS') },
    });
  }

  async sendEmail(options: SendEmailOptions): Promise<void> {
    const html = this.renderTemplate(options.template, options.context);
    await this.transporter.sendMail({
      from: this.config.get('EMAIL_FROM', 'noreply@example.com'),
      to: options.to,
      subject: options.subject,
      html,
    });
    this.logger.log(`Email sent: "${options.subject}" → ${options.to}`);
  }

  async sendSms(to: string, message: string): Promise<void> {
    // Stub — replace with Twilio client in production
    this.logger.log(`SMS stub: to=${to} message="${message}"`);
  }

  private renderTemplate(template: string, context: Record<string, unknown>): string {
    // Simple template stub — use Handlebars in production
    return `<html><body><h2>${template}</h2><pre>${JSON.stringify(context, null, 2)}</pre></body></html>`;
  }
}
```

### RabbitMQ setup with DLX: `rabbitmq-setup.sh`
```bash
#!/bin/bash
# Run after RabbitMQ starts to set up DLX

RABBITMQ_HOST=${RABBITMQ_HOST:-localhost}

# Create DLX exchange
curl -s -u guest:guest -X PUT "http://$RABBITMQ_HOST:15672/api/exchanges/%2F/notifications.dlx" \
  -H "Content-Type: application/json" \
  -d '{"type": "direct", "durable": true}'

# Create dead letter queue
curl -s -u guest:guest -X PUT "http://$RABBITMQ_HOST:15672/api/queues/%2F/notifications.dead" \
  -H "Content-Type: application/json" \
  -d '{"durable": true}'

# Bind DLQ to DLX
curl -s -u guest:guest -X POST "http://$RABBITMQ_HOST:15672/api/bindings/%2F/e/notifications.dlx/q/notifications.dead" \
  -H "Content-Type: application/json" \
  -d '{"routing_key": "notifications.dead"}'

echo "RabbitMQ DLX setup complete"
```

## Deliverables

- [ ] `POST /users` in core-service emits `user.registered` to RabbitMQ via `client.emit()`
- [ ] notification-ms subscribes with `@EventPattern('user.registered')` and sends welcome email
- [ ] `channel.ack(originalMsg)` called on successful processing
- [ ] `channel.nack(originalMsg, false, false)` called on failure (routes to DLX)
- [ ] After 3 failures, message moves to `notifications.dead` DLQ (inspectable in RabbitMQ dashboard)
- [ ] `noAck: false` set in both core-service publisher and notification-ms consumer
- [ ] High-value orders (`total > 100`) trigger both email AND SMS handler
- [ ] `notification-ms` can be scaled to 2 replicas (competing consumers on same queue)
- [ ] Shared event types live in `libs/events` imported by both apps
- [ ] `docker-compose up` starts RabbitMQ + core-service + notification-ms (2 replicas)
