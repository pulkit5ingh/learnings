## Goal
Split a monolith into two communicating microservices using NestJS's built-in TCP transport, understanding the difference between request-response and fire-and-forget patterns.

## Background

NestJS microservices are independent processes that communicate via messages over a transport layer. Unlike HTTP (which is request-response only), message transports support two patterns:
- **Request-Response** (`@MessagePattern`) — the caller sends a message and waits for a reply. Useful when you need data back.
- **Event (fire-and-forget)** (`@EventPattern`) — the caller emits an event and doesn't wait. Useful for notifications, domain events, and side effects.

The **TCP transport** is the simplest — direct TCP socket connection. Not production-ready for distributed systems (no resilience, no persistence) but perfect for learning the message patterns before adding a real transport (Redis, RabbitMQ, gRPC).

`ClientProxy` is the client-side proxy NestJS provides. `client.send<TResult>(pattern, data)` returns an Observable for request-response. `client.emit<T>(pattern, data)` fires an event with no response.

### 2. Users microservice — bootstrap

**`apps/users-ms/src/main.ts`**
```typescript
import { NestFactory } from '@nestjs/core';
import { MicroserviceOptions, Transport } from '@nestjs/microservices';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.createMicroservice<MicroserviceOptions>(AppModule, {
    transport: Transport.TCP,
    options: {
      host: '0.0.0.0',
      port: 3001,    // TCP port, not HTTP
    },
  });

  await app.listen();
  console.log('Users microservice listening on TCP port 3001');
}
bootstrap();
```

### 3. Users microservice — message handlers

**`apps/users-ms/src/users/users.controller.ts`**
```typescript
import { Controller, Logger } from '@nestjs/common';
import { MessagePattern, EventPattern, Payload } from '@nestjs/microservices';
import { UsersService } from './users.service';

@Controller()
export class UsersMsController {
  private readonly logger = new Logger(UsersMsController.name);

  constructor(private readonly usersService: UsersService) {}

  // Request-Response: caller waits for reply
  @MessagePattern({ cmd: 'find_all_users' })
  findAll() {
    return this.usersService.findAll();
  }

  @MessagePattern({ cmd: 'find_user' })
  findOne(@Payload() data: { id: number }) {
    return this.usersService.findOne(data.id);
  }

  @MessagePattern({ cmd: 'create_user' })
  create(@Payload() data: { name: string; email: string; password: string }) {
    return this.usersService.create(data);
  }

  @MessagePattern({ cmd: 'update_user' })
  update(@Payload() data: { id: number; name?: string; email?: string }) {
    const { id, ...updateData } = data;
    return this.usersService.update(id, updateData);
  }

  @MessagePattern({ cmd: 'delete_user' })
  remove(@Payload() data: { id: number }) {
    return this.usersService.remove(data.id);
  }

  // Event (fire-and-forget): no reply sent
  @EventPattern('user_logged_in')
  async handleUserLogin(@Payload() data: { userId: number; ip: string; timestamp: string }) {
    this.logger.log(`User ${data.userId} logged in from ${data.ip}`);
    await this.usersService.recordLogin(data.userId, data.ip);
    // No return value needed
  }

  @EventPattern('user_password_changed')
  async handlePasswordChange(@Payload() data: { userId: number }) {
    await this.usersService.invalidateAllSessions(data.userId);
    this.logger.log(`Sessions invalidated for user ${data.userId}`);
  }
}
```

### 4. API Gateway — HTTP API that calls the microservice

**`apps/gateway/src/app.module.ts`**
```typescript
import { Module } from '@nestjs/common';
import { ClientsModule, Transport } from '@nestjs/microservices';
import { UsersController } from './users/users.controller';

export const USERS_SERVICE = 'USERS_SERVICE';

@Module({
  imports: [
    ClientsModule.register([
      {
        name: USERS_SERVICE,
        transport: Transport.TCP,
        options: { host: 'localhost', port: 3001 },
      },
    ]),
  ],
  controllers: [UsersController],
})
export class AppModule {}
```

### 5. API Gateway — controller using ClientProxy

**`apps/gateway/src/users/users.controller.ts`**
```typescript
import {
  Controller, Get, Post, Patch, Delete,
  Param, Body, Inject, ParseIntPipe, Req,
} from '@nestjs/common';
import { ClientProxy } from '@nestjs/microservices';
import { lastValueFrom } from 'rxjs';
import { timeout, catchError } from 'rxjs/operators';
import { throwError } from 'rxjs';
import { USERS_SERVICE } from '../app.module';

@Controller('users')
export class UsersController {
  constructor(
    @Inject(USERS_SERVICE) private readonly usersClient: ClientProxy,
  ) {}

  @Get()
  async findAll() {
    return lastValueFrom(
      this.usersClient
        .send({ cmd: 'find_all_users' }, {})
        .pipe(timeout(5000)),
    );
  }

  @Get(':id')
  async findOne(@Param('id', ParseIntPipe) id: number) {
    return lastValueFrom(
      this.usersClient.send({ cmd: 'find_user' }, { id }).pipe(timeout(5000)),
    );
  }

  @Post()
  async create(@Body() body: { name: string; email: string; password: string }) {
    return lastValueFrom(
      this.usersClient.send({ cmd: 'create_user' }, body).pipe(timeout(5000)),
    );
  }

  @Patch(':id')
  async update(@Param('id', ParseIntPipe) id: number, @Body() body: any) {
    return lastValueFrom(
      this.usersClient.send({ cmd: 'update_user' }, { id, ...body }).pipe(timeout(5000)),
    );
  }

  @Delete(':id')
  async remove(@Param('id', ParseIntPipe) id: number) {
    return lastValueFrom(
      this.usersClient.send({ cmd: 'delete_user' }, { id }).pipe(timeout(5000)),
    );
  }

  @Post(':id/login-event')
  emitLoginEvent(@Param('id', ParseIntPipe) id: number, @Req() req: any) {
    // Fire-and-forget — no await needed
    this.usersClient.emit('user_logged_in', {
      userId: id,
      ip: req.ip,
      timestamp: new Date().toISOString(),
    });
    return { received: true };
  }
}
```

### 6. Error propagation across microservices

```typescript
// In the microservice controller, throw exceptions normally:
@MessagePattern({ cmd: 'find_user' })
findOne(@Payload() data: { id: number }) {
  // If this throws NotFoundException, it propagates back to the gateway
  return this.usersService.findOne(data.id);
}

// In the gateway, errors are thrown as RpcException:
import { RpcException } from '@nestjs/microservices';

// Catch RpcException in a filter:
@Catch(RpcException)
export class RpcExceptionFilter implements ExceptionFilter {
  catch(exception: RpcException, host: ArgumentsHost) {
    const ctx = host.switchToHttp();
    const response = ctx.getResponse<Response>();
    const error = exception.getError() as any;

    response.status(error.statusCode ?? 500).json({
      message: error.message ?? 'Microservice error',
      status: error.statusCode ?? 500,
    });
  }
}
```
