# A6.4 — Rate Limiting

**Type:** Assignment

## Goal
Protect API endpoints from abuse with rate limiting using `@nestjs/throttler`, configure per-route limits, back the limiter with Redis, and implement API-key based rate limits.

## Background

Rate limiting prevents a single client from overwhelming your API with requests. The `@nestjs/throttler` package provides a `ThrottlerGuard` that tracks request counts by a client identifier (default: IP address) in a store (default: in-memory Map).

**Key concepts:**
- **TTL** — time window in seconds. Requests within this window are counted.
- **Limit** — max requests allowed within the TTL window.
- **Throttler names** — you can define multiple named throttlers (e.g., `short` and `long`) and stack them for multi-window rate limiting.
- **ThrottlerGuard** — applied globally or per-controller/route. Returns 429 Too Many Requests when the limit is exceeded.
- **Custom key generator** — override the `generateKey` method to rate-limit by API key, user ID, or any request attribute.

For production, replace the in-memory store with a Redis-backed store so rate limit counters persist across server restarts and are shared across multiple instances.

### 2. Register ThrottlerModule with Redis storage

**`src/app.module.ts`**
```typescript
import { ThrottlerModule } from '@nestjs/throttler';
import { ThrottlerStorageRedisService } from '@nest-lab/throttler-storage-redis';
import { Redis } from 'ioredis';
import { ConfigService } from '@nestjs/config';

@Module({
  imports: [
    ThrottlerModule.forRootAsync({
      imports: [ConfigModule],
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        throttlers: [
          {
            name: 'short',
            ttl: 1,      // 1 second
            limit: 10,   // 10 requests per second
          },
          {
            name: 'medium',
            ttl: 60,     // 1 minute
            limit: 100,  // 100 requests per minute
          },
          {
            name: 'long',
            ttl: 3600,   // 1 hour
            limit: 1000, // 1000 requests per hour
          },
        ],
        storage: new ThrottlerStorageRedisService(
          new Redis({
            host: config.get('REDIS_HOST', 'localhost'),
            port: config.get<number>('REDIS_PORT', 6379),
            password: config.get('REDIS_PASSWORD'),
          }),
        ),
        skipIf: (ctx) => {
          // Skip rate limiting in test environment
          return process.env.NODE_ENV === 'test';
        },
      }),
    }),
  ],
})
export class AppModule {}
```

### 4. Skip throttling on certain routes

```typescript
import { SkipThrottle, Throttle } from '@nestjs/throttler';

@Controller('auth')
export class AuthController {
  @Post('login')
  @Throttle({ short: { limit: 5, ttl: 60 } })   // stricter: 5/minute for login
  login(@Body() dto: LoginDto) { ... }

  @Post('register')
  @Throttle({ medium: { limit: 3, ttl: 3600 } }) // 3 registrations per hour per IP
  register(@Body() dto: RegisterDto) { ... }

  @Get('health')
  @SkipThrottle()    // no rate limiting on health checks
  health() { return 'ok'; }
}
```

### 5. Custom ThrottlerGuard with API key tracking

```typescript
import { Injectable, ExecutionContext } from '@nestjs/common';
import { ThrottlerGuard } from '@nestjs/throttler';
import { Request } from 'express';

@Injectable()
export class ApiKeyThrottlerGuard extends ThrottlerGuard {
  protected async generateKey(
    context: ExecutionContext,
    suffix: string,
    throttlerName: string,
  ): Promise<string> {
    const request = context.switchToHttp().getRequest<Request>();

    // Rate limit by API key if present, otherwise by IP
    const apiKey = request.headers['x-api-key'] as string;
    if (apiKey) {
      return `api_key:${apiKey}:${throttlerName}`;
    }

    // Rate limit by user ID if authenticated
    const userId = (request as any).user?.id;
    if (userId) {
      return `user:${userId}:${throttlerName}`;
    }

    // Fall back to IP-based limiting
    const ip = request.ip ?? 'unknown';
    return `ip:${ip}:${throttlerName}`;
  }

  protected getTracker(req: Request): Promise<string> {
    // The value returned here is combined with the suffix in generateKey
    const apiKey = req.headers['x-api-key'] as string;
    const userId = (req as any).user?.id;
    return Promise.resolve(apiKey ?? userId?.toString() ?? req.ip ?? 'unknown');
  }
}
```

Register as `APP_GUARD`:
```typescript
{ provide: APP_GUARD, useClass: ApiKeyThrottlerGuard }
```

### 6. Per-API-key dynamic limits

```typescript
@Injectable()
export class DynamicRateLimitGuard extends ThrottlerGuard {
  constructor(
    options: ThrottlerModuleOptions,
    storageService: ThrottlerStorage,
    private readonly apiKeyService: ApiKeyService,
  ) {
    super(options, storageService, new Reflector());
  }

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest<Request>();
    const apiKey = request.headers['x-api-key'] as string;

    if (apiKey) {
      // Fetch the rate limit configured for this API key
      const keyConfig = await this.apiKeyService.getConfig(apiKey);
      if (keyConfig) {
        // Override throttler options for this request
        (request as any)._rateLimit = {
          ttl: keyConfig.ttl,
          limit: keyConfig.limit,
        };
      }
    }

    return super.canActivate(context);
  }
}
```

### 7. Rate limit response headers

Add informative headers to all responses showing limit status:

```typescript
@Injectable()
export class RateLimitHeadersInterceptor implements NestInterceptor {
  intercept(context: ExecutionContext, next: CallHandler): Observable<any> {
    return next.handle().pipe(
      tap(() => {
        const response = context.switchToHttp().getResponse<Response>();
        const request = context.switchToHttp().getRequest<Request>();

        // ThrottlerGuard stores this on the request
        const throttleData = (request as any).throttleData;
        if (throttleData) {
          response.setHeader('X-RateLimit-Limit', throttleData.limit);
          response.setHeader('X-RateLimit-Remaining', throttleData.remaining);
          response.setHeader('X-RateLimit-Reset', throttleData.resetTime);
        }
      }),
    );
  }
}
```
