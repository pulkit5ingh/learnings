## Goal
Add multi-layer caching to a NestJS API using the built-in cache manager with Redis as the backing store, including automatic route caching and manual programmatic cache control.

## Background

Caching reduces latency and database load by storing the results of expensive operations and reusing them for repeated requests. NestJS provides `@nestjs/cache-manager` which wraps the `cache-manager` library and supports multiple stores (in-memory, Redis, etc.).

**Two levels of caching in NestJS:**
1. **Route caching** — The `CacheInterceptor` (from `@nestjs/cache-manager`) automatically caches the full response of GET endpoints. Use `@CacheKey('custom-key')` and `@CacheTTL(seconds)` to control the key and expiry.
2. **Programmatic caching** — Inject `CACHE_MANAGER` token and call `cache.get(key)`, `cache.set(key, value, ttl)`, and `cache.del(key)` directly in your services.

**Cache invalidation** is the hard problem: when underlying data changes, the cache must be updated or deleted. Strategies:
- TTL-based: let entries expire naturally. Simple but may serve stale data.
- Event-based: delete or update the cache when a write operation occurs.
- Tags/patterns: group related cache keys so all can be invalidated at once. Redis SCAN or key-prefix conventions support this.

### 2. Register CacheModule globally

**`src/app.module.ts`**
```typescript
import { CacheModule } from '@nestjs/cache-manager';
import { redisStore } from 'cache-manager-ioredis-yet';
import { ConfigModule, ConfigService } from '@nestjs/config';

@Module({
  imports: [
    ConfigModule.forRoot({ isGlobal: true }),

    CacheModule.registerAsync({
      isGlobal: true,
      imports: [ConfigModule],
      inject: [ConfigService],
      useFactory: async (config: ConfigService) => ({
        store: await redisStore({
          host: config.get('REDIS_HOST', 'localhost'),
          port: config.get<number>('REDIS_PORT', 6379),
          password: config.get('REDIS_PASSWORD'),
          ttl: 60,   // default TTL in seconds
        }),
      }),
    }),
  ],
})
export class AppModule {}
```

### 3. Route-level cache with CacheInterceptor

```typescript
import { CacheInterceptor, CacheKey, CacheTTL } from '@nestjs/cache-manager';
import { UseInterceptors, Controller, Get, Param } from '@nestjs/common';

@Controller('products')
@UseInterceptors(CacheInterceptor)   // applied to all routes in this controller
export class ProductsController {
  constructor(private readonly productsService: ProductsService) {}

  // Cached for 5 minutes — key is auto-generated from URL
  @Get()
  @CacheTTL(300)
  findAll() {
    return this.productsService.findAll();
  }

  // Custom cache key (same key regardless of URL variations)
  @Get('featured')
  @CacheKey('featured-products')
  @CacheTTL(600)   // 10 minutes
  getFeatured() {
    return this.productsService.getFeatured();
  }

  // No caching for this route — use @CacheTTL(0) or exclude the interceptor
  @Get(':id')
  @CacheTTL(0)
  findOne(@Param('id') id: string) {
    return this.productsService.findOne(+id);
  }
}
```

### 4. Programmatic cache in service

**`src/products/products.service.ts`**
```typescript
import { Inject, Injectable, Logger } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Product } from './entities/product.entity';

@Injectable()
export class ProductsService {
  private readonly logger = new Logger(ProductsService.name);
  private readonly CACHE_PREFIX = 'product:';
  private readonly LIST_KEY = 'products:all';

  constructor(
    @Inject(CACHE_MANAGER) private readonly cache: Cache,
    @InjectRepository(Product)
    private readonly productRepo: Repository<Product>,
  ) {}

  async findAll(): Promise<Product[]> {
    const cached = await this.cache.get<Product[]>(this.LIST_KEY);
    if (cached) {
      this.logger.debug('Cache HIT: products list');
      return cached;
    }

    this.logger.debug('Cache MISS: fetching products from DB');
    const products = await this.productRepo.find({ order: { createdAt: 'DESC' } });
    await this.cache.set(this.LIST_KEY, products, 120); // 2 minutes
    return products;
  }

  async findOne(id: number): Promise<Product> {
    const cacheKey = `${this.CACHE_PREFIX}${id}`;
    const cached = await this.cache.get<Product>(cacheKey);
    if (cached) return cached;

    const product = await this.productRepo.findOneOrFail({ where: { id } });
    await this.cache.set(cacheKey, product, 300); // 5 minutes
    return product;
  }

  async create(dto: CreateProductDto): Promise<Product> {
    const product = await this.productRepo.save(this.productRepo.create(dto));
    await this.invalidateListCache();
    return product;
  }

  async update(id: number, dto: UpdateProductDto): Promise<Product> {
    await this.productRepo.update(id, dto);
    const updated = await this.productRepo.findOneOrFail({ where: { id } });

    // Invalidate specific item and list cache
    await this.cache.del(`${this.CACHE_PREFIX}${id}`);
    await this.invalidateListCache();
    return updated;
  }

  async remove(id: number): Promise<void> {
    await this.productRepo.delete(id);
    await this.cache.del(`${this.CACHE_PREFIX}${id}`);
    await this.invalidateListCache();
  }

  private async invalidateListCache(): Promise<void> {
    await this.cache.del(this.LIST_KEY);
    this.logger.debug('Cache INVALIDATED: products list');
  }
}
```

### 5. Cache warmup on app start

```typescript
import { Injectable, OnApplicationBootstrap } from '@nestjs/common';
import { Inject } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';

@Injectable()
export class CacheWarmupService implements OnApplicationBootstrap {
  constructor(
    @Inject(CACHE_MANAGER) private readonly cache: Cache,
    private readonly productsService: ProductsService,
  ) {}

  async onApplicationBootstrap(): Promise<void> {
    // Pre-warm popular caches after app starts
    try {
      await this.productsService.findAll();  // triggers DB fetch + cache set
      console.log('[Cache] Warmed up products cache');
    } catch (err) {
      console.warn('[Cache] Warmup failed:', err.message);
    }
  }
}
```

### 6. Redis key pattern deletion (bulk invalidation)

```typescript
import { Injectable, Inject } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';
import Redis from 'ioredis';

@Injectable()
export class CacheService {
  constructor(@Inject(CACHE_MANAGER) private readonly cache: Cache) {}

  private getRedisClient(): Redis {
    // Access the underlying ioredis client
    return (this.cache as any).store.client as Redis;
  }

  async invalidateByPattern(pattern: string): Promise<number> {
    const client = this.getRedisClient();
    const keys: string[] = [];
    let cursor = '0';

    do {
      const [nextCursor, found] = await client.scan(cursor, 'MATCH', pattern, 'COUNT', 100);
      cursor = nextCursor;
      keys.push(...found);
    } while (cursor !== '0');

    if (keys.length === 0) return 0;
    await client.del(...keys);
    return keys.length;
  }
}
```

Usage:
```typescript
// Invalidate all product caches
await cacheService.invalidateByPattern('product:*');
// Invalidate a specific user's data caches
await cacheService.invalidateByPattern(`user:${userId}:*`);
```
