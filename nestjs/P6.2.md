# P6.2 — Cache-Heavy Product API

**Phase:** 6 — Caching, Queues & Performance
**Topic:** Redis caching, cache invalidation, rate limiting per API key
**Type:** Project

---

## Goal

Build a product API with Redis-backed response caching, intelligent cache invalidation on updates, and per-API-key rate limiting using `@nestjs/throttler`.

## What to Build

- Product CRUD with PostgreSQL + TypeORM
- `@CacheKey` + `@CacheTTL` on read endpoints (Redis, 5-minute TTL)
- Manual cache invalidation when a product is updated or deleted
- `ApiKey` entity + `ApiKeyGuard` extracting key from `X-API-Key` header
- Per-API-key rate limiting: 100 requests/minute for standard, 1000/minute for premium
- `ThrottlerGuard` using Redis storage for distributed rate limiting
- Cache warm-up: pre-populate cache for top-N products on startup
- Cache key strategy: `products:all`, `products:{id}`, `products:category:{id}`

### `src/api-keys/guards/api-key.guard.ts`
```typescript
import { Injectable, CanActivate, ExecutionContext, UnauthorizedException, ForbiddenException } from '@nestjs/common';
import { ApiKeysService } from '../api-keys.service';

@Injectable()
export class ApiKeyGuard implements CanActivate {
  constructor(private readonly apiKeysService: ApiKeysService) {}

  async canActivate(context: ExecutionContext): Promise<boolean> {
    const request = context.switchToHttp().getRequest();
    const key = request.headers['x-api-key'];

    if (!key) throw new UnauthorizedException('API key required (X-API-Key header)');

    const apiKey = await this.apiKeysService.validate(key as string);
    if (!apiKey) throw new UnauthorizedException('Invalid API key');
    if (!apiKey.active) throw new ForbiddenException('API key has been deactivated');

    request.apiKey = apiKey; // attach to request for downstream use
    return true;
  }
}
```

### `src/api-keys/api-keys.service.ts`
```typescript
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { ApiKey, ApiKeyTier } from './api-key.entity';
import { randomBytes } from 'crypto';

@Injectable()
export class ApiKeysService {
  constructor(@InjectRepository(ApiKey) private repo: Repository<ApiKey>) {}

  async create(owner: string, tier: ApiKeyTier = ApiKeyTier.STANDARD): Promise<ApiKey> {
    const key = `ak_${tier === ApiKeyTier.PREMIUM ? 'prem' : 'std'}_${randomBytes(24).toString('hex')}`;
    return this.repo.save(this.repo.create({ key, owner, tier }));
  }

  async validate(key: string): Promise<ApiKey | null> {
    return this.repo.findOne({ where: { key } });
  }

  async revoke(id: string): Promise<void> {
    await this.repo.update(id, { active: false });
  }

  findAll(): Promise<ApiKey[]> {
    return this.repo.find({ order: { createdAt: 'DESC' } });
  }
}
```

### `src/cache/cache.service.ts`
```typescript
import { Injectable, Inject, Logger } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';

@Injectable()
export class CacheService {
  private readonly logger = new Logger(CacheService.name);

  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  async get<T>(key: string): Promise<T | null> {
    return (await this.cacheManager.get<T>(key)) ?? null;
  }

  async set(key: string, value: unknown, ttlSeconds = 300): Promise<void> {
    await this.cacheManager.set(key, value, ttlSeconds * 1000);
  }

  async del(key: string): Promise<void> {
    await this.cacheManager.del(key);
  }

  async delMany(keys: string[]): Promise<void> {
    await Promise.all(keys.map((k) => this.cacheManager.del(k)));
  }

  productKey(id: string): string { return `products:${id}`; }
  productsAllKey(): string { return 'products:all'; }
  productsByCategoryKey(categoryId: string): string { return `products:category:${categoryId}`; }
}
```

### `src/products/products.service.ts`
```typescript
import { Injectable, NotFoundException, OnModuleInit, Logger } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Product } from './entities/product.entity';
import { CacheService } from '../cache/cache.service';

const CACHE_TTL = 300; // 5 minutes

@Injectable()
export class ProductsService implements OnModuleInit {
  private readonly logger = new Logger(ProductsService.name);

  constructor(
    @InjectRepository(Product) private repo: Repository<Product>,
    private readonly cache: CacheService,
  ) {}

  async onModuleInit(): Promise<void> {
    await this.warmCache();
  }

  private async warmCache(): Promise<void> {
    this.logger.log('Warming product cache...');
    const topProducts = await this.repo.find({
      where: { active: true },
      order: { viewCount: 'DESC' },
      take: 20,
    });
    for (const product of topProducts) {
      await this.cache.set(this.cache.productKey(product.id), product, CACHE_TTL);
    }
    await this.cache.set(this.cache.productsAllKey(), topProducts, CACHE_TTL);
    this.logger.log(`Cache warmed with ${topProducts.length} products`);
  }

  async create(data: Partial<Product>): Promise<Product> {
    const product = await this.repo.save(this.repo.create(data));
    await this.cache.del(this.cache.productsAllKey());
    if (product.categoryId) {
      await this.cache.del(this.cache.productsByCategoryKey(product.categoryId));
    }
    return product;
  }

  async findAll(): Promise<Product[]> {
    const cached = await this.cache.get<Product[]>(this.cache.productsAllKey());
    if (cached) return cached;

    const products = await this.repo.find({ where: { active: true }, order: { createdAt: 'DESC' } });
    await this.cache.set(this.cache.productsAllKey(), products, CACHE_TTL);
    return products;
  }

  async findOne(id: string): Promise<Product> {
    const cached = await this.cache.get<Product>(this.cache.productKey(id));
    if (cached) return cached;

    const product = await this.repo.findOne({ where: { id } });
    if (!product) throw new NotFoundException(`Product ${id} not found`);

    // Increment view count asynchronously
    this.repo.increment({ id }, 'viewCount', 1).catch(() => {});

    await this.cache.set(this.cache.productKey(id), product, CACHE_TTL);
    return product;
  }

  async findByCategory(categoryId: string): Promise<Product[]> {
    const cacheKey = this.cache.productsByCategoryKey(categoryId);
    const cached = await this.cache.get<Product[]>(cacheKey);
    if (cached) return cached;

    const products = await this.repo.find({ where: { categoryId, active: true } });
    await this.cache.set(cacheKey, products, CACHE_TTL);
    return products;
  }

  async update(id: string, data: Partial<Product>): Promise<Product> {
    const existing = await this.findOne(id);
    await this.repo.update(id, data);

    // Invalidate all related cache keys
    await this.cache.delMany([
      this.cache.productKey(id),
      this.cache.productsAllKey(),
      this.cache.productsByCategoryKey(existing.categoryId),
    ]);

    return this.repo.findOne({ where: { id } }) as Promise<Product>;
  }

  async remove(id: string): Promise<void> {
    const product = await this.findOne(id);
    await this.repo.delete(id);
    await this.cache.delMany([
      this.cache.productKey(id),
      this.cache.productsAllKey(),
      this.cache.productsByCategoryKey(product.categoryId),
    ]);
  }
}
```

### `src/products/products.controller.ts`
```typescript
import { Controller, Get, Post, Patch, Delete, Param, Body, UseGuards, Req, HttpCode, HttpStatus } from '@nestjs/common';
import { Throttle } from '@nestjs/throttler';
import { ProductsService } from './products.service';
import { ApiKeyGuard } from '../api-keys/guards/api-key.guard';
import { ApiKey, ApiKeyTier } from '../api-keys/api-key.entity';

@Controller('products')
@UseGuards(ApiKeyGuard)
export class ProductsController {
  constructor(private readonly productsService: ProductsService) {}

  @Get()
  @Throttle({ default: { ttl: 60000, limit: 100 } })
  findAll() { return this.productsService.findAll(); }

  @Get('category/:categoryId')
  @Throttle({ default: { ttl: 60000, limit: 100 } })
  findByCategory(@Param('categoryId') categoryId: string) {
    return this.productsService.findByCategory(categoryId);
  }

  @Get(':id')
  findOne(@Param('id') id: string) { return this.productsService.findOne(id); }

  @Post()
  @HttpCode(HttpStatus.CREATED)
  create(@Body() body: any) { return this.productsService.create(body); }

  @Patch(':id')
  update(@Param('id') id: string, @Body() body: any) {
    return this.productsService.update(id, body);
  }

  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)
  remove(@Param('id') id: string) { return this.productsService.remove(id); }
}
```

### `src/throttling/throttle-config.service.ts`
```typescript
import { Injectable } from '@nestjs/common';
import { ThrottlerOptionsFactory, ThrottlerModuleOptions } from '@nestjs/throttler';
import { ThrottlerStorageRedisService } from 'nestjs-throttler-storage-redis';
import { ConfigService } from '@nestjs/config';

@Injectable()
export class ThrottleConfigService implements ThrottlerOptionsFactory {
  constructor(private readonly config: ConfigService) {}

  createThrottlerOptions(): ThrottlerModuleOptions {
    return {
      throttlers: [
        { name: 'default', ttl: 60000, limit: 100 },
        { name: 'premium', ttl: 60000, limit: 1000 },
      ],
      storage: new ThrottlerStorageRedisService({
        host: this.config.get('REDIS_HOST', 'localhost'),
        port: this.config.get('REDIS_PORT', 6379),
      }),
    };
  }
}
```

### `src/app.module.ts`
```typescript
import { Module } from '@nestjs/common';
import { CacheModule } from '@nestjs/cache-manager';
import { ThrottlerModule } from '@nestjs/throttler';
import { ConfigModule, ConfigService } from '@nestjs/config';
import { redisStore } from 'cache-manager-redis-yet';
import { ProductsModule } from './products/products.module';
import { ApiKeysModule } from './api-keys/api-keys.module';
import { ThrottleConfigService } from './throttling/throttle-config.service';

@Module({
  imports: [
    ConfigModule.forRoot({ isGlobal: true }),
    CacheModule.registerAsync({
      isGlobal: true,
      inject: [ConfigService],
      useFactory: async (config: ConfigService) => ({
        store: await redisStore({
          socket: { host: config.get('REDIS_HOST', 'localhost'), port: config.get<number>('REDIS_PORT', 6379) },
        }),
        ttl: 300_000, // default 5 min
      }),
    }),
    ThrottlerModule.forRootAsync({
      useClass: ThrottleConfigService,
    }),
    ProductsModule,
    ApiKeysModule,
  ],
})
export class AppModule {}
```

## Deliverables

- [ ] `GET /products` returns cached result on second call (no DB query)
- [ ] `PATCH /products/:id` invalidates `products:{id}`, `products:all`, and `products:category:{id}`
- [ ] `GET /products` without `X-API-Key` header returns 401
- [ ] Invalid or inactive API key returns 401 / 403 respectively
- [ ] Standard API key is rate-limited at 100 req/min (returns 429 when exceeded)
- [ ] Rate limits are stored in Redis (work across multiple app instances)
- [ ] Cache warm-up runs on module init and pre-populates top-20 products
- [ ] `CacheService.productKey()` and friends prevent cache key typos
- [ ] `PATCH` and `DELETE` trigger targeted invalidation, not full cache flush
- [ ] `Product.viewCount` is incremented asynchronously (fire-and-forget) on each `findOne`
