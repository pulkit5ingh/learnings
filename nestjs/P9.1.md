# P9.1 — Real-Time Collaboration API

**Phase:** 9 — Advanced & Production
**Topic:** WebSocket gateway, rooms, presence, Redis pub/sub
**Type:** Project

---

## Goal

Build a real-time collaboration API where users join document rooms, see who is online (presence), send cursor positions, and receive live document updates — all over WebSockets with Redis pub/sub for horizontal scaling.

## What to Build

- `@WebSocketGateway` with `socket.io` adapter
- Rooms: join/leave a document room by ID
- Presence: broadcast online users in a room
- Live cursor positions: user broadcasts their cursor, others receive it
- Document updates: broadcast text changes to all room members
- WebSocket authentication (JWT token in handshake)
- Redis pub/sub adapter (`@socket.io/redis-adapter`) for multi-instance scaling
- Persistent room membership tracking in Redis

### `src/main.ts`
```typescript
import { NestFactory } from '@nestjs/core';
import { AppModule } from './app.module';
import { IoAdapter } from '@nestjs/platform-socket.io';
import { createAdapter } from '@socket.io/redis-adapter';
import { createClient } from 'redis';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  const pubClient = createClient({ url: process.env.REDIS_URL ?? 'redis://localhost:6379' });
  const subClient = pubClient.duplicate();
  await Promise.all([pubClient.connect(), subClient.connect()]);

  const ioAdapter = new IoAdapter(app);
  (ioAdapter as any).createIOServer = function (port: number, options: any) {
    const server = (IoAdapter.prototype as any).createIOServer.call(this, port, options);
    server.adapter(createAdapter(pubClient, subClient));
    return server;
  };
  app.useWebSocketAdapter(ioAdapter);

  await app.listen(3000);
  console.log('Real-time collaboration API running on port 3000');
}
bootstrap();
```

### `src/auth/ws-auth.guard.ts`
```typescript
import { CanActivate, ExecutionContext, Injectable } from '@nestjs/common';
import { JwtService } from '@nestjs/jwt';
import { WsException } from '@nestjs/websockets';
import { Socket } from 'socket.io';
import { ConfigService } from '@nestjs/config';

@Injectable()
export class WsAuthGuard implements CanActivate {
  constructor(
    private readonly jwtService: JwtService,
    private readonly config: ConfigService,
  ) {}

  canActivate(context: ExecutionContext): boolean {
    const client: Socket = context.switchToWs().getClient();
    const token =
      client.handshake.auth?.token ??
      client.handshake.headers?.authorization?.replace('Bearer ', '');

    if (!token) throw new WsException('Missing authentication token');

    try {
      const payload = this.jwtService.verify(token, {
        secret: this.config.getOrThrow('JWT_ACCESS_SECRET'),
      });
      client.data.user = payload;
      return true;
    } catch {
      throw new WsException('Invalid or expired token');
    }
  }
}
```

import { Injectable, Inject } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';

export interface PresenceUser {
  userId: string;
  name: string;
  socketId: string;
  joinedAt: string;
}

@Injectable()
export class PresenceService {
  private readonly TTL = 3_600_000; // 1 hour

  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  private roomKey(roomId: string): string { return `presence:room:${roomId}`; }

  async joinRoom(roomId: string, user: PresenceUser): Promise<PresenceUser[]> {
    const users = await this.getRoomUsers(roomId);
    const filtered = users.filter((u) => u.userId !== user.userId);
    const updated = [...filtered, user];
    await this.cacheManager.set(this.roomKey(roomId), updated, this.TTL);
    return updated;
  }

  async leaveRoom(roomId: string, userId: string): Promise<PresenceUser[]> {
    const users = await this.getRoomUsers(roomId);
    const updated = users.filter((u) => u.userId !== userId);
    await this.cacheManager.set(this.roomKey(roomId), updated, this.TTL);
    return updated;
  }

  async getRoomUsers(roomId: string): Promise<PresenceUser[]> {
    return (await this.cacheManager.get<PresenceUser[]>(this.roomKey(roomId))) ?? [];
  }

  async leaveAllRooms(userId: string, socketId: string): Promise<void> {

  }
}

### `src/gateway/collab.gateway.ts`
```typescript
import {
  WebSocketGateway,
  WebSocketServer,
  SubscribeMessage,
  OnGatewayConnection,
  OnGatewayDisconnect,
  ConnectedSocket,
  MessageBody,
  WsException,
} from '@nestjs/websockets';
import { UseGuards, Logger } from '@nestjs/common';
import { Server, Socket } from 'socket.io';
import { JwtService } from '@nestjs/jwt';
import { ConfigService } from '@nestjs/config';
import { WsAuthGuard } from '../auth/ws-auth.guard';
import { PresenceService } from '../presence/presence.service';
import { JoinRoomDto } from './dto/join-room.dto';
import { CursorPositionDto } from './dto/cursor-position.dto';
import { DocumentUpdateDto } from './dto/document-update.dto';

@WebSocketGateway({
  cors: { origin: '*', credentials: true },
  namespace: '/collab',
})
export class CollabGateway implements OnGatewayConnection, OnGatewayDisconnect {
  @WebSocketServer()
  server: Server;

  private readonly logger = new Logger(CollabGateway.name);
  // Track which rooms each socket has joined
  private socketRooms = new Map<string, Set<string>>();

  constructor(
    private readonly presenceService: PresenceService,
    private readonly jwtService: JwtService,
    private readonly config: ConfigService,
  ) {}

  async handleConnection(client: Socket): Promise<void> {
    try {
      const token = client.handshake.auth?.token;
      if (!token) throw new Error('No token');

      const payload = this.jwtService.verify(token, {
        secret: this.config.getOrThrow('JWT_ACCESS_SECRET'),
      });
      client.data.user = payload;
      this.socketRooms.set(client.id, new Set());
      this.logger.log(`Client connected: ${client.id} (user: ${payload.sub})`);
    } catch (error) {
      this.logger.warn(`Connection rejected: ${error.message}`);
      client.emit('error', { message: 'Authentication failed' });
      client.disconnect();
    }
  }

  async handleDisconnect(client: Socket): Promise<void> {
    const rooms = this.socketRooms.get(client.id) ?? new Set();
    for (const roomId of rooms) {
      const onlineUsers = await this.presenceService.leaveRoom(roomId, client.data.user?.sub);
      this.server.to(roomId).emit('presence:update', { roomId, users: onlineUsers });
    }
    this.socketRooms.delete(client.id);
    this.logger.log(`Client disconnected: ${client.id}`);
  }

  @SubscribeMessage('room:join')
  async handleJoinRoom(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: JoinRoomDto,
  ) {
    const { roomId } = data;
    await client.join(roomId);
    this.socketRooms.get(client.id)?.add(roomId);

    const onlineUsers = await this.presenceService.joinRoom(roomId, {
      userId: client.data.user.sub,
      name: client.data.user.name ?? 'Unknown',
      socketId: client.id,
      joinedAt: new Date().toISOString(),
    });

    // Notify all room members of updated presence
    this.server.to(roomId).emit('presence:update', { roomId, users: onlineUsers });

    // Confirm to joining user
    client.emit('room:joined', { roomId, users: onlineUsers });
    this.logger.log(`User ${client.data.user.sub} joined room ${roomId}`);
  }

  @SubscribeMessage('room:leave')
  async handleLeaveRoom(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { roomId: string },
  ) {
    const { roomId } = data;
    await client.leave(roomId);
    this.socketRooms.get(client.id)?.delete(roomId);

    const onlineUsers = await this.presenceService.leaveRoom(roomId, client.data.user.sub);
    this.server.to(roomId).emit('presence:update', { roomId, users: onlineUsers });
    client.emit('room:left', { roomId });
  }

  @SubscribeMessage('cursor:move')
  handleCursorMove(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: CursorPositionDto,
  ) {
    // Broadcast cursor to all OTHER users in room (not sender)
    client.to(data.roomId).emit('cursor:update', {
      userId: client.data.user.sub,
      name: client.data.user.name,
      line: data.line,
      column: data.column,
    });
  }

  @SubscribeMessage('document:update')
  handleDocumentUpdate(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: DocumentUpdateDto,
  ) {
    // Broadcast delta to all OTHER users in room
    client.to(data.roomId).emit('document:patch', {
      userId: client.data.user.sub,
      delta: data.delta,
      version: data.version,
    });
  }

  @SubscribeMessage('presence:get')
  async handleGetPresence(
    @ConnectedSocket() client: Socket,
    @MessageBody() data: { roomId: string },
  ) {
    const users = await this.presenceService.getRoomUsers(data.roomId);
    client.emit('presence:current', { roomId: data.roomId, users });
  }
}
```

### Client-side usage example
```typescript
// Browser / Node.js client
import { io } from 'socket.io-client';

const socket = io('http://localhost:3000/collab', {
  auth: { token: 'your-jwt-access-token' },
});

socket.on('connect', () => {
  console.log('Connected:', socket.id);

  // Join a document room
  socket.emit('room:join', { roomId: 'document-abc123' });
});

socket.on('room:joined', ({ roomId, users }) => {
  console.log(`Joined ${roomId}, online users:`, users);
});

socket.on('presence:update', ({ roomId, users }) => {
  console.log(`Presence update in ${roomId}:`, users.map((u: any) => u.name));
});

socket.on('cursor:update', ({ userId, name, line, column }) => {
  console.log(`${name} moved cursor to line ${line}, col ${column}`);
});

socket.on('document:patch', ({ userId, delta, version }) => {
  console.log(`Doc patch from ${userId}:`, delta);
});

// Send cursor position
socket.emit('cursor:move', { roomId: 'document-abc123', line: 10, column: 5 });

// Send a text change (Quill delta format)
socket.emit('document:update', {
  roomId: 'document-abc123',
  delta: { ops: [{ retain: 10 }, { insert: 'Hello ' }] },
  version: 1,
});
```

## Deliverables

- [ ] `@WebSocketGateway` mounted at `/collab` namespace
- [ ] JWT token extracted from `socket.handshake.auth.token` on connection
- [ ] Invalid token causes immediate `client.disconnect()` with error event
- [ ] `room:join` event adds socket to Socket.IO room and updates presence in Redis
- [ ] `room:leave` and `handleDisconnect` both remove user from presence
- [ ] `presence:update` event broadcast to all room members on any join/leave
- [ ] `cursor:move` event forwarded to all OTHER sockets in the room (not sender)
- [ ] `document:update` delta broadcast to all OTHER sockets in the room
- [ ] Redis pub/sub adapter enables presence sync across multiple server instances
- [ ] `GET /collab/rooms/:roomId/presence` HTTP endpoint returns current online users
