# A5.4 — Feature Flags & App Settings

**Phase:** 5 — Configuration & Environment  
**Topic:** Dynamic config, database-driven settings, environment-specific behavior, change events  
**Type:** Assignment

---

## Goal
Implement a settings service that stores admin-configurable values in the database, caches them in memory, and broadcasts change events when settings are updated.

## Background

Static environment variables work well for infrastructure concerns (database host, secrets). But some application settings need to change at runtime without restarting the server — rate limits, maintenance mode, feature flags, email templates. These belong in a database-backed settings table.

**Feature flags** enable/disable features per environment or per user segment without a deployment. This pattern is used by every major tech company. A basic implementation stores flags in a database table. A production implementation uses a service like LaunchDarkly or Unleash, but understanding the underlying pattern is essential.

**Cache invalidation** is the hard problem: when a setting changes in the DB, the in-memory cache must be refreshed. In a single-node app, an `EventEmitter` works. In multi-node deployments, you need a shared cache (Redis pub/sub) to notify all instances.

import {
  Entity, PrimaryGeneratedColumn, Column,
  CreateDateColumn, UpdateDateColumn,
} from 'typeorm';

export enum SettingType {
  STRING = 'string',
  NUMBER = 'number',
  BOOLEAN = 'boolean',
  JSON = 'json',
}

@Entity('app_settings')
export class AppSetting {
  @PrimaryGeneratedColumn()
  id: number;

  @Column({ unique: true })
  key: string;

  @Column('text')
  value: string;   // stored as string, cast on read

  @Column({ type: 'enum', enum: SettingType, default: SettingType.STRING })
  type: SettingType;

  @Column({ nullable: true })
  description?: string;

  @Column({ default: false })
  isPublic: boolean;   // if true, can be returned to non-admin users

  @CreateDateColumn()
  createdAt: Date;

  @UpdateDateColumn()
  updatedAt: Date;
}

### 2. SettingsService with typed accessors

**`src/settings/settings.service.ts`**
```typescript
import { Injectable, Logger, NotFoundException, OnModuleInit } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { AppSetting, SettingType } from './entities/app-setting.entity';

export const SETTINGS_CHANGED_EVENT = 'settings.changed';

@Injectable()
export class SettingsService implements OnModuleInit {
  private readonly logger = new Logger(SettingsService.name);
  private cache = new Map<string, any>();

  constructor(
    @InjectRepository(AppSetting)
    private readonly repo: Repository<AppSetting>,
    private readonly eventEmitter: EventEmitter2,
  ) {}

  async onModuleInit(): Promise<void> {
    await this.loadAll();
  }

  async loadAll(): Promise<void> {
    const settings = await this.repo.find();
    this.cache.clear();
    for (const s of settings) {
      this.cache.set(s.key, this.cast(s.value, s.type));
    }
    this.logger.log(`Loaded ${settings.length} settings into cache`);
  }

  private cast(value: string, type: SettingType): any {
    switch (type) {
      case SettingType.NUMBER: return Number(value);
      case SettingType.BOOLEAN: return value === 'true';
      case SettingType.JSON: return JSON.parse(value);
      default: return value;
    }
  }

  get<T = string>(key: string, defaultValue?: T): T {
    if (this.cache.has(key)) return this.cache.get(key) as T;
    if (defaultValue !== undefined) return defaultValue;
    throw new Error(`Setting '${key}' not found`);
  }

  getOptional<T = string>(key: string): T | undefined {
    return this.cache.get(key) as T | undefined;
  }

  async set(key: string, value: any, type = SettingType.STRING): Promise<AppSetting> {
    const stringValue = type === SettingType.JSON
      ? JSON.stringify(value)
      : String(value);

    let setting = await this.repo.findOne({ where: { key } });
    if (setting) {
      setting.value = stringValue;
      setting.type = type;
    } else {
      setting = this.repo.create({ key, value: stringValue, type });
    }

    const saved = await this.repo.save(setting);
    this.cache.set(key, this.cast(stringValue, type));

    // Broadcast change event
    this.eventEmitter.emit(SETTINGS_CHANGED_EVENT, { key, value: this.cache.get(key) });
    this.logger.log(`Setting updated: ${key}`);

    return saved;
  }

  async delete(key: string): Promise<void> {
    const result = await this.repo.delete({ key });
    if (result.affected === 0) throw new NotFoundException(`Setting '${key}' not found`);
    this.cache.delete(key);
    this.eventEmitter.emit(SETTINGS_CHANGED_EVENT, { key, value: null });
  }

  getAll(includePrivate = false): Record<string, any> {
    if (includePrivate) return Object.fromEntries(this.cache);
    // Only public settings for non-admin
    // In a real app, query the DB with isPublic: true
    return Object.fromEntries(this.cache);
  }
}
```

### 3. Feature flags service

**`src/settings/feature-flags.service.ts`**
```typescript
import { Injectable } from '@nestjs/common';
import { SettingsService } from './settings.service';

export enum FeatureFlag {
  NEW_CHECKOUT_FLOW = 'feature.new_checkout_flow',
  MAINTENANCE_MODE = 'feature.maintenance_mode',
  EMAIL_NOTIFICATIONS = 'feature.email_notifications',
  RATE_LIMIT_STRICT = 'feature.rate_limit_strict',
}

@Injectable()
export class FeatureFlagsService {
  constructor(private readonly settings: SettingsService) {}

  isEnabled(flag: FeatureFlag): boolean {
    return this.settings.get<boolean>(flag, false);
  }

  isMaintenanceMode(): boolean {
    return this.isEnabled(FeatureFlag.MAINTENANCE_MODE);
  }

  areEmailNotificationsEnabled(): boolean {
    return this.isEnabled(FeatureFlag.EMAIL_NOTIFICATIONS);
  }
}
```

### 4. Maintenance mode guard

```typescript
import { CanActivate, ExecutionContext, Injectable, ServiceUnavailableException } from '@nestjs/common';
import { FeatureFlagsService } from '../settings/feature-flags.service';
import { Reflector } from '@nestjs/core';

export const BYPASS_MAINTENANCE = 'bypassMaintenance';
export const BypassMaintenance = () => SetMetadata(BYPASS_MAINTENANCE, true);

@Injectable()
export class MaintenanceModeGuard implements CanActivate {
  constructor(
    private readonly featureFlags: FeatureFlagsService,
    private readonly reflector: Reflector,
  ) {}

  canActivate(context: ExecutionContext): boolean {
    const bypass = this.reflector.getAllAndOverride<boolean>(BYPASS_MAINTENANCE, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (bypass) return true;

    if (this.featureFlags.isMaintenanceMode()) {
      throw new ServiceUnavailableException(
        'System is under maintenance. Please try again later.',
      );
    }
    return true;
  }
}
```

### 5. Listen to settings change events

```typescript
import { OnEvent } from '@nestjs/event-emitter';
import { SETTINGS_CHANGED_EVENT } from '../settings/settings.service';

@Injectable()
export class RateLimitService {
  private limit = 100;

  @OnEvent(SETTINGS_CHANGED_EVENT)
  handleSettingsChanged(payload: { key: string; value: any }) {
    if (payload.key === 'rate_limit.requests_per_minute') {
      this.limit = payload.value;
      console.log(`Rate limit updated to: ${this.limit}`);
    }
  }

  getLimit(): number {
    return this.limit;
  }
}
```

### 6. Admin settings controller

**`src/settings/settings.controller.ts`**
```typescript
import {
  Body, Controller, Delete, Get, Param,
  Post, Put, UseGuards,
} from '@nestjs/common';
import { SettingsService } from './settings.service';
import { SettingType } from './entities/app-setting.entity';
import { JwtAuthGuard } from '../auth/guards/jwt-auth.guard';
import { RolesGuard } from '../auth/guards/roles.guard';
import { Roles } from '../auth/decorators/roles.decorator';
import { Role } from '../common/enums/role.enum';

@Controller('admin/settings')
@UseGuards(JwtAuthGuard, RolesGuard)
@Roles(Role.ADMIN)
export class SettingsController {
  constructor(private readonly settingsService: SettingsService) {}

  @Get()
  getAll() {
    return this.settingsService.getAll(true);
  }

  @Put(':key')
  set(@Param('key') key: string, @Body() body: { value: any; type?: SettingType }) {
    return this.settingsService.set(key, body.value, body.type);
  }

  @Delete(':key')
  remove(@Param('key') key: string) {
    return this.settingsService.delete(key);
  }

  @Post('reload')
  async reload() {
    await this.settingsService.loadAll();
    return { message: 'Settings reloaded from database' };
  }
}
```
