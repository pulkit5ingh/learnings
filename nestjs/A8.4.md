## Goal
Implement gRPC-based microservice communication using Protocol Buffers, typed service definitions, and streaming RPCs.

## Background

gRPC (Google Remote Procedure Call) uses Protocol Buffers (protobuf) as its Interface Definition Language (IDL) and wire format. Unlike REST (JSON over HTTP/1.1) or TCP/Redis (JSON strings), gRPC provides:
- **Strongly typed contracts** — the `.proto` file is the API contract, shared between client and server.
- **Efficient binary serialisation** — protobuf is 3–10× smaller than JSON.
- **Four communication patterns** — unary, server streaming, client streaming, and bidirectional streaming.
- **HTTP/2** — multiplexing, header compression, and built-in support for streaming.

In NestJS, `@nestjs/microservices` includes gRPC support. You define a `.proto` file, use `@GrpcMethod('ServiceName', 'MethodName')` on handlers, and generate TypeScript types from the proto using `protoc` or `ts-proto`.

### 2. Proto file definition

**`proto/users.proto`**
```protobuf
syntax = "proto3";

package users;

service UsersService {
  rpc FindOne (FindOneRequest) returns (User);
  rpc FindAll (FindAllRequest) returns (UsersResponse);
  rpc Create (CreateUserRequest) returns (User);
  rpc Update (UpdateUserRequest) returns (User);
  rpc Delete (DeleteUserRequest) returns (DeleteUserResponse);

  // Server-side streaming: stream user events to the client
  rpc SubscribeToEvents (EventSubscribeRequest) returns (stream UserEvent);

  // Bidirectional streaming: bulk create users
  rpc BulkCreate (stream CreateUserRequest) returns (stream User);
}

message User {
  int32 id = 1;
  string name = 2;
  string email = 3;
  bool isActive = 4;
  string createdAt = 5;
}

message FindOneRequest {
  int32 id = 1;
}

message FindAllRequest {
  string search = 1;
  int32 page = 2;
  int32 limit = 3;
}

message UsersResponse {
  repeated User users = 1;
  int32 total = 2;
}

message CreateUserRequest {
  string name = 1;
  string email = 2;
  string password = 3;
}

message UpdateUserRequest {
  int32 id = 1;
  string name = 2;
  string email = 3;
}

message DeleteUserRequest {
  int32 id = 1;
}

message DeleteUserResponse {
  bool success = 1;
}

message EventSubscribeRequest {
  int32 userId = 1;
}

message UserEvent {
  string type = 1;
  string payload = 2;
  string timestamp = 3;
}
```

### 3. Users gRPC microservice

**`apps/users-grpc/src/main.ts`**
```typescript
import { NestFactory } from '@nestjs/core';
import { MicroserviceOptions, Transport } from '@nestjs/microservices';
import { join } from 'path';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.createMicroservice<MicroserviceOptions>(AppModule, {
    transport: Transport.GRPC,
    options: {
      url: '0.0.0.0:5000',
      package: 'users',
      protoPath: join(__dirname, '../../proto/users.proto'),
      loader: {
        keepCase: true,
        longs: String,
        enums: String,
        defaults: true,
        oneofs: true,
      },
    },
  });

  await app.listen();
  console.log('Users gRPC service running on port 5000');
}
bootstrap();
```

### 4. gRPC controller (service handler)

**`apps/users-grpc/src/users/users.grpc.controller.ts`**
```typescript
import { Controller } from '@nestjs/common';
import { GrpcMethod, GrpcStreamMethod } from '@nestjs/microservices';
import { Observable, Subject } from 'rxjs';
import { UsersService } from './users.service';

@Controller()
export class UsersGrpcController {
  constructor(private readonly usersService: UsersService) {}

  // Unary RPC
  @GrpcMethod('UsersService', 'FindOne')
  async findOne(data: { id: number }) {
    return this.usersService.findOne(data.id);
  }

  @GrpcMethod('UsersService', 'FindAll')
  async findAll(data: { search?: string; page?: number; limit?: number }) {
    const users = await this.usersService.findAll(data.search);
    return { users, total: users.length };
  }

  @GrpcMethod('UsersService', 'Create')
  async create(data: { name: string; email: string; password: string }) {
    return this.usersService.create(data);
  }

  // Server-streaming RPC: stream user events
  @GrpcMethod('UsersService', 'SubscribeToEvents')
  subscribeToEvents(data: { userId: number }): Observable<any> {
    const subject = new Subject<any>();

    // Simulate streaming events
    const interval = setInterval(() => {
      subject.next({
        type: 'HEARTBEAT',
        payload: JSON.stringify({ userId: data.userId }),
        timestamp: new Date().toISOString(),
      });
    }, 1000);

    // Complete the stream after 10 events
    let count = 0;
    const eventInterval = setInterval(() => {
      subject.next({
        type: 'USER_ACTIVITY',
        payload: JSON.stringify({ userId: data.userId, activity: 'page_view' }),
        timestamp: new Date().toISOString(),
      });
      count++;
      if (count >= 10) {
        clearInterval(interval);
        clearInterval(eventInterval);
        subject.complete();
      }
    }, 500);

    return subject.asObservable();
  }

  // Bidirectional streaming RPC: bulk create
  @GrpcStreamMethod('UsersService', 'BulkCreate')
  bulkCreate(messages: Observable<{ name: string; email: string; password: string }>): Observable<any> {
    const subject = new Subject<any>();

    messages.subscribe({
      next: async (data) => {
        const user = await this.usersService.create(data);
        subject.next(user);
      },
      complete: () => subject.complete(),
      error: (err) => subject.error(err),
    });

    return subject.asObservable();
  }
}
```

**`src/users/users.service.ts`** in the gateway:
```typescript
import { Injectable, OnModuleInit, Inject } from '@nestjs/common';
import { ClientGrpc } from '@nestjs/microservices';
import { Observable, lastValueFrom } from 'rxjs';

interface UsersGrpcService {
  findOne(data: { id: number }): Observable<any>;
  findAll(data: { search?: string }): Observable<{ users: any[]; total: number }>;
  create(data: { name: string; email: string; password: string }): Observable<any>;
  subscribeToEvents(data: { userId: number }): Observable<any>;
}

@Injectable()
export class UsersService implements OnModuleInit {
  private usersGrpcService: UsersGrpcService;

  constructor(@Inject('USERS_GRPC') private readonly client: ClientGrpc) {}

  onModuleInit() {
    this.usersGrpcService = this.client.getService<UsersGrpcService>('UsersService');
  }

  async findOne(id: number) {
    return lastValueFrom(this.usersGrpcService.findOne({ id }));
  }

  async findAll(search?: string) {
    return lastValueFrom(this.usersGrpcService.findAll({ search }));
  }

  async create(data: { name: string; email: string; password: string }) {
    return lastValueFrom(this.usersGrpcService.create(data));
  }

  streamUserEvents(userId: number): Observable<any> {
    return this.usersGrpcService.subscribeToEvents({ userId });
  }
}
```

### 6. Streaming in an HTTP endpoint (SSE)

```typescript
import { Controller, Get, Param, Res, ParseIntPipe } from '@nestjs/common';
import { Response } from 'express';

@Controller('users')
export class UsersController {
  // Stream gRPC events to HTTP client via Server-Sent Events
  @Get(':id/events')
  streamEvents(
    @Param('id', ParseIntPipe) id: number,
    @Res() res: Response,
  ) {
    res.setHeader('Content-Type', 'text/event-stream');
    res.setHeader('Cache-Control', 'no-cache');
    res.setHeader('Connection', 'keep-alive');

    const stream = this.usersService.streamUserEvents(id);
    const subscription = stream.subscribe({
      next: (event) => res.write(`data: ${JSON.stringify(event)}\n\n`),
      complete: () => res.end(),
      error: (err) => {
        res.write(`data: ${JSON.stringify({ error: err.message })}\n\n`);
        res.end();
      },
    });

    res.on('close', () => subscription.unsubscribe());
  }
}
```

### 7. gRPC error handling

```typescript
import { status } from '@grpc/grpc-js';
import { RpcException } from '@nestjs/microservices';

// In the gRPC controller:
@GrpcMethod('UsersService', 'FindOne')
async findOne(data: { id: number }) {
  const user = await this.usersService.findOne(data.id).catch(() => null);
  if (!user) {
    throw new RpcException({
      code: status.NOT_FOUND,
      message: `User ${data.id} not found`,
    });
  }
  return user;
}
```
