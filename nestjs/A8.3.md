## Goal
Use RabbitMQ as the message broker for microservice communication, configure durable queues with manual acknowledgement, and set up a dead letter exchange for failed messages.

## Background

RabbitMQ is a production-grade message broker with rich routing capabilities. Unlike Redis pub/sub (which loses messages if no subscriber is listening), RabbitMQ persists messages in durable queues. If a microservice goes down, messages accumulate in the queue and are processed when it comes back up.

**Key RabbitMQ concepts:**
- **Exchange** — receives messages and routes them to queues based on routing keys.
- **Queue** — stores messages until a consumer processes them.
- **Binding** — connects an exchange to a queue with a routing key.
- **Durable** — survives broker restart.
- **noAck: false** — manual acknowledgement mode; the broker waits for `channel.ack()` before removing the message. If the consumer crashes before acking, the message is requeued.
- **Dead Letter Exchange (DLX)** — when a message fails (rejected or TTL expired), RabbitMQ routes it to a DLX. You bind a dead letter queue to the DLX to inspect failed messages.

### 3. Users microservice with RMQ transport

**`apps/users-ms/src/main.ts`**
```typescript
import { NestFactory } from '@nestjs/core';
import { MicroserviceOptions, Transport } from '@nestjs/microservices';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.createMicroservice<MicroserviceOptions>(AppModule, {
    transport: Transport.RMQ,
    options: {
      urls: [process.env.RABBITMQ_URL ?? 'amqp://admin:admin@localhost:5672'],
      queue: 'users_queue',
      queueOptions: {
        durable: true,        // queue survives broker restart
        arguments: {
          'x-dead-letter-exchange': 'users_dlx',   // failed messages go here
          'x-message-ttl': 30_000,                 // messages expire after 30s if not processed
        },
      },
      noAck: false,           // manual acknowledgement
      prefetchCount: 5,       // process max 5 messages concurrently
    },
  });

  await app.listen();
  console.log('Users service listening on RabbitMQ queue: users_queue');
}
bootstrap();
```

### 4. Message patterns with acknowledgement context

```typescript
import { Controller, Logger } from '@nestjs/common';
import { MessagePattern, EventPattern, Payload, Ctx, RmqContext } from '@nestjs/microservices';

@Controller()
export class UsersMsController {
  private readonly logger = new Logger(UsersMsController.name);

  @MessagePattern({ cmd: 'find_user' })
  async findUser(
    @Payload() data: { id: number },
    @Ctx() context: RmqContext,   // access the raw channel and message
  ) {
    const channel = context.getChannelRef();
    const originalMsg = context.getMessage();

    try {
      const user = await this.usersService.findOne(data.id);
      channel.ack(originalMsg);   // acknowledge: message processed successfully
      return user;
    } catch (err) {
      this.logger.error(`Failed to find user ${data.id}:`, err.message);
      // Nack and don't requeue → message goes to DLX
      channel.nack(originalMsg, false, false);
      throw err;
    }
  }

  @EventPattern('user_registered')
  async handleUserRegistered(
    @Payload() data: { userId: number; email: string },
    @Ctx() context: RmqContext,
  ) {
    const channel = context.getChannelRef();
    const msg = context.getMessage();

    try {
      await this.emailService.sendWelcomeEmail(data.email);
      channel.ack(msg);
    } catch (err) {
      this.logger.error('Failed to send welcome email:', err.message);
      // Nack with requeue: retry later
      channel.nack(msg, false, true);
    }
  }
}
```

### 6. Dead Letter Exchange setup

Create a separate consumer for the dead letter queue (failed messages):

**`apps/dlq-consumer/src/main.ts`**
```typescript
const app = await NestFactory.createMicroservice<MicroserviceOptions>(AppModule, {
  transport: Transport.RMQ,
  options: {
    urls: ['amqp://admin:admin@localhost:5672'],
    queue: 'users_dlq',   // The dead letter queue
    queueOptions: {
      durable: true,
      arguments: {
        'x-dead-letter-exchange': '',  // no further routing
      },
    },
    noAck: false,
  },
});
```

**`apps/dlq-consumer/src/dlq.processor.ts`**
```typescript
@Controller()
export class DlqProcessor {
  private readonly logger = new Logger(DlqProcessor.name);

  @MessagePattern('*')  // catch any pattern
  async handleDeadLetter(
    @Payload() data: any,
    @Ctx() context: RmqContext,
  ) {
    const channel = context.getChannelRef();
    const msg = context.getMessage();
    const headers = msg.properties.headers;

    this.logger.error('Dead letter received:', {
      routingKey: msg.fields.routingKey,
      deathReason: headers['x-death']?.[0]?.reason,
      originalQueue: headers['x-death']?.[0]?.queue,
      data,
    });

    // Store in a database for manual review
    await this.dlqService.save({
      pattern: msg.fields.routingKey,
      payload: JSON.stringify(data),
      reason: headers['x-death']?.[0]?.reason,
      failedAt: new Date(),
    });

    channel.ack(msg);  // acknowledge the DLQ message
  }
}
```

### 7. Declare RabbitMQ topology via admin

Use the RabbitMQ Management API or `amqplib` to set up exchanges and bindings:

```typescript
import * as amqp from 'amqplib';

async function setupRabbitMQTopology() {
  const connection = await amqp.connect('amqp://admin:admin@localhost:5672');
  const channel = await connection.createChannel();

  // Main exchanges
  await channel.assertExchange('users', 'direct', { durable: true });
  await channel.assertExchange('users_dlx', 'direct', { durable: true });

  // Main queue bound to exchange
  await channel.assertQueue('users_queue', {
    durable: true,
    arguments: { 'x-dead-letter-exchange': 'users_dlx' },
  });
  await channel.bindQueue('users_queue', 'users', 'users_queue');

  // Dead letter queue bound to DLX
  await channel.assertQueue('users_dlq', { durable: true });
  await channel.bindQueue('users_dlq', 'users_dlx', 'users_queue');

  await channel.close();
  await connection.close();
  console.log('RabbitMQ topology set up');
}
```
