# P5.2 — Settings Service

**Phase:** 5 — Configuration & Environment
**Topic:** Database-driven settings, Redis cache, config change events
**Type:** Project

---

## Goal

Build an admin-configurable settings service where settings are stored in the database, cached in Redis, and emit events on change so other services can react without polling.

## What to Build

- `AppSetting` entity: key, value (JSON), data type, description, last updated by
- `SettingsService` with typed get/set methods and Redis-backed caching
- Cache invalidation on setting update
- `EventEmitter2` emitting `setting.changed` events with old/new values
- Admin-only REST API to list, update, and reset settings to defaults
- `@InjectSetting(key)` param decorator as a DI shortcut
- Default settings seeder that runs on startup (uses upsert)
- Setting groups/categories for organisational clarity

```typescript
import { Entity, PrimaryColumn, Column, UpdateDateColumn } from 'typeorm';

export enum SettingType {
  STRING = 'string',
  NUMBER = 'number',
  BOOLEAN = 'boolean',
  JSON = 'json',
}

export enum SettingGroup {
  GENERAL = 'general',
  EMAIL = 'email',
  SECURITY = 'security',
  FEATURES = 'features',
}

@Entity('app_settings')
export class AppSetting {
  @PrimaryColumn()
  key: string;

  @Column({ type: 'text' })
  value: string; // stored as JSON string

  @Column({ type: 'enum', enum: SettingType, default: SettingType.STRING })
  type: SettingType;

  @Column({ type: 'enum', enum: SettingGroup, default: SettingGroup.GENERAL })
  group: SettingGroup;

  @Column({ nullable: true })
  description: string;

  @Column({ type: 'text' })
  defaultValue: string;

  @Column({ nullable: true })
  updatedBy: string; // user ID of last editor

  @UpdateDateColumn()
  updatedAt: Date;
}
```

### `src/cache/cache.service.ts`
```typescript
import { Injectable, Inject } from '@nestjs/common';
import { CACHE_MANAGER } from '@nestjs/cache-manager';
import { Cache } from 'cache-manager';

@Injectable()
export class CacheService {
  constructor(@Inject(CACHE_MANAGER) private cacheManager: Cache) {}

  async get<T>(key: string): Promise<T | null> {
    return (await this.cacheManager.get<T>(key)) ?? null;
  }

  async set(key: string, value: unknown, ttlSeconds = 300): Promise<void> {
    await this.cacheManager.set(key, value, ttlSeconds * 1000);
  }

  async del(key: string): Promise<void> {
    await this.cacheManager.del(key);
  }

  async delByPattern(pattern: string): Promise<void> {
    // For Redis store, use scan-based deletion
    const store = (this.cacheManager as any).store;
    if (store?.keys) {
      const keys: string[] = await store.keys(pattern);
      await Promise.all(keys.map((k) => this.cacheManager.del(k)));
    }
  }
}
```

### `src/settings/settings.service.ts`
```typescript
import { Injectable, Logger, NotFoundException, OnModuleInit } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { EventEmitter2 } from '@nestjs/event-emitter';
import { AppSetting, SettingType } from './entities/app-setting.entity';
import { CacheService } from '../cache/cache.service';
import { SettingChangedEvent } from './events/setting-changed.event';
import { SettingsSeeder } from './settings.seeder';
import { UpdateSettingDto } from './dto/update-setting.dto';

const CACHE_KEY = (key: string) => `setting:${key}`;
const CACHE_TTL = 600; // 10 minutes

@Injectable()
export class SettingsService implements OnModuleInit {
  private readonly logger = new Logger(SettingsService.name);

  constructor(
    @InjectRepository(AppSetting) private repo: Repository<AppSetting>,
    private readonly cache: CacheService,
    private readonly eventEmitter: EventEmitter2,
    private readonly seeder: SettingsSeeder,
  ) {}

  async onModuleInit(): Promise<void> {
    await this.seeder.seed();
    this.logger.log('Settings seeded/verified');
  }

  async get<T = string>(key: string): Promise<T> {
    const cached = await this.cache.get<T>(CACHE_KEY(key));
    if (cached !== null) return cached;

    const setting = await this.repo.findOne({ where: { key } });
    if (!setting) throw new NotFoundException(`Setting "${key}" not found`);

    const value = this.deserialize<T>(setting.value, setting.type);
    await this.cache.set(CACHE_KEY(key), value, CACHE_TTL);
    return value;
  }

  async set(key: string, raw: unknown, updatedBy: string): Promise<AppSetting> {
    const setting = await this.repo.findOne({ where: { key } });
    if (!setting) throw new NotFoundException(`Setting "${key}" not found`);

    const oldValue = this.deserialize(setting.value, setting.type);
    const newValueStr = JSON.stringify(raw);

    await this.repo.update({ key }, { value: newValueStr, updatedBy });
    await this.cache.del(CACHE_KEY(key));

    this.eventEmitter.emit('setting.changed', new SettingChangedEvent(key, oldValue, raw, updatedBy));

    return this.repo.findOne({ where: { key } }) as Promise<AppSetting>;
  }

  async reset(key: string, updatedBy: string): Promise<AppSetting> {
    const setting = await this.repo.findOne({ where: { key } });
    if (!setting) throw new NotFoundException(`Setting "${key}" not found`);
    return this.set(key, JSON.parse(setting.defaultValue), updatedBy);
  }

  async findAll(group?: string): Promise<AppSetting[]> {
    const where: any = group ? { group } : {};
    return this.repo.find({ where, order: { group: 'ASC', key: 'ASC' } });
  }

  private deserialize<T>(value: string, type: SettingType): T {
    try {
      const parsed = JSON.parse(value);
      if (type === SettingType.NUMBER) return Number(parsed) as unknown as T;
      if (type === SettingType.BOOLEAN) return Boolean(parsed) as unknown as T;
      return parsed as T;
    } catch {
      return value as unknown as T;
    }
  }
}
```

### `src/settings/settings.seeder.ts`
```typescript
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { AppSetting, SettingType, SettingGroup } from './entities/app-setting.entity';

const DEFAULT_SETTINGS: Omit<AppSetting, 'updatedAt'>[] = [
  { key: 'app.maintenance_mode', value: 'false', defaultValue: 'false', type: SettingType.BOOLEAN, group: SettingGroup.GENERAL, description: 'Enable maintenance mode', updatedBy: 'system' },
  { key: 'app.max_upload_size_mb', value: '10', defaultValue: '10', type: SettingType.NUMBER, group: SettingGroup.GENERAL, description: 'Max upload size in MB', updatedBy: 'system' },
  { key: 'email.from_address', value: '"noreply@example.com"', defaultValue: '"noreply@example.com"', type: SettingType.STRING, group: SettingGroup.EMAIL, description: 'Sender email address', updatedBy: 'system' },
  { key: 'email.smtp_enabled', value: 'true', defaultValue: 'true', type: SettingType.BOOLEAN, group: SettingGroup.EMAIL, description: 'Enable SMTP email sending', updatedBy: 'system' },
  { key: 'security.max_login_attempts', value: '5', defaultValue: '5', type: SettingType.NUMBER, group: SettingGroup.SECURITY, description: 'Max failed login attempts before lockout', updatedBy: 'system' },
  { key: 'features.registration_open', value: 'true', defaultValue: 'true', type: SettingType.BOOLEAN, group: SettingGroup.FEATURES, description: 'Allow new user registrations', updatedBy: 'system' },
];

@Injectable()
export class SettingsSeeder {
  constructor(@InjectRepository(AppSetting) private repo: Repository<AppSetting>) {}

  async seed(): Promise<void> {
    for (const setting of DEFAULT_SETTINGS) {
      await this.repo
        .createQueryBuilder()
        .insert()
        .into(AppSetting)
        .values(setting)
        .orIgnore() // Don't overwrite existing settings
        .execute();
    }
  }
}
```

### `src/settings/settings.controller.ts`
```typescript
import { Controller, Get, Put, Post, Param, Body, Query } from '@nestjs/common';
import { SettingsService } from './settings.service';
import { UpdateSettingDto } from './dto/update-setting.dto';

@Controller('admin/settings')
export class SettingsController {
  constructor(private readonly settingsService: SettingsService) {}

  @Get()
  findAll(@Query('group') group?: string) {
    return this.settingsService.findAll(group);
  }

  @Get(':key')
  get(@Param('key') key: string) {
    return this.settingsService.get(key);
  }

  @Put(':key')
  update(@Param('key') key: string, @Body() dto: UpdateSettingDto) {
    return this.settingsService.set(key, dto.value, 'admin'); // replace 'admin' with req.user.id
  }

  @Post(':key/reset')
  reset(@Param('key') key: string) {
    return this.settingsService.reset(key, 'admin');
  }
}
```

### Listening to settings change events
```typescript
// In any service or module
import { OnEvent } from '@nestjs/event-emitter';
import { SettingChangedEvent } from '../settings/events/setting-changed.event';

@Injectable()
export class FeatureFlagService {
  private maintenanceMode = false;

  @OnEvent('setting.changed')
  onSettingChanged(event: SettingChangedEvent) {
    if (event.key === 'app.maintenance_mode') {
      this.maintenanceMode = Boolean(event.newValue);
      console.log(`Maintenance mode changed to: ${this.maintenanceMode}`);
    }
  }

  isMaintenanceMode() { return this.maintenanceMode; }
}
```

```typescript
import { Module } from '@nestjs/common';
import { EventEmitterModule } from '@nestjs/event-emitter';
import { CacheModule } from '@nestjs/cache-manager';
import { redisStore } from 'cache-manager-redis-yet';
import { ConfigService } from '@nestjs/config';
import { SettingsModule } from './settings/settings.module';

@Module({
  imports: [
    EventEmitterModule.forRoot({ wildcard: false, maxListeners: 20 }),
    CacheModule.registerAsync({
      isGlobal: true,
      inject: [ConfigService],
      useFactory: async (config: ConfigService) => ({
        store: await redisStore({
          socket: { host: config.get('REDIS_HOST'), port: config.get('REDIS_PORT') },
          ttl: 300,
        }),
      }),
    }),
    SettingsModule,
  ],
})
export class AppModule {}
```

## Deliverables

- [ ] `AppSetting` entity uses `key` as primary key (no auto-generated UUID)
- [ ] `SettingsSeeder.seed()` uses `orIgnore()` so existing values are not overwritten on restart
- [ ] `SettingsService.get()` returns value from Redis cache on second call (cache hit)
- [ ] `SettingsService.set()` invalidates the Redis cache entry for that key
- [ ] `setting.changed` event is emitted synchronously after update (before HTTP response)
- [ ] `@OnEvent('setting.changed')` listener in another service reacts to the event
- [ ] `GET /admin/settings?group=security` returns only security group settings
- [ ] `POST /admin/settings/app.max_upload_size_mb/reset` restores default value
- [ ] Number settings deserialized as `number`, boolean settings as `boolean`
- [ ] Setting groups displayed in `GET /admin/settings` response for organisational clarity
