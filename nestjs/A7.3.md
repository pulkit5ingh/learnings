# A7.3 — E2E Testing

**Phase:** 7 — Testing  
**Topic:** @nestjs/testing, real database, test app bootstrap, cleanup  
**Type:** Assignment

---

## Goal
Write end-to-end tests that spin up a full NestJS app with a real database, execute HTTP requests, and clean up after each test.

## Background

E2E tests test the full request stack: middleware, guards, pipes, controllers, services, and the database. They are slower than unit tests (require a real DB connection) but catch integration problems that unit tests miss — missing `imports`, wrong module wiring, broken migrations, ORM query bugs.

The standard approach is to create a test database (separate from dev/prod), run migrations before the test suite, seed necessary data in `beforeEach`, and truncate tables in `afterEach`. Use an in-memory SQLite database or a real PostgreSQL database running in Docker.

**Key patterns:**
- `createTestingModule` + `createNestApplication` — builds the full app.
- `moduleRef.get(Service)` — access providers directly for setup/assertions.
- Database cleanup via raw SQL `TRUNCATE` or TypeORM `dataSource.synchronize({ dropTables: true })`.
- Generate JWT tokens for protected route tests without a real login flow.

## Tasks

### 1. E2E test configuration

**`test/jest-e2e.json`**
```json
{
  "moduleFileExtensions": ["js", "json", "ts"],
  "rootDir": ".",
  "testEnvironment": "node",
  "testRegex": ".e2e-spec.ts$",
  "transform": { "^.+\\.(t|j)s$": "ts-jest" },
  "testTimeout": 30000
}
```

**`package.json`** — add e2e script:
```json
{
  "scripts": {
    "test:e2e": "jest --config ./test/jest-e2e.json --forceExit --detectOpenHandles"
  }
}
```

### 2. Test app factory

**`test/test-app.ts`**
```typescript
import { INestApplication, ValidationPipe } from '@nestjs/common';
import { Test } from '@nestjs/testing';
import { TypeOrmModule } from '@nestjs/typeorm';
import { JwtService } from '@nestjs/jwt';
import { AppModule } from '../src/app.module';
import { User } from '../src/users/entities/user.entity';
import { DataSource } from 'typeorm';

export async function createTestApp(): Promise<{
  app: INestApplication;
  dataSource: DataSource;
  jwtService: JwtService;
}> {
  const moduleRef = await Test.createTestingModule({
    imports: [
      // Override the TypeORM config to use a test database
      TypeOrmModule.forRoot({
        type: 'sqlite',
        database: ':memory:',
        entities: [__dirname + '/../src/**/*.entity{.ts,.js}'],
        synchronize: true,   // safe in tests — recreates schema on each run
        dropSchema: true,    // start fresh
      }),
      // Import all other app modules (without the TypeORM forRoot from AppModule)
      // In practice, you'd use a ConfigModule override or env var
    ],
  })
    .overrideProvider(TypeOrmModule)  // if AppModule registers TypeORM, override it
    .useValue({})
    .compile();

  const app = moduleRef.createNestApplication();
  app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));
  await app.init();

  return {
    app,
    dataSource: moduleRef.get(DataSource),
    jwtService: moduleRef.get(JwtService),
  };
}
```

### 3. Full E2E test with real DB (SQLite in-memory)

**`test/users.e2e-spec.ts`**
```typescript
import { INestApplication } from '@nestjs/common';
import { Test, TestingModule } from '@nestjs/testing';
import * as request from 'supertest';
import { TypeOrmModule } from '@nestjs/typeorm';
import { DataSource } from 'typeorm';
import { ConfigModule } from '@nestjs/config';
import { UsersModule } from '../src/users/users.module';
import { User } from '../src/users/entities/user.entity';

describe('Users (e2e)', () => {
  let app: INestApplication;
  let dataSource: DataSource;

  beforeAll(async () => {
    const moduleRef: TestingModule = await Test.createTestingModule({
      imports: [
        ConfigModule.forRoot({ envFilePath: '.env.test', isGlobal: true }),
        TypeOrmModule.forRoot({
          type: 'sqlite',
          database: ':memory:',
          entities: [User],
          synchronize: true,
        }),
        UsersModule,
      ],
    }).compile();

    app = moduleRef.createNestApplication();
    app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));
    await app.init();

    dataSource = moduleRef.get(DataSource);
  });

  afterAll(async () => {
    await app.close();
  });

  afterEach(async () => {
    // Clean up after each test
    await dataSource.query('DELETE FROM users');
  });

  describe('/users (POST)', () => {
    it('should create a user', async () => {
      const res = await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!' })
        .expect(201);

      expect(res.body).toMatchObject({
        id: expect.any(Number),
        name: 'Alice',
        email: 'alice@example.com',
      });
      expect(res.body).not.toHaveProperty('passwordHash');
    });

    it('should return 409 for duplicate email', async () => {
      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!' })
        .expect(201);

      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice 2', email: 'alice@example.com', password: 'Password1!' })
        .expect(409);
    });
  });

  describe('/users (GET)', () => {
    beforeEach(async () => {
      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!' });
      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Bob', email: 'bob@example.com', password: 'Password1!' });
    });

    it('should return all users', async () => {
      const res = await request(app.getHttpServer())
        .get('/users')
        .expect(200);

      expect(res.body).toHaveLength(2);
    });

    it('should filter by search', async () => {
      const res = await request(app.getHttpServer())
        .get('/users?search=alice')
        .expect(200);

      expect(res.body).toHaveLength(1);
      expect(res.body[0].email).toBe('alice@example.com');
    });
  });

  describe('/users/:id (GET)', () => {
    let userId: number;

    beforeEach(async () => {
      const res = await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!' });
      userId = res.body.id;
    });

    it('should return a user by id', async () => {
      const res = await request(app.getHttpServer())
        .get(`/users/${userId}`)
        .expect(200);

      expect(res.body.id).toBe(userId);
    });

    it('should return 404 for non-existent user', () => {
      return request(app.getHttpServer())
        .get('/users/99999')
        .expect(404);
    });
  });
});
```

### 4. E2E test for authenticated endpoints

```typescript
describe('Posts E2E (authenticated)', () => {
  let app: INestApplication;
  let jwtService: JwtService;
  let authToken: string;
  let testUserId: number;

  beforeAll(async () => {
    // ... setup app
    jwtService = moduleRef.get(JwtService);
  });

  beforeEach(async () => {
    // Create a user in the DB
    const createRes = await request(app.getHttpServer())
      .post('/auth/register')
      .send({ name: 'Test', email: 'test@test.com', password: 'Password1!' });
    testUserId = createRes.body.id;

    // Generate a valid JWT (without going through login flow)
    authToken = jwtService.sign({ sub: testUserId, email: 'test@test.com', roles: ['user'] });
  });

  it('should create a post when authenticated', () => {
    return request(app.getHttpServer())
      .post('/posts')
      .set('Authorization', `Bearer ${authToken}`)
      .send({ title: 'Test Post', content: 'Content here' })
      .expect(201);
  });

  it('should return 401 without auth token', () => {
    return request(app.getHttpServer())
      .post('/posts')
      .send({ title: 'Test', content: 'Content' })
      .expect(401);
  });
});
```

### 5. Database cleanup patterns

```typescript
// Option 1: DELETE in afterEach (preserves auto-increment counters)
afterEach(async () => {
  await dataSource.query('DELETE FROM posts');
  await dataSource.query('DELETE FROM users');
  // Order matters — respect foreign keys
});

// Option 2: TRUNCATE with cascade (PostgreSQL)
afterEach(async () => {
  const entities = dataSource.entityMetadatas;
  for (const entity of entities.reverse()) {
    await dataSource.query(`TRUNCATE "${entity.tableName}" CASCADE`);
  }
});

// Option 3: Transaction rollback (fastest but complex)
let queryRunner: QueryRunner;
beforeEach(async () => {
  queryRunner = dataSource.createQueryRunner();
  await queryRunner.startTransaction();
});
afterEach(async () => {
  await queryRunner.rollbackTransaction();
  await queryRunner.release();
});
```
