# A3.3 — Roles & Permissions

**Phase:** 3 — Authentication & Guards  
**Topic:** Custom @Roles() decorator, RolesGuard, SetMetadata, CASL for fine-grained permissions  
**Type:** Assignment

---

## Goal
Implement role-based access control (RBAC) with a `@Roles()` decorator and `RolesGuard`, then extend to fine-grained attribute-based access control (ABAC) using the CASL library.

## Background

RBAC assigns permissions to roles, and roles to users. A user with role `admin` can do everything; a user with role `moderator` can edit but not delete; a plain `user` can only manage their own resources.

CASL (Comprehensive Access-Control Library) extends this to attribute-based checks: "a user can update a Post *only if* they are the author." CASL defines **abilities** as a set of rules: `can('action', 'subject')`. In NestJS, you create an `AbilityFactory` (or `CaslAbilityFactory`) that builds the ability set for a given user, then use it in guards or service methods.

**Two levels of permission checking:**
1. **Guard-level** — coarse-grained; runs before the handler; checks roles or action+subject from metadata.
2. **Service-level** — fine-grained; runs after loading the resource; checks ownership or attribute conditions.

### 4. RolesGuard

**`src/auth/guards/roles.guard.ts`**
```typescript
import {
  CanActivate, ExecutionContext,
  ForbiddenException, Injectable,
} from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { Role, ROLES_KEY } from '../decorators/roles.decorator';

@Injectable()
export class RolesGuard implements CanActivate {
  constructor(private reflector: Reflector) {}

  canActivate(context: ExecutionContext): boolean {
    const requiredRoles = this.reflector.getAllAndOverride<Role[]>(ROLES_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);

    if (!requiredRoles?.length) return true;

    const { user } = context.switchToHttp().getRequest();
    if (!user) throw new ForbiddenException('Not authenticated');

    const hasRole = requiredRoles.some(r => (user.roles as Role[]).includes(r));
    if (!hasRole) {
      throw new ForbiddenException(`Access denied. Required roles: ${requiredRoles}`);
    }
    return true;
  }
}
```

### 5. CASL ability factory

**`src/casl/casl-ability.factory.ts`**
```typescript
import { Injectable } from '@nestjs/common';
import {
  AbilityBuilder,
  createMongoAbility,
  MongoAbility,
  InferSubjects,
} from '@casl/ability';
import { User } from '../users/entities/user.entity';
import { Post } from '../posts/entities/post.entity';
import { Role } from '../common/enums/role.enum';

type Subjects = InferSubjects<typeof User | typeof Post> | 'all';

export type Action = 'manage' | 'create' | 'read' | 'update' | 'delete';

export type AppAbility = MongoAbility<[Action, Subjects]>;

@Injectable()
export class CaslAbilityFactory {
  createForUser(user: User): AppAbility {
    const { can, cannot, build } = new AbilityBuilder<AppAbility>(createMongoAbility);

    if (user.roles.includes(Role.ADMIN)) {
      can('manage', 'all');  // Admin can do everything
    } else if (user.roles.includes(Role.MODERATOR)) {
      can('read', 'all');
      can('update', Post);   // Moderator can update any post
      can('delete', Post);
      cannot('delete', User); // But cannot delete users
    } else {
      // Regular user
      can('read', 'all');
      can('create', Post);
      can('update', Post, { 'author.id': user.id });   // only own posts
      can('delete', Post, { 'author.id': user.id });
    }

    return build();
  }
}
```

### 6. Policy decorator and guard

**`src/casl/policies.decorator.ts`**
```typescript
import { SetMetadata } from '@nestjs/common';
import { AppAbility } from './casl-ability.factory';

export interface PolicyHandler {
  handle(ability: AppAbility): boolean;
}

export type PolicyHandlerCallback = (ability: AppAbility) => boolean;

export const CHECK_POLICIES_KEY = 'check_policies';
export const CheckPolicies = (...handlers: PolicyHandlerCallback[]) =>
  SetMetadata(CHECK_POLICIES_KEY, handlers);
```

**`src/casl/policies.guard.ts`**
```typescript
import { CanActivate, ExecutionContext, Injectable } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { CaslAbilityFactory } from './casl-ability.factory';
import { CHECK_POLICIES_KEY, PolicyHandlerCallback } from './policies.decorator';

@Injectable()
export class PoliciesGuard implements CanActivate {
  constructor(
    private readonly reflector: Reflector,
    private readonly caslAbilityFactory: CaslAbilityFactory,
  ) {}

  canActivate(context: ExecutionContext): boolean {
    const handlers = this.reflector.get<PolicyHandlerCallback[]>(
      CHECK_POLICIES_KEY,
      context.getHandler(),
    );

    if (!handlers?.length) return true;

    const { user } = context.switchToHttp().getRequest();
    const ability = this.caslAbilityFactory.createForUser(user);

    return handlers.every(handler => handler(ability));
  }
}
```

### 7. Using policies on routes

```typescript
import { Action } from '../casl/casl-ability.factory';
import { Post } from './entities/post.entity';

@Controller('posts')
@UseGuards(JwtAuthGuard, PoliciesGuard)
export class PostsController {
  // Anyone authenticated can read
  @Get()
  findAll() { ... }

  // Must be able to create Post
  @Post()
  @CheckPolicies((ability) => ability.can('create', Post))
  create(@Body() dto: CreatePostDto, @Req() req: Request) {
    return this.postsService.create((req as any).user.id, dto);
  }

  // Must be able to update Post — ownership check happens in service
  @Patch(':id')
  @CheckPolicies((ability) => ability.can('update', Post))
  update(@Param('id', ParseIntPipe) id: number, @Body() dto: UpdatePostDto) {
    return this.postsService.update(id, dto);
  }

  // Admin only route (via RolesGuard)
  @Delete(':id')
  @Roles(Role.ADMIN)
  @UseGuards(RolesGuard)
  remove(@Param('id', ParseIntPipe) id: number) {
    return this.postsService.remove(id);
  }
}
```

### 8. Service-level ownership check

```typescript
// posts.service.ts
async update(postId: number, userId: number, dto: UpdatePostDto): Promise<Post> {
  const post = await this.postsRepository.findOne({
    where: { id: postId },
    relations: { author: true },
  });
  if (!post) throw new NotFoundException();

  // Attribute-based check: user must own the post
  if (post.author.id !== userId) {
    throw new ForbiddenException('You can only update your own posts');
  }

  Object.assign(post, dto);
  return this.postsRepository.save(post);
}
```
