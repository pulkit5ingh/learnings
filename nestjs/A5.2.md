# A5.2 — Database Config Patterns

**Phase:** 5 — Configuration & Environment  
**Topic:** TypeOrmModule.forRootAsync, async config with ConfigService, multiple databases  
**Type:** Assignment

---

## Goal
Wire TypeORM and other modules asynchronously using ConfigService so that all configuration flows from validated environment variables.

## Background

Most NestJS library modules (TypeORM, JWT, Bull, Mailer) expose both `forRoot(options)` (synchronous) and `forRootAsync(asyncOptions)` (asynchronous) factory methods. The async form accepts a `useFactory` function that can receive injected providers — typically `ConfigService` — so that module configuration comes from environment variables rather than hardcoded values.

The `forRootAsync` pattern is required whenever your configuration is not known at import time (i.e., depends on env vars, which are loaded by `ConfigModule`). The `imports` array inside `forRootAsync` ensures `ConfigModule` is available when the factory runs.

For **multiple databases**, TypeORM supports multiple named connections. Each has a unique `name` (default name is `'default'`). When injecting repositories from a non-default connection, use `@InjectRepository(Entity, 'connectionName')`.

## Tasks

### 1. TypeORM forRootAsync

**`src/app.module.ts`**
```typescript
import { Module } from '@nestjs/common';
import { TypeOrmModule } from '@nestjs/typeorm';
import { ConfigModule, ConfigService } from '@nestjs/config';

@Module({
  imports: [
    ConfigModule.forRoot({ isGlobal: true }),

    TypeOrmModule.forRootAsync({
      imports: [ConfigModule],   // ensure ConfigModule is available
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        type: 'postgres',
        host: config.get<string>('DB_HOST'),
        port: config.get<number>('DB_PORT', 5432),
        username: config.get<string>('DB_USER'),
        password: config.get<string>('DB_PASS'),
        database: config.get<string>('DB_NAME'),
        autoLoadEntities: true,
        synchronize: config.get<string>('NODE_ENV') === 'development',
        logging: config.get<string>('NODE_ENV') === 'development',
        ssl: config.get<string>('NODE_ENV') === 'production'
          ? { rejectUnauthorized: false }
          : undefined,
        extra: {
          max: config.get<number>('DB_POOL_MAX', 10),
          idleTimeoutMillis: 30_000,
          connectionTimeoutMillis: 2_000,
        },
      }),
    }),
  ],
})
export class AppModule {}
```

### 2. Using namespaced config with forRootAsync

If you used `registerAs('database', ...)` from A5.1:
```typescript
import { databaseConfig } from './config/database.config';
import { ConfigType } from '@nestjs/config';

TypeOrmModule.forRootAsync({
  inject: [databaseConfig.KEY],
  useFactory: (dbConfig: ConfigType<typeof databaseConfig>) => ({
    type: 'postgres',
    host: dbConfig.host,
    port: dbConfig.port,
    username: dbConfig.username,
    password: dbConfig.password,
    database: dbConfig.database,
    autoLoadEntities: true,
    synchronize: dbConfig.synchronize,
    ssl: dbConfig.ssl,
  }),
})
```

### 3. JwtModule forRootAsync

```typescript
import { JwtModule } from '@nestjs/jwt';
import { ConfigModule, ConfigService } from '@nestjs/config';

JwtModule.registerAsync({
  imports: [ConfigModule],
  inject: [ConfigService],
  useFactory: (config: ConfigService) => ({
    secret: config.get<string>('JWT_SECRET'),
    signOptions: {
      expiresIn: config.get<string>('JWT_EXPIRES_IN', '15m'),
      issuer: config.get<string>('APP_NAME', 'api'),
    },
  }),
})
```

### 4. Multiple databases

Scenario: primary PostgreSQL for application data, secondary read-replica for analytics.

**`src/app.module.ts`**
```typescript
TypeOrmModule.forRootAsync({
  // name: 'default' (omitted = default)
  inject: [ConfigService],
  useFactory: (config: ConfigService) => ({
    type: 'postgres',
    host: config.get('DB_HOST'),
    // ...
    name: 'primary',
    autoLoadEntities: true,
  }),
}),

TypeOrmModule.forRootAsync({
  inject: [ConfigService],
  useFactory: (config: ConfigService) => ({
    type: 'postgres',
    host: config.get('DB_REPLICA_HOST'),
    port: config.get<number>('DB_REPLICA_PORT', 5432),
    username: config.get('DB_REPLICA_USER'),
    password: config.get('DB_REPLICA_PASS'),
    database: config.get('DB_NAME'),
    name: 'replica',       // ← named connection
    entities: [User, Post], // explicit for non-default connections
    synchronize: false,
    logging: ['query'],
  }),
}),
```

**In the analytics module:**
```typescript
@Module({
  imports: [
    TypeOrmModule.forFeature([Post], 'replica'),  // register entity for 'replica' connection
  ],
  providers: [AnalyticsService],
})
export class AnalyticsModule {}
```

**`src/analytics/analytics.service.ts`**
```typescript
import { Injectable } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Post } from '../posts/entities/post.entity';

@Injectable()
export class AnalyticsService {
  constructor(
    @InjectRepository(Post, 'replica')   // ← specify connection name
    private readonly postReplicaRepo: Repository<Post>,
  ) {}

  async getPostStats() {
    return this.postReplicaRepo
      .createQueryBuilder('post')
      .select('DATE(post.createdAt)', 'date')
      .addSelect('COUNT(*)', 'count')
      .groupBy('DATE(post.createdAt)')
      .orderBy('date', 'DESC')
      .limit(30)
      .getRawMany();
  }
}
```

### 5. DataSource injection for raw queries

```typescript
import { InjectDataSource } from '@nestjs/typeorm';
import { DataSource } from 'typeorm';

@Injectable()
export class ReportService {
  constructor(
    @InjectDataSource()                    // default connection
    private readonly dataSource: DataSource,
    @InjectDataSource('replica')           // named connection
    private readonly replicaDataSource: DataSource,
  ) {}

  async runMigrations(): Promise<void> {
    await this.dataSource.runMigrations();
  }

  async rawQuery<T>(sql: string, params?: any[]): Promise<T[]> {
    return this.replicaDataSource.query(sql, params);
  }
}
```

### 6. Testing async config modules

In tests, use `ConfigModule.forRoot({ envFilePath: '.env.test' })` and override the factory:
```typescript
const testModule = await Test.createTestingModule({
  imports: [
    ConfigModule.forRoot({ envFilePath: '.env.test', isGlobal: true }),
    TypeOrmModule.forRootAsync({
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        type: 'sqlite',     // SQLite in-memory for tests
        database: ':memory:',
        autoLoadEntities: true,
        synchronize: true,
      }),
    }),
  ],
}).compile();
```
