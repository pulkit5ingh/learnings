# P6.1 — Email Notification System

**Phase:** 6 — Caching, Queues & Performance
**Topic:** Bull queues, Handlebars templates, retry, DLQ, Flower dashboard
**Type:** Project

---

## Goal

Build an email notification system that queues outbound emails via Bull, renders them with Handlebars templates, supports retry with exponential backoff, and routes permanently failed jobs to a Dead Letter Queue.

## What to Build

- `EmailQueue` with Bull: jobs for welcome email, password reset, order confirmation
- Handlebars templates for each email type
- `EmailProcessor` handling jobs, calling Nodemailer/SendGrid
- Retry strategy: 3 attempts, exponential backoff (1s, 2s, 4s)
- DLQ: `email-failed` queue receiving jobs that exhausted retries
- `BullModule` with Redis connection
- Flower dashboard for monitoring queues (via `bull-board`)
- `NotificationService` as a public facade — callers don't touch the queue directly
- `EmailLog` entity persisting every send attempt and its status

  @Column()
  to: string;

  @Column()
  subject: string;

  @Column()
  jobType: string;

  @Column({ nullable: true })

  @Column({ default: 0 })

}

### `src/email/email.service.ts`
```typescript
import { Injectable, Logger } from '@nestjs/common';
import * as nodemailer from 'nodemailer';
import * as Handlebars from 'handlebars';
import { readFileSync } from 'fs';
import { join } from 'path';
import { ConfigService } from '@nestjs/config';

export interface SendEmailOptions {
  to: string;
  subject: string;
  template: string;
  context: Record<string, unknown>;
}

@Injectable()
export class EmailService {
  private readonly logger = new Logger(EmailService.name);
  private transporter: nodemailer.Transporter;
  private templates = new Map<string, HandlebarsTemplateDelegate>();

  constructor(private readonly config: ConfigService) {
    this.transporter = nodemailer.createTransport({
      host: config.get('SMTP_HOST', 'smtp.ethereal.email'),
      port: config.get('SMTP_PORT', 587),
      auth: {
        user: config.get('SMTP_USER'),
        pass: config.get('SMTP_PASS'),
      },
    });
  }

  private getTemplate(name: string): HandlebarsTemplateDelegate {
    if (!this.templates.has(name)) {
      const path = join(__dirname, 'templates', `${name}.hbs`);
      const source = readFileSync(path, 'utf-8');
      this.templates.set(name, Handlebars.compile(source));
    }
    return this.templates.get(name)!;
  }

  async send(options: SendEmailOptions): Promise<void> {
    const template = this.getTemplate(options.template);
    const html = template(options.context);

    await this.transporter.sendMail({
      from: this.config.get('EMAIL_FROM', 'noreply@example.com'),
      to: options.to,
      subject: options.subject,
      html,
    });

    this.logger.log(`Email sent: ${options.subject} → ${options.to}`);
  }
}
```

### `src/email/email.processor.ts`
```typescript
import { Processor, Process, OnQueueFailed, OnQueueCompleted } from '@nestjs/bull';
import { Logger } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Job } from 'bull';
import { EmailService } from './email.service';
import { EmailLog, EmailStatus } from './email-log.entity';
import { EMAIL_QUEUE, EmailJobType } from './constants';

export interface EmailJobData {
  logId: string;
  to: string;
  name: string;
  [key: string]: unknown;
}

@Processor(EMAIL_QUEUE)
export class EmailProcessor {
  private readonly logger = new Logger(EmailProcessor.name);

  constructor(
    private readonly emailService: EmailService,
    @InjectRepository(EmailLog) private readonly logRepo: Repository<EmailLog>,
  ) {}

  @Process(EmailJobType.WELCOME)
  async handleWelcome(job: Job<EmailJobData>): Promise<void> {
    await this.emailService.send({
      to: job.data.to,
      subject: 'Welcome to the platform!',
      template: 'welcome',
      context: { name: job.data.name, verifyUrl: job.data.verifyUrl, appName: 'MyApp' },
    });
  }

  @Process(EmailJobType.PASSWORD_RESET)
  async handlePasswordReset(job: Job<EmailJobData>): Promise<void> {
    await this.emailService.send({
      to: job.data.to,
      subject: 'Password Reset Request',
      template: 'password-reset',
      context: { name: job.data.name, resetUrl: job.data.resetUrl },
    });
  }

  @Process(EmailJobType.ORDER_CONFIRMATION)
  async handleOrderConfirmation(job: Job<EmailJobData>): Promise<void> {
    await this.emailService.send({
      to: job.data.to,
      subject: `Order Confirmed #${job.data.orderId}`,
      template: 'order-confirmation',
      context: { name: job.data.name, orderId: job.data.orderId, items: job.data.items, total: job.data.total },
    });
  }

  @OnQueueCompleted()
  async onCompleted(job: Job<EmailJobData>): Promise<void> {
    await this.logRepo.update(job.data.logId, { status: EmailStatus.SENT, attemptsMade: job.attemptsMade });
  }

  @OnQueueFailed()
  async onFailed(job: Job<EmailJobData>, error: Error): Promise<void> {
    this.logger.error(`Job ${job.id} failed (attempt ${job.attemptsMade}): ${error.message}`);
    await this.logRepo.update(job.data.logId, {
      status: job.attemptsMade >= (job.opts.attempts ?? 3) ? EmailStatus.DLQ : EmailStatus.FAILED,
      errorMessage: error.message,
      attemptsMade: job.attemptsMade,
    });
  }
}
```

### `src/notifications/notifications.service.ts`
```typescript
import { Injectable } from '@nestjs/common';
import { InjectQueue } from '@nestjs/bull';
import { InjectRepository } from '@nestjs/typeorm';
import { Queue } from 'bull';
import { Repository } from 'typeorm';
import { EmailLog, EmailStatus } from '../email/email-log.entity';
import { EMAIL_QUEUE, EmailJobType } from '../email/constants';

const RETRY_OPTIONS = {
  attempts: 3,
  backoff: { type: 'exponential' as const, delay: 1000 },
  removeOnComplete: 100,
  removeOnFail: false,
};

@Injectable()
export class NotificationsService {
  constructor(
    @InjectQueue(EMAIL_QUEUE) private emailQueue: Queue,
    @InjectRepository(EmailLog) private logRepo: Repository<EmailLog>,
  ) {}

  async sendWelcomeEmail(to: string, name: string, verifyUrl: string): Promise<void> {
    const log = await this.logRepo.save(
      this.logRepo.create({ to, subject: 'Welcome to the platform!', jobType: EmailJobType.WELCOME, status: EmailStatus.QUEUED }),
    );
    await this.emailQueue.add(EmailJobType.WELCOME, { logId: log.id, to, name, verifyUrl }, RETRY_OPTIONS);
  }

  async sendPasswordResetEmail(to: string, name: string, resetUrl: string): Promise<void> {
    const log = await this.logRepo.save(
      this.logRepo.create({ to, subject: 'Password Reset', jobType: EmailJobType.PASSWORD_RESET, status: EmailStatus.QUEUED }),
    );
    await this.emailQueue.add(EmailJobType.PASSWORD_RESET, { logId: log.id, to, name, resetUrl }, RETRY_OPTIONS);
  }

  async sendOrderConfirmation(to: string, name: string, orderId: string, items: any[], total: number): Promise<void> {
    const log = await this.logRepo.save(
      this.logRepo.create({ to, subject: `Order #${orderId}`, jobType: EmailJobType.ORDER_CONFIRMATION, status: EmailStatus.QUEUED }),
    );
    await this.emailQueue.add(EmailJobType.ORDER_CONFIRMATION, { logId: log.id, to, name, orderId, items, total }, RETRY_OPTIONS);
  }

  async getEmailLogs() {
    return this.logRepo.find({ order: { createdAt: 'DESC' }, take: 100 });
  }
}
```

import { Module } from '@nestjs/common';
import { BullModule } from '@nestjs/bull';
import { TypeOrmModule } from '@nestjs/typeorm';
import { ConfigService } from '@nestjs/config';
import { EmailService } from './email.service';
import { EmailProcessor } from './email.processor';
import { EmailLog } from './email-log.entity';
import { EMAIL_QUEUE } from './constants';

@Module({
  imports: [
    BullModule.registerQueueAsync({
      name: EMAIL_QUEUE,
      inject: [ConfigService],
      useFactory: (config: ConfigService) => ({
        redis: { host: config.get('REDIS_HOST'), port: config.get('REDIS_PORT') },
        defaultJobOptions: { attempts: 3, backoff: { type: 'exponential', delay: 1000 } },
      }),
    }),
    TypeOrmModule.forFeature([EmailLog]),
  ],
  providers: [EmailService, EmailProcessor],
  exports: [BullModule, EmailService],
})
export class EmailModule {}

## Deliverables

- [ ] `NotificationsService.sendWelcomeEmail()` adds a job to Bull queue and persists an `EmailLog` row
- [ ] Failed jobs retry up to 3 times with exponential backoff (1s, 2s, 4s)
- [ ] After 3 failures, `EmailLog.status` is set to `'dlq'`
- [ ] `EmailProcessor` has separate `@Process()` handlers for each email type
- [ ] Handlebars templates are compiled once and cached in memory
- [ ] `/admin/queues` (bull-board) shows queue stats in browser
- [ ] `GET /notifications/logs` returns 100 most recent email log rows
- [ ] SMTP credentials come from environment variables, not hard-coded
- [ ] Queue uses `removeOnComplete: 100` to cap completed job history
- [ ] `EmailProcessor.onFailed` logs the error and updates the `EmailLog` record
