# P4.2 — Audit Logging Interceptor

**Phase:** 4 — Pipes, Interceptors & Filters
**Topic:** Interceptors, request/response lifecycle, DB persistence, timing
**Type:** Project

---

## Goal

Build an audit logging interceptor that records every API request and response — including timing, user ID, endpoint, status code, and request/response body — to a database table.

## What to Build

- `AuditLogInterceptor` capturing: method, path, status, duration, user ID, IP, request body, response body
- `AuditLog` entity stored in PostgreSQL via TypeORM
- `@SkipAudit()` decorator for health check or high-volume routes
- Async logging (non-blocking) so audit I/O doesn't slow down responses
- `GET /admin/audit-logs` endpoint with filters (userId, method, path, date range)
- Sensitive field scrubbing (remove `password`, `token`, `secret` from logged bodies)
- Log retention: auto-delete logs older than 90 days via a scheduled job

### `src/audit/interceptors/audit-log.interceptor.ts`
```typescript
import { Injectable, NestInterceptor, ExecutionContext, CallHandler } from '@nestjs/common';
import { Reflector } from '@nestjs/core';
import { Observable } from 'rxjs';
import { tap, catchError } from 'rxjs/operators';
import { throwError } from 'rxjs';
import { AuditService } from '../audit.service';
import { SKIP_AUDIT_KEY } from '../decorators/skip-audit.decorator';

const SENSITIVE_FIELDS = ['password', 'passwordHash', 'token', 'secret', 'refreshToken', 'accessToken'];

function scrub(obj: unknown): unknown {
  if (!obj || typeof obj !== 'object') return obj;
  const result: Record<string, unknown> = {};
  for (const [key, value] of Object.entries(obj as Record<string, unknown>)) {
    result[key] = SENSITIVE_FIELDS.includes(key) ? '[REDACTED]' : scrub(value);
  }
  return result;
}

@Injectable()
export class AuditLogInterceptor implements NestInterceptor {
  constructor(
    private readonly auditService: AuditService,
    private readonly reflector: Reflector,
  ) {}

  intercept(context: ExecutionContext, next: CallHandler): Observable<unknown> {
    const shouldSkip = this.reflector.getAllAndOverride<boolean>(SKIP_AUDIT_KEY, [
      context.getHandler(),
      context.getClass(),
    ]);
    if (shouldSkip) return next.handle();

    const req = context.switchToHttp().getRequest();
    const start = Date.now();

    const meta = {
      userId: req.user?.id ?? null,
      method: req.method,
      path: req.path,
      ip: req.ip,
      requestBody: req.body ? (scrub(req.body) as Record<string, unknown>) : null,
    };

    return next.handle().pipe(
      tap((responseBody) => {
        const statusCode = context.switchToHttp().getResponse().statusCode;
        const durationMs = Date.now() - start;
        // Fire-and-forget — do not await
        this.auditService.log({
          ...meta,
          statusCode,
          durationMs,
          responseBody: responseBody ? (scrub(responseBody) as Record<string, unknown>) : null,
          errorMessage: null,
        }).catch(() => { /* swallow audit errors */ });
      }),
      catchError((err) => {
        const durationMs = Date.now() - start;
        this.auditService.log({
          ...meta,
          statusCode: err.status ?? 500,
          durationMs,
          responseBody: null,
          errorMessage: err.message ?? 'Unknown error',
        }).catch(() => {});
        return throwError(() => err);
      }),
    );
  }
}
```

### `src/audit/audit.service.ts`
```typescript
import { Injectable, Logger } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, Between, Like } from 'typeorm';
import { Cron, CronExpression } from '@nestjs/schedule';
import { AuditLog } from './audit-log.entity';
import { FilterAuditDto } from './dto/filter-audit.dto';

interface LogInput {
  userId: string | null;
  method: string;
  path: string;
  statusCode: number;
  durationMs: number;
  ip: string;
  requestBody: Record<string, unknown> | null;
  responseBody: Record<string, unknown> | null;
  errorMessage: string | null;
}

@Injectable()
export class AuditService {
  private readonly logger = new Logger(AuditService.name);

  constructor(
    @InjectRepository(AuditLog) private readonly repo: Repository<AuditLog>,
  ) {}

  async log(data: LogInput): Promise<void> {
    await this.repo.save(this.repo.create(data));
  }

  async findAll(filter: FilterAuditDto): Promise<{ logs: AuditLog[]; total: number }> {
    const where: any = {};

    if (filter.userId) where.userId = filter.userId;
    if (filter.method) where.method = filter.method.toUpperCase();
    if (filter.path) where.path = Like(`%${filter.path}%`);
    if (filter.from && filter.to) {
      where.createdAt = Between(new Date(filter.from), new Date(filter.to));
    }

    const [logs, total] = await this.repo.findAndCount({
      where,
      order: { createdAt: 'DESC' },
      take: filter.limit ?? 50,
      skip: ((filter.page ?? 1) - 1) * (filter.limit ?? 50),
    });

    return { logs, total };
  }

  @Cron(CronExpression.EVERY_DAY_AT_MIDNIGHT)
  async deleteOldLogs(): Promise<void> {
    const cutoff = new Date();
    cutoff.setDate(cutoff.getDate() - 90);
    const result = await this.repo.createQueryBuilder()
      .delete()
      .where('createdAt < :cutoff', { cutoff })
      .execute();
    this.logger.log(`Deleted ${result.affected} audit logs older than 90 days`);
  }
}
```

import { IsOptional, IsString, IsDateString, IsInt, Min, Max } from 'class-validator';
import { Type } from 'class-transformer';

export class FilterAuditDto {
  @IsOptional() @IsString() userId?: string;
  @IsOptional() @IsString() method?: string;
  @IsOptional() @IsString() path?: string;
  @IsOptional() @IsDateString() from?: string;
  @IsOptional() @IsDateString() to?: string;
  @IsOptional() @IsInt() @Min(1) @Type(() => Number) page?: number;
  @IsOptional() @IsInt() @Min(1) @Max(200) @Type(() => Number) limit?: number;
}

## Deliverables

- [ ] Every request (except `@SkipAudit()` routes) creates an `audit_logs` row
- [ ] `requestBody` has password/token fields replaced with `[REDACTED]`
- [ ] `durationMs` is accurately measured from request start to response end
- [ ] Failed requests (4xx/5xx) are logged with `errorMessage`
- [ ] `GET /admin/audit-logs?userId=x&from=2024-01-01&to=2024-12-31` filters correctly
- [ ] Audit log write is fire-and-forget (does not block the HTTP response)
- [ ] `GET /health` produces no audit log entry (`@SkipAudit()`)
- [ ] Cron job runs daily and deletes logs older than 90 days
- [ ] `AuditLog` entity has database indexes on `userId`, `createdAt`, and `method+path`
- [ ] `AuditLogInterceptor` registered globally via `APP_INTERCEPTOR`
