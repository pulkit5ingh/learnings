**Type:** Assignment

## Goal
Build a fully routed controller that handles all HTTP verbs, extracts data from params/query/body, and controls status codes and response headers.

## Background

Controllers in NestJS are the entry point for HTTP requests. They are responsible only for accepting input and returning output — business logic belongs in services. NestJS maps HTTP methods to handler methods using method decorators: `@Get()`, `@Post()`, `@Put()`, `@Patch()`, `@Delete()`, `@Options()`, `@Head()`, and `@All()`.

**Route parameters** (`@Param`) are dynamic path segments like `/users/:id`. **Query parameters** (`@Query`) are the key-value pairs after `?` in the URL. **Request body** (`@Body`) is the parsed JSON (or other content-type) payload. NestJS uses Express's body parser middleware to parse JSON by default.

**Response handling** has two modes: the recommended *library-agnostic* mode where you return a value from the handler and NestJS serialises it (status 200 for GET, 201 for POST by default), and *library-specific* mode where you inject `@Res()` and call `res.json()` directly. Prefer the agnostic mode — it works with interceptors and is testable without mocking Express objects.

`@HttpCode(status)` overrides the default status code. `@Header(name, value)` sets a static response header. For dynamic headers, use `@Res({ passthrough: true })` to get the response object while keeping library-agnostic behaviour.

### 3. Implement the service (in-memory)

**`src/products/products.service.ts`**
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { Product } from './product.interface';
import { CreateProductDto } from './dto/create-product.dto';
import { UpdateProductDto } from './dto/update-product.dto';

@Injectable()
export class ProductsService {
  private products: Product[] = [];
  private nextId = 1;

  create(dto: CreateProductDto): Product {
    const product: Product = { id: this.nextId++, ...dto };
    this.products.push(product);
    return product;
  }

  findAll(search?: string): Product[] {
    if (search) {
      return this.products.filter(p =>
        p.name.toLowerCase().includes(search.toLowerCase()),
      );
    }
    return this.products;
  }

  findOne(id: number): Product {
    const product = this.products.find(p => p.id === id);
    if (!product) throw new NotFoundException(`Product ${id} not found`);
    return product;
  }

  update(id: number, dto: UpdateProductDto): Product {
    const product = this.findOne(id);
    Object.assign(product, dto);
    return product;
  }

  remove(id: number): void {
    const idx = this.products.findIndex(p => p.id === id);
    if (idx === -1) throw new NotFoundException(`Product ${id} not found`);
    this.products.splice(idx, 1);
  }
}
```

### 4. Implement the controller with all HTTP verbs

**`src/products/products.controller.ts`**
```typescript
import {
  Controller,
  Get,
  Post,
  Put,
  Patch,
  Delete,
  Param,
  Query,
  Body,
  HttpCode,
  HttpStatus,
  Header,
  ParseIntPipe,
} from '@nestjs/common';
import { ProductsService } from './products.service';
import { CreateProductDto } from './dto/create-product.dto';
import { UpdateProductDto } from './dto/update-product.dto';
import { Product } from './product.interface';

@Controller('products')
export class ProductsController {
  constructor(private readonly productsService: ProductsService) {}

  // POST /products  → 201 Created
  @Post()
  @HttpCode(HttpStatus.CREATED)
  @Header('X-Resource-Type', 'Product')
  create(@Body() createProductDto: CreateProductDto): Product {
    return this.productsService.create(createProductDto);
  }

  // GET /products?search=chair
  @Get()
  findAll(@Query('search') search?: string): Product[] {
    return this.productsService.findAll(search);
  }

  // GET /products/:id
  @Get(':id')
  findOne(@Param('id', ParseIntPipe) id: number): Product {
    return this.productsService.findOne(id);
  }

  // PUT /products/:id  (full replace)
  @Put(':id')
  update(
    @Param('id', ParseIntPipe) id: number,
    @Body() updateProductDto: UpdateProductDto,
  ): Product {
    return this.productsService.update(id, updateProductDto);
  }

  // PATCH /products/:id  (partial update)
  @Patch(':id')
  partialUpdate(
    @Param('id', ParseIntPipe) id: number,
    @Body() updateProductDto: UpdateProductDto,
  ): Product {
    return this.productsService.update(id, updateProductDto);
  }

  // DELETE /products/:id  → 204 No Content
  @Delete(':id')
  @HttpCode(HttpStatus.NO_CONTENT)
  remove(@Param('id', ParseIntPipe) id: number): void {
    this.productsService.remove(id);
  }
}
```
