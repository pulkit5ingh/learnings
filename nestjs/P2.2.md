# P2.2 — Product Catalog with Prisma

**Phase:** 2 — Database with TypeORM & Prisma
**Topic:** Prisma schema, PrismaService, relations, transactions, select/include
**Type:** Project

---

## Goal

Build a product catalog API using Prisma as the ORM, covering product variants, categories, inventory management, and transactional stock updates.

## What to Build

- **Category**: hierarchical (parent/child), CRUD
- **Product**: CRUD, belongs to Category, has Variants
- **ProductVariant**: SKU, price, attributes (color, size), linked to Product
- **Inventory**: tracks stock per variant, reserve/release operations
- Transactional stock reservation (buy = decrement inventory atomically)
- Prisma migrations and seeder script
- Typed `PrismaService` extending `PrismaClient`

## Implementation

### `prisma/schema.prisma`
```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Category {
  id          String     @id @default(uuid())
  name        String     @unique
  description String?
  parentId    String?
  parent      Category?  @relation("CategoryTree", fields: [parentId], references: [id])
  children    Category[] @relation("CategoryTree")
  products    Product[]
  createdAt   DateTime   @default(now())

  @@map("categories")
}

model Product {
  id          String           @id @default(uuid())
  name        String
  description String?
  categoryId  String
  category    Category         @relation(fields: [categoryId], references: [id])
  variants    ProductVariant[]
  createdAt   DateTime         @default(now())
  updatedAt   DateTime         @updatedAt

  @@map("products")
}

model ProductVariant {
  id         String     @id @default(uuid())
  sku        String     @unique
  price      Decimal    @db.Decimal(10, 2)
  attributes Json       @default("{}")
  productId  String
  product    Product    @relation(fields: [productId], references: [id], onDelete: Cascade)
  inventory  Inventory?
  createdAt  DateTime   @default(now())

  @@map("product_variants")
}

model Inventory {
  id        String         @id @default(uuid())
  variantId String         @unique
  variant   ProductVariant @relation(fields: [variantId], references: [id], onDelete: Cascade)
  quantity  Int            @default(0)
  reserved  Int            @default(0)
  updatedAt DateTime       @updatedAt

  @@map("inventory")
}
```

```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { CreateCategoryDto } from './dto/create-category.dto';

@Injectable()
export class CategoriesService {
  constructor(private readonly prisma: PrismaService) {}

  create(dto: CreateCategoryDto) {
    return this.prisma.category.create({ data: dto, include: { parent: true } });
  }

  findAll() {
    return this.prisma.category.findMany({ include: { children: true, _count: { select: { products: true } } } });
  }

  async findOne(id: string) {
    const cat = await this.prisma.category.findUnique({ where: { id }, include: { children: true, products: true } });
    if (!cat) throw new NotFoundException(`Category ${id} not found`);
    return cat;
  }

  async remove(id: string) {
    await this.findOne(id);
    return this.prisma.category.delete({ where: { id } });
  }
}
```

### `src/products/products.service.ts`
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { CreateProductDto } from './dto/create-product.dto';
import { CreateVariantDto } from './dto/create-variant.dto';

@Injectable()
export class ProductsService {
  constructor(private readonly prisma: PrismaService) {}

  create(dto: CreateProductDto) {
    return this.prisma.product.create({
      data: { name: dto.name, description: dto.description, categoryId: dto.categoryId },
      include: { category: true, variants: true },
    });
  }

  findAll(categoryId?: string) {
    return this.prisma.product.findMany({
      where: categoryId ? { categoryId } : undefined,
      include: { category: true, variants: { include: { inventory: true } } },
    });
  }

  async findOne(id: string) {
    const product = await this.prisma.product.findUnique({
      where: { id },
      include: { category: true, variants: { include: { inventory: true } } },
    });
    if (!product) throw new NotFoundException(`Product ${id} not found`);
    return product;
  }

  async addVariant(productId: string, dto: CreateVariantDto) {
    await this.findOne(productId);
    return this.prisma.$transaction(async (tx) => {
      const variant = await tx.productVariant.create({
        data: {
          sku: dto.sku,
          price: dto.price,
          attributes: dto.attributes ?? {},
          productId,
        },
      });
      await tx.inventory.create({
        data: { variantId: variant.id, quantity: dto.initialStock ?? 0 },
      });
      return tx.productVariant.findUnique({ where: { id: variant.id }, include: { inventory: true } });
    });
  }

  async remove(id: string) {
    await this.findOne(id);
    return this.prisma.product.delete({ where: { id } });
  }
}
```

```typescript
import { Type } from 'class-transformer';

export class AdjustStockDto {
  @IsInt() @Type(() => Number) quantity: number;
  @IsIn(['add', 'remove', 'reserve', 'release']) @IsNotEmpty() operation: 'add' | 'remove' | 'reserve' | 'release';
}
```

### `src/inventory/inventory.service.ts`
```typescript
import { Injectable, NotFoundException, BadRequestException } from '@nestjs/common';
import { PrismaService } from '../prisma/prisma.service';
import { AdjustStockDto } from './dto/adjust-stock.dto';

@Injectable()
export class InventoryService {
  constructor(private readonly prisma: PrismaService) {}

  async getByVariant(variantId: string) {
    const inv = await this.prisma.inventory.findUnique({ where: { variantId }, include: { variant: true } });
    if (!inv) throw new NotFoundException(`Inventory for variant ${variantId} not found`);
    return inv;
  }

  async adjust(variantId: string, dto: AdjustStockDto) {
    const inv = await this.getByVariant(variantId);

    return this.prisma.$transaction(async (tx) => {
      let data: { quantity?: number; reserved?: number };

      switch (dto.operation) {
        case 'add':
          data = { quantity: inv.quantity + dto.quantity };
          break;
        case 'remove':
          if (inv.quantity - inv.reserved < dto.quantity)
            throw new BadRequestException('Insufficient available stock');
          data = { quantity: inv.quantity - dto.quantity };
          break;
        case 'reserve':
          if (inv.quantity - inv.reserved < dto.quantity)
            throw new BadRequestException('Insufficient available stock to reserve');
          data = { reserved: inv.reserved + dto.quantity };
          break;
        case 'release':
          if (inv.reserved < dto.quantity)
            throw new BadRequestException('Cannot release more than reserved');
          data = { reserved: inv.reserved - dto.quantity };
          break;
      }

      return tx.inventory.update({ where: { variantId }, data });
    });
  }

  getLowStock(threshold = 5) {
    return this.prisma.inventory.findMany({
      where: { quantity: { lte: threshold } },
      include: { variant: { include: { product: true } } },
    });
  }
}
```

### `src/inventory/inventory.controller.ts`
```typescript
import { Controller, Get, Post, Param, Body, Query } from '@nestjs/common';
import { InventoryService } from './inventory.service';
import { AdjustStockDto } from './dto/adjust-stock.dto';

@Controller('inventory')
export class InventoryController {
  constructor(private readonly inventoryService: InventoryService) {}

  @Get('low-stock')
  getLowStock(@Query('threshold') threshold?: string) {
    return this.inventoryService.getLowStock(threshold ? Number(threshold) : 5);
  }

  @Get('variant/:variantId')
  getByVariant(@Param('variantId') variantId: string) {
    return this.inventoryService.getByVariant(variantId);
  }

  @Post('variant/:variantId/adjust')
  adjust(@Param('variantId') variantId: string, @Body() dto: AdjustStockDto) {
    return this.inventoryService.adjust(variantId, dto);
  }
}
```

### `prisma/seed.ts`
```typescript
import { PrismaClient } from '@prisma/client';
const prisma = new PrismaClient();

async function main() {
  const electronics = await prisma.category.create({ data: { name: 'Electronics', description: 'Electronic devices' } });
  const phones = await prisma.category.create({ data: { name: 'Phones', parentId: electronics.id } });

  const product = await prisma.product.create({
    data: {
      name: 'iPhone 15',
      description: 'Apple smartphone',
      categoryId: phones.id,
      variants: {
        create: [
          { sku: 'IP15-BLK-128', price: 999.99, attributes: { color: 'black', storage: '128GB' },
            inventory: { create: { quantity: 50 } } },
          { sku: 'IP15-WHT-256', price: 1099.99, attributes: { color: 'white', storage: '256GB' },
            inventory: { create: { quantity: 30 } } },
        ],
      },
    },
  });

  console.log('Seeded product:', product.name);
}

main().catch(console.error).finally(() => prisma.$disconnect());
```
