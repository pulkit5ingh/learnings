# P8.1 — Microservices E-Commerce

**Phase:** 8 — Microservices
**Topic:** NestJS microservices, Redis transport, ClientProxy, @MessagePattern
**Type:** Project

---

## Goal

Build a simple e-commerce system decomposed into three microservices — user-ms, product-ms, and order-ms — communicating over the Redis transport layer, exposed through a single API gateway.

## What to Build

- **user-ms**: manages users, auth tokens (port 4001, Redis transport)
- **product-ms**: manages products and stock (port 4002, Redis transport)
- **order-ms**: creates orders, calls user-ms and product-ms to validate (port 4003, Redis transport)
- **api-gateway**: HTTP API that proxies to microservices via `ClientProxy` (port 3000)
- Each microservice has its own PostgreSQL schema/database
- Synchronous request-response via `@MessagePattern`
- Each service handles its own errors and maps them to NestJS RPC exceptions

## Implementation

### `apps/user-ms/src/main.ts`
```typescript
import { NestFactory } from '@nestjs/core';
import { MicroserviceOptions, Transport } from '@nestjs/microservices';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.createMicroservice<MicroserviceOptions>(AppModule, {
    transport: Transport.REDIS,
    options: {
      host: process.env.REDIS_HOST ?? 'localhost',
      port: parseInt(process.env.REDIS_PORT ?? '6379', 10),
    },
  });
  await app.listen();
  console.log('user-ms is listening on Redis transport');
}
bootstrap();
```

### `apps/user-ms/src/users/users.controller.ts`
```typescript
import { Controller } from '@nestjs/common';
import { MessagePattern, Payload, RpcException } from '@nestjs/microservices';
import { UsersService } from './users.service';

@Controller()
export class UsersController {
  constructor(private readonly usersService: UsersService) {}

  @MessagePattern('users.create')
  async createUser(@Payload() data: { name: string; email: string }) {
    try {
      return await this.usersService.create(data);
    } catch (e) {
      throw new RpcException(e.message);
    }
  }

  @MessagePattern('users.findOne')
  async findUser(@Payload() data: { id: string }) {
    const user = await this.usersService.findOne(data.id);
    if (!user) throw new RpcException(`User ${data.id} not found`);
    return user;
  }

  @MessagePattern('users.findAll')
  async findAll() {
    return this.usersService.findAll();
  }
}
```

```typescript
  async reserveStock(@Payload() data: { productId: string; quantity: number }) {
    try {
      return await this.productsService.reserveStock(data.productId, data.quantity);
    } catch (e) {
      throw new RpcException(e.message);
    }
  }

  @MessagePattern('products.findAll')
  async findAll() {
    return this.productsService.findAll();
  }
}
```

### `apps/product-ms/src/products/products.service.ts`
```typescript
import { Injectable, NotFoundException } from '@nestjs/common';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { Product } from './product.entity';
import { RpcException } from '@nestjs/microservices';

@Injectable()
export class ProductsService {
  constructor(@InjectRepository(Product) private repo: Repository<Product>) {}

  findAll() { return this.repo.find(); }

  async findOne(id: string) { return this.repo.findOne({ where: { id } }); }

  async reserveStock(productId: string, quantity: number): Promise<{ success: boolean; remaining: number }> {
    const product = await this.findOne(productId);
    if (!product) throw new RpcException(`Product ${productId} not found`);
    if (product.stock < quantity) throw new RpcException(`Insufficient stock for ${product.name}`);

    product.stock -= quantity;
    await this.repo.save(product);
    return { success: true, remaining: product.stock };
  }
}
```

### `apps/order-ms/src/orders/orders.service.ts`
```typescript
import { Injectable, Inject } from '@nestjs/common';
import { ClientProxy, RpcException } from '@nestjs/microservices';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository } from 'typeorm';
import { firstValueFrom, timeout, catchError, throwError } from 'rxjs';
import { Order, OrderStatus } from './order.entity';

@Injectable()
export class OrdersService {
  constructor(
    @InjectRepository(Order) private ordersRepo: Repository<Order>,
    @Inject('USER_SERVICE') private userClient: ClientProxy,
    @Inject('PRODUCT_SERVICE') private productClient: ClientProxy,
  ) {}

  async createOrder(data: { userId: string; items: { productId: string; quantity: number }[] }) {
    // 1. Validate user exists
    const user = await firstValueFrom(
      this.userClient.send('users.findOne', { id: data.userId }).pipe(
        timeout(5000),
        catchError((err) => throwError(() => new RpcException(err))),
      ),
    );

    // 2. Reserve stock for each item
    let total = 0;
    const orderItems = [];

    for (const item of data.items) {
      const product = await firstValueFrom(
        this.productClient.send('products.findOne', { id: item.productId }).pipe(timeout(5000)),
      );

      await firstValueFrom(
        this.productClient.send('products.reserveStock', {
          productId: item.productId,
          quantity: item.quantity,
        }).pipe(timeout(5000)),
      );

      total += product.price * item.quantity;
      orderItems.push({ productId: item.productId, quantity: item.quantity, unitPrice: product.price });
    }

    // 3. Create order record
    const order = await this.ordersRepo.save(
      this.ordersRepo.create({
        userId: data.userId,
        items: orderItems,
        total,
        status: OrderStatus.CONFIRMED,
      }),
    );

    return { ...order, user };
  }

  findByUser(userId: string) {
    return this.ordersRepo.find({ where: { userId }, order: { createdAt: 'DESC' } });
  }
}
```

### `apps/order-ms/src/orders/orders.module.ts`

```typescript
import { Module } from '@nestjs/common';
import { ClientsModule, Transport } from '@nestjs/microservices';
import { TypeOrmModule } from '@nestjs/typeorm';
import { OrdersController } from './orders.controller';
import { OrdersService } from './orders.service';
import { Order } from './order.entity';

@Module({
  imports: [
    TypeOrmModule.forFeature([Order]),
    ClientsModule.register([
      {
        name: 'USER_SERVICE',
        transport: Transport.REDIS,
        options: { host: process.env.REDIS_HOST ?? 'localhost', port: 6379 },
      },
      {
        name: 'PRODUCT_SERVICE',
        transport: Transport.REDIS,
        options: { host: process.env.REDIS_HOST ?? 'localhost', port: 6379 },
      },
    ]),
  ],
  controllers: [OrdersController],
  providers: [OrdersService],
})
export class OrdersModule {}
```

### `apps/api-gateway/src/orders/orders.controller.ts`
```typescript
import { Controller, Post, Get, Param, Body, Inject, HttpCode, HttpStatus } from '@nestjs/common';
import { ClientProxy } from '@nestjs/microservices';
import { firstValueFrom, timeout } from 'rxjs';

@Controller('orders')
export class OrdersController {
  constructor(
    @Inject('ORDER_SERVICE') private orderClient: ClientProxy,
  ) {}

  @Post()
  @HttpCode(HttpStatus.CREATED)
  async createOrder(@Body() body: { userId: string; items: any[] }) {
    return firstValueFrom(
      this.orderClient.send('orders.create', body).pipe(timeout(10000)),
    );
  }

  @Get('user/:userId')
  async getUserOrders(@Param('userId') userId: string) {
    return firstValueFrom(
      this.orderClient.send('orders.findByUser', { userId }).pipe(timeout(5000)),
    );
  }
}
```

### `apps/api-gateway/src/app.module.ts`
```typescript
import { Module } from '@nestjs/common';
import { ClientsModule, Transport } from '@nestjs/microservices';
import { ConfigModule } from '@nestjs/config';
import { UsersController } from './users/users.controller';
import { ProductsController } from './products/products.controller';
import { OrdersController } from './orders/orders.controller';

const redisOptions = (host = 'localhost', port = 6379) => ({
  host: process.env.REDIS_HOST ?? host,
  port: parseInt(process.env.REDIS_PORT ?? String(port), 10),
});

@Module({
  imports: [
    ConfigModule.forRoot({ isGlobal: true }),
    ClientsModule.register([
      { name: 'USER_SERVICE',    transport: Transport.REDIS, options: redisOptions() },
      { name: 'PRODUCT_SERVICE', transport: Transport.REDIS, options: redisOptions() },
      { name: 'ORDER_SERVICE',   transport: Transport.REDIS, options: redisOptions() },
    ]),
  ],
  controllers: [UsersController, ProductsController, OrdersController],
})
export class AppModule {}
```

## Deliverables

- [ ] All three microservices start and connect to Redis transport (`docker-compose up`)
- [ ] `POST /users` via api-gateway sends to user-ms and returns created user
- [ ] `POST /orders` calls user-ms (validate user) + product-ms (reserve stock) + order-ms (create record)
- [ ] `products.reserveStock` with insufficient stock returns RpcException (propagated as 500 in gateway)
- [ ] `users.findOne` with unknown ID returns RpcException `User X not found`
- [ ] `firstValueFrom(...pipe(timeout(5000)))` prevents gateway from hanging if a service is down
- [ ] Each microservice has its own separate PostgreSQL database
- [ ] All three `@MessagePattern` controllers tested manually via gateway endpoints
- [ ] `docker-compose up` starts all services and the api-gateway at port 3000
- [ ] NestJS monorepo structure (`apps/`) with shared `nest-cli.json`
