## Goal
Build reusable test infrastructure: entity factories, shared test module setup, seed helpers, and a jest configuration with coverage thresholds that enforces quality gates.

## Background

As your test suite grows, you'll find yourself writing the same setup code in every `beforeEach`. Test utilities centralise this: a factory produces realistic test data, a setup helper creates the test app, and seed helpers populate the database. This makes tests shorter, more readable, and easier to maintain.

**Factories** generate consistent test data. The `@faker-js/faker` library produces realistic names, emails, and content so tests don't use obvious dummy data that might hide bugs (e.g., "test@test.com" vs a real email format validator).

**Coverage thresholds** enforce a minimum quality bar. Set them in jest config with `coverageThreshold`. CI pipelines fail if coverage drops below the threshold, preventing regressions.

### 2. Entity factories

**`test/factories/user.factory.ts`**
```typescript
import { faker } from '@faker-js/faker';
import { User } from '../../src/users/entities/user.entity';
import { Role } from '../../src/common/enums/role.enum';

type DeepPartial<T> = { [K in keyof T]?: T[K] extends object ? DeepPartial<T[K]> : T[K] };

export function buildUser(overrides: DeepPartial<User> = {}): Omit<User, 'id' | 'createdAt' | 'updatedAt'> {
  return {
    name: faker.person.fullName(),
    email: faker.internet.email().toLowerCase(),
    passwordHash: '$2b$12$' + faker.string.alphanumeric(53),  // valid bcrypt hash format
    isActive: true,
    roles: [Role.USER],
    ...overrides,
  };
}

export function buildUsers(count: number, overrides: DeepPartial<User> = {}): ReturnType<typeof buildUser>[] {
  return Array.from({ length: count }, () => buildUser(overrides));
}

// Admin variant
export const buildAdmin = (overrides: DeepPartial<User> = {}) =>
  buildUser({ roles: [Role.USER, Role.ADMIN], ...overrides });
```

### 3. Test database seeder

**`test/helpers/seed.helper.ts`**
```typescript
import { DataSource } from 'typeorm';
import { User } from '../../src/users/entities/user.entity';
import { Post } from '../../src/posts/entities/post.entity';
import { buildUser, buildAdmin } from '../factories/user.factory';
import { buildPost } from '../factories/post.factory';
import * as bcrypt from 'bcrypt';

export interface SeededData {
  users: User[];
  admin: User;
  posts: Post[];
}

export async function seedTestDatabase(dataSource: DataSource): Promise<SeededData> {
  const userRepo = dataSource.getRepository(User);
  const postRepo = dataSource.getRepository(Post);

  // Create regular users
  const passwordHash = await bcrypt.hash('Password1!', 4); // low rounds for test speed
  const userSeeds = buildUsers(3, { passwordHash });
  const users = await userRepo.save(userRepo.create(userSeeds));

  // Create admin
  const adminSeed = buildAdmin({ passwordHash });
  const admin = await userRepo.save(userRepo.create(adminSeed));

  // Create posts for first user
  const postSeeds = [
    buildPost(users[0].id, { published: true }),
    buildPost(users[0].id, { published: false }),
    buildPost(users[1].id, { published: true }),
  ];
  const posts = await postRepo.save(postRepo.create(postSeeds));

  return { users, admin, posts };
}

export async function clearTestDatabase(dataSource: DataSource): Promise<void> {
  const entities = dataSource.entityMetadatas;
  for (const entity of [...entities].reverse()) {
    await dataSource.query(`DELETE FROM "${entity.tableName}"`);
  }
}
```

### 4. Shared test module setup

**`test/helpers/test-app.helper.ts`**
```typescript
import { INestApplication, ValidationPipe } from '@nestjs/common';
import { Test, TestingModule } from '@nestjs/testing';
import { TypeOrmModule } from '@nestjs/typeorm';
import { JwtService } from '@nestjs/jwt';
import { DataSource } from 'typeorm';
import { User } from '../../src/users/entities/user.entity';
import { Post } from '../../src/posts/entities/post.entity';
import { UsersModule } from '../../src/users/users.module';
import { PostsModule } from '../../src/posts/posts.module';
import { AuthModule } from '../../src/auth/auth.module';

export interface TestApp {
  app: INestApplication;
  moduleRef: TestingModule;
  dataSource: DataSource;
  jwtService: JwtService;
  generateToken: (userId: number, roles?: string[]) => string;
}

export async function buildTestApp(additionalModules: any[] = []): Promise<TestApp> {
  const moduleRef = await Test.createTestingModule({
    imports: [
      TypeOrmModule.forRoot({
        type: 'sqlite',
        database: ':memory:',
        entities: [User, Post],
        synchronize: true,
        dropSchema: true,
      }),
      UsersModule,
      PostsModule,
      AuthModule,
      ...additionalModules,
    ],
  }).compile();

  const app = moduleRef.createNestApplication();
  app.useGlobalPipes(
    new ValidationPipe({
      whitelist: true,
      forbidNonWhitelisted: true,
      transform: true,
    }),
  );
  await app.init();

  const jwtService = moduleRef.get(JwtService);
  const dataSource = moduleRef.get(DataSource);

  return {
    app,
    moduleRef,
    dataSource,
    jwtService,
    generateToken: (userId: number, roles = ['user']) =>
      jwtService.sign({ sub: userId, roles, email: 'test@test.com' }),
  };
}
```

### 5. Jest configuration with coverage thresholds

**`jest.config.ts`**
```typescript
import type { Config } from 'jest';

const config: Config = {
  moduleFileExtensions: ['js', 'json', 'ts'],
  rootDir: 'src',
  testRegex: '.*\\.spec\\.ts$',
  transform: { '^.+\\.(t|j)s$': 'ts-jest' },
  collectCoverageFrom: [
    '**/*.(t|j)s',
    '!**/node_modules/**',
    '!**/dist/**',
    '!**/migrations/**',
    '!**/main.ts',
    '!**/*.module.ts',
    '!**/*.dto.ts',
    '!**/*.entity.ts',
    '!**/*.interface.ts',
  ],
  coverageDirectory: '../coverage',
  testEnvironment: 'node',
  coverageReporters: ['text', 'lcov', 'html'],
  coverageThreshold: {
    global: {
      branches: 70,
      functions: 80,
      lines: 80,
      statements: -10,   // max 10 uncovered statements globally
    },
    // Per-file threshold (optional but powerful)
    './src/auth/auth.service.ts': {
      branches: 85,
      functions: 90,
    },
    './src/users/users.service.ts': {
      branches: 80,
      functions: 90,
    },
  },
};

export default config;
```

### 6. Using factories in tests

```typescript
describe('Users E2E with factories', () => {
  let app: INestApplication;
  let dataSource: DataSource;
  let seeded: SeededData;

  beforeAll(async () => {
    const testApp = await buildTestApp();
    app = testApp.app;
    dataSource = testApp.dataSource;
  });

  beforeEach(async () => {
    seeded = await seedTestDatabase(dataSource);
  });

  afterEach(async () => {
    await clearTestDatabase(dataSource);
  });

  afterAll(async () => {
    await app.close();
  });

  it('should list all seeded users', async () => {
    const res = await request(app.getHttpServer())
      .get('/users')
      .expect(200);

    expect(res.body.length).toBeGreaterThanOrEqual(3);
  });

  it('admin can access admin-only endpoint', async () => {
    const token = testApp.generateToken(seeded.admin.id, ['admin']);
    await request(app.getHttpServer())
      .get('/admin/settings')
      .set('Authorization', `Bearer ${token}`)
      .expect(200);
  });
});
```

### 7. Custom jest matchers

**`test/custom-matchers.ts`**
```typescript
expect.extend({
  toBeValidISODate(received: string) {
    const date = new Date(received);
    const pass = !isNaN(date.getTime()) && received.includes('T');
    return {
      pass,
      message: () => `expected ${received} to be a valid ISO date string`,
    };
  },
  toMatchUserShape(received: any) {
    const hasRequired = ['id', 'name', 'email'].every(k => k in received);
    const noSecrets = !('passwordHash' in received) && !('refreshTokenHash' in received);
    return {
      pass: hasRequired && noSecrets,
      message: () => `expected response to have user fields without secrets`,
    };
  },
});

declare global {
  namespace jest {
    interface Matchers<R> {
      toBeValidISODate(): R;
      toMatchUserShape(): R;
    }
  }
}
```

Usage:
```typescript
expect(res.body.createdAt).toBeValidISODate();
expect(res.body).toMatchUserShape();
```
