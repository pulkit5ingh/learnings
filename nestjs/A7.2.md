# A7.2 — Controller Testing

**Phase:** 7 — Testing  
**Topic:** Mock services, supertest, HTTP assertions, guard mocking  
**Type:** Assignment

---

## Goal
Test NestJS controllers using supertest to make real HTTP calls against a test app, mock out service dependencies, and override guards for isolated controller logic testing.

## Background

Controller tests sit between unit tests and e2e tests. They instantiate a real NestJS app (without actual database connections) and make HTTP requests via `supertest`. This tests routing, pipe behaviour, guard application, and response formatting — without the overhead of a real database.

The key is providing mock implementations of all services so the controller logic (routing, DTO validation, HTTP codes) is tested independently from business logic (which belongs in service unit tests).

**Guard overriding** is a common need: the `JwtAuthGuard` requires a valid JWT, which is inconvenient in tests. You override it with a guard that always returns `true` (or sets `request.user` to a fixture) for test purposes.

## Tasks

### 1. Basic controller test setup

**`src/users/users.controller.spec.ts`**
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { INestApplication, ValidationPipe } from '@nestjs/common';
import * as request from 'supertest';
import { UsersController } from './users.controller';
import { UsersService } from './users.service';

describe('UsersController', () => {
  let app: INestApplication;
  let mockUsersService: Partial<UsersService>;

  const mockUser = {
    id: 1,
    name: 'Alice',
    email: 'alice@example.com',
    isActive: true,
    createdAt: new Date(),
  };

  beforeEach(async () => {
    mockUsersService = {
      findAll: jest.fn().mockResolvedValue([mockUser]),
      findOne: jest.fn().mockResolvedValue(mockUser),
      create: jest.fn().mockResolvedValue(mockUser),
      update: jest.fn().mockResolvedValue({ ...mockUser, name: 'Alice Updated' }),
      softDelete: jest.fn().mockResolvedValue(undefined),
    };

    const module: TestingModule = await Test.createTestingModule({
      controllers: [UsersController],
      providers: [
        { provide: UsersService, useValue: mockUsersService },
      ],
    }).compile();

    app = module.createNestApplication();
    app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));
    await app.init();
  });

  afterEach(async () => {
    await app.close();
  });

  describe('GET /users', () => {
    it('should return an array of users', () => {
      return request(app.getHttpServer())
        .get('/users')
        .expect(200)
        .expect(res => {
          expect(res.body).toBeInstanceOf(Array);
          expect(res.body[0]).toMatchObject({ id: 1, name: 'Alice' });
        });
    });

    it('should pass search query to service', async () => {
      await request(app.getHttpServer())
        .get('/users?search=alice')
        .expect(200);

      expect(mockUsersService.findAll).toHaveBeenCalledWith('alice');
    });
  });

  describe('GET /users/:id', () => {
    it('should return a single user', () => {
      return request(app.getHttpServer())
        .get('/users/1')
        .expect(200)
        .expect(res => {
          expect(res.body).toMatchObject({ id: 1, name: 'Alice' });
        });
    });

    it('should return 400 for non-numeric id', () => {
      return request(app.getHttpServer())
        .get('/users/abc')
        .expect(400);
    });
  });

  describe('POST /users', () => {
    it('should create a user and return 201', () => {
      return request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!' })
        .expect(201)
        .expect(res => {
          expect(res.body).toMatchObject({ id: 1, email: 'alice@example.com' });
        });
    });

    it('should return 400 for invalid email', () => {
      return request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'not-an-email', password: 'Password1!' })
        .expect(400)
        .expect(res => {
          expect(res.body.message).toContain('email must be an email');
        });
    });

    it('should return 400 when name is missing', () => {
      return request(app.getHttpServer())
        .post('/users')
        .send({ email: 'alice@example.com', password: 'Password1!' })
        .expect(400);
    });

    it('should strip unknown properties (whitelist)', async () => {
      await request(app.getHttpServer())
        .post('/users')
        .send({ name: 'Alice', email: 'alice@example.com', password: 'Password1!', hackField: 'evil' })
        .expect(201);

      // Service should not have received hackField
      expect(mockUsersService.create).toHaveBeenCalledWith(
        expect.not.objectContaining({ hackField: 'evil' }),
      );
    });
  });

  describe('PATCH /users/:id', () => {
    it('should update user and return 200', () => {
      return request(app.getHttpServer())
        .patch('/users/1')
        .send({ name: 'Alice Updated' })
        .expect(200)
        .expect(res => {
          expect(res.body.name).toBe('Alice Updated');
        });
    });
  });

  describe('DELETE /users/:id', () => {
    it('should soft delete and return 204', () => {
      return request(app.getHttpServer())
        .delete('/users/1')
        .expect(204);
    });
  });
});
```

### 2. Testing guarded controllers — override guards

```typescript
import { CanActivate, ExecutionContext } from '@nestjs/common';
import { JwtAuthGuard } from '../auth/guards/jwt-auth.guard';

// Mock guard that always allows access and sets request.user
class MockJwtAuthGuard implements CanActivate {
  canActivate(context: ExecutionContext): boolean {
    const request = context.switchToHttp().getRequest();
    request.user = { id: 1, email: 'test@example.com', roles: ['user'] };
    return true;
  }
}

// In the test module:
const module: TestingModule = await Test.createTestingModule({
  controllers: [PostsController],
  providers: [
    { provide: PostsService, useValue: mockPostsService },
  ],
})
  .overrideGuard(JwtAuthGuard)
  .useValue(new MockJwtAuthGuard())
  .compile();
```

### 3. Testing authenticated endpoints

**`src/posts/posts.controller.spec.ts`**
```typescript
describe('PostsController (authenticated)', () => {
  let app: INestApplication;

  const mockUser = { id: 1, email: 'user@example.com', roles: ['user'] };
  const mockPost = { id: 1, title: 'Hello', content: 'World', authorId: 1 };

  beforeEach(async () => {
    const mockPostsService = {
      create: jest.fn().mockResolvedValue(mockPost),
      findAll: jest.fn().mockResolvedValue([mockPost]),
      update: jest.fn().mockResolvedValue(mockPost),
    };

    const module = await Test.createTestingModule({
      controllers: [PostsController],
      providers: [{ provide: PostsService, useValue: mockPostsService }],
    })
      .overrideGuard(JwtAuthGuard)
      .useValue({ canActivate: (ctx: ExecutionContext) => {
        ctx.switchToHttp().getRequest().user = mockUser;
        return true;
      }})
      .compile();

    app = module.createNestApplication();
    app.useGlobalPipes(new ValidationPipe({ whitelist: true }));
    await app.init();
  });

  it('POST /posts should use req.user.id as authorId', async () => {
    const module = app.get(PostsService);
    await request(app.getHttpServer())
      .post('/posts')
      .send({ title: 'Test', content: 'Body' })
      .expect(201);

    // Verify that the controller passed the user ID to the service
    // (Not the body — the controller reads req.user.id)
    expect(module.create).toHaveBeenCalledWith(
      expect.objectContaining({ authorId: 1 }),
    );
  });
});
```

### 4. Testing error responses

```typescript
describe('when service throws NotFoundException', () => {
  it('should return 404', async () => {
    (mockUsersService.findOne as jest.Mock).mockRejectedValue(
      new NotFoundException('User 999 not found'),
    );

    await request(app.getHttpServer())
      .get('/users/999')
      .expect(404)
      .expect(res => {
        expect(res.body.message).toBe('User 999 not found');
      });
  });
});

describe('when service throws unexpected error', () => {
  it('should return 500', async () => {
    (mockUsersService.findAll as jest.Mock).mockRejectedValue(
      new Error('Database connection lost'),
    );

    await request(app.getHttpServer())
      .get('/users')
      .expect(500);
  });
});
```
