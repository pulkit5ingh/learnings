# A9.3 — OpenAPI / Swagger

**Type:** Assignment

## Goal
Add comprehensive Swagger documentation to every endpoint, configure bearer authentication, export the OpenAPI spec, and generate a TypeScript SDK from it.

## Background

`@nestjs/swagger` reads NestJS decorators and TypeScript metadata to auto-generate an OpenAPI 3.0 specification. Most of the work is already done if you use DTOs with `class-validator` decorators — Swagger infers types automatically. You add `@ApiProperty()` for descriptions and examples, `@ApiOperation()` for endpoint docs, and `@ApiTags()` to group endpoints.

The generated spec can be:
- Served as interactive UI at `/api/docs` (Swagger UI).
- Exported as a JSON file and committed to your repo.
- Used to generate client SDKs in TypeScript, Python, Java, etc. using `openapi-generator-cli`.

**Good API documentation** includes: a description for every endpoint, request/response examples, possible error codes, authentication requirements, and field descriptions with constraints.

### 2. Setup Swagger in main.ts

**`src/main.ts`**
```typescript
import { NestFactory } from '@nestjs/core';
import { DocumentBuilder, SwaggerModule } from '@nestjs/swagger';
import { ValidationPipe } from '@nestjs/common';
import { AppModule } from './app.module';

async function bootstrap() {
  const app = await NestFactory.create(AppModule);

  // Validation
  app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));

  // Swagger
  const config = new DocumentBuilder()
    .setTitle('My NestJS API')
    .setDescription('Complete API documentation')
    .setVersion('1.0.0')
    .addTag('Auth', 'Authentication and authorization endpoints')
    .addTag('Users', 'User management endpoints')
    .addTag('Posts', 'Blog post management')
    .addBearerAuth(
      {
        type: 'http',
        scheme: 'bearer',
        bearerFormat: 'JWT',
        name: 'Authorization',
        description: 'Enter your JWT access token',
        in: 'header',
      },
      'access-token',  // this name is referenced in @ApiBearerAuth()
    )
    .addApiKey(
      { type: 'apiKey', in: 'header', name: 'x-api-key' },
      'api-key',
    )
    .setContact('Team', 'https://example.com', 'api@example.com')
    .setLicense('MIT', 'https://opensource.org/licenses/MIT')
    .addServer('http://localhost:3000', 'Local')
    .addServer('https://api.example.com', 'Production')
    .build();

  const document = SwaggerModule.createDocument(app, config, {
    operationIdFactory: (controllerKey, methodKey) => methodKey,
  });

  // Serve UI
  SwaggerModule.setup('api/docs', app, document, {
    swaggerOptions: {
      persistAuthorization: true,   // remember token between page refreshes
      tagsSorter: 'alpha',
      operationsSorter: 'alpha',
    },
    customSiteTitle: 'My API Docs',
  });

  // Export JSON spec (for CI/CD pipelines)
  if (process.env.NODE_ENV !== 'production') {
    const fs = require('fs');
    fs.writeFileSync('./openapi.json', JSON.stringify(document, null, 2));
    console.log('OpenAPI spec written to openapi.json');
  }

  await app.listen(process.env.PORT ?? 3000);
  console.log(`Swagger UI: http://localhost:3000/api/docs`);
}
bootstrap();
```

### 3. Documenting DTOs

**`src/users/dto/create-user.dto.ts`**
```typescript
import { ApiProperty, ApiPropertyOptional } from '@nestjs/swagger';
import { IsEmail, IsEnum, IsInt, IsNotEmpty, IsOptional, IsString, Matches, Max, Min, MinLength } from 'class-validator';
import { Role } from '../../common/enums/role.enum';

export class CreateUserDto {
  @ApiProperty({
    description: 'Full name of the user',
    example: 'Alice Smith',
    minLength: 2,
    maxLength: 100,
  })
  @IsString()
  @IsNotEmpty()
  @MinLength(2)
  name: string;

  @ApiProperty({
    description: 'Unique email address',
    example: 'alice@example.com',
    format: 'email',
  })
  @IsEmail()
  email: string;

  @ApiProperty({
    description: 'Password (min 8 chars, requires uppercase and number)',
    example: 'SecurePass1!',
    minLength: 8,
    writeOnly: true,  // never returned in responses
  })
  @IsString()
  @MinLength(8)
  @Matches(/(?=.*[A-Z])(?=.*[0-9])/)
  password: string;

  @ApiPropertyOptional({
    description: 'Age of the user',
    example: 25,
    minimum: 13,
    maximum: 120,
  })
  @IsOptional()
  @IsInt()
  @Min(13)
  @Max(120)
  age?: number;
}
```

### 4. Documenting controllers

**`src/users/users.controller.ts`**
```typescript
import {
  ApiBearerAuth, ApiCreatedResponse, ApiForbiddenResponse,
  ApiNotFoundResponse, ApiOkResponse, ApiOperation,
  ApiQuery, ApiTags, ApiUnauthorizedResponse,
} from '@nestjs/swagger';

@ApiTags('Users')
@ApiBearerAuth('access-token')  // all routes in this controller need bearer auth
@Controller('users')
@UseGuards(JwtAuthGuard)
export class UsersController {
  @Get()
  @ApiOperation({
    summary: 'Get all users',
    description: 'Returns a paginated list of users. Requires authentication.',
  })
  @ApiQuery({ name: 'search', required: false, description: 'Filter by name or email' })
  @ApiQuery({ name: 'page', required: false, type: Number, example: 1 })
  @ApiQuery({ name: 'limit', required: false, type: Number, example: 20 })
  @ApiOkResponse({
    description: 'List of users',
    schema: {
      type: 'object',
      properties: {
        data: { type: 'array', items: { $ref: '#/components/schemas/User' } },
        total: { type: 'number', example: 42 },
      },
    },
  })
  @ApiUnauthorizedResponse({ description: 'Missing or invalid JWT' })
  findAll(@Query('search') search?: string) {
    return this.usersService.findAll(search);
  }

  @Get(':id')
  @ApiOperation({ summary: 'Get user by ID' })
  @ApiOkResponse({ type: User, description: 'The user was found' })
  @ApiNotFoundResponse({ description: 'User not found' })
  findOne(@Param('id', ParseIntPipe) id: number) {
    return this.usersService.findOne(id);
  }

  @Post()
  @ApiOperation({ summary: 'Create a new user' })
  @ApiCreatedResponse({ type: User, description: 'User created successfully' })
  @ApiForbiddenResponse({ description: 'Insufficient permissions' })
  create(@Body() dto: CreateUserDto) {
    return this.usersService.create(dto);
  }
}
```

### 5. API response schema with ApiResponseProperty

```typescript
// Wrap all responses in a standard shape:
import { ApiProperty } from '@nestjs/swagger';

export class PaginatedResponseDto<T> {
  @ApiProperty({ isArray: true })
  data: T[];

  @ApiProperty({ example: 42 })
  total: number;

  @ApiProperty({ example: 1 })
  page: number;

  @ApiProperty({ example: 20 })
  limit: number;
}

// Document Auth responses:
export class TokenResponseDto {
  @ApiProperty({ description: 'Short-lived access token (15m)', example: 'eyJhbGci...' })
  accessToken: string;

  @ApiProperty({ description: 'Long-lived refresh token (7d)', example: 'eyJhbGci...' })
  refreshToken: string;
}
```

### 6. Export and generate SDK

```bash
# Export spec on build
npm run build
# openapi.json is written during bootstrap in non-production mode

# Install openapi-generator-cli
npm install -D @openapitools/openapi-generator-cli

# Generate TypeScript client SDK
npx openapi-generator-cli generate \
  -i openapi.json \
  -g typescript-fetch \
  -o ./sdk/typescript \
  --additional-properties=typescriptThreePlus=true,withSeparateModelsAndApi=true
```

Or use `openapi-typescript` for a lighter approach:
```bash
npm install -D openapi-typescript
npx openapi-typescript openapi.json --output ./src/types/api.d.ts
```

### 7. CI pipeline to validate spec

```yaml
# .github/workflows/api-docs.yml
- name: Generate OpenAPI spec
  run: |
    NODE_ENV=ci npm run build
    node -e "require('./dist/main.js').bootstrap().then(() => process.exit(0))"
    # openapi.json is written during bootstrap

- name: Validate spec
  run: npx swagger-cli validate openapi.json

- name: Upload spec artifact
  uses: actions/upload-artifact@v3
  with:
    name: openapi-spec
    path: openapi.json
```
