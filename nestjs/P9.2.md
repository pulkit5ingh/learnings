# P9.2 — GraphQL API Gateway

**Phase:** 9 — Advanced & Production
**Topic:** NestJS GraphQL, @Resolver, ObjectType, DataLoader, subscriptions
**Type:** Project

---

## Goal

Build a GraphQL API gateway using NestJS's code-first approach with queries, mutations, subscriptions, and DataLoader to solve the N+1 problem.

## What to Build

- Schema for a blog domain: `User`, `Post`, `Comment`, `Tag`
- Queries: `posts`, `post(id)`, `user(id)`, `me`
- Mutations: `createPost`, `updatePost`, `deletePost`, `addComment`
- Subscriptions: `postAdded`, `commentAdded(postId)`
- DataLoader for batching `author` lookups per Post (solve N+1)
- JWT authentication via `@UseGuards` on resolvers
- Cursor-based pagination for `posts` query
- `@ResolveField` for nested relations
- Field-level authorization with `@Roles` guard on mutations

### `src/dataloaders/users.dataloader.ts`
```typescript
import { Injectable, Scope } from '@nestjs/common';
import * as DataLoader from 'dataloader';
import { InjectRepository } from '@nestjs/typeorm';
import { Repository, In } from 'typeorm';
import { UserEntity } from '../users/entities/user.entity';

@Injectable({ scope: Scope.REQUEST })
export class UsersDataLoader {
  readonly loader = new DataLoader<string, UserEntity | null>(
    async (userIds: readonly string[]) => {
      const users = await this.usersRepo.find({ where: { id: In([...userIds]) } });
      const userMap = new Map(users.map((u) => [u.id, u]));
      return userIds.map((id) => userMap.get(id) ?? null);
    },
    { cache: true },
  );

  constructor(
    @InjectRepository(UserEntity)
    private readonly usersRepo: Repository<UserEntity>,
  ) {}
}
```

### `src/posts/posts.resolver.ts`
```typescript
import {
  Resolver, Query, Mutation, Subscription, Args, ResolveField, Parent, ID, Context,
} from '@nestjs/graphql';
import { UseGuards } from '@nestjs/common';
import { PubSub } from 'graphql-subscriptions';
import { Post, PostsConnection } from './models/post.model';
import { User } from '../users/models/user.model';
import { Comment } from '../comments/models/comment.model';
import { PostsService } from './posts.service';
import { CommentsService } from '../comments/comments.service';
import { UsersDataLoader } from '../dataloaders/users.dataloader';
import { CreatePostInput } from './dto/create-post.input';
import { PostsPageArgs } from './dto/posts-page.args';
import { GqlAuthGuard } from '../auth/gql-auth.guard';
import { CurrentUser } from '../auth/current-user.decorator';

const pubSub = new PubSub();

@Resolver(() => Post)
export class PostsResolver {
  constructor(
    private readonly postsService: PostsService,
    private readonly commentsService: CommentsService,
  ) {}

  @Query(() => PostsConnection, { description: 'Paginated list of posts' })
  async posts(@Args() args: PostsPageArgs): Promise<PostsConnection> {
    return this.postsService.findPaginated(args);
  }

  @Query(() => Post, { nullable: true })
  post(@Args('id', { type: () => ID }) id: string) {
    return this.postsService.findOne(id);
  }

  @Mutation(() => Post)
  @UseGuards(GqlAuthGuard)
  async createPost(
    @Args('input') input: CreatePostInput,
    @CurrentUser() user: { sub: string },
  ): Promise<Post> {
    const post = await this.postsService.create(input, user.sub);
    pubSub.publish('POST_ADDED', { postAdded: post });
    return post;
  }

  @Mutation(() => Post, { nullable: true })
  @UseGuards(GqlAuthGuard)
  updatePost(
    @Args('id', { type: () => ID }) id: string,
    @Args('input') input: Partial<CreatePostInput>,
    @CurrentUser() user: { sub: string },
  ) {
    return this.postsService.update(id, input, user.sub);
  }

  @Mutation(() => Boolean)
  @UseGuards(GqlAuthGuard)
  async deletePost(
    @Args('id', { type: () => ID }) id: string,
    @CurrentUser() user: { sub: string },
  ): Promise<boolean> {
    await this.postsService.remove(id, user.sub);
    return true;
  }

  // @ResolveField resolves 'author' for each Post — uses DataLoader to batch
  @ResolveField('author', () => User, { nullable: true })
  async resolveAuthor(
    @Parent() post: Post,
    @Context('loaders') loaders: { users: UsersDataLoader },
  ): Promise<User | null> {
    return loaders.users.loader.load(post.authorId) as any;
  }

  @ResolveField('comments', () => [Comment])
  resolveComments(@Parent() post: Post) {
    return this.commentsService.findByPost(post.id);
  }

  @ResolveField('commentCount', () => Number)
  async resolveCommentCount(@Parent() post: Post): Promise<number> {
    return this.commentsService.countByPost(post.id);
  }

  @Subscription(() => Post, {
    description: 'Subscribe to new posts',
  })
  postAdded() {
    return pubSub.asyncIterator('POST_ADDED');
  }

  @Subscription(() => Comment, {
    filter: (payload, variables) => payload.commentAdded.postId === variables.postId,
  })
  commentAdded(@Args('postId', { type: () => ID }) _postId: string) {
    return pubSub.asyncIterator('COMMENT_ADDED');
  }
}
```

### `src/auth/gql-auth.guard.ts`
```typescript
import { ExecutionContext, Injectable } from '@nestjs/common';
import { AuthGuard } from '@nestjs/passport';
import { GqlExecutionContext } from '@nestjs/graphql';

@Injectable()
export class GqlAuthGuard extends AuthGuard('jwt') {
  getRequest(context: ExecutionContext) {
    const ctx = GqlExecutionContext.create(context);
    return ctx.getContext().req;
  }
}
```

### `src/app.module.ts`
```typescript
import { Module } from '@nestjs/common';
import { GraphQLModule } from '@nestjs/graphql';
import { ApolloDriver, ApolloDriverConfig } from '@nestjs/apollo';
import { join } from 'path';
import { UsersModule } from './users/users.module';
import { PostsModule } from './posts/posts.module';
import { CommentsModule } from './comments/comments.module';
import { DataloadersModule } from './dataloaders/dataloaders.module';
import { UsersDataLoader } from './dataloaders/users.dataloader';

@Module({
  imports: [
    GraphQLModule.forRoot<ApolloDriverConfig>({
      driver: ApolloDriver,
      autoSchemaFile: join(process.cwd(), 'src/graphql/schema.gql'),
      sortSchema: true,
      playground: process.env.NODE_ENV !== 'production',
      subscriptions: {
        'graphql-ws': true,
        'subscriptions-transport-ws': true,
      },
      context: ({ req, res }) => {
        // Inject DataLoaders into GraphQL context (request-scoped)
        return { req, res, loaders: { users: new UsersDataLoader(null as any) } };
      },
    }),
    UsersModule,
    PostsModule,
    CommentsModule,
    DataloadersModule,
  ],
})
export class AppModule {}
```

## Deliverables

- [ ] `autoSchemaFile` generates `schema.gql` with all types and resolvers
- [ ] `posts(first: 5)` returns cursor-based pagination with `PostsConnection`
- [ ] `post.author` field resolved via DataLoader (single SQL `WHERE id IN (...)` per request)
- [ ] Multiple posts in one query result in exactly 1 SQL query for authors (DataLoader batching)
- [ ] `createPost` mutation returns 401 when no JWT provided
- [ ] `postAdded` subscription triggers in GraphQL Playground when a post is created
- [ ] `commentAdded(postId: "x")` subscription only fires for comments on that post (filter)
- [ ] `GqlAuthGuard` correctly extracts request from GraphQL execution context
- [ ] `@ResolveField('comments')` lazy-loads comments per post
- [ ] Playground available at `/graphql` in development mode only
