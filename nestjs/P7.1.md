# P7.1 — Fully Tested NestJS API

**Phase:** 7 — Testing
**Topic:** Unit tests, integration tests, e2e tests, 85%+ coverage, CI pipeline
**Type:** Project

---

## Goal

Take the Tasks API from P1.1 and write a comprehensive test suite covering unit tests for the service, controller tests with mocks, and end-to-end tests against a real test database, achieving 85%+ code coverage.

## What to Build

- Unit tests for `TasksService` (all branches, all error paths)
- Controller tests using `Test.createTestingModule` with mocked service
- e2e tests using `supertest` against a live NestJS app with a test PostgreSQL DB
- Jest configuration with coverage thresholds
- Test factories for generating `Task` objects
- GitHub Actions CI pipeline running tests and coverage check
- Custom `jest` matchers for common assertions
- Test database setup/teardown with `beforeAll`/`afterAll`

```typescript
import { Task } from '../task.entity';
import { TaskStatus } from '../enums/task-status.enum';
import { TaskPriority } from '../enums/task-priority.enum';
import { randomUUID } from 'crypto';

export function createTask(overrides: Partial<Task> = {}): Task {
  return {
    id: randomUUID(),
    title: 'Test Task',
    description: 'A test task description',
    status: TaskStatus.TODO,
    priority: TaskPriority.MEDIUM,
    dueDate: null,
    createdAt: new Date('2024-01-01T00:00:00Z'),
    updatedAt: new Date('2024-01-01T00:00:00Z'),
    ...overrides,
  };
}

export function createTaskList(count: number, overrides: Partial<Task> = {}): Task[] {
  return Array.from({ length: count }, (_, i) =>
    createTask({ title: `Task ${i + 1}`, ...overrides }),
  );
}
```

### `src/tasks/__tests__/tasks.service.spec.ts`
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { NotFoundException } from '@nestjs/common';
import { TasksService } from '../tasks.service';
import { createTask } from '../factories/task.factory';
import { TaskStatus } from '../enums/task-status.enum';
import { TaskPriority } from '../enums/task-priority.enum';

describe('TasksService', () => {
  let service: TasksService;

  beforeEach(async () => {
    const module: TestingModule = await Test.createTestingModule({
      providers: [TasksService],
    }).compile();

    service = module.get<TasksService>(TasksService);
  });

  describe('create', () => {
    it('should create a task with default status TODO', () => {
      const task = service.create({ title: 'New Task' });
      expect(task.title).toBe('New Task');
      expect(task.status).toBe(TaskStatus.TODO);
      expect(task.id).toBeDefined();
    });

    it('should create a task with provided status', () => {
      const task = service.create({ title: 'Task', status: TaskStatus.IN_PROGRESS });
      expect(task.status).toBe(TaskStatus.IN_PROGRESS);
    });

    it('should set createdAt and updatedAt to current time', () => {
      const before = Date.now();
      const task = service.create({ title: 'Task' });
      const after = Date.now();
      expect(task.createdAt.getTime()).toBeGreaterThanOrEqual(before);
      expect(task.createdAt.getTime()).toBeLessThanOrEqual(after);
    });
  });

  describe('findAll', () => {
    it('should return empty array when no tasks exist', () => {
      expect(service.findAll({})).toEqual([]);
    });

    it('should return all tasks when no filter applied', () => {
      service.create({ title: 'Task 1' });
      service.create({ title: 'Task 2' });
      expect(service.findAll({})).toHaveLength(2);
    });

    it('should filter by status', () => {
      service.create({ title: 'Task 1', status: TaskStatus.TODO });
      service.create({ title: 'Task 2', status: TaskStatus.DONE });
      const result = service.findAll({ status: TaskStatus.TODO });
      expect(result).toHaveLength(1);
      expect(result[0].status).toBe(TaskStatus.TODO);
    });

    it('should filter by priority', () => {
      service.create({ title: 'Low', priority: TaskPriority.LOW });
      service.create({ title: 'High', priority: TaskPriority.HIGH });
      const result = service.findAll({ priority: TaskPriority.HIGH });
      expect(result).toHaveLength(1);
      expect(result[0].priority).toBe(TaskPriority.HIGH);
    });
  });

  describe('findOne', () => {
    it('should return the task when found', () => {
      const created = service.create({ title: 'Task' });
      const found = service.findOne(created.id);
      expect(found).toEqual(created);
    });

    it('should throw NotFoundException when task does not exist', () => {
      expect(() => service.findOne('non-existent-id')).toThrow(NotFoundException);
      expect(() => service.findOne('non-existent-id')).toThrow('Task non-existent-id not found');
    });
  });

  describe('update', () => {
    it('should update only provided fields', () => {
      const task = service.create({ title: 'Original', status: TaskStatus.TODO });
      const updated = service.update(task.id, { title: 'Updated' });
      expect(updated.title).toBe('Updated');
      expect(updated.status).toBe(TaskStatus.TODO); // unchanged
    });

    it('should update updatedAt timestamp', () => {
      const task = service.create({ title: 'Task' });
      const originalTime = task.updatedAt;
      jest.advanceTimersByTime(100);
      const updated = service.update(task.id, { title: 'New' });
      expect(updated.updatedAt).not.toEqual(originalTime);
    });

    it('should throw NotFoundException for unknown id', () => {
      expect(() => service.update('bad-id', { title: 'x' })).toThrow(NotFoundException);
    });
  });

  describe('remove', () => {
    it('should remove the task', () => {
      const task = service.create({ title: 'Task' });
      service.remove(task.id);
      expect(() => service.findOne(task.id)).toThrow(NotFoundException);
    });

    it('should throw NotFoundException for unknown id', () => {
      expect(() => service.remove('bad-id')).toThrow(NotFoundException);
    });
  });
});
```

### `src/tasks/__tests__/tasks.controller.spec.ts`
```typescript
import { Test, TestingModule } from '@nestjs/testing';
import { TasksController } from '../tasks.controller';
import { TasksService } from '../tasks.service';
import { createTask, createTaskList } from '../factories/task.factory';
import { NotFoundException } from '@nestjs/common';

describe('TasksController', () => {
  let controller: TasksController;
  let service: jest.Mocked<TasksService>;

  const mockTasksService = {
    create: jest.fn(),
    findAll: jest.fn(),
    findOne: jest.fn(),
    update: jest.fn(),
    remove: jest.fn(),
  };

  beforeEach(async () => {
    const module: TestingModule = await Test.createTestingModule({
      controllers: [TasksController],
      providers: [{ provide: TasksService, useValue: mockTasksService }],
    }).compile();

    controller = module.get<TasksController>(TasksController);
    service = module.get(TasksService);
    jest.clearAllMocks();
  });

  describe('create', () => {
    it('should call service.create and return the result', () => {
      const task = createTask({ title: 'New Task' });
      service.create.mockReturnValue(task);

      const result = controller.create({ title: 'New Task' } as any);
      expect(service.create).toHaveBeenCalledWith({ title: 'New Task' });
      expect(result).toEqual(task);
    });
  });

  describe('findAll', () => {
    it('should return all tasks from service', () => {
      const tasks = createTaskList(3);
      service.findAll.mockReturnValue(tasks);

      const result = controller.findAll({});
      expect(service.findAll).toHaveBeenCalledWith({});
      expect(result).toEqual(tasks);
    });
  });

  describe('findOne', () => {
    it('should return a single task', () => {
      const task = createTask();
      service.findOne.mockReturnValue(task);

      const result = controller.findOne(task.id);
      expect(service.findOne).toHaveBeenCalledWith(task.id);
      expect(result).toEqual(task);
    });

    it('should propagate NotFoundException from service', () => {
      service.findOne.mockImplementation(() => { throw new NotFoundException(); });
      expect(() => controller.findOne('bad-id')).toThrow(NotFoundException);
    });
  });

  describe('update', () => {
    it('should call service.update with id and dto', () => {
      const task = createTask({ title: 'Updated' });
      service.update.mockReturnValue(task);

      const result = controller.update(task.id, { title: 'Updated' } as any);
      expect(service.update).toHaveBeenCalledWith(task.id, { title: 'Updated' });
      expect(result).toEqual(task);
    });
  });

  describe('remove', () => {
    it('should call service.remove with id', () => {
      service.remove.mockReturnValue(undefined as any);
      controller.remove('some-id');
      expect(service.remove).toHaveBeenCalledWith('some-id');
    });
  });
});
```

```typescript
import { INestApplication, ValidationPipe } from '@nestjs/common';
import { Test, TestingModule } from '@nestjs/testing';
import { TypeOrmModule } from '@nestjs/typeorm';
import { AppModule } from '../../src/app.module';

export async function createTestApp(): Promise<INestApplication> {
  const moduleFixture: TestingModule = await Test.createTestingModule({
    imports: [AppModule],
  })
    .overrideModule(TypeOrmModule)
    .useModule(
      TypeOrmModule.forRoot({
        type: 'postgres',
        url: process.env.TEST_DATABASE_URL ?? 'postgresql://postgres:postgres@localhost:5432/tasks_test',
        entities: [__dirname + '/../../src/**/*.entity{.ts,.js}'],
        synchronize: true,
        dropSchema: false,
      }),
    )
    .compile();

  const app = moduleFixture.createNestApplication();
  app.useGlobalPipes(new ValidationPipe({ whitelist: true, transform: true }));
  await app.init();
  return app;
}
```

### `test/tasks.e2e-spec.ts`
```typescript
import * as request from 'supertest';
import { INestApplication } from '@nestjs/common';
import { createTestApp } from './helpers/app.helper';
import { TaskStatus } from '../src/tasks/enums/task-status.enum';

describe('Tasks (e2e)', () => {
  let app: INestApplication;

  beforeAll(async () => {
    app = await createTestApp();
  });

  afterAll(async () => {
    await app.close();
  });

  describe('POST /api/tasks', () => {
    it('should create a task and return 201', () => {
      return request(app.getHttpServer())
        .post('/api/tasks')
        .send({ title: 'E2E Task', priority: 'HIGH' })
        .expect(201)
        .expect(({ body }) => {
          expect(body.id).toBeDefined();
          expect(body.title).toBe('E2E Task');
          expect(body.status).toBe(TaskStatus.TODO);
        });
    });

    it('should return 400 when title is missing', () => {
      return request(app.getHttpServer())
        .post('/api/tasks')
        .send({ priority: 'HIGH' })
        .expect(400);
    });

    it('should return 400 for extra unrecognised fields', () => {
      return request(app.getHttpServer())
        .post('/api/tasks')
        .send({ title: 'Task', unknownField: 'foo' })
        .expect(400);
    });
  });

  describe('GET /api/tasks', () => {
    it('should return an array of tasks', () => {
      return request(app.getHttpServer())
        .get('/api/tasks')
        .expect(200)
        .expect(({ body }) => {
          expect(Array.isArray(body)).toBe(true);
        });
    });

    it('should filter by status', async () => {
      const created = await request(app.getHttpServer())
        .post('/api/tasks')
        .send({ title: 'Filter Task', status: 'DONE' })
        .then((r) => r.body);

      return request(app.getHttpServer())
        .get('/api/tasks?status=DONE')
        .expect(200)
        .expect(({ body }) => {
          expect(body.some((t: any) => t.id === created.id)).toBe(true);
        });
    });
  });

  describe('GET /api/tasks/:id', () => {
    it('should return 404 for unknown id', () => {
      return request(app.getHttpServer())
        .get('/api/tasks/non-existent-id')
        .expect(404);
    });
  });

  describe('PATCH /api/tasks/:id', () => {
    it('should update a task', async () => {
      const task = await request(app.getHttpServer())
        .post('/api/tasks')
        .send({ title: 'Patchable Task' })
        .then((r) => r.body);

      return request(app.getHttpServer())
        .patch(`/api/tasks/${task.id}`)
        .send({ status: 'IN_PROGRESS' })
        .expect(200)
        .expect(({ body }) => {
          expect(body.status).toBe('IN_PROGRESS');
          expect(body.title).toBe('Patchable Task'); // unchanged
        });
    });
  });

  describe('DELETE /api/tasks/:id', () => {
    it('should delete a task and return 204', async () => {
      const task = await request(app.getHttpServer())
        .post('/api/tasks')
        .send({ title: 'Deletable Task' })
        .then((r) => r.body);

      await request(app.getHttpServer()).delete(`/api/tasks/${task.id}`).expect(204);
      await request(app.getHttpServer()).get(`/api/tasks/${task.id}`).expect(404);
    });
  });
});
```
