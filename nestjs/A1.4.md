# A1.4 — Modules

**Phase:** 1 — NestJS Fundamentals  
**Topic:** Feature modules, shared modules, global modules, dynamic modules, forRoot/forRootAsync pattern  
**Type:** Assignment

## Goal
Understand every module variant in NestJS and implement the `forRoot`/`forRootAsync` pattern used by all major NestJS packages.

## Background

Modules are the unit of organisation in NestJS. A well-structured NestJS application has a module per domain slice (users, orders, products), with cross-cutting concerns (logging, config, database) in shared or global modules.

**Feature module** — owns a specific domain. Imports what it needs, exports only what other modules need. This is the default type you create with `nest g module`.

**Shared module** — exports its providers so other modules can use them simply by importing the shared module. Every NestJS module is a singleton: importing the same shared module in two places does not create two instances of its providers. Example: a `DatabaseModule` that exports a TypeORM `DataSource` provider.

**Global module** — decorated with `@Global()`. Its exports are available application-wide without any module importing it. Use sparingly (config, logging) — overuse creates hidden, hard-to-trace dependencies.

**Dynamic module** — a module whose `providers`, `imports`, and `exports` are computed at runtime based on options passed in. The canonical pattern is `SomeModule.forRoot(options)` for synchronous configuration and `SomeModule.forRootAsync({ useFactory: ... })` for async configuration (e.g., when options depend on another provider like `ConfigService`). This is how `TypeOrmModule`, `JwtModule`, `BullModule`, etc., all work internally.

### 2. Shared module

**`src/shared/logger/logger.service.ts`**
```typescript
import { Injectable } from '@nestjs/common';

@Injectable()
export class LoggerService {
  log(context: string, message: string): void {
    const ts = new Date().toISOString();
    console.log(`[${ts}] [${context}] ${message}`);
  }

  error(context: string, message: string, trace?: string): void {
    const ts = new Date().toISOString();
    console.error(`[${ts}] [${context}] ERROR: ${message}`, trace ?? '');
  }
}
```

**`src/shared/logger/logger.module.ts`**
```typescript
import { Module } from '@nestjs/common';
import { LoggerService } from './logger.service';

@Module({
  providers: [LoggerService],
  exports: [LoggerService],   // ← exported so importing modules can inject it
})
export class LoggerModule {}
```

**`src/orders/orders.module.ts`** (imports shared module):
```typescript
import { Module } from '@nestjs/common';
import { LoggerModule } from '../shared/logger/logger.module';
import { OrdersController } from './orders.controller';
import { OrdersService } from './orders.service';

@Module({
  imports: [LoggerModule],     // ← imports the shared module
  controllers: [OrdersController],
  providers: [OrdersService],
})
export class OrdersModule {}
```

**`src/orders/orders.service.ts`**
```typescript
import { Injectable } from '@nestjs/common';
import { LoggerService } from '../shared/logger/logger.service';

@Injectable()
export class OrdersService {
  constructor(private readonly logger: LoggerService) {}

  createOrder(data: any) {
    this.logger.log('OrdersService', `Creating order: ${JSON.stringify(data)}`);
    return { id: 1, ...data };
  }
}
```

### 3. Global module

**`src/shared/app-config/app-config.module.ts`**
```typescript
import { Global, Module } from '@nestjs/common';

export const APP_VERSION = 'APP_VERSION';

@Global()
@Module({
  providers: [
    { provide: APP_VERSION, useValue: '1.0.0' },
  ],
  exports: [APP_VERSION],
})
export class AppConfigModule {}
```

Register in `AppModule.imports`. Now any module can inject `APP_VERSION` without importing `AppConfigModule`.

```typescript
// In any service, anywhere in the app:
import { Injectable, Inject } from '@nestjs/common';
import { APP_VERSION } from '../shared/app-config/app-config.module';

@Injectable()
export class AnyService {
  constructor(@Inject(APP_VERSION) private readonly version: string) {}

  getVersion() { return this.version; }
}
```

### 4. Dynamic module with forRoot

**`src/http-client/http-client.module.ts`**
```typescript
import { DynamicModule, Module } from '@nestjs/common';
import { HttpClientService } from './http-client.service';

export interface HttpClientOptions {
  baseUrl: string;
  timeout: number;
}

export const HTTP_CLIENT_OPTIONS = 'HTTP_CLIENT_OPTIONS';

@Module({})
export class HttpClientModule {
  static forRoot(options: HttpClientOptions): DynamicModule {
    return {
      module: HttpClientModule,
      providers: [
        { provide: HTTP_CLIENT_OPTIONS, useValue: options },
        HttpClientService,
      ],
      exports: [HttpClientService],
    };
  }
}
```

**`src/http-client/http-client.service.ts`**
```typescript
import { Injectable, Inject } from '@nestjs/common';
import { HTTP_CLIENT_OPTIONS, HttpClientOptions } from './http-client.module';

@Injectable()
export class HttpClientService {
  constructor(
    @Inject(HTTP_CLIENT_OPTIONS) private readonly options: HttpClientOptions,
  ) {}

  async get(path: string): Promise<string> {
    const url = `${this.options.baseUrl}${path}`;
    console.log(`[HttpClientService] GET ${url} (timeout: ${this.options.timeout}ms)`);
    return url; // placeholder — replace with actual fetch call
  }
}
```

Register in `AppModule`:
```typescript
HttpClientModule.forRoot({ baseUrl: 'https://api.example.com', timeout: 5000 })
```

### 5. Dynamic module with forRootAsync

This is the production pattern — options come from `ConfigService`:

```typescript
import { DynamicModule, Module, ModuleMetadata } from '@nestjs/common';
import { HttpClientService } from './http-client.service';
import { HTTP_CLIENT_OPTIONS, HttpClientOptions } from './http-client.module';

export interface HttpClientAsyncOptions extends Pick<ModuleMetadata, 'imports'> {
  useFactory: (...args: any[]) => HttpClientOptions | Promise<HttpClientOptions>;
  inject?: any[];
}

@Module({})
export class HttpClientModule {
  static forRoot(options: HttpClientOptions): DynamicModule {
    return {
      module: HttpClientModule,
      providers: [
        { provide: HTTP_CLIENT_OPTIONS, useValue: options },
        HttpClientService,
      ],
      exports: [HttpClientService],
    };
  }

  static forRootAsync(asyncOptions: HttpClientAsyncOptions): DynamicModule {
    return {
      module: HttpClientModule,
      imports: asyncOptions.imports ?? [],
      providers: [
        {
          provide: HTTP_CLIENT_OPTIONS,
          useFactory: asyncOptions.useFactory,
          inject: asyncOptions.inject ?? [],
        },
        HttpClientService,
      ],
      exports: [HttpClientService],
    };
  }
}
```

Usage (options from ConfigService):
```typescript
import { ConfigModule, ConfigService } from '@nestjs/config';

HttpClientModule.forRootAsync({
  imports: [ConfigModule],
  inject: [ConfigService],
  useFactory: (config: ConfigService): HttpClientOptions => ({
    baseUrl: config.get<string>('API_BASE_URL')!,
    timeout: config.get<number>('API_TIMEOUT', 5000),
  }),
})
```

### 6. Re-export pattern

A module can re-export modules it imports, so consumers of that module get all the exports:

```typescript
@Module({
  imports: [LoggerModule, AppConfigModule],
  exports: [LoggerModule, AppConfigModule],  // consumers of SharedModule get both
})
export class SharedModule {}
```
