# P9.1 — Real-time Chat App

**Phase:** 9 — Advanced Patterns  
**Topic:** Socket.io rooms, MongoDB message history, typing indicators  
**Type:** Project

---

## Goal

Build a real-time chat application with Socket.io: multiple rooms, persistent message history in MongoDB, online presence tracking, and typing indicators.

## What to Build

- REST API: `GET /api/rooms`, `POST /api/rooms`, `GET /api/rooms/:id/messages`
- Socket.io events: `room:join`, `room:leave`, `message:send`, `message:new`, `typing:start`, `typing:stop`, `presence:update`
- Online users tracked in Redis (sorted set per room)
- Message history loaded via REST; new messages delivered via Socket.io
- Typing indicator debounced and auto-cleared after 3 s

// src/middleware/socketAuth.js
import jwt from 'jsonwebtoken';
import User from '../models/User.js';

export const socketAuth = async (socket, next) => {
  const token = socket.handshake.auth.token;
  if (!token) return next(new Error('Authentication required'));

  try {
    const payload = jwt.verify(token, process.env.JWT_ACCESS_SECRET);
    const user = await User.findById(payload.sub).select('name avatar').lean();
    if (!user) return next(new Error('User not found'));
    socket.data.user = { id: user._id.toString(), name: user.name, avatar: user.avatar };
    next();
  } catch {
    next(new Error('Invalid token'));
  }
};

```js
// src/socket/presenceHandlers.js
import { redis } from '../cache.js';

const PRESENCE_KEY = (roomId) => `presence:${roomId}`;
const SCORE = () => Date.now();

export const trackJoin = async (io, socket, roomId) => {
  const { id, name, avatar } = socket.data.user;

  // Redis sorted set: score = last seen timestamp
  await redis.zadd(PRESENCE_KEY(roomId), SCORE(), JSON.stringify({ id, name, avatar }));

  // Notify others in the room
  socket.to(roomId).emit('presence:update', { userId: id, status: 'online' });

  // Return current online users to the joining socket
  const members = await getOnlineUsers(roomId);
  socket.emit('presence:init', members);
};

export const trackLeave = async (io, socket, roomId) => {
  const { id } = socket.data.user;
  await redis.zrem(PRESENCE_KEY(roomId), JSON.stringify({ id, name: socket.data.user.name, avatar: socket.data.user.avatar }));
  io.to(roomId).emit('presence:update', { userId: id, status: 'offline' });
};

export const getOnlineUsers = async (roomId) => {
  const members = await redis.zrange(PRESENCE_KEY(roomId), 0, -1);
  return members.map((m) => JSON.parse(m));
};
```

```js
// src/socket/messageHandlers.js
import Message from '../models/Message.js';

const typingTimers = new Map();

export const registerMessageHandlers = (io, socket) => {
  const { user } = socket.data;

  socket.on('message:send', async ({ roomId, body }, ack) => {
    if (!socket.rooms.has(roomId)) {
      return ack?.({ error: 'Not in room' });
    }

    try {
      const message = await Message.create({ room: roomId, author: user.id, body });
      await message.populate('author', 'name avatar');

      // Broadcast to all clients in the room (including sender)
      io.to(roomId).emit('message:new', {
        id:        message._id,
        body:      message.body,
        author:    { id: user.id, name: user.name, avatar: user.avatar },
        roomId,
        createdAt: message.createdAt,
      });

      ack?.({ ok: true, messageId: message._id });
    } catch (err) {
      ack?.({ error: 'Failed to send message' });
    }
  });

  socket.on('typing:start', ({ roomId }) => {
    socket.to(roomId).emit('typing:update', { userId: user.id, name: user.name, isTyping: true });

    // Auto-clear typing indicator after 3 s (in case client doesn't send typing:stop)
    const key = `${socket.id}:${roomId}`;
    clearTimeout(typingTimers.get(key));
    typingTimers.set(key, setTimeout(() => {
      socket.to(roomId).emit('typing:update', { userId: user.id, name: user.name, isTyping: false });
      typingTimers.delete(key);
    }, 3000));
  });

  socket.on('typing:stop', ({ roomId }) => {
    const key = `${socket.id}:${roomId}`;
    clearTimeout(typingTimers.get(key));
    typingTimers.delete(key);
    socket.to(roomId).emit('typing:update', { userId: user.id, name: user.name, isTyping: false });
  });
};
```

```js
// src/socket/roomHandlers.js
import { trackJoin, trackLeave } from './presenceHandlers.js';

export const registerRoomHandlers = (io, socket) => {
  socket.on('room:join', async (roomId, ack) => {
    await socket.join(roomId);
    await trackJoin(io, socket, roomId);
    ack?.({ ok: true });
  });

  socket.on('room:leave', async (roomId) => {
    await socket.leave(roomId);
    await trackLeave(io, socket, roomId);
  });

  socket.on('disconnecting', async () => {
    for (const roomId of socket.rooms) {
      if (roomId !== socket.id) {
        await trackLeave(io, socket, roomId);
      }
    }
  });
};
```

```js
// src/socket/index.js
import { socketAuth } from '../middleware/socketAuth.js';
import { registerRoomHandlers } from './roomHandlers.js';
import { registerMessageHandlers } from './messageHandlers.js';

export const setupSocket = (io) => {
  io.use(socketAuth);

  io.on('connection', (socket) => {
    console.log(`[Socket] ${socket.data.user.name} connected (${socket.id})`);

    registerRoomHandlers(io, socket);
    registerMessageHandlers(io, socket);

    socket.on('error', (err) => console.error(`[Socket] Error for ${socket.id}:`, err.message));
  });
};
```

```js
// src/routes/messages.js — load history via REST
import { Router } from 'express';
import Message from '../models/Message.js';
import { authenticate } from '../middleware/authenticate.js';

const router = Router({ mergeParams: true });

router.get('/', authenticate, async (req, res) => {
  const { before, limit = 50 } = req.query;
  const filter = { room: req.params.roomId, deletedAt: null };
  if (before) filter.createdAt = { $lt: new Date(before) };

  const messages = await Message.find(filter)
    .populate('author', 'name avatar')
    .sort({ createdAt: -1 })
    .limit(Number(limit));

  res.json({ data: messages.reverse(), hasMore: messages.length === Number(limit) });
});

export default router;
```

## Deliverables

- [ ] `socketAuth` middleware verifies JWT and attaches `socket.data.user`
- [ ] `room:join` joins the Socket.io room and tracks presence in Redis
- [ ] `message:send` saves to MongoDB and broadcasts to all room members
- [ ] `message:send` accepts an `ack` callback (client confirms delivery)
- [ ] `typing:start` broadcasts to others; auto-clears after 3 s if `typing:stop` not received
- [ ] `disconnecting` removes presence for all rooms the socket was in
- [ ] `GET /api/rooms/:roomId/messages` supports cursor pagination via `?before=` timestamp
- [ ] Online users list served to joining client via `presence:init` event
