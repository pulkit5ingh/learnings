## Goal
Build a centralized error handling system that distinguishes operational errors from programming bugs and returns consistent error responses.

## Background

### Operational vs Programmer Errors
This distinction, made famous by Node.js contributor Joyent, is crucial for building reliable systems.

**Operational errors** are expected failures in a correctly written program: a user provides a wrong password, a database is temporarily unavailable, a network request times out, a file doesn't exist. These are part of normal operation and should be handled gracefully — return a useful error response to the client, log at `warn` level.

**Programmer errors** are bugs: calling a function with the wrong argument type, reading a property of `undefined`, off-by-one errors in loops. These should crash the process (or at least be logged prominently) because the application is in an unknown state.

In Express, this distinction guides the error handler: operational errors get client-friendly messages; programmer errors get generic "Internal server error" to avoid leaking implementation details.

### async/await Error Handling in Express 4
Express 4 does not catch rejected promises automatically. Every async route handler needs a try/catch:

```js
app.get('/route', async (req, res, next) => {
  try {
    const data = await fetchData();
    res.json(data);
  } catch (err) {
    next(err);
  }
});
```

This boilerplate is noisy. An `asyncWrapper` higher-order function eliminates it:

```js
const asyncWrapper = (fn) => (req, res, next) => {
  Promise.resolve(fn(req, res, next)).catch(next);
};
```

Express 5 (in beta) handles this automatically.

### RFC 7807 — Problem Details
RFC 7807 defines a standard JSON error response format used by modern APIs:
```json
{
  "type": "https://api.example.com/errors/not-found",
  "title": "Resource Not Found",
  "status": 404,
  "detail": "User with ID 42 was not found",
  "instance": "/users/42"
}
```

## Tasks

### 1. AppError Class

```js
// src/utils/AppError.js
class AppError extends Error {
  constructor(message, statusCode, options = {}) {
    super(message);

    this.name = 'AppError';
    this.statusCode = statusCode;
    this.status = `${statusCode}`.startsWith('4') ? 'fail' : 'error';
    this.isOperational = true; // mark as expected error

    // Optional extras
    this.code = options.code;     // machine-readable error code
    this.field = options.field;   // field that caused the error
    this.details = options.details; // additional context

    Error.captureStackTrace(this, this.constructor);
  }
}

// Predefined operational error factories
AppError.notFound = (resource, id) =>
  new AppError(`${resource} with ID ${id} not found`, 404, { code: 'NOT_FOUND' });

AppError.badRequest = (message, details) =>
  new AppError(message, 400, { code: 'BAD_REQUEST', details });

AppError.unauthorized = (message = 'Authentication required') =>
  new AppError(message, 401, { code: 'UNAUTHORIZED' });

AppError.forbidden = (message = 'Access denied') =>
  new AppError(message, 403, { code: 'FORBIDDEN' });

AppError.conflict = (message, field) =>
  new AppError(message, 409, { code: 'CONFLICT', field });

AppError.unprocessable = (message, details) =>
  new AppError(message, 422, { code: 'UNPROCESSABLE', details });

AppError.tooManyRequests = (message = 'Rate limit exceeded') =>
  new AppError(message, 429, { code: 'RATE_LIMITED' });

module.exports = AppError;
```

### 3. Error Handler Middleware

```js
// src/middleware/errorHandler.js
const AppError = require('../utils/AppError');

// Handle specific error types from libraries
const handleMongooseCastError = (err) =>
  AppError.badRequest(`Invalid value for field: ${err.path}`);

const handleMongooseDuplicateKey = (err) => {
  const field = Object.keys(err.keyValue)[0];
  return AppError.conflict(`${field} already exists`, field);
};

const handleMongooseValidation = (err) => {
  const messages = Object.values(err.errors).map(e => e.message);
  return AppError.unprocessable('Validation failed', messages);
};

const handleJWTError = () =>
  AppError.unauthorized('Invalid token');

const handleJWTExpiredError = () =>
  AppError.unauthorized('Token has expired');

const handlePgUniqueViolation = (err) => {
  const match = err.detail?.match(/Key \((.+?)\)=/);
  const field = match?.[1] ?? 'field';
  return AppError.conflict(`${field} already exists`, field);
};

// Transform known library errors into AppErrors
const transformError = (err) => {
  if (err.name === 'CastError') return handleMongooseCastError(err);
  if (err.code === 11000) return handleMongooseDuplicateKey(err);
  if (err.name === 'ValidationError') return handleMongooseValidation(err);
  if (err.name === 'JsonWebTokenError') return handleJWTError();
  if (err.name === 'TokenExpiredError') return handleJWTExpiredError();
  if (err.code === '23505') return handlePgUniqueViolation(err); // pg unique violation
  return err;
};

// RFC 7807 response format
const sendErrorResponse = (err, req, res) => {
  const isOperational = err instanceof AppError && err.isOperational;

  if (process.env.NODE_ENV === 'development') {
    return res.status(err.statusCode ?? 500).json({
      type: `https://api.example.com/errors/${err.code ?? 'internal'}`,
      title: err.message,
      status: err.statusCode ?? 500,
      detail: err.message,
      instance: req.path,
      code: err.code,
      details: err.details,
      stack: err.stack,
    });
  }

  // Production: don't leak programmer error details
  if (isOperational) {
    return res.status(err.statusCode).json({
      type: `https://api.example.com/errors/${err.code ?? 'error'}`,
      title: err.message,
      status: err.statusCode,
      detail: err.message,
      instance: req.path,
      ...(err.details && { details: err.details }),
    });
  }

  // Programmer error in production — generic message
  console.error('[PROGRAMMER ERROR]', err);
  return res.status(500).json({
    type: 'https://api.example.com/errors/internal',
    title: 'Internal Server Error',
    status: 500,
    detail: 'An unexpected error occurred',
    instance: req.path,
  });
};

const errorHandler = (err, req, res, next) => {
  const error = transformError(err);
  error.statusCode = error.statusCode ?? 500;

  console.error(`[${new Date().toISOString()}] ${req.method} ${req.path} - ${error.statusCode}: ${error.message}`);

  sendErrorResponse(error, req, res);
};

const notFound = (req, res, next) => {
  next(AppError.notFound('Route', `${req.method} ${req.path}`));
};

module.exports = { errorHandler, notFound };
```

### 4. Usage in Routes

```js
// src/routes/users.js
const router = require('express').Router();
const AppError = require('../utils/AppError');
const asyncWrapper = require('../utils/asyncWrapper');

const users = [{ id: 1, name: 'Alice', email: 'alice@test.com' }];

router.get('/:id', asyncWrapper(async (req, res) => {
  const id = parseInt(req.params.id);

  if (isNaN(id)) {
    throw AppError.badRequest('ID must be a number');
  }

  const user = users.find(u => u.id === id);

  if (!user) {
    throw AppError.notFound('User', id); // operational error
  }

  res.json(user);
}));

router.post('/', asyncWrapper(async (req, res) => {
  const { name, email } = req.body;

  if (!name || !email) {
    throw AppError.badRequest('name and email are required');
  }

  if (users.some(u => u.email === email)) {
    throw AppError.conflict('Email already exists', 'email');
  }

  const user = { id: users.length + 1, name, email };
  users.push(user);

  res.status(201).json(user);
}));

// Simulate programmer error
router.get('/crash', asyncWrapper(async (req, res) => {
  const obj = null;
  return obj.property; // TypeError — programmer error, not caught by us intentionally
}));

module.exports = router;
```

### 5. Uncaught Exception Handlers

```js
// src/server.js
require('dotenv').config();
const app = require('./app');

const server = app.listen(process.env.PORT ?? 3000, () => {
  console.log(`Server on http://localhost:${process.env.PORT ?? 3000}`);
});

// Unhandled promise rejections (async programmer errors)
process.on('unhandledRejection', (err) => {
  console.error('[UNHANDLED REJECTION]', err.message);
  console.error(err.stack);
  server.close(() => process.exit(1)); // graceful shutdown
});

// Synchronous uncaught errors
process.on('uncaughtException', (err) => {
  console.error('[UNCAUGHT EXCEPTION]', err.message);
  console.error(err.stack);
  process.exit(1); // must exit — state is corrupted
});
```
