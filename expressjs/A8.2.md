# A8.2 — Environment Configuration

**Phase:** 8 — Deployment & DevOps
**Topic:** dotenv, per-environment config, secrets, 12-factor app
**Type:** Assignment

---

## Goal
Build a robust configuration management system that loads environment-specific settings, validates required variables at startup, and never leaks secrets.

## Background

### The 12-Factor App: Config
The third factor of the Twelve-Factor App methodology states: "Store config in the environment." Config includes database URLs, API keys, port numbers, feature flags — anything that varies between deployments. By keeping config in environment variables (not in code or config files checked into source control), you can deploy the same codebase to development, staging, and production by changing only the environment.

The `.env` file is a convenience for local development only. Never commit `.env` to git. Always commit `.env.example` with placeholder values — this documents what variables are required without exposing real values.

### Validation at Startup
Instead of getting cryptic errors when a required variable is missing or malformed (e.g., MongoDB connection fails with a timeout), validate all required config at application startup. If any variable is missing or invalid, crash immediately with a clear error message. Fail fast and loudly.

### Secrets Management
In production, secrets come from secret management systems, not `.env` files:
- AWS Secrets Manager / Parameter Store
- HashiCorp Vault
- GitHub Actions secrets (injected as environment variables)
- Railway/Render/Heroku environment variable settings

The application code should not know or care where the secret comes from — only that it's available as an environment variable.

### 2. Config Module with Validation

```js
// src/config/index.js
require('dotenv').config();

// Validators
const required = (name) => {
  const value = process.env[name];
  if (!value) throw new Error(`Missing required environment variable: ${name}`);
  return value;
};

const optional = (name, defaultValue) => process.env[name] ?? defaultValue;

const requiredInt = (name, min, max) => {
  const raw = required(name);
  const value = parseInt(raw, 10);
  if (isNaN(value)) throw new Error(`${name} must be an integer, got: "${raw}"`);
  if (min !== undefined && value < min) throw new Error(`${name} must be >= ${min}, got: ${value}`);
  if (max !== undefined && value > max) throw new Error(`${name} must be <= ${max}, got: ${value}`);
  return value;
};

const optionalInt = (name, defaultValue) => {
  const raw = process.env[name];
  if (!raw) return defaultValue;
  const value = parseInt(raw, 10);
  if (isNaN(value)) throw new Error(`${name} must be an integer, got: "${raw}"`);
  return value;
};

const optionalBool = (name, defaultValue = false) => {
  const raw = process.env[name];
  if (!raw) return defaultValue;
  if (!['true', 'false', '1', '0'].includes(raw.toLowerCase())) {
    throw new Error(`${name} must be true/false/1/0, got: "${raw}"`);
  }
  return ['true', '1'].includes(raw.toLowerCase());
};

const oneOf = (name, allowed, defaultValue) => {
  const value = process.env[name] ?? defaultValue;
  if (!allowed.includes(value)) {
    throw new Error(`${name} must be one of [${allowed.join(', ')}], got: "${value}"`);
  }
  return value;
};

// --- Config object ---
let config;

try {
  config = {
    env: oneOf('NODE_ENV', ['development', 'test', 'production'], 'development'),
    port: optionalInt('PORT', 3000),
    isDev: (process.env.NODE_ENV ?? 'development') === 'development',
    isProd: process.env.NODE_ENV === 'production',
    isTest: process.env.NODE_ENV === 'test',

    db: {
      uri: required('MONGODB_URI'),
      poolSize: optionalInt('DB_POOL_SIZE', 10),
    },

    redis: {
      url: optional('REDIS_URL', 'redis://localhost:6379'),
    },

    jwt: {
      accessSecret: required('ACCESS_TOKEN_SECRET'),
      refreshSecret: required('REFRESH_TOKEN_SECRET'),
      accessExpiry: optional('ACCESS_TOKEN_EXPIRY', '15m'),
      refreshExpiry: optional('REFRESH_TOKEN_EXPIRY', '7d'),
    },

    cors: {
      allowedOrigins: (process.env.ALLOWED_ORIGINS ?? 'http://localhost:3000')
        .split(',')
        .map(o => o.trim()),
    },

    email: {
      from: optional('EMAIL_FROM', 'noreply@example.com'),
      fromName: optional('EMAIL_FROM_NAME', 'App'),
      smtpHost: optional('SMTP_HOST', 'localhost'),
      smtpPort: optionalInt('SMTP_PORT', 587),
      smtpUser: optional('SMTP_USER', ''),
      smtpPass: optional('SMTP_PASS', ''),
    },

    aws: {
      region: optional('AWS_REGION', 'us-east-1'),
      bucket: optional('S3_BUCKET', ''),
    },

    stripe: {
      secretKey: optional('STRIPE_SECRET_KEY', ''),
      webhookSecret: optional('STRIPE_WEBHOOK_SECRET', ''),
    },

    logging: {
      level: optional('LOG_LEVEL', process.env.NODE_ENV === 'production' ? 'info' : 'debug'),
    },

    rateLimit: {
      windowMs: optionalInt('RATE_LIMIT_WINDOW_MS', 15 * 60 * 1000),
      max: optionalInt('RATE_LIMIT_MAX', 100),
    },
  };

  // Validate secrets are long enough
  if (config.jwt.accessSecret.length < 32) {
    throw new Error('ACCESS_TOKEN_SECRET must be at least 32 characters');
  }
  if (config.jwt.refreshSecret.length < 32) {
    throw new Error('REFRESH_TOKEN_SECRET must be at least 32 characters');
  }

  console.log(`Config loaded: env=${config.env}, port=${config.port}`);
} catch (err) {
  console.error('[CONFIG ERROR]', err.message);
  process.exit(1); // Fail fast
}

module.exports = config;
```

### 5. Config Usage Across App

```js
// src/app.js — import config instead of process.env directly
const config = require('./config');

const app = express();

app.use(cors({ origin: config.cors.allowedOrigins }));

// src/config/database.js
const config = require('./index');
mongoose.connect(config.db.uri, { maxPoolSize: config.db.poolSize });

// src/middleware/rateLimiter.js
const config = require('../config');
const limiter = rateLimit({
  windowMs: config.rateLimit.windowMs,
  max: config.rateLimit.max,
});
```

### 7. Docker Environment Variables

```dockerfile

# Pass secrets at runtime:
# docker run -e ACCESS_TOKEN_SECRET=... my-api

```

```yaml
# In docker-compose.yml — reference from .env file

  api:
    environment:
      ACCESS_TOKEN_SECRET: ${ACCESS_TOKEN_SECRET}
      REFRESH_TOKEN_SECRET: ${REFRESH_TOKEN_SECRET}
    env_file:
      - .env  # Load non-secret defaults from .env file
```
