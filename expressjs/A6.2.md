## Goal
Send transactional emails using Nodemailer and SendGrid, render HTML templates with Handlebars, and queue emails with Bull to decouple delivery from the request/response cycle.

## Background

### Transactional Email
Transactional emails are triggered by user actions: welcome email, password reset, order confirmation, invoice. They differ from bulk/marketing email. Use a dedicated email service (SendGrid, AWS SES, Mailgun, Postmark) rather than your own SMTP server — they handle deliverability, bounces, unsubscribes, and provide delivery analytics.

Nodemailer is the Node.js SMTP client. It connects to any SMTP server (Gmail, SendGrid, AWS SES) and sends messages. For SendGrid's HTTP API (faster, better deliverability metrics), use `@sendgrid/mail` directly.

### HTML Email Templates
HTML emails are different from web HTML — many clients (Outlook) don't support modern CSS. Use inline styles, simple table layouts, and test with Litmus or Email on Acid. Handlebars is a lightweight templating engine that compiles templates to HTML functions.

### Email Queue with Bull
Sending email synchronously in a request handler causes problems: if the email service is slow or down, your API response is delayed. If the request fails after the email sends, you might send duplicate emails. A queue (Bull/BullMQ over Redis) decouples email sending: your API adds a job to the queue and responds immediately. A separate worker process processes jobs, retrying on failure.

// src/templates/email.templates.js
const Handlebars = require('handlebars');
const path = require('node:path');
const fs = require('node:fs/promises');

// Template cache
const templateCache = new Map();

const loadTemplate = async (name) => {
  if (templateCache.has(name)) return templateCache.get(name);

  const templatePath = path.join(__dirname, `${name}.hbs`);

  try {
    const source = await fs.readFile(templatePath, 'utf8');
    const compiled = Handlebars.compile(source);
    templateCache.set(name, compiled);
    return compiled;
  } catch {
    // Inline template if file not found
    return null;
  }
};

  const template = TEMPLATES[name];

### 4. Email Service

```js
// src/services/email.service.js
const { transporter } = require('../config/mailer');
const { renderTemplate } = require('../templates/email.templates');

const FROM = `${process.env.EMAIL_FROM_NAME ?? 'App'} <${process.env.EMAIL_FROM}>`;

const sendEmail = async ({ to, subject, template, data, attachments }) => {
  const html = renderTemplate(template, data);

  const info = await transporter.sendMail({
    from: FROM,
    to,
    subject,
    html,
    // Plain text fallback
    text: html.replace(/<[^>]*>/g, '').replace(/\s+/g, ' ').trim(),
    attachments,
  });

  console.log(`Email sent: ${info.messageId} → ${to}`);
  return info;
};

const sendWelcome = (user, verificationToken) =>
  sendEmail({
    to: user.email,
    subject: `Welcome to ${process.env.APP_NAME ?? 'App'}!`,
    template: 'welcome',
    data: {
      name: user.name,
      email: user.email,
      appName: process.env.APP_NAME ?? 'App',
      verificationUrl: verificationToken
        ? `${process.env.APP_URL}/verify-email?token=${verificationToken}`
        : null,
    },
  });

const sendPasswordReset = (user, resetToken) =>
  sendEmail({
    to: user.email,
    subject: 'Password Reset Request',
    template: 'passwordReset',
    data: {
      name: user.name,
      resetUrl: `${process.env.APP_URL}/reset-password?token=${resetToken}`,
      expiresIn: '1 hour',
    },
  });

const sendOrderConfirmation = (user, order) =>
  sendEmail({
    to: user.email,
    subject: `Order #${order.id} Confirmed`,
    template: 'orderConfirmation',
    data: {
      customerName: user.name,
      orderId: order.id,
      items: order.items,
      total: order.total.toFixed(2),
    },
  });

module.exports = { sendEmail, sendWelcome, sendPasswordReset, sendOrderConfirmation };
```

### 5. Bull Email Queue

```js
// src/queues/emailQueue.js
const Bull = require('bull');
const emailService = require('../services/email.service');

const emailQueue = new Bull('email', {
  redis: process.env.REDIS_URL ?? 'redis://localhost:6379',
  defaultJobOptions: {
    attempts: 3,           // retry 3 times on failure
    backoff: {
      type: 'exponential',
      delay: 2000,         // 2s, 4s, 8s
    },
    removeOnComplete: 100, // keep last 100 completed jobs
    removeOnFail: 50,      // keep last 50 failed jobs
  },
});

// Queue a welcome email
const queueWelcomeEmail = (user, verificationToken) =>
  emailQueue.add('welcome', { user, verificationToken }, { priority: 1 });

// Queue a password reset email (high priority)
const queuePasswordReset = (user, resetToken) =>
  emailQueue.add('passwordReset', { user, resetToken }, { priority: 10 });

const queueOrderConfirmation = (user, order) =>
  emailQueue.add('orderConfirmation', { user, order });

// Worker: process jobs
emailQueue.process('welcome', async (job) => {
  const { user, verificationToken } = job.data;
  await emailService.sendWelcome(user, verificationToken);
});

emailQueue.process('passwordReset', async (job) => {
  const { user, resetToken } = job.data;
  await emailService.sendPasswordReset(user, resetToken);
});

emailQueue.process('orderConfirmation', async (job) => {
  const { user, order } = job.data;
  await emailService.sendOrderConfirmation(user, order);
});

// Event listeners for monitoring
emailQueue.on('completed', (job) => {
  console.log(`Email job ${job.id} (${job.name}) completed`);
});

emailQueue.on('failed', (job, err) => {
  console.error(`Email job ${job.id} (${job.name}) failed: ${err.message}`);
});

emailQueue.on('stalled', (job) => {
  console.warn(`Email job ${job.id} stalled`);
});

module.exports = { emailQueue, queueWelcomeEmail, queuePasswordReset, queueOrderConfirmation };
```

router.post('/register', async (req, res, next) => {
