# P1.2 — Notes API

**Phase:** 1 — Node.js & Express Fundamentals  
**Topic:** CRUD with tags, search, full middleware stack  
**Type:** Project

---

## Goal

Build a Notes REST API with tagging, full-text search, and a complete middleware stack (logging, rate limiting, CORS, request IDs).

## What to Build

- Full CRUD for notes (`title`, `body`, `tags[]`, `pinned`, `color`)
- `GET /api/notes?tag=work&search=meeting&pinned=true` — combined filters
- `GET /api/notes/tags` — list all unique tags in use
- `PATCH /api/notes/:id/pin` — toggle pinned status
- Request logging middleware (method, URL, status, duration)
- Request ID middleware (attach `x-request-id` to every response)
- Rate limiting (100 req/15 min per IP)

## Implementation

```js
// src/store/notes.js
import { randomUUID } from 'node:crypto';

const notes = [];

export const findAll = ({ tag, search, pinned } = {}) => {
  let result = [...notes];
  if (tag) result = result.filter((n) => n.tags.includes(tag));
  if (search) {
    const q = search.toLowerCase();
    result = result.filter(
      (n) => n.title.toLowerCase().includes(q) || n.body.toLowerCase().includes(q),
    );
  }
  if (pinned !== undefined) result = result.filter((n) => n.pinned === (pinned === 'true'));
  return result.sort((a, b) => (b.pinned ? 1 : 0) - (a.pinned ? 1 : 0));
};

export const getAllTags = () => [...new Set(notes.flatMap((n) => n.tags))].sort();

export const findById = (id) => notes.find((n) => n.id === id);

export const create = ({ title, body = '', tags = [], color = 'white' }) => {
  const note = { id: randomUUID(), title, body, tags, color, pinned: false, createdAt: new Date().toISOString() };
  notes.push(note);
  return note;
};

export const update = (id, data) => {
  const idx = notes.findIndex((n) => n.id === id);
  if (idx === -1) return null;
  notes[idx] = { ...notes[idx], ...data, updatedAt: new Date().toISOString() };
  return notes[idx];
};

export const togglePin = (id) => {
  const note = notes.find((n) => n.id === id);
  if (!note) return null;
  note.pinned = !note.pinned;
  return note;
};

export const remove = (id) => {
  const idx = notes.findIndex((n) => n.id === id);
  if (idx === -1) return false;
  notes.splice(idx, 1);
  return true;
};
```

```js
// src/middleware/requestId.js
import { randomUUID } from 'node:crypto';
export const requestId = (req, res, next) => {
  req.id = req.headers['x-request-id'] ?? randomUUID();
  res.set('x-request-id', req.id);
  next();
};
```

```js
// src/middleware/requestLogger.js
export const requestLogger = (req, res, next) => {
  const start = Date.now();
  res.on('finish', () => {
    const ms = Date.now() - start;
    console.log(`[${req.id}] ${req.method} ${req.originalUrl} ${res.statusCode} ${ms}ms`);
  });
  next();
};
```

```js
// src/middleware/rateLimiter.js
// Simple in-memory rate limiter without external deps
const store = new Map();

export const rateLimiter = ({ windowMs = 15 * 60 * 1000, max = 100 } = {}) =>
  (req, res, next) => {
    const key = req.ip;
    const now = Date.now();
    const record = store.get(key) ?? { count: 0, resetAt: now + windowMs };

    if (now > record.resetAt) {
      record.count = 0;
      record.resetAt = now + windowMs;
    }

    record.count++;
    store.set(key, record);

    res.set('X-RateLimit-Limit', String(max));
    res.set('X-RateLimit-Remaining', String(Math.max(0, max - record.count)));
    res.set('X-RateLimit-Reset', String(Math.ceil(record.resetAt / 1000)));

    if (record.count > max) {
      return res.status(429).json({ error: 'Too many requests, please slow down.' });
    }
    next();
  };
```

```js
// src/controllers/notesController.js
import * as store from '../store/notes.js';

export const listNotes = (req, res) => {
  const { tag, search, pinned } = req.query;
  res.json(store.findAll({ tag, search, pinned }));
};

export const listTags = (_req, res) => res.json(store.getAllTags());

export const getNote = (req, res) => {
  const note = store.findById(req.params.id);
  if (!note) return res.status(404).json({ error: 'Note not found' });
  res.json(note);
};

export const createNote = (req, res) => {
  const { title, body, tags, color } = req.body;
  if (!title?.trim()) return res.status(422).json({ error: 'title is required' });
  res.status(201).json(store.create({ title, body, tags, color }));
};

export const updateNote = (req, res) => {
  if (!Object.keys(req.body).length) return res.status(422).json({ error: 'No fields provided' });
  const note = store.update(req.params.id, req.body);
  if (!note) return res.status(404).json({ error: 'Note not found' });
  res.json(note);
};

export const togglePin = (req, res) => {
  const note = store.togglePin(req.params.id);
  if (!note) return res.status(404).json({ error: 'Note not found' });
  res.json(note);
};

export const deleteNote = (req, res) => {
  if (!store.remove(req.params.id)) return res.status(404).json({ error: 'Note not found' });
  res.status(204).send();
};
```

```js
import { Router } from 'express';
import * as ctrl from '../controllers/notesController.js';
const router = Router();

router.get('/', ctrl.listNotes);
router.get('/tags', ctrl.listTags);

router.post('/', ctrl.createNote);
router.patch('/:id', ctrl.updateNote);
router.patch('/:id/pin', ctrl.togglePin);
router.delete('/:id', ctrl.deleteNote);

export default router;
```

```js
// src/app.js
import express from 'express';
import { requestId } from './middleware/requestId.js';
import { requestLogger } from './middleware/requestLogger.js';
import { rateLimiter } from './middleware/rateLimiter.js';
import notesRouter from './routes/notes.js';
import { notFound } from './middleware/notFound.js';
import { errorHandler } from './middleware/errorHandler.js';

const app = express();

app.use(requestId);
app.use(requestLogger);
app.use(rateLimiter({ windowMs: 15 * 60 * 1000, max: 100 }));
app.use(express.json());

app.use('/api/notes', notesRouter);

app.use(notFound);
app.use(errorHandler);

export default app;
```
