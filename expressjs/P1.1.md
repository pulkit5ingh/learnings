# P1.1 — Recipe API

**Phase:** 1 — Node.js & Express Fundamentals  
**Topic:** CRUD, in-memory store, express.Router  
**Type:** Project

---

## Goal

Build a fully functional Recipe REST API using an in-memory array as the data store, organized with express.Router and a controller pattern.

## What to Build

- `GET /api/recipes` — list all recipes, supports `?category=` and `?search=` query filters
- `GET /api/recipes/:id` — get one recipe by ID
- `POST /api/recipes` — create a recipe
- `PUT /api/recipes/:id` — replace a recipe
- `PATCH /api/recipes/:id` — partial update
- `DELETE /api/recipes/:id` — delete a recipe
- Request body validation (manual, no library)
- Consistent JSON error responses

## Implementation

```js
// src/store/recipes.js
import { randomUUID } from 'node:crypto';

const recipes = [];

export const findAll = ({ category, search } = {}) => {
  let result = [...recipes];
  if (category) result = result.filter((r) => r.category === category);
  if (search) {
    const q = search.toLowerCase();
    result = result.filter(
      (r) => r.title.toLowerCase().includes(q) || r.description.toLowerCase().includes(q),
    );
  }
  return result;
};

export const findById = (id) => recipes.find((r) => r.id === id);

export const create = (data) => {
  const recipe = { id: randomUUID(), ...data, createdAt: new Date().toISOString() };
  recipes.push(recipe);
  return recipe;
};

export const update = (id, data) => {
  const idx = recipes.findIndex((r) => r.id === id);
  if (idx === -1) return null;
  recipes[idx] = { ...recipes[idx], ...data, updatedAt: new Date().toISOString() };
  return recipes[idx];
};

export const remove = (id) => {
  const idx = recipes.findIndex((r) => r.id === id);
  if (idx === -1) return false;
  recipes.splice(idx, 1);
  return true;
};
```

```js
// src/controllers/recipesController.js
import * as store from '../store/recipes.js';

const REQUIRED = ['title', 'ingredients', 'instructions', 'category'];

const validate = (body) => {
  const missing = REQUIRED.filter((k) => !body[k]);
  return missing.length ? `Missing fields: ${missing.join(', ')}` : null;
};

export const listRecipes = (req, res) => {
  const { category, search } = req.query;
  res.json(store.findAll({ category, search }));
};

export const getRecipe = (req, res) => {
  const recipe = store.findById(req.params.id);
  if (!recipe) return res.status(404).json({ error: 'Recipe not found' });
  res.json(recipe);
};

export const createRecipe = (req, res) => {
  const error = validate(req.body);
  if (error) return res.status(422).json({ error });
  const recipe = store.create(req.body);
  res.status(201).json(recipe);
};

export const replaceRecipe = (req, res) => {
  const error = validate(req.body);
  if (error) return res.status(422).json({ error });
  const recipe = store.update(req.params.id, req.body);
  if (!recipe) return res.status(404).json({ error: 'Recipe not found' });
  res.json(recipe);
};

export const patchRecipe = (req, res) => {
  if (!Object.keys(req.body).length) return res.status(422).json({ error: 'No fields to update' });
  const recipe = store.update(req.params.id, req.body);
  if (!recipe) return res.status(404).json({ error: 'Recipe not found' });
  res.json(recipe);
};

export const deleteRecipe = (req, res) => {
  const deleted = store.remove(req.params.id);
  if (!deleted) return res.status(404).json({ error: 'Recipe not found' });
  res.status(204).send();
};
```

```js
// src/routes/recipes.js
import { Router } from 'express';
import * as ctrl from '../controllers/recipesController.js';

const router = Router();

router.get('/', ctrl.listRecipes);
router.get('/:id', ctrl.getRecipe);
router.post('/', ctrl.createRecipe);
router.put('/:id', ctrl.replaceRecipe);
router.patch('/:id', ctrl.patchRecipe);
router.delete('/:id', ctrl.deleteRecipe);

export default router;
```

```js
// src/app.js
import express from 'express';
import recipesRouter from './routes/recipes.js';
import { notFound } from './middleware/notFound.js';
import { errorHandler } from './middleware/errorHandler.js';

const app = express();
app.use(express.json());

app.use('/api/recipes', recipesRouter);

app.use(notFound);
app.use(errorHandler);

export default app;
```

```js
// src/middleware/notFound.js
export const notFound = (_req, res) => res.status(404).json({ error: 'Route not found' });

// src/middleware/errorHandler.js
export const errorHandler = (err, _req, res, _next) => {
  console.error(err);
  res.status(err.status ?? 500).json({ error: err.message ?? 'Internal server error' });
};
```

```js
// src/server.js
import app from './app.js';
const PORT = process.env.PORT ?? 4000;
app.listen(PORT, () => console.log(`Recipe API running on :${PORT}`));
```
