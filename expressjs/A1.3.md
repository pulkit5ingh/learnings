## Goal
Understand how Express middleware works, write custom middleware, and use essential third-party middleware correctly.

## Background

### The Middleware Pipeline
Middleware in Express is a function with the signature `(req, res, next)`. When Express receives a request, it runs middleware functions in the order they were registered with `app.use()`. Each middleware can:
1. Execute any code
2. Modify `req` and `res`
3. End the request-response cycle (call `res.json()`, `res.send()`, etc.)
4. Call `next()` to pass control to the next middleware

If a middleware doesn't call `next()` and doesn't send a response, the request hangs. This is a very common bug — always ensure every code path either calls `next()` or sends a response.

Middleware can be scoped: `app.use('/api', middleware)` only runs that middleware for routes starting with `/api`. Route-specific middleware is passed as extra arguments: `app.get('/route', middleware1, middleware2, handler)`.

### Error Middleware
Express has a special error-handling middleware signature: `(err, req, res, next)` — four arguments. Express detects the 4-argument signature and only calls this middleware when an error has been passed via `next(err)`. Error middleware must be registered last, after all other `app.use()` calls and route definitions.

To trigger error middleware: call `next(new Error('something went wrong'))` or `next(err)` from any regular middleware. In async handlers you must catch errors and pass them to `next` — Express does not automatically catch promise rejections in Express 4. (Express 5, currently in beta, handles this automatically.)

### Middleware Order Matters
A common mistake: registering `express.json()` after route definitions. Because middleware runs in registration order, any route registered before `express.json()` won't have `req.body` populated. Always register body-parsing middleware before routes.

```js
// middleware/logger.js
const logger = (req, res, next) => {
  const start = Date.now();

  // Wait for response to finish, then log
  res.on('finish', () => {
    const duration = Date.now() - start;
    const log = [
      new Date().toISOString(),
      req.method,
      req.originalUrl,
      res.statusCode,
      `${duration}ms`,
    ].join(' | ');

    console.log(log);
  });

  next();
};

module.exports = logger;
```

### 2. Request ID Middleware

```js
// middleware/requestId.js
const { randomUUID } = require('node:crypto');

const requestId = (req, res, next) => {
  const id = req.headers['x-request-id'] ?? randomUUID();
  req.id = id;
  res.set('X-Request-Id', id);
  next();
};

module.exports = requestId;
```

### 3. Custom Authentication Guard

```js
// middleware/authGuard.js
const authGuard = (req, res, next) => {
  const authHeader = req.headers['authorization'];

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({
      error: 'Missing or invalid Authorization header',
      hint: 'Use: Authorization: Bearer <token>',
    });
  }

  const token = authHeader.slice(7);

  // Fake token check — we'll use real JWTs in Phase 3
  if (token !== 'secret-token') {
    return res.status(403).json({ error: 'Invalid token' });
  }

  req.user = { id: 1, role: 'admin' }; // attach user to request
  next();
};

module.exports = authGuard;
```

### 4. Error Middleware

```js
// middleware/errorHandler.js
const errorHandler = (err, req, res, next) => {
  console.error(`[ERROR] ${req.id} ${err.stack}`);

  // Handle specific error types
  if (err.name === 'SyntaxError' && err.type === 'entity.parse.failed') {
    return res.status(400).json({ error: 'Invalid JSON in request body' });
  }

  const statusCode = err.statusCode ?? err.status ?? 500;
  const message = err.isOperational ? err.message : 'Internal server error';

  res.status(statusCode).json({
    error: message,
    requestId: req.id,
    ...(process.env.NODE_ENV === 'development' && { stack: err.stack }),
  });
};

// 404 handler — place before errorHandler
const notFound = (req, res, next) => {
  const err = new Error(`Route ${req.method} ${req.path} not found`);
  err.statusCode = 404;
  err.isOperational = true;
  next(err);
};

module.exports = { errorHandler, notFound };
```

### 5. Assemble the Middleware Stack

```js
// app.js
const express = require('express');
const morgan = require('morgan');
const helmet = require('helmet');
const cors = require('cors');
const compression = require('compression');

const logger = require('./middleware/logger');
const requestId = require('./middleware/requestId');
const authGuard = require('./middleware/authGuard');
const { errorHandler, notFound } = require('./middleware/errorHandler');

const app = express();

// --- Security & Global middleware (runs for ALL requests) ---
app.use(requestId);           // assign request ID first
app.use(helmet());            // sets secure HTTP headers
app.use(cors({                // allow cross-origin requests
  origin: process.env.ALLOWED_ORIGIN ?? '*',
  methods: ['GET', 'POST', 'PUT', 'PATCH', 'DELETE'],
}));
app.use(compression());       // gzip responses
app.use(morgan('dev'));        // third-party request logger (replaces our custom one in dev)
app.use(logger);              // our custom logger (also runs)
app.use(express.json({ limit: '10kb' }));  // parse JSON, limit body size
app.use(express.urlencoded({ extended: false }));

// --- Public routes ---
app.get('/health', (req, res) => {
  res.json({ status: 'ok', uptime: process.uptime() });
});

// --- Protected routes (authGuard applied to this sub-path) ---
app.get('/protected', authGuard, (req, res) => {
  res.json({ message: 'You are authenticated', user: req.user });
});

// Async route — must catch errors manually in Express 4
app.get('/async-error', async (req, res, next) => {
  try {
    throw new Error('Something went wrong in async handler');
  } catch (err) {
    next(err); // IMPORTANT: pass to error middleware
  }
});

// Trigger a formatted error
app.get('/error', (req, res, next) => {
  const err = new Error('Intentional error');
  err.statusCode = 422;
  err.isOperational = true;
  next(err);
});

// --- Error handling (must be last) ---
app.use(notFound);
app.use(errorHandler);

module.exports = app;
```

```js
// server.js
const app = require('./app');
app.listen(3000, () => console.log('Listening on http://localhost:3000'));
```
