# A9.2 — GraphQL with Apollo Server

**Phase:** 9 — Advanced Patterns  
**Topic:** GraphQL API with Apollo Server integrated into Express  
**Type:** Assignment

---

## Goal

Build a GraphQL API using Apollo Server mounted on Express, define a schema with typeDefs and resolvers, pass context, solve N+1 with DataLoader, and explore GraphQL Playground.

## Background

REST uses multiple URLs; GraphQL uses one endpoint (`/graphql`) where the client describes exactly what data it needs. Apollo Server is the most popular Node.js GraphQL runtime. Mounting it on Express lets you keep existing REST routes and middleware.

Key concepts:
- **typeDefs** — SDL schema: types, queries, mutations
- **resolvers** — functions that fulfil each field
- **context** — per-request object (user, db, loaders) shared across resolvers
- **DataLoader** — batches & caches DB calls within a single request to avoid N+1

### 2. Define schema (typeDefs)

```js
// src/graphql/schema.js
import { gql } from 'graphql-tag';

export const typeDefs = gql`
  type Author {
    id: ID!
    name: String!
    email: String!
    posts: [Post!]!
  }

  type Post {
    id: ID!
    title: String!
    body: String!
    published: Boolean!
    author: Author!
    createdAt: String!
  }

  type Query {
    posts(published: Boolean, limit: Int, offset: Int): [Post!]!
    post(id: ID!): Post
    authors: [Author!]!
    author(id: ID!): Author
  }

  type Mutation {
    createPost(title: String!, body: String!, authorId: ID!): Post!
    publishPost(id: ID!): Post!
    deletePost(id: ID!): Boolean!
  }
`;
```

### 3. Write resolvers

```js
// src/graphql/resolvers.js
import { Post, Author } from '../models/index.js';
import { GraphQLError } from 'graphql';

export const resolvers = {
  Query: {
    posts: async (_parent, { published, limit = 20, offset = 0 }) => {
      const filter = published !== undefined ? { published } : {};
      return Post.find(filter).skip(offset).limit(limit).sort({ createdAt: -1 });
    },

    post: async (_parent, { id }) => {
      const post = await Post.findById(id);
      if (!post) throw new GraphQLError('Post not found', { extensions: { code: 'NOT_FOUND' } });
      return post;
    },

    authors: () => Author.find(),
    author: (_parent, { id }) => Author.findById(id),
  },

  Mutation: {
    createPost: async (_parent, { title, body, authorId }, { user }) => {
      if (!user) throw new GraphQLError('Unauthenticated', { extensions: { code: 'UNAUTHENTICATED' } });

      const author = await Author.findById(authorId);
      if (!author) throw new GraphQLError('Author not found', { extensions: { code: 'NOT_FOUND' } });

      return Post.create({ title, body, authorId, published: false });
    },

    publishPost: async (_parent, { id }, { user }) => {
      if (!user) throw new GraphQLError('Unauthenticated', { extensions: { code: 'UNAUTHENTICATED' } });
      const post = await Post.findByIdAndUpdate(id, { published: true }, { new: true });
      if (!post) throw new GraphQLError('Post not found', { extensions: { code: 'NOT_FOUND' } });
      return post;
    },

    deletePost: async (_parent, { id }, { user }) => {
      if (!user) throw new GraphQLError('Unauthenticated', { extensions: { code: 'UNAUTHENTICATED' } });
      const result = await Post.findByIdAndDelete(id);
      return result !== null;
    },
  },

  // Field-level resolvers — called for each Post instance
  Post: {
    // Without DataLoader this would fire one query per post (N+1)
    author: (post, _args, { loaders }) => loaders.author.load(post.authorId.toString()),
  },

  Author: {
    posts: (author, _args, { loaders }) => loaders.postsByAuthor.load(author.id.toString()),
  },
};
```

### 4. Create DataLoaders (solve N+1)

```js
// src/graphql/loaders.js
import DataLoader from 'dataloader';
import { Author, Post } from '../models/index.js';

// Called once per batch of IDs collected during a single request tick
const batchAuthors = async (ids) => {
  const authors = await Author.find({ _id: { $in: ids } });
  const map = Object.fromEntries(authors.map((a) => [a.id, a]));
  return ids.map((id) => map[id] ?? new Error(`Author ${id} not found`));
};

const batchPostsByAuthor = async (authorIds) => {
  const posts = await Post.find({ authorId: { $in: authorIds } });
  return authorIds.map((id) => posts.filter((p) => p.authorId.toString() === id));
};

// Call this once per request — DataLoader is NOT shared across requests
export const createLoaders = () => ({
  author: new DataLoader(batchAuthors),
  postsByAuthor: new DataLoader(batchPostsByAuthor),
});
```

### 5. Mount Apollo Server on Express

```js
// src/server.js
import express from 'express';
import cors from 'cors';
import { json } from 'body-parser';
import { ApolloServer } from '@apollo/server';
import { expressMiddleware } from '@apollo/server/express4';
import { typeDefs } from './graphql/schema.js';
import { resolvers } from './graphql/resolvers.js';
import { createLoaders } from './graphql/loaders.js';
import jwt from 'jsonwebtoken';

const app = express();

// Build and start Apollo Server first
const apolloServer = new ApolloServer({ typeDefs, resolvers });
await apolloServer.start();

// Mount at /graphql — AFTER app.use(json()) and app.use(cors())
app.use(
  '/graphql',
  cors(),
  json(),
  expressMiddleware(apolloServer, {
    context: async ({ req }) => {
      // Decode JWT and attach user to context
      let user = null;
      const authHeader = req.headers.authorization ?? '';
      if (authHeader.startsWith('Bearer ')) {
        try {
          user = jwt.verify(authHeader.slice(7), process.env.JWT_SECRET);
        } catch {
          // invalid token — user stays null, resolvers decide if auth is needed
        }
      }

      return {
        user,
        loaders: createLoaders(),   // fresh loaders per request
      };
    },
  }),
);

// Keep existing REST routes working alongside GraphQL
app.get('/health', (_req, res) => res.json({ ok: true }));

app.listen(4000, () => {
  console.log('Server ready at http://localhost:4000/graphql');
  console.log('GraphQL Playground: http://localhost:4000/graphql (GET in browser)');
});
```

### 7. Error handling pattern

```js
import { ApolloServer, ApolloServerPlugin } from '@apollo/server';

// Plugin that logs errors
const errorLoggingPlugin = {
  async requestDidStart() {
    return {
      async didEncounterErrors({ errors }) {
        errors.forEach((err) => console.error('[GraphQL Error]', err));
      },
    };
  },
};

const server = new ApolloServer({
  typeDefs,
  resolvers,
  plugins: [errorLoggingPlugin],
  formatError: (formattedError) => {
    // Don't expose internal server errors in production
    if (formattedError.extensions?.code === 'INTERNAL_SERVER_ERROR' && process.env.NODE_ENV === 'production') {
      return { message: 'Internal server error', extensions: { code: 'INTERNAL_SERVER_ERROR' } };
    }
    return formattedError;
  },
});
```
