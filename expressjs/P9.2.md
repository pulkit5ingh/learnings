# P9.2 — Microservices Order System

**Phase:** 9 — Advanced Patterns  
**Topic:** 4 services + Bull queues + inter-service HTTP  
**Type:** Project

---

## Goal

Build a minimal e-commerce microservices system with four independent services: User, Product, Order, and Notification — communicating via inter-service HTTP and Bull queues.

## What to Build

- **user-service** (port 3001): register, login, profile
- **product-service** (port 3002): product catalog, inventory decrement
- **order-service** (port 3003): create order (calls user-service + product-service), publish job
- **notification-service** (port 3004): consumes order queue, sends email
- Bull queue: `order-service` produces `order.created` job; `notification-service` consumes it
- Inter-service auth: shared `INTERNAL_API_KEY` header
- `docker-compose.yml` to run all four services + Redis + MongoDB

## Implementation

```js
// services/order-service/src/services/orderService.js
import axios from 'axios';
import axiosRetry from 'axios-retry';
import Order from '../models/Order.js';
import { orderQueue } from '../queues/orderQueue.js';
import { AppError } from '../lib/AppError.js';

// Internal HTTP clients with timeout + retry
const makeInternalClient = (baseURL) => {
  const client = axios.create({
    baseURL,
    timeout: 3000,
    headers: { 'x-internal-key': process.env.INTERNAL_API_KEY },
  });
  axiosRetry(client, { retries: 2, retryDelay: axiosRetry.exponentialDelay });
  return client;
};

const userClient    = makeInternalClient(process.env.USER_SERVICE_URL    ?? 'http://user-service:3001');
const productClient = makeInternalClient(process.env.PRODUCT_SERVICE_URL ?? 'http://product-service:3002');

export const createOrder = async ({ userId, items }) => {
  // 1. Verify user exists
  let user;
  try {
    ({ data: user } = await userClient.get(`/users/${userId}`));
  } catch {
    throw new AppError('User not found', 404);
  }

  // 2. Validate products and get prices (parallel)
  const productResults = await Promise.allSettled(
    items.map((item) => productClient.get(`/products/${item.productId}`)),
  );

  const resolvedItems = [];
  for (let i = 0; i < items.length; i++) {
    const result = productResults[i];
    if (result.status === 'rejected') {
      throw new AppError(`Product ${items[i].productId} not found`, 404);
    }
    const product = result.value.data;
    if (product.inventory < items[i].quantity) {
      throw new AppError(`Insufficient inventory for ${product.name}`, 409);
    }
    resolvedItems.push({
      productId: product.id,
      name:      product.name,
      price:     product.price,
      quantity:  items[i].quantity,
    });
  }

  // 3. Decrement inventory (parallel)
  await Promise.all(
    resolvedItems.map((item) =>
      productClient.post(`/products/${item.productId}/decrement`, { quantity: item.quantity }),
    ),
  );

  // 4. Create order record
  const total = resolvedItems.reduce((sum, i) => sum + i.price * i.quantity, 0);
  const order = await Order.create({
    userId,
    items: resolvedItems,
    total,
    status: 'pending',
  });

  // 5. Publish to Bull queue — notification-service picks this up
  await orderQueue.add('order.created', {
    orderId:     order._id,
    userId,
    userEmail:   user.email,
    userName:    user.name,
    items:       resolvedItems,
    total,
    createdAt:   order.createdAt,
  });

  return order;
};
```

```js
// services/order-service/src/queues/orderQueue.js
import Bull from 'bull';

export const orderQueue = new Bull('orders', {
  redis: { host: process.env.REDIS_HOST ?? 'redis', port: 6379 },
  defaultJobOptions: {
    attempts: 3,
    backoff: { type: 'exponential', delay: 3000 },
    removeOnComplete: 200,
  },
});

orderQueue.on('failed', (job, err) => {
  console.error(`[Order Queue] Job ${job.id} failed:`, err.message);
});
```

```js
// services/notification-service/src/consumers/orderConsumer.js
import Bull from 'bull';
import { sendOrderConfirmation } from '../services/emailService.js';

const orderQueue = new Bull('orders', {
  redis: { host: process.env.REDIS_HOST ?? 'redis', port: 6379 },
});

export const startOrderConsumer = () => {
  orderQueue.process('order.created', async (job) => {
    const { orderId, userEmail, userName, items, total } = job.data;
    console.log(`[Notification] Processing order ${orderId} for ${userEmail}`);

    await sendOrderConfirmation({ to: userEmail, name: userName, orderId, items, total });
    console.log(`[Notification] Email sent for order ${orderId}`);
  });

  orderQueue.on('failed', (job, err) => {
    console.error(`[Notification Consumer] Job ${job.id} failed (attempt ${job.attemptsMade}):`, err.message);
  });

  console.log('[Notification] Order consumer started');
};
```

```js
// services/user-service/src/middleware/internalAuth.js
export const internalAuth = (req, res, next) => {
  if (req.headers['x-internal-key'] !== process.env.INTERNAL_API_KEY) {
    return res.status(401).json({ error: 'Unauthorized internal request' });
  }
  next();
};
```
