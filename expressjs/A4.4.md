## Goal
Set up production-quality structured logging with Winston, request correlation IDs, and separate log destinations per environment.

## Background

### Structured Logging
`console.log('User logged in: ' + email)` is human-readable but not machine-searchable. Structured logging emits JSON where each log entry has consistent fields: timestamp, level, message, and contextual metadata. A log aggregator (Datadog, CloudWatch, ELK Stack) can then filter, group, and alert on specific fields.

```json
{"timestamp":"2025-01-01T10:00:00Z","level":"info","message":"User logged in","userId":42,"email":"alice@example.com","requestId":"abc-123"}
```

### Winston
Winston is the most popular Node.js logging library. Key concepts:
- **Logger** — the main object you log through
- **Transport** — a destination for log output (console, file, HTTP endpoint, database)
- **Level** — severity from most to least: `error`, `warn`, `info`, `http`, `verbose`, `debug`, `silly`. Set a minimum level; only logs at that level and above are written.
- **Format** — how log entries are serialized (`printf`, `json`, `combine`, `colorize`, `timestamp`)

In development: colorized, human-readable output. In production: JSON to stdout or files, which are consumed by log aggregators.

### Correlation IDs
When a single user action results in multiple log entries (incoming request, DB query, outgoing API call, response), a correlation ID (also called request ID or trace ID) links them together. Without it, finding all logs for a specific request is nearly impossible.

Assign a UUID at the start of each request, attach it to the logger context, and include it in every log entry for that request's lifecycle.

### 2. Winston Logger

```js
// src/config/logger.js
const winston = require('winston');
const path = require('node:path');

const { combine, timestamp, json, colorize, printf, errors } = winston.format;

const isDev = process.env.NODE_ENV !== 'production';

// Custom format for development: colorized and readable
const devFormat = combine(
  colorize({ all: true }),
  timestamp({ format: 'HH:mm:ss' }),
  errors({ stack: true }),
  printf(({ timestamp, level, message, requestId, ...meta }) => {
    const rid = requestId ? ` [${requestId}]` : '';
    const metaStr = Object.keys(meta).length ? ' ' + JSON.stringify(meta) : '';
    return `${timestamp}${rid} ${level}: ${message}${metaStr}`;
  })
);

// JSON format for production — consumed by log aggregators
const prodFormat = combine(
  timestamp(),
  errors({ stack: true }),
  json()
);

// --- Transports ---
const transports = [];

// Console transport (always)
transports.push(
  new winston.transports.Console({
    format: isDev ? devFormat : prodFormat,
  })
);

// File transports (production only)
if (!isDev) {
  // Combined log
  transports.push(
    new winston.transports.File({
      filename: path.join('logs', 'combined.log'),
      format: prodFormat,
      maxsize: 10 * 1024 * 1024, // 10MB
      maxFiles: 5,
    })
  );

  // Error log
  transports.push(
    new winston.transports.File({
      level: 'error',
      filename: path.join('logs', 'error.log'),
      format: prodFormat,
      maxsize: 10 * 1024 * 1024,
      maxFiles: 5,
    })
  );
}

const logger = winston.createLogger({
  level: process.env.LOG_LEVEL ?? (isDev ? 'debug' : 'info'),
  transports,
  // Prevent process crash on logger error
  exitOnError: false,
});

// Stream for Morgan HTTP logging
logger.stream = {
  write: (message) => logger.http(message.trim()),
};

module.exports = logger;
```

### 3. Request Logger Middleware

```js
// src/middleware/requestLogger.js
const { randomUUID } = require('node:crypto');
const morgan = require('morgan');
const logger = require('../config/logger');

// Assign correlation ID to each request
const correlationId = (req, res, next) => {
  const id = req.headers['x-request-id'] ?? randomUUID();
  req.requestId = id;
  res.setHeader('X-Request-Id', id);

  // Child logger with requestId baked in
  req.log = logger.child({ requestId: id });

  next();
};

// Morgan format using correlation ID
const httpLogger = morgan(
  ':method :url :status :res[content-length] - :response-time ms',
  {
    stream: {
      write: (message) => {
        // Log at 'http' level via Winston
        logger.http(message.trim());
      },
    },
    skip: (req) => req.path === '/health', // don't log health checks
  }
);

module.exports = { correlationId, httpLogger };
```

### 4. Usage Throughout the App

```js
// src/routes/users.js
const router = require('express').Router();
const logger = require('../config/logger');

const users = [{ id: 1, name: 'Alice', email: 'alice@test.com' }];

router.get('/', (req, res) => {
  // Use req.log (child logger with requestId)
  req.log.info('Fetching users list', { count: users.length });
  res.json({ data: users });
});

router.get('/:id', (req, res) => {
  const id = parseInt(req.params.id);
  const user = users.find(u => u.id === id);

  if (!user) {
    req.log.warn('User not found', { userId: id });
    return res.status(404).json({ error: 'User not found' });
  }

  req.log.debug('User found', { userId: id, email: user.email });
  res.json(user);
});

router.post('/', (req, res) => {
  const { name, email } = req.body;

  if (!name || !email) {
    req.log.warn('Invalid user creation attempt', { body: req.body });
    return res.status(400).json({ error: 'name and email required' });
  }

  const user = { id: users.length + 1, name, email };
  users.push(user);

  req.log.info('User created', { userId: user.id, email });
  res.status(201).json(user);
});

module.exports = router;
```

### 5. Error Handler with Logging

```js
// src/middleware/errorHandler.js
const logger = require('../config/logger');

const errorHandler = (err, req, res, next) => {
  const statusCode = err.statusCode ?? 500;
  const isServerError = statusCode >= 500;

  // Log level based on severity
  const logMethod = isServerError ? 'error' : 'warn';

  req.log[logMethod]('Request error', {
    statusCode,
    message: err.message,
    code: err.code,
    ...(isServerError && { stack: err.stack }),
  });

  res.status(statusCode).json({
    error: err.isOperational ? err.message : 'Internal server error',
    requestId: req.requestId,
    ...(process.env.NODE_ENV === 'development' && { stack: err.stack }),
  });
};

module.exports = errorHandler;
```

### 6. App Assembly

```js
// src/app.js
const express = require('express');
const logger = require('./config/logger');
const { correlationId, httpLogger } = require('./middleware/requestLogger');
const errorHandler = require('./middleware/errorHandler');
const usersRouter = require('./routes/users');

const app = express();

app.use(express.json());
app.use(correlationId); // FIRST — everything after needs req.log
app.use(httpLogger);

app.get('/health', (req, res) => {
  res.json({ status: 'ok', env: process.env.NODE_ENV });
});

app.use('/api/users', usersRouter);

app.use((req, res) => {
  req.log.warn('Route not found', { method: req.method, path: req.path });
  res.status(404).json({ error: 'Not found' });
});

app.use(errorHandler);

// Log unhandled rejections
process.on('unhandledRejection', (err) => {
  logger.error('Unhandled rejection', { error: err.message, stack: err.stack });
  process.exit(1);
});

module.exports = app;
```

## Deliverables

- [ ] Winston logger uses `devFormat` (colorized) in development, JSON in production
- [ ] Log level is configurable via `LOG_LEVEL` environment variable
- [ ] Each request gets a unique `requestId` attached to `req.log`
- [ ] `X-Request-Id` header appears in every response
- [ ] Morgan logs HTTP requests through Winston at `http` level
- [ ] Health check endpoint is excluded from Morgan logging
- [ ] Error handler logs server errors at `error` level, client errors at `warn`
- [ ] Production mode writes to `logs/combined.log` and `logs/error.log`
- [ ] `logger.child({ requestId })` is used so all logs from a request share the same ID
