## Goal
Use the `pg` library to run raw SQL queries against PostgreSQL, manage connection pools, and handle transactions correctly.

## Background

### Why Raw SQL?
ORMs like Sequelize abstract SQL behind JavaScript methods, which is convenient but introduces overhead and can generate inefficient queries. For performance-critical paths, complex queries, or when you need full SQL control (window functions, CTEs, EXPLAIN ANALYZE), writing raw SQL is better. The `pg` library (node-postgres) is the foundation that Sequelize and Prisma build on.

### Connection Pooling
Opening a new database connection for each request is expensive — each connection requires TCP handshake, authentication, and server-side state. A connection pool maintains a set of open connections and reuses them. `pg.Pool` manages this automatically. Key pool settings: `max` (default 10) limits concurrent connections; `idleTimeoutMillis` closes idle connections; `connectionTimeoutMillis` rejects requests that wait too long for a connection.

### Parameterized Queries and SQL Injection
Never concatenate user input into SQL strings. This opens the door for SQL injection attacks. Parameterized queries pass values separately from the query template:

```js
// WRONG (vulnerable):
const result = await pool.query(`SELECT * FROM users WHERE email = '${email}'`);

// CORRECT (safe):
const result = await pool.query('SELECT * FROM users WHERE email = $1', [email]);
```

PostgreSQL uses `$1`, `$2`, etc. as placeholders. The driver sends query and parameters separately; the database never interprets parameter values as SQL.

### Transactions
A transaction groups multiple SQL statements into an atomic unit. Either all statements succeed (COMMIT), or all are rolled back (ROLLBACK) on error. In `pg`, you acquire a client from the pool and manually control the transaction:
```
BEGIN → execute statements → COMMIT (or ROLLBACK on error) → release client
```
Always release the client in a `finally` block, even if the transaction fails. Unreleased clients exhaust the pool.

### 2. Pool Setup

```js
// src/config/db.js
const { Pool } = require('pg');

const pool = new Pool({
  host: process.env.DB_HOST ?? 'localhost',
  port: parseInt(process.env.DB_PORT ?? '5432'),
  database: process.env.DB_NAME ?? 'blog_dev',
  user: process.env.DB_USER ?? 'postgres',
  password: process.env.DB_PASS ?? 'secret',
  max: 10,                      // max pool size
  idleTimeoutMillis: 30_000,    // close idle connections after 30s
  connectionTimeoutMillis: 2_000, // fail if no connection in 2s
});

pool.on('error', (err) => {
  console.error('Unexpected pool error:', err.message);
});

// Convenience query helper
const query = (text, params) => pool.query(text, params);

// Transaction helper
const withTransaction = async (fn) => {
  const client = await pool.connect();
  try {
    await client.query('BEGIN');
    const result = await fn(client);
    await client.query('COMMIT');
    return result;
  } catch (err) {
    await client.query('ROLLBACK');
    throw err;
  } finally {
    client.release(); // ALWAYS release
  }
};

module.exports = { pool, query, withTransaction };
```

### 3. Account Queries

```js
// src/models/account.queries.js
const { query, withTransaction } = require('../config/db');

const findAll = async ({ limit = 20, offset = 0 } = {}) => {
  const { rows } = await query(
    'SELECT id, owner_name, email, balance, created_at FROM accounts ORDER BY id LIMIT $1 OFFSET $2',
    [limit, offset]
  );
  return rows;
};

const findById = async (id) => {
  const { rows } = await query(
    'SELECT * FROM accounts WHERE id = $1',
    [id]
  );
  return rows[0] ?? null;
};

const findByEmail = async (email) => {
  const { rows } = await query(
    'SELECT * FROM accounts WHERE email = $1',
    [email]
  );
  return rows[0] ?? null;
};

const create = async ({ ownerName, email, initialBalance = 0 }) => {
  const { rows } = await query(
    `INSERT INTO accounts (owner_name, email, balance)
     VALUES ($1, $2, $3)
     RETURNING *`,
    [ownerName, email, initialBalance]
  );
  return rows[0];
};

// Transfer money between accounts — must be atomic
const transfer = async ({ fromId, toId, amount, description }) => {
  return withTransaction(async (client) => {
    // Check source account has sufficient balance
    const { rows: [from] } = await client.query(
      'SELECT id, balance FROM accounts WHERE id = $1 FOR UPDATE', // lock row
      [fromId]
    );

    if (!from) throw Object.assign(new Error('Source account not found'), { statusCode: 404 });
    if (parseFloat(from.balance) < amount) {
      throw Object.assign(new Error('Insufficient balance'), { statusCode: 422 });
    }

    // Check destination exists
    const { rows: [to] } = await client.query(
      'SELECT id FROM accounts WHERE id = $1',
      [toId]
    );
    if (!to) throw Object.assign(new Error('Destination account not found'), { statusCode: 404 });

    // Debit source
    await client.query(
      'UPDATE accounts SET balance = balance - $1, updated_at = NOW() WHERE id = $2',
      [amount, fromId]
    );

    // Credit destination
    await client.query(
      'UPDATE accounts SET balance = balance + $1, updated_at = NOW() WHERE id = $2',
      [amount, toId]
    );

    // Record transaction
    const { rows: [txn] } = await client.query(
      `INSERT INTO transactions (from_account_id, to_account_id, amount, description)
       VALUES ($1, $2, $3, $4) RETURNING *`,
      [fromId, toId, amount, description]
    );

    return txn;
  });
};

const getTransactionHistory = async (accountId) => {
  const { rows } = await query(
    `SELECT t.*, 
            a1.owner_name AS from_name,
            a2.owner_name AS to_name
     FROM transactions t
     LEFT JOIN accounts a1 ON t.from_account_id = a1.id
     LEFT JOIN accounts a2 ON t.to_account_id = a2.id
     WHERE t.from_account_id = $1 OR t.to_account_id = $1
     ORDER BY t.created_at DESC`,
    [accountId]
  );
  return rows;
};

module.exports = { findAll, findById, findByEmail, create, transfer, getTransactionHistory };
```

### 4. Routes

```js
// src/routes/accounts.js
const router = require('express').Router();
const accounts = require('../models/account.queries');

router.get('/', async (req, res, next) => {
  try {
    const { limit = 20, page = 1 } = req.query;
    const offset = (parseInt(page) - 1) * parseInt(limit);
    const data = await accounts.findAll({ limit: parseInt(limit), offset });
    res.json({ data });
  } catch (err) { next(err); }
});

router.get('/:id', async (req, res, next) => {
  try {
    const account = await accounts.findById(parseInt(req.params.id));
    if (!account) return res.status(404).json({ error: 'Account not found' });
    res.json(account);
  } catch (err) { next(err); }
});

router.post('/', async (req, res, next) => {
  try {
    const { ownerName, email, initialBalance } = req.body;
    if (!ownerName || !email) {
      return res.status(400).json({ error: 'ownerName and email required' });
    }
    const account = await accounts.create({ ownerName, email, initialBalance });
    res.status(201).json(account);
  } catch (err) {
    if (err.code === '23505') { // unique violation
      return res.status(409).json({ error: 'Email already exists' });
    }
    next(err);
  }
});

router.post('/transfer', async (req, res, next) => {
  try {
    const { fromId, toId, amount, description } = req.body;
    if (!fromId || !toId || !amount) {
      return res.status(400).json({ error: 'fromId, toId, amount required' });
    }
    const txn = await accounts.transfer({ fromId, toId, amount: parseFloat(amount), description });
    res.status(201).json({ message: 'Transfer successful', transaction: txn });
  } catch (err) {
    if (err.statusCode) return res.status(err.statusCode).json({ error: err.message });
    next(err);
  }
});

router.get('/:id/history', async (req, res, next) => {
  try {
    const history = await accounts.getTransactionHistory(parseInt(req.params.id));
    res.json({ data: history });
  } catch (err) { next(err); }
});

module.exports = router;
```
