## Goal
Scale a Node.js application across all CPU cores using the cluster module and manage processes with PM2 for production deployments.

## Background

### Node.js Single-Thread Limitation
Node.js runs JavaScript on a single thread. A server with 8 CPU cores runs only one of those cores unless you explicitly create multiple processes. The `cluster` module forks the process into a master (coordinator) and N worker processes, each running the same server code on a separate CPU core. The OS distributes incoming connections across workers.

Workers share no memory. State must live in an external store (Redis, database) — not in module-level variables. This is why in-memory rate limiters and sessions require Redis-backed stores when clustering.

### PM2
PM2 is a production process manager for Node.js. It wraps the cluster module with better ergonomics: auto-restart on crash, log management, monitoring dashboard, and zero-downtime reload via `pm2 reload` (restarts workers one at a time while others keep serving traffic).

Key PM2 features:
- `cluster` mode: automatically creates `os.cpus().length` workers
- `fork` mode: single process (default)
- `pm2 reload app` — zero-downtime restart (rolling restart of workers)
- `pm2 save` + `pm2 startup` — start PM2 on system boot
- `pm2 monit` — real-time dashboard
- `pm2 logs` — stream logs

### Memory and CPU Limits
PM2 can restart a worker when it exceeds a memory threshold (`max_memory_restart: '500M'`) or after a set number of requests (`restart_delay`). This prevents memory leaks from accumulating indefinitely.

### 2. Manual Cluster Module

```js
// cluster.js — run this instead of server.js to use cluster
const cluster = require('node:cluster');
const { cpus } = require('node:os');
const process = require('node:process');

const NUM_CPUS = cpus().length;

if (cluster.isPrimary) {
  console.log(`Primary ${process.pid} started`);
  console.log(`Forking ${NUM_CPUS} workers...`);

  // Fork workers
  for (let i = 0; i < NUM_CPUS; i++) {
    cluster.fork();
  }

  // Replace a dead worker
  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died (code: ${code}, signal: ${signal})`);
    console.log('Forking replacement worker...');
    cluster.fork();
  });

  cluster.on('online', (worker) => {
    console.log(`Worker ${worker.process.pid} is online`);
  });

  // Graceful shutdown: send SIGTERM to all workers
  process.on('SIGTERM', () => {
    console.log('Primary received SIGTERM, shutting down workers...');
    for (const worker of Object.values(cluster.workers)) {
      worker.kill('SIGTERM');
    }
  });

} else {
  // Workers run the actual server
  const app = require('./src/app');
  const PORT = process.env.PORT ?? 3000;

  const server = app.listen(PORT, () => {
    console.log(`Worker ${process.pid} listening on port ${PORT}`);
  });

  // Graceful shutdown for worker
  process.on('SIGTERM', () => {
    console.log(`Worker ${process.pid} shutting down...`);
    server.close(() => {
      console.log(`Worker ${process.pid} closed`);
      process.exit(0);
    });
  });
}
```

### 3. App with Worker ID in Response

```js
// src/app.js
const express = require('express');
const process = require('node:process');

const app = express();
app.use(express.json());

app.get('/health', (req, res) => {
  res.json({
    status: 'ok',
    pid: process.pid,
    uptime: process.uptime(),
    memory: process.memoryUsage(),
  });
});

// CPU-intensive work (blocks this worker's event loop)
app.get('/cpu-work', (req, res) => {
  const n = parseInt(req.query.n ?? '1000000');
  let sum = 0;
  for (let i = 0; i < n; i++) sum += i;
  res.json({ pid: process.pid, sum, n });
});

module.exports = app;
```

### 4. PM2 Ecosystem Config

```js
// ecosystem.config.js
module.exports = {
  apps: [
    {
      name: 'api-server',
      script: './src/server.js',
      instances: 'max',        // use all CPU cores
      exec_mode: 'cluster',    // cluster mode
      watch: false,
      env: {
        NODE_ENV: 'production',
        PORT: 3000,
      },
      env_development: {
        NODE_ENV: 'development',
        PORT: 3000,
        LOG_LEVEL: 'debug',
      },
      // Auto-restart policies
      max_memory_restart: '500M',
      restart_delay: 4000,      // wait 4s before restart
      max_restarts: 10,         // stop trying after 10 crashes
      min_uptime: '10s',        // must stay up for 10s to count as successful start

      // Logging
      log_file: './logs/pm2-combined.log',
      out_file: './logs/pm2-out.log',
      error_file: './logs/pm2-error.log',
      log_date_format: 'YYYY-MM-DD HH:mm:ss Z',
      merge_logs: true,         // combine logs from all workers

      // Graceful shutdown
      kill_timeout: 5000,       // wait 5s for graceful shutdown before SIGKILL
      listen_timeout: 3000,     // wait 3s for server to start listening

      // Node.js flags
      node_args: '--max-old-space-size=512',
    },
  ],
};
```

### 5. PM2 Commands

```bash
# Start in development

# Start in production
npx pm2 start ecosystem.config.js --env production

# Zero-downtime reload (rolling restart)
npx pm2 reload api-server

# Full restart (brief downtime)
npx pm2 restart api-server

# Scale to 4 workers
npx pm2 scale api-server 4

# Real-time monitoring dashboard
npx pm2 monit

# Stream logs
npx pm2 logs api-server

# List all processes
npx pm2 list

# Show details for a process
npx pm2 show api-server

# Stop all
npx pm2 stop all

# Save process list (so pm2 startup can restore it)
npx pm2 save

# Auto-start PM2 on system boot
npx pm2 startup
# Run the command it outputs as sudo

# Delete a process from PM2
npx pm2 delete api-server
```
