# P6.1 — File Upload Service

**Phase:** 6 — File Handling & External Services  
**Topic:** multer, S3 upload, sharp thumbnail, signed URLs, metadata in DB  
**Type:** Project

---

## Goal

Build a file upload microservice: accept image uploads, resize to multiple sizes with sharp, store originals and thumbnails in S3, persist metadata in MongoDB, and return pre-signed download URLs.

## What to Build

- `POST /api/uploads` — upload one image (multipart/form-data)
- `GET /api/uploads` — list user's uploads (paginated)
- `GET /api/uploads/:id` — get upload details + fresh signed URLs
- `DELETE /api/uploads/:id` — remove from S3 and DB
- Accepted: JPEG, PNG, WebP — max 10 MB
- Three sizes generated: original, medium (800px), thumbnail (200px)
- All sizes stored in S3; signed URLs valid 1 hour

## Implementation

```js
// src/middleware/multerConfig.js
import multer from 'multer';
import { AppError } from '../lib/AppError.js';

const ALLOWED_TYPES = ['image/jpeg', 'image/png', 'image/webp'];
const MAX_SIZE = 10 * 1024 * 1024; // 10 MB

export const upload = multer({
  storage: multer.memoryStorage(), // keep in buffer — we'll process with sharp before S3
  limits: { fileSize: MAX_SIZE, files: 1 },
  fileFilter(_req, file, callback) {
    if (ALLOWED_TYPES.includes(file.mimetype)) {
      callback(null, true);
    } else {
      callback(new AppError('Unsupported file type. Use JPEG, PNG, or WebP.', 415));
    }
  },
});
```

```js
// src/services/imageService.js
import sharp from 'sharp';

const SIZES = {
  original:  null,               // no resize
  medium:    { width: 800 },
  thumbnail: { width: 200, height: 200, fit: 'cover' },
};

/**
 * Returns { original, medium, thumbnail } — each a Buffer of WebP data.
 * Converting to WebP reduces size by ~30% vs JPEG.
 */
export const processImage = async (buffer) => {
  const results = {};

  for (const [name, options] of Object.entries(SIZES)) {
    let pipeline = sharp(buffer);
    if (options) pipeline = pipeline.resize(options);
    results[name] = await pipeline.webp({ quality: 85 }).toBuffer();
  }

  // Also extract metadata from the original
  const meta = await sharp(buffer).metadata();
  results.meta = { width: meta.width, height: meta.height, format: meta.format, size: buffer.length };

  return results;
};
```

```js
// src/services/s3Service.js
import { S3Client, PutObjectCommand, DeleteObjectCommand, GetObjectCommand } from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import { randomUUID } from 'node:crypto';

const s3 = new S3Client({
  region: process.env.AWS_REGION ?? 'us-east-1',
  credentials: {
    accessKeyId:     process.env.AWS_ACCESS_KEY_ID,
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
  },
});

const BUCKET = process.env.S3_BUCKET;

export const uploadToS3 = async (buffer, { folder = 'uploads', contentType = 'image/webp' } = {}) => {
  const key = `${folder}/${randomUUID()}.webp`;

  await s3.send(new PutObjectCommand({
    Bucket:      BUCKET,
    Key:         key,
    Body:        buffer,
    ContentType: contentType,
    // No public ACL — access via signed URLs only
  }));

  return key;
};

export const deleteFromS3 = async (key) => {
  await s3.send(new DeleteObjectCommand({ Bucket: BUCKET, Key: key }));
};

export const getSignedDownloadUrl = (key, expiresInSeconds = 3600) =>
  getSignedUrl(s3, new GetObjectCommand({ Bucket: BUCKET, Key: key }), { expiresIn: expiresInSeconds });
```

```js
// src/models/Upload.js
import mongoose from 'mongoose';

const uploadSchema = new mongoose.Schema(
  {
    user:    { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
    originalName: String,
    mimeType: String,
    meta:    {
      width:  Number,
      height: Number,
      format: String,
      size:   Number,   // bytes of original
    },
    s3Keys: {
      original:  { type: String, required: true },
      medium:    { type: String, required: true },
      thumbnail: { type: String, required: true },
    },
  },
  { timestamps: true },
);

uploadSchema.index({ user: 1, createdAt: -1 });

export default mongoose.model('Upload', uploadSchema);
```

```js
// src/controllers/uploadsController.js
import { processImage } from '../services/imageService.js';
import { uploadToS3, deleteFromS3, getSignedDownloadUrl } from '../services/s3Service.js';
import Upload from '../models/Upload.js';

export const createUpload = async (req, res) => {
  if (!req.file) return res.status(400).json({ error: 'No file uploaded' });

  const { original, medium, thumbnail, meta } = await processImage(req.file.buffer);

  const [originalKey, mediumKey, thumbnailKey] = await Promise.all([
    uploadToS3(original,   { folder: 'originals' }),
    uploadToS3(medium,     { folder: 'medium' }),
    uploadToS3(thumbnail,  { folder: 'thumbnails' }),
  ]);

  const upload = await Upload.create({
    user:         req.user._id,
    originalName: req.file.originalname,
    mimeType:     req.file.mimetype,
    meta,
    s3Keys: { original: originalKey, medium: mediumKey, thumbnail: thumbnailKey },
  });

  const urls = await generateSignedUrls(upload.s3Keys);
  res.status(201).json({ ...upload.toObject(), urls });
};

export const listUploads = async (req, res) => {
  const { page = 1, limit = 20 } = req.query;
  const [uploads, total] = await Promise.all([
    Upload.find({ user: req.user._id }).sort({ createdAt: -1 }).skip((page - 1) * limit).limit(Number(limit)),
    Upload.countDocuments({ user: req.user._id }),
  ]);

  const withUrls = await Promise.all(uploads.map(async (u) => ({
    ...u.toObject(),
    urls: await generateSignedUrls(u.s3Keys),
  })));

  res.json({ data: withUrls, meta: { total, page: Number(page) } });
};

export const getUpload = async (req, res) => {
  const upload = await Upload.findOne({ _id: req.params.id, user: req.user._id });
  if (!upload) return res.status(404).json({ error: 'Upload not found' });
  const urls = await generateSignedUrls(upload.s3Keys);
  res.json({ ...upload.toObject(), urls });
};

export const deleteUpload = async (req, res) => {
  const upload = await Upload.findOneAndDelete({ _id: req.params.id, user: req.user._id });
  if (!upload) return res.status(404).json({ error: 'Upload not found' });

  await Promise.all([
    deleteFromS3(upload.s3Keys.original),
    deleteFromS3(upload.s3Keys.medium),
    deleteFromS3(upload.s3Keys.thumbnail),
  ]);

  res.status(204).send();
};

const generateSignedUrls = async (keys) => ({
  original:  await getSignedDownloadUrl(keys.original),
  medium:    await getSignedDownloadUrl(keys.medium),
  thumbnail: await getSignedDownloadUrl(keys.thumbnail),
});
```

## Deliverables

- [ ] `multerConfig` accepts only JPEG/PNG/WebP, max 10 MB; rejects others with 415
- [ ] `processImage` returns original, medium (800px), and thumbnail (200px) WebP buffers
- [ ] All three sizes uploaded to S3 in parallel with `Promise.all`
- [ ] `Upload` model stores S3 keys and original file metadata
- [ ] `createUpload` returns the new document with fresh signed URLs
- [ ] `getUpload` returns fresh signed URLs (not stored in DB)
- [ ] `deleteUpload` removes all three S3 objects and the DB record
- [ ] Signed URLs are scoped per-user (user cannot access other users' uploads)
