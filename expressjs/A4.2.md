## Goal
Use express-validator to build inline validation middleware chains and understand when to choose it over Joi.

## Background

### express-validator vs Joi
express-validator is built on the `validator.js` library and integrates directly with Express middleware. Instead of a separate schema object, validation is expressed as middleware functions chained together.

When to choose express-validator:
- You prefer inline validation co-located with routes
- You want to use built-in sanitizers (`.trim()`, `.escape()`, `.normalizeEmail()`)
- Lighter footprint than Joi for simple cases

When to choose Joi:
- You want centralized, reusable schemas
- Complex nested validation, conditional schemas
- Schema doubles as documentation

Both approaches are valid. Many teams use Joi for complex models and express-validator for simple endpoint validation.

### Validation Chain
`body('email')` returns a `ValidationChain` — middleware that validates `req.body.email`. Call `.isEmail()`, `.isLength()`, `.notEmpty()`, `.custom()` on it to add validators. Multiple chains are passed as an array to the route.

`validationResult(req)` collects all errors from all chains into a `Result` object. Call `.isEmpty()` to check for errors, `.array()` to get the error list.

### Custom Validators
`.custom(fn)` accepts an async function. Throw an error or return a rejected Promise to fail validation. This is how you check uniqueness (does this email already exist in the database?) in the validation layer.

### 2. Validation Result Extractor

```js
// src/middleware/handleValidation.js
const { validationResult } = require('express-validator');

const handleValidation = (req, res, next) => {
  const errors = validationResult(req);

  if (!errors.isEmpty()) {
    return res.status(400).json({
      error: 'Validation failed',
      details: errors.array().map(e => ({
        field: e.path,
        message: e.msg,
        value: e.value,
      })),
    });
  }

  next();
};

module.exports = handleValidation;
```

### 3. User Validation Chains

```js
// src/validators/user.validators.js
const { body, param, query } = require('express-validator');
const userStore = require('../data/users');

// Reusable chain builders

const validateName = () =>
  body('name')
    .trim()
    .notEmpty().withMessage('Name is required')
    .isLength({ min: 2, max: 50 }).withMessage('Name must be 2–50 characters');

const validateEmail = () =>
  body('email')
    .trim()
    .notEmpty().withMessage('Email is required')
    .isEmail().withMessage('Must be a valid email')
    .normalizeEmail()
    .custom(async (email) => {
      const existing = userStore.findByEmail(email);
      if (existing) throw new Error('Email is already in use');
    });

const validateEmailOptional = () =>
  body('email')
    .optional()
    .trim()
    .isEmail().withMessage('Must be a valid email')
    .normalizeEmail();

const validatePassword = () =>
  body('password')
    .notEmpty().withMessage('Password is required')
    .isLength({ min: 8 }).withMessage('Password must be at least 8 characters')
    .matches(/[A-Z]/).withMessage('Password must contain an uppercase letter')
    .matches(/[a-z]/).withMessage('Password must contain a lowercase letter')
    .matches(/\d/).withMessage('Password must contain a digit');

const validateAge = () =>
  body('age')
    .optional()
    .isInt({ min: 0, max: 150 }).withMessage('Age must be an integer between 0 and 150')
    .toInt();

const validateRole = () =>
  body('role')
    .optional()
    .isIn(['user', 'moderator']).withMessage('Role must be user or moderator');

// Param validator
const validateIdParam = () =>
  param('id')
    .isInt({ min: 1 }).withMessage('ID must be a positive integer')
    .toInt();

// Query validators
const validateListQuery = () => [
  query('page').optional().isInt({ min: 1 }).toInt().default(1),
  query('limit').optional().isInt({ min: 1, max: 100 }).toInt().default(20),
  query('sort').optional().isIn(['name', '-name', 'createdAt', '-createdAt']),
];

// Combined validator sets for routes
const registerValidators = [
  validateName(),
  validateEmail(),
  validatePassword(),
  validateAge(),
  validateRole(),
];

const updateValidators = [
  validateIdParam(),
  validateName().optional(),
  validateEmailOptional(),
  validateAge(),
];

module.exports = {
  registerValidators,
  updateValidators,
  validateIdParam,
  validateListQuery,
};
```

### 4. Custom Sanitizer and Cross-Field Validation

```js
// src/validators/product.validators.js
const { body } = require('express-validator');

const createProductValidators = [
  body('name')
    .trim()
    .escape()  // HTML entity encode: < → &lt;
    .notEmpty().withMessage('Product name is required')
    .isLength({ max: 100 }),

  body('price')
    .isFloat({ min: 0.01 }).withMessage('Price must be a positive number')
    .toFloat(),

  body('salePrice')
    .optional()
    .isFloat({ min: 0 }).withMessage('Sale price must be non-negative')
    .toFloat()
    .custom((salePrice, { req }) => {
      // Cross-field validation: salePrice must be less than price
      if (salePrice >= req.body.price) {
        throw new Error('Sale price must be less than the regular price');
      }
      return true;
    }),

  body('category')
    .isIn(['electronics', 'clothing', 'food', 'books'])
    .withMessage('Invalid category'),

  body('tags')
    .optional()
    .isArray().withMessage('Tags must be an array')
    .custom((tags) => {
      if (tags.some(t => typeof t !== 'string')) {
        throw new Error('All tags must be strings');
      }
      return true;
    }),

  body('tags.*')
    .optional()
    .trim()
    .toLowerCase()
    .isLength({ min: 1, max: 30 }).withMessage('Each tag must be 1–30 characters'),

  body('stock')
    .isInt({ min: 0 }).withMessage('Stock must be a non-negative integer')
    .toInt(),
];

module.exports = { createProductValidators };
```

### 5. Routes

```js
// src/routes/users.js
const router = require('express').Router();
const handleValidation = require('../middleware/handleValidation');
const { registerValidators, updateValidators, validateIdParam, validateListQuery } = require('../validators/user.validators');

const users = [];
let nextId = 1;

// Validation chains are spread into route, followed by handleValidation
router.get('/',
  ...validateListQuery(),
  handleValidation,
  (req, res) => {
    res.json({ page: req.query.page, limit: req.query.limit, data: users });
  }
);

router.post('/',
  ...registerValidators,
  handleValidation,
  (req, res) => {
    const { name, email, age, role = 'user' } = req.body;
    const user = { id: nextId++, name, email, age, role };
    users.push(user);
    res.status(201).json(user);
  }
);

router.patch('/:id',
  ...updateValidators,
  handleValidation,
  (req, res) => {
    const index = users.findIndex(u => u.id === req.params.id);
    if (index === -1) return res.status(404).json({ error: 'User not found' });
    Object.assign(users[index], req.body);
    res.json(users[index]);
  }
);

module.exports = router;
```
