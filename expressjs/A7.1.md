## Goal
Write unit tests with Jest covering synchronous code, async functions, edge cases, and generate a coverage report.

## Background

### What to Test
Test behavior, not implementation. A test should describe what the function does (its contract), not how it does it. If you refactor the implementation without changing the behavior, all tests should still pass.

Write tests for: pure utility functions, validation logic, data transformations, business rules (pricing calculations, permission checks). Don't test: third-party library internals, simple property access, database queries without mocking.

### Test Structure: AAA
Arrange, Act, Assert:
1. **Arrange**: set up test data and preconditions
2. **Act**: call the function being tested
3. **Assert**: verify the result

### Coverage
Jest measures line, branch, function, and statement coverage. Coverage tells you which code was executed during tests — not whether the tests are good. 100% coverage with bad assertions is worthless. Aim for 80%+ coverage on business logic; don't chase 100% on every file.

```js
const isStrongPassword = (password) => {
  if (password.length < 8) return { valid: false, reason: 'Too short' };
  if (!/[A-Z]/.test(password)) return { valid: false, reason: 'Missing uppercase' };
  if (!/[a-z]/.test(password)) return { valid: false, reason: 'Missing lowercase' };
```

```js
// src/utils/pricing.js
const calculateDiscount = (price, discountPercent) => {
  if (price < 0) throw new Error('Price cannot be negative');
  if (discountPercent < 0 || discountPercent > 100) {
    throw new Error('Discount must be between 0 and 100');
  }
  return parseFloat((price * (1 - discountPercent / 100)).toFixed(2));
};

const calculateTax = (price, taxRate) => parseFloat((price * taxRate).toFixed(2));

const calculateTotal = ({ items, discountPercent = 0, taxRate = 0 }) => {
  const subtotal = items.reduce((sum, item) => sum + item.price * item.quantity, 0);
  const discounted = calculateDiscount(subtotal, discountPercent);
  const tax = calculateTax(discounted, taxRate);
  return {
    subtotal: parseFloat(subtotal.toFixed(2)),
    discount: parseFloat((subtotal - discounted).toFixed(2)),
    tax,
    total: parseFloat((discounted + tax).toFixed(2)),
  };
};

module.exports = { calculateDiscount, calculateTax, calculateTotal };
```

```js
// src/utils/async.js
const delay = (ms) => new Promise(resolve => setTimeout(resolve, ms));

const fetchUserById = async (id, db) => {
  if (!Number.isInteger(id) || id <= 0) {
    throw new Error('ID must be a positive integer');
  }
  const user = await db.findById(id);
  if (!user) throw new Error(`User ${id} not found`);
  return user;
};

const retryAsync = async (fn, maxAttempts = 3) => {
  let lastError;
  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    try {
      return await fn();
    } catch (err) {
      lastError = err;
      if (attempt < maxAttempts) await delay(10); // small delay in tests
    }
  }
  throw lastError;
};

module.exports = { delay, fetchUserById, retryAsync };
```

```js
// tests/async.test.js
const { fetchUserById, retryAsync } = require('../src/utils/async');

describe('fetchUserById', () => {
  let mockDb;

  beforeEach(() => {
    mockDb = { findById: jest.fn() }; // fresh mock before each test
  });

  it('returns user when found', async () => {
    const user = { id: 1, name: 'Alice' };
    mockDb.findById.mockResolvedValue(user);

    const result = await fetchUserById(1, mockDb);
    expect(result).toEqual(user);
    expect(mockDb.findById).toHaveBeenCalledWith(1);
  });

  it('throws when user not found', async () => {
    mockDb.findById.mockResolvedValue(null);
    await expect(fetchUserById(1, mockDb)).rejects.toThrow('User 1 not found');
  });

  it('throws on invalid ID', async () => {
    await expect(fetchUserById(-1, mockDb)).rejects.toThrow('positive integer');
    await expect(fetchUserById('abc', mockDb)).rejects.toThrow('positive integer');
    expect(mockDb.findById).not.toHaveBeenCalled();
  });
});

describe('retryAsync', () => {
  it('returns result on first success', async () => {
    const fn = jest.fn().mockResolvedValue('success');
    const result = await retryAsync(fn);
    expect(result).toBe('success');
    expect(fn).toHaveBeenCalledTimes(1);
  });

  it('retries on failure and succeeds eventually', async () => {
    const fn = jest.fn()
      .mockRejectedValueOnce(new Error('fail 1'))
      .mockRejectedValueOnce(new Error('fail 2'))
      .mockResolvedValue('success on 3rd');

    const result = await retryAsync(fn, 3);
    expect(result).toBe('success on 3rd');
    expect(fn).toHaveBeenCalledTimes(3);
  });

  it('throws after max retries', async () => {
    const fn = jest.fn().mockRejectedValue(new Error('always fails'));
    await expect(retryAsync(fn, 3)).rejects.toThrow('always fails');
    expect(fn).toHaveBeenCalledTimes(3);
  });
});
```
