## Goal

Build a blog API using strict test-driven development: write a failing test, implement the minimum code to make it pass, refactor, then move to the next test. All implementation is driven by tests.

## What to Build

- Posts: CRUD with slug, author, tags, published state
- Comments: add and list per post
- All business logic in service layer (easy to unit test)
- Aim for 90%+ coverage with the TDD approach
- Final coverage report in `coverage/`

### Step 1 — Write a failing test first

```js
// tests/unit/postsService.test.js  (written BEFORE postsService.js exists)
import { jest } from '@jest/globals';

jest.unstable_mockModule('../../src/models/Post.js', () => ({
  default: {
    create:           jest.fn(),
    find:             jest.fn(),
    findOne:          jest.fn(),
    findByIdAndUpdate: jest.fn(),
    findByIdAndDelete: jest.fn(),
  },
}));

const Post = (await import('../../src/models/Post.js')).default;
const { createPost, getPostBySlug, publishPost } = await import('../../src/services/postsService.js');

describe('postsService.createPost', () => {
  beforeEach(() => jest.clearAllMocks());

  it('creates a post with auto-generated slug', async () => {
    const input = { title: 'Hello World', body: 'Content', authorId: 'author123' };
    const saved = { _id: 'post1', ...input, slug: 'hello-world', published: false };
    Post.create.mockResolvedValueOnce(saved);

    const result = await createPost(input);

    expect(Post.create).toHaveBeenCalledWith(
      expect.objectContaining({ slug: 'hello-world' }),
    );
    expect(result.slug).toBe('hello-world');
  });

  it('throws if title is missing', async () => {
    await expect(createPost({ body: 'No title' })).rejects.toThrow(/title/i);
  });

  it('throws if slug already exists', async () => {
    Post.create.mockRejectedValueOnce(Object.assign(new Error('duplicate'), { code: 11000 }));
    await expect(createPost({ title: 'Dup', body: 'body', authorId: 'a' })).rejects.toThrow(/slug.*taken/i);
  });
});

describe('postsService.publishPost', () => {
  it('sets published=true and publishedAt', async () => {
    const updated = { _id: 'p1', published: true, publishedAt: new Date() };
    Post.findByIdAndUpdate.mockResolvedValueOnce(updated);

    const result = await publishPost('p1');
    expect(result.published).toBe(true);
    expect(Post.findByIdAndUpdate).toHaveBeenCalledWith(
      'p1',
      expect.objectContaining({ published: true, publishedAt: expect.any(Date) }),
      expect.any(Object),
    );
  });

  it('throws NotFoundError if post does not exist', async () => {
    Post.findByIdAndUpdate.mockResolvedValueOnce(null);
    await expect(publishPost('nonexistent')).rejects.toThrow(/not found/i);
  });
});
```

### Step 2 — Implement the minimum code to pass

```js
// src/services/postsService.js  (written AFTER tests are failing)
import Post from '../models/Post.js';
import { AppError } from '../lib/AppError.js';

const slugify = (title) =>
  title.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '');

export const createPost = async ({ title, body, authorId, tags = [] }) => {
  if (!title) throw new AppError('title is required', 422);
  if (!body)  throw new AppError('body is required', 422);

  const slug = slugify(title);

  try {
    return await Post.create({ title, body, authorId, tags, slug, published: false });
  } catch (err) {
    if (err.code === 11000) throw new AppError('Slug already taken — choose a different title', 409);
    throw err;
  }
};

export const getPostBySlug = async (slug) => {
  const post = await Post.findOne({ slug, published: true }).populate('authorId', 'name email');
  if (!post) throw new AppError('Post not found', 404);
  return post;
};

export const publishPost = async (id) => {
  const post = await Post.findByIdAndUpdate(
    id,
    { published: true, publishedAt: new Date() },
    { new: true },
  );
  if (!post) throw new AppError('Post not found', 404);
  return post;
};

export const listPosts = async ({ tag, page = 1, limit = 10 } = {}) => {
  const filter = { published: true };
  if (tag) filter.tags = tag;
  return Post.find(filter).sort({ publishedAt: -1 }).skip((page - 1) * limit).limit(Number(limit));
};

export const updatePost = async (id, data) => {
  if (data.title) data.slug = slugify(data.title);
  const post = await Post.findByIdAndUpdate(id, data, { new: true, runValidators: true });
  if (!post) throw new AppError('Post not found', 404);
  return post;
};

export const deletePost = async (id) => {
  const post = await Post.findByIdAndDelete(id);
  if (!post) throw new AppError('Post not found', 404);
};
```

### Step 3 — Integration tests (Red → Green)

```js
// tests/integration/posts.test.js
import request from 'supertest';
import app from '../../src/app.js';
import Post from '../../src/models/Post.js';
import { createTestUser, getAuthToken } from '../helpers/auth.js';

let token;
let userId;

beforeAll(async () => {
  ({ userId } = await createTestUser());
  token = await getAuthToken(userId);
});

describe('POST /api/posts', () => {
  it('creates a post', async () => {
    const res = await request(app)
      .post('/api/posts')
      .set('Authorization', `Bearer ${token}`)
      .send({ title: 'TDD is great', body: 'Here is why...' })
      .expect(201);

    expect(res.body.slug).toBe('tdd-is-great');
    expect(res.body.published).toBe(false);
  });

  it('returns 422 without title', async () => {
    await request(app)
      .post('/api/posts')
      .set('Authorization', `Bearer ${token}`)
      .send({ body: 'No title' })
      .expect(422);
  });
});

describe('PATCH /api/posts/:id/publish', () => {
  it('publishes a post', async () => {
    const post = await Post.create({ title: 'Draft', body: 'body', authorId: userId, slug: 'draft', published: false });

    const res = await request(app)
      .patch(`/api/posts/${post._id}/publish`)
      .set('Authorization', `Bearer ${token}`)
      .expect(200);

    expect(res.body.published).toBe(true);
    expect(res.body.publishedAt).toBeDefined();
  });
});
```
