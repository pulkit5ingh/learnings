## Goal
Implement a complete JWT authentication flow with access tokens, refresh tokens, and a Redis-based token blacklist.

## Background

### JSON Web Tokens
A JWT consists of three Base64URL-encoded parts separated by dots: `header.payload.signature`. The header specifies the algorithm (HS256, RS256, etc.). The payload contains claims — `sub` (subject/user ID), `iat` (issued at), `exp` (expiration). The signature is a keyed hash of the header and payload that verifies the token hasn't been tampered with.

JWTs are self-contained: the server can verify authenticity without a database lookup, which makes them stateless and fast. The tradeoff: you can't invalidate a JWT before it expires without additional infrastructure (like a blacklist).

### Access and Refresh Tokens
A short-lived access token (15 minutes) minimizes exposure if stolen. A long-lived refresh token (7–30 days) allows the client to obtain a new access token without re-authenticating. The refresh token is stored securely (HttpOnly cookie or secure storage) and exchanged at `/auth/refresh` for a new access token.

On logout, the refresh token should be invalidated. Since JWTs can't be "deleted," you maintain a blacklist. Redis is ideal: store the token's JTI (JWT ID) with a TTL matching the token's expiry. On each request, check if the JTI is in the blacklist.

### Token Rotation
On each `/auth/refresh` call, issue a new refresh token AND blacklist the old one. This prevents refresh token reuse attacks. If an attacker replays a stolen refresh token, the legitimate user's next refresh will also fail (since the original was blacklisted on rotation), triggering a forced re-login.

### 2. Token Service

```js
// src/services/token.service.js
const jwt = require('jsonwebtoken');
const { randomUUID } = require('node:crypto');
const redis = require('../config/redis');

const sign = (payload, secret, expiresIn) => {
  return jwt.sign(
    { ...payload, jti: randomUUID() }, // jti = JWT ID, unique per token
    secret,
    { expiresIn }
  );
};

const generateTokens = (userId) => {
  const accessToken = sign(
    { sub: userId, type: 'access' },
    process.env.ACCESS_TOKEN_SECRET,
    process.env.ACCESS_TOKEN_EXPIRY ?? '15m'
  );

  const refreshToken = sign(
    { sub: userId, type: 'refresh' },
    process.env.REFRESH_TOKEN_SECRET,
    process.env.REFRESH_TOKEN_EXPIRY ?? '7d'
  );

  return { accessToken, refreshToken };
};

const verifyAccess = (token) => {
  return jwt.verify(token, process.env.ACCESS_TOKEN_SECRET);
};

const verifyRefresh = (token) => {
  return jwt.verify(token, process.env.REFRESH_TOKEN_SECRET);
};

// Blacklist a token by its JTI until it expires
const blacklistToken = async (decoded) => {
  const now = Math.floor(Date.now() / 1000);
  const ttl = decoded.exp - now; // seconds until expiry
  if (ttl > 0) {
    await redis.set(`blacklist:${decoded.jti}`, '1', 'EX', ttl);
  }
};

const isBlacklisted = async (jti) => {
  const result = await redis.get(`blacklist:${jti}`);
  return result !== null;
};

module.exports = { generateTokens, verifyAccess, verifyRefresh, blacklistToken, isBlacklisted };
```

### 5. Auth Middleware

```js
// src/middleware/authenticate.js
const tokenService = require('../services/token.service');

const authenticate = async (req, res, next) => {
  const authHeader = req.headers['authorization'];

  if (!authHeader?.startsWith('Bearer ')) {
    return res.status(401).json({ error: 'Missing Bearer token' });
  }

  const token = authHeader.slice(7);

  try {
    const decoded = tokenService.verifyAccess(token);

    // Check blacklist
    if (await tokenService.isBlacklisted(decoded.jti)) {
      return res.status(401).json({ error: 'Token has been revoked' });
    }

    req.user = { id: decoded.sub };
    next();
  } catch (err) {
    if (err.name === 'TokenExpiredError') {
      return res.status(401).json({ error: 'Token expired', code: 'TOKEN_EXPIRED' });
    }
    if (err.name === 'JsonWebTokenError') {
      return res.status(401).json({ error: 'Invalid token' });
    }
    next(err);
  }
};

module.exports = authenticate;
```

### 6. Auth Routes

```js
// src/routes/auth.js
const router = require('express').Router();
const bcrypt = require('bcryptjs');
const userStore = require('../data/users');
const tokenService = require('../services/token.service');
const authenticate = require('../middleware/authenticate');

// Register
router.post('/register', async (req, res, next) => {
  try {
    const { name, email, password } = req.body;

    if (!name || !email || !password) {
      return res.status(400).json({ error: 'name, email, password required' });
    }

    if (userStore.findByEmail(email)) {
      return res.status(409).json({ error: 'Email already registered' });
    }

    const user = await userStore.create({ name, email, password });
    const tokens = tokenService.generateTokens(user.id);

    res.status(201).json({
      message: 'Registered successfully',
      user: { id: user.id, name: user.name, email: user.email },
      ...tokens,
    });
  } catch (err) { next(err); }
});

// Login
router.post('/login', async (req, res, next) => {
  try {
    const { email, password } = req.body;

    if (!email || !password) {
      return res.status(400).json({ error: 'email and password required' });
    }

    const user = userStore.findByEmail(email);

    // Constant-time comparison avoids timing attacks
    const passwordMatch = user ? await bcrypt.compare(password, user.password) : false;

    if (!user || !passwordMatch) {
      // Same error for missing user or wrong password — don't reveal which
      return res.status(401).json({ error: 'Invalid credentials' });
    }

    const tokens = tokenService.generateTokens(user.id);

    res.json({
      user: { id: user.id, name: user.name, email: user.email },
      ...tokens,
    });
  } catch (err) { next(err); }
});

// Refresh — token rotation
router.post('/refresh', async (req, res, next) => {
  try {
    const { refreshToken } = req.body;

    if (!refreshToken) {
      return res.status(400).json({ error: 'refreshToken required' });
    }

    let decoded;
    try {
      decoded = tokenService.verifyRefresh(refreshToken);
    } catch {
      return res.status(401).json({ error: 'Invalid or expired refresh token' });
    }

    if (decoded.type !== 'refresh') {
      return res.status(401).json({ error: 'Wrong token type' });
    }

    // Check if this refresh token was already used (or logout blacklisted it)
    if (await tokenService.isBlacklisted(decoded.jti)) {
      // Possible token reuse attack — log this
      console.warn(`[SECURITY] Refresh token reuse detected for user ${decoded.sub}`);
      return res.status(401).json({ error: 'Refresh token already used or revoked' });
    }

    // Blacklist the old refresh token (rotation)
    await tokenService.blacklistToken(decoded);

    // Issue new pair
    const tokens = tokenService.generateTokens(decoded.sub);
    res.json(tokens);
  } catch (err) { next(err); }
});

// Logout
router.post('/logout', authenticate, async (req, res, next) => {
  try {
    const token = req.headers['authorization'].slice(7);
    const decoded = tokenService.verifyAccess(token);
    await tokenService.blacklistToken(decoded);

    res.json({ message: 'Logged out successfully' });
  } catch (err) { next(err); }
});

// Protected test route
router.get('/me', authenticate, (req, res) => {
  const user = userStore.findById(req.user.id);
  if (!user) return res.status(404).json({ error: 'User not found' });
  res.json({ id: user.id, name: user.name, email: user.email });
});

module.exports = router;
```
