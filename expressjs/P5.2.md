# P5.2 — PM2 Production Setup

**Phase:** 5 — Performance & Caching  
**Topic:** Cluster mode, ecosystem.config.js, health checks, log management

## Goal

Configure an Express API for production with PM2 cluster mode, zero-downtime reloads, structured log management, and health check endpoints.

## What to Build

- `ecosystem.config.js` with cluster mode, env vars, and log paths
- Graceful shutdown on `SIGINT`/`SIGTERM` (drain in-flight requests, close DB)
- `GET /health` with liveness and readiness checks
- Node.js cluster module usage (understand before PM2 abstracts it)
- PM2 monitoring with `pm2 monit` and custom metrics via `@pm2/io`

module.exports = {
  apps: [
    {
      name: 'api',
      script: './src/server.js',
      interpreter: 'node',
      interpreter_args: '--experimental-vm-modules',

      // Cluster mode: spawn one worker per CPU core
      instances: 'max',          // or a fixed number e.g. 4

      // Environment

        NODE_ENV: 'development',
        PORT: 4000,

        PORT: 4000,

      // Logs

```js
// src/server.js
import app from './app.js';
import mongoose from 'mongoose';
import { redis } from './cache.js';
import { logger } from './lib/logger.js';

const PORT = process.env.PORT ?? 4000;
let server;

const start = async () => {
  await mongoose.connect(process.env.MONGODB_URI);
  logger.info('MongoDB connected');

  server = app.listen(PORT, () => {
    logger.info(`Worker ${process.pid} listening on :${PORT}`);

    // Signal PM2 that this worker is ready (used with wait_ready: true)
    if (process.send) process.send('ready');
  });
};

const shutdown = async (signal) => {
  logger.info(`${signal} received — starting graceful shutdown`);

  // Stop accepting new connections
  server.close(async () => {
    logger.info('HTTP server closed');

    await Promise.allSettled([
      mongoose.connection.close(),
      redis.quit(),
    ]);

    logger.info('All connections closed — exiting');
    process.exit(0);
  });

  // Force-kill if shutdown takes too long
  setTimeout(() => {
    logger.error('Forced shutdown after timeout');
    process.exit(1);
  }, 10_000);
};

process.on('SIGTERM', () => shutdown('SIGTERM'));
process.on('SIGINT',  () => shutdown('SIGINT'));

start().catch((err) => { logger.error('Startup failed', { err }); process.exit(1); });
```

```js
// src/routes/health.js
import { Router } from 'express';
import mongoose from 'mongoose';
import { redis } from '../cache.js';
import os from 'node:os';

const router = Router();

/**
 * Liveness probe — is the process alive?
 * Kubernetes/ECS calls this; if it fails, the container is restarted.
 */
router.get('/health/live', (_req, res) => res.json({ status: 'ok', pid: process.pid }));

/**
 * Readiness probe — is the process ready to serve traffic?
 * Load balancer removes the instance from rotation if this fails.
 */
router.get('/health/ready', async (_req, res) => {
  const checks = {};
  let healthy = true;

  // MongoDB
  try {
    await mongoose.connection.db.admin().ping();
    checks.mongodb = 'ok';
  } catch (err) {
    checks.mongodb = err.message;
    healthy = false;
  }

  // Redis
  try {
    await redis.ping();
    checks.redis = 'ok';
  } catch (err) {
    checks.redis = err.message;
    healthy = false;
  }

  res.status(healthy ? 200 : 503).json({
    status: healthy ? 'ok' : 'degraded',
    checks,
    pid:     process.pid,
    uptime:  process.uptime(),
    memory:  process.memoryUsage(),
    cpuLoad: os.loadavg(),
  });
});

export default router;
```

```js
// src/cluster.js — understand cluster BEFORE using PM2
import cluster from 'node:cluster';
import { cpus } from 'node:os';
import process from 'node:process';

const NUM_WORKERS = cpus().length;

if (cluster.isPrimary) {
  console.log(`Primary ${process.pid} started — forking ${NUM_WORKERS} workers`);

  for (let i = 0; i < NUM_WORKERS; i++) {
    cluster.fork();
  }

  cluster.on('exit', (worker, code, signal) => {
    console.log(`Worker ${worker.process.pid} died (${signal ?? code}) — restarting`);
    cluster.fork();
  });
} else {
  // Each worker runs the Express app independently
  const { default: app } = await import('./app.js');
  app.listen(process.env.PORT ?? 4000, () => {
    console.log(`Worker ${process.pid} listening`);
    process.send?.('ready');
  });
}
```

### PM2 commands

```bash
# Start in production cluster mode
pm2 start ecosystem.config.cjs --env production

# Zero-downtime reload (rolling restart, no dropped connections)
pm2 reload api

# Real-time monitoring dashboard
pm2 monit

# Persistent process list (survives server reboot)
pm2 save
pm2 startup   # generates init script

# Logs
pm2 logs api --lines 100
pm2 logs api --err    # errors only
```
