## Background

### Mongoose vs the MongoDB Driver
The native `mongodb` Node.js driver gives you full control but requires manual schema enforcement and validation. Mongoose adds a schema layer on top: you define the shape of your documents upfront, including field types, defaults, validators, and hooks (middleware). This makes models self-documenting and prevents invalid data from reaching the database.

Mongoose uses a connection model: you call `mongoose.connect()` once at startup, and all subsequent model operations use that shared connection. Mongoose internally pools connections, so you don't manage individual database connections.

### Schemas, Models, and Documents
A `Schema` is a blueprint. A `Model` is a constructor compiled from a schema. A `Document` is an instance of a model — it represents one MongoDB document and provides instance methods like `.save()`, `.toJSON()`, `.toObject()`.

Key schema options:
- `required: true` — field must be present
- `default: value` — applied on save if field is absent
- `unique: true` — creates a MongoDB unique index
- `index: true` — creates a regular index
- `trim: true` — removes leading/trailing whitespace
- `enum: [...]` — value must be in the list

### Virtuals, Methods, and Statics
- **Virtuals** are computed properties that exist on documents but aren't persisted to MongoDB. Useful for combining fields (e.g., `fullName` from `firstName` + `lastName`) or URL generation.
- **Instance methods** (`schema.methods.methodName`) are available on document instances.
- **Static methods** (`schema.statics.methodName`) are available on the Model class.

```js
// src/config/database.js
const mongoose = require('mongoose');

const connectDB = async () => {
  try {
    const conn = await mongoose.connect(process.env.MONGODB_URI, {
      // mongoose 7+ has sensible defaults; these are for clarity
    });
    console.log(`MongoDB connected: ${conn.connection.host}`);
  } catch (err) {
    console.error('MongoDB connection error:', err.message);
    process.exit(1);
  }
};

// Log connection events
mongoose.connection.on('disconnected', () => console.log('MongoDB disconnected'));
mongoose.connection.on('error', (err) => console.error('MongoDB error:', err));

module.exports = connectDB;
```

### 3. User Schema with Virtuals and Methods

```js
// src/models/User.js
const mongoose = require('mongoose');
const { Schema } = mongoose;

const userSchema = new Schema(
  {
    firstName: {
      type: String,
      required: [true, 'First name is required'],
      trim: true,
      maxlength: [50, 'First name cannot exceed 50 characters'],
    },
    lastName: {
      type: String,
      required: [true, 'Last name is required'],
      trim: true,
      maxlength: [50, 'Last name cannot exceed 50 characters'],
    },
    email: {
      type: String,
      required: [true, 'Email is required'],
      unique: true,
      lowercase: true,
      trim: true,
      match: [/^\S+@\S+\.\S+$/, 'Please provide a valid email'],
    },
    age: {
      type: Number,
      min: [0, 'Age cannot be negative'],
      max: [150, 'Age seems unrealistic'],
    },
    role: {
      type: String,
      enum: ['user', 'moderator', 'admin'],
      default: 'user',
    },
    isActive: {
      type: Boolean,
      default: true,
    },
    tags: {
      type: [String],
      default: [],
    },
  },
  {
    timestamps: true, // adds createdAt and updatedAt automatically
    toJSON: { virtuals: true },
    toObject: { virtuals: true },
  }
);

// --- Indexes ---
userSchema.index({ email: 1 }); // compound or additional indexes here

// --- Virtual: fullName ---
userSchema.virtual('fullName').get(function () {
  return `${this.firstName} ${this.lastName}`;
});

// --- Virtual: profileUrl ---
userSchema.virtual('profileUrl').get(function () {
  return `/users/${this._id}`;
});

// --- Instance method: check if admin ---
userSchema.methods.isAdmin = function () {
  return this.role === 'admin';
};

// --- Instance method: summarize ---
userSchema.methods.toSummary = function () {
  return {
    id: this._id,
    fullName: this.fullName,
    email: this.email,
    role: this.role,
  };
};

// --- Static: find active users ---
userSchema.statics.findActive = function (filter = {}) {
  return this.find({ ...filter, isActive: true });
};

// --- Static: find by email ---
userSchema.statics.findByEmail = function (email) {
  return this.findOne({ email: email.toLowerCase() });
};

// --- Pre-save hook: normalize tags ---
userSchema.pre('save', function (next) {
  if (this.isModified('tags')) {
    this.tags = [...new Set(this.tags.map(t => t.toLowerCase().trim()))];
  }
  next();
});

// --- toJSON transform: remove __v, rename _id to id ---
userSchema.set('toJSON', {
  virtuals: true,
  transform(doc, ret) {
    ret.id = ret._id;
    delete ret._id;
    delete ret.__v;
    return ret;
  },
});

const User = mongoose.model('User', userSchema);
module.exports = User;
```

### 4. CRUD Operations

```js
// src/routes/users.js
const router = require('express').Router();
const User = require('../models/User');

// GET all users
router.get('/', async (req, res, next) => {
  try {
    const { role, isActive, sort = '-createdAt' } = req.query;
    const filter = {};
    if (role) filter.role = role;
    if (isActive !== undefined) filter.isActive = isActive === 'true';

    const users = await User.find(filter).sort(sort).lean();
    res.json({ count: users.length, data: users });
  } catch (err) {
    next(err);
  }
});

// GET one user
router.get('/:id', async (req, res, next) => {
  try {
    const user = await User.findById(req.params.id);
    if (!user) return res.status(404).json({ error: 'User not found' });
    res.json(user); // toJSON transform applies automatically
  } catch (err) {
    next(err);
  }
});

// POST create
router.post('/', async (req, res, next) => {
  try {
    const user = new User(req.body);
    await user.save(); // triggers pre-save hooks and validation
    res.status(201).json(user);
  } catch (err) {
    if (err.name === 'ValidationError') {
      return res.status(400).json({
        error: 'Validation failed',
        details: Object.values(err.errors).map(e => e.message),
      });
    }
    if (err.code === 11000) {
      return res.status(409).json({ error: 'Email already exists' });
    }
    next(err);
  }
});

// PATCH update
router.patch('/:id', async (req, res, next) => {
  try {
    const user = await User.findByIdAndUpdate(
      req.params.id,
      req.body,
      { new: true, runValidators: true } // return updated doc, run validators
    );
    if (!user) return res.status(404).json({ error: 'User not found' });
    res.json(user);
  } catch (err) {
    next(err);
  }
});

// DELETE
router.delete('/:id', async (req, res, next) => {
  try {
    const user = await User.findByIdAndDelete(req.params.id);
    if (!user) return res.status(404).json({ error: 'User not found' });
    res.json({ message: 'Deleted', user });
  } catch (err) {
    next(err);
  }
});

// Use static method
router.get('/active/all', async (req, res, next) => {
  try {
    const users = await User.findActive();
    res.json({ count: users.length, data: users.map(u => u.toSummary()) });
  } catch (err) {
    next(err);
  }
});

module.exports = router;
```
