## Goal
Integrate Stripe for one-time payments and subscription billing, handle payment webhooks, and use idempotency keys to prevent duplicate charges.

## Background

### Stripe's Payment Flow
Stripe uses a server-side API with client-side elements. The recommended flow:

1. Server creates a `PaymentIntent` (one-time) or `CheckoutSession`
2. Client confirms payment using Stripe.js / Elements (PCI-compliant card collection happens on Stripe's servers)
3. Stripe processes payment and sends a webhook event to your server
4. Your server updates order status on `payment_intent.succeeded`

Never process raw card numbers yourself — always use Stripe Elements or Stripe Checkout. This keeps you out of PCI scope.

### Idempotency Keys
If a network error occurs during a Stripe API call, you don't know if the charge was created. Retrying might create a duplicate charge. Stripe supports idempotency keys: send the same key with the same request, and Stripe returns the same result without creating a duplicate. Generate idempotency keys per user-action: a UUID stored with the order before the Stripe call.

### Subscriptions
Stripe subscriptions consist of:
- **Product**: what you're selling (e.g., "Pro Plan")
- **Price**: billing interval and amount (e.g., $29/month)
- **Subscription**: a Customer + Price combination with automatic renewal

Stripe sends `invoice.payment_succeeded` when subscription renews, `customer.subscription.deleted` when cancelled.

### 3. Payment Service

```js
// src/services/payment.service.js
const stripe = require('../config/stripe');
const { v4: uuidv4 } = require('uuid');

// Create a Stripe Checkout Session (hosted payment page)
const createCheckoutSession = async ({ userId, items, successUrl, cancelUrl }) => {
  const idempotencyKey = uuidv4(); // Store this with the order

  const lineItems = items.map(item => ({
    price_data: {
      currency: 'usd',
      product_data: {
        name: item.name,
        description: item.description,
        images: item.imageUrl ? [item.imageUrl] : [],
      },
      unit_amount: Math.round(item.price * 100), // cents
    },
    quantity: item.quantity,
  }));

  const session = await stripe.checkout.sessions.create({
    mode: 'payment',
    line_items: lineItems,
    success_url: `${successUrl}?session_id={CHECKOUT_SESSION_ID}`,
    cancel_url: cancelUrl,
    client_reference_id: userId.toString(),
    metadata: {
      userId: userId.toString(),
      idempotencyKey,
    },
  }, {
    idempotencyKey, // Stripe idempotency key
  });

  return { sessionId: session.id, url: session.url, idempotencyKey };
};

// Create a PaymentIntent (for custom checkout UI)
const createPaymentIntent = async ({ amount, currency = 'usd', userId, orderId }) => {
  const idempotencyKey = `pi-${orderId}-${userId}`;

  const paymentIntent = await stripe.paymentIntents.create({
    amount: Math.round(amount * 100),
    currency,
    automatic_payment_methods: { enabled: true },
    metadata: { userId: userId.toString(), orderId },
  }, {
    idempotencyKey,
  });

  return {
    clientSecret: paymentIntent.client_secret,
    paymentIntentId: paymentIntent.id,
  };
};

// Create a Stripe Customer
const createCustomer = async ({ name, email, userId }) => {
  const customer = await stripe.customers.create({
    name,
    email,
    metadata: { userId: userId.toString() },
  });
  return customer;
};

// Create a subscription
const createSubscription = async ({ customerId, priceId, trialDays = 0 }) => {
  const params = {
    customer: customerId,
    items: [{ price: priceId }],
    payment_behavior: 'default_incomplete',
    payment_settings: { save_default_payment_method: 'on_subscription' },
    expand: ['latest_invoice.payment_intent'],
  };

  if (trialDays > 0) {
    params.trial_period_days = trialDays;
  }

  const subscription = await stripe.subscriptions.create(params);

  return {
    subscriptionId: subscription.id,
    clientSecret: subscription.latest_invoice.payment_intent?.client_secret,
    status: subscription.status,
  };
};

const cancelSubscription = async (subscriptionId) => {
  return stripe.subscriptions.cancel(subscriptionId);
};

module.exports = {
  createCheckoutSession,
  createPaymentIntent,
  createCustomer,
  createSubscription,
  cancelSubscription,
};
```

### 4. Payment Routes

```js
// src/routes/payments.js
const router = require('express').Router();
const express = require('express');
const stripe = require('../config/stripe');
const paymentService = require('../services/payment.service');

// Create checkout session
router.post('/checkout', async (req, res, next) => {
  try {
    const { items } = req.body;
    const userId = req.user?.id ?? 1;

    if (!items?.length) {
      return res.status(400).json({ error: 'items required' });
    }

    const session = await paymentService.createCheckoutSession({
      userId,
      items,
      successUrl: `${process.env.CLIENT_URL}/success`,
      cancelUrl: `${process.env.CLIENT_URL}/cancel`,
    });

    res.json({ url: session.url, sessionId: session.sessionId });
  } catch (err) { next(err); }
});

// Create payment intent (for Stripe Elements)
router.post('/payment-intent', async (req, res, next) => {
  try {
    const { amount, orderId } = req.body;
    const userId = req.user?.id ?? 1;

    const result = await paymentService.createPaymentIntent({ amount, userId, orderId });
    res.json(result);
  } catch (err) { next(err); }
});

// Subscribe
router.post('/subscribe', async (req, res, next) => {
  try {
    const { email, name, priceId } = req.body;
    const userId = req.user?.id ?? 1;

    // Create or retrieve Stripe customer
    const customer = await paymentService.createCustomer({ name, email, userId });

    const subscription = await paymentService.createSubscription({
      customerId: customer.id,
      priceId,
      trialDays: 14,
    });

    res.json(subscription);
  } catch (err) { next(err); }
});

// WEBHOOK — must use raw body
router.post('/webhook',
  express.raw({ type: 'application/json' }),
  (req, res) => {
    let event;

    try {
      event = stripe.webhooks.constructEvent(
        req.body,
        req.headers['stripe-signature'],
        process.env.STRIPE_WEBHOOK_SECRET
      );
    } catch (err) {
      console.error(`Webhook signature error: ${err.message}`);
      return res.status(400).send(`Webhook error: ${err.message}`);
    }

    console.log(`Stripe event: ${event.type} [${event.id}]`);

    // Handle events
    switch (event.type) {
      case 'checkout.session.completed': {
        const session = event.data.object;
        console.log(`Checkout complete: ${session.id}, customer: ${session.customer_email}`);
        // TODO: fulfill order, send confirmation email
        break;
      }
      case 'payment_intent.succeeded': {
        const pi = event.data.object;
        console.log(`Payment succeeded: ${pi.id}, amount: ${pi.amount / 100} ${pi.currency.toUpperCase()}`);
        // TODO: update order status to 'paid'
        break;
      }
      case 'payment_intent.payment_failed': {
        const pi = event.data.object;
        console.log(`Payment failed: ${pi.id}, reason: ${pi.last_payment_error?.message}`);
        // TODO: notify user
        break;
      }
      case 'invoice.payment_succeeded': {
        const invoice = event.data.object;
        console.log(`Subscription renewed: ${invoice.subscription}`);
        // TODO: extend subscription period in your DB
        break;
      }
      case 'customer.subscription.deleted': {
        const subscription = event.data.object;
        console.log(`Subscription cancelled: ${subscription.id}`);
        // TODO: downgrade user
        break;
      }
      default:
        console.log(`Unhandled event: ${event.type}`);
    }

    res.json({ received: true });
  }
);

module.exports = router;
```
