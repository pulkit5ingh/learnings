# A7.4 — Integration and E2E Tests

**Phase:** 7 — Testing
**Topic:** Real database in tests, test data factories, test isolation, CI test setup
**Type:** Assignment

---

## Goal
Write integration tests that use a real (in-memory) database through the full stack, build test factories for consistent seed data, and configure CI to run tests automatically.

## Background

### Integration vs Unit Tests
Unit tests test a single function in isolation — fast but test each piece individually. Integration tests test multiple layers together (route → controller → service → database) — slower but verify the full flow works end-to-end.

A useful rule: unit tests for complex business logic (pricing, permissions), integration tests for API endpoints. If an integration test covers a simple endpoint fully, you don't need separate unit tests for the controller.

### Test Data Factories
Instead of copy-pasting `{ name: 'Alice', email: 'alice@test.com', ... }` in every test, define factory functions that generate valid objects with sensible defaults. Call the factory with overrides for specific test cases.

A factory for User might look like:
```js
const userFactory = (overrides = {}) => ({
  name: 'Test User',
  email: `user-${Date.now()}@test.com`, // unique email per call
  password: 'Password1',
  role: 'user',
  ...overrides,
});
```

Using unique emails (with timestamp or counter) prevents collision when multiple tests create users.

### CI Configuration
GitHub Actions runs your test suite automatically on every push and pull request. The workflow installs dependencies, starts any required services (Redis in this case), and runs `npm test`.

### 2. Test Factories

```js
// tests/factories/user.factory.js
const { faker } = require('@faker-js/faker');
const User = require('../../src/models/User');

const userFactory = (overrides = {}) => ({
  name: faker.person.fullName(),
  email: faker.internet.email().toLowerCase(),
  password: 'Password1',
  role: 'user',
  ...overrides,
});

// Creates user in DB and returns the document
const createUser = async (overrides = {}) => {
  return User.create(userFactory(overrides));
};

// Creates multiple users
const createUsers = async (count, overrides = {}) => {
  return Promise.all(Array.from({ length: count }, () => createUser(overrides)));
};

module.exports = { userFactory, createUser, createUsers };
```

```js
// tests/factories/post.factory.js
const { faker } = require('@faker-js/faker');
const Post = require('../../src/models/Post');

const postFactory = (userId, overrides = {}) => ({
  title: faker.lorem.sentence(),
  body: faker.lorem.paragraphs(2),
  category: faker.helpers.arrayElement(['tech', 'lifestyle', 'news']),
  tags: faker.helpers.arrayElements(['js', 'node', 'express', 'api', 'web'], 2),
  authorId: userId,
  published: true,
  ...overrides,
});

const createPost = async (userId, overrides = {}) => {
  return Post.create(postFactory(userId, overrides));
};

module.exports = { postFactory, createPost };
```

### 3. Integration Test: Auth Flow

```js
// tests/integration/auth.test.js
const request = require('supertest');
const app = require('../../src/app');
const { createUser } = require('../factories/user.factory');

describe('Auth Integration', () => {
  describe('POST /api/auth/register', () => {
    it('registers a new user and returns tokens', async () => {
      const res = await request(app)
        .post('/api/auth/register')
        .send({
          name: 'Test User',
          email: 'test@integration.com',
          password: 'Password1',
        });

      expect(res.status).toBe(201);
      expect(res.body.accessToken).toBeDefined();
      expect(res.body.refreshToken).toBeDefined();
      expect(res.body.user.password).toBeUndefined();
    });

    it('prevents duplicate email registration', async () => {
      await createUser({ email: 'taken@test.com' });

      const res = await request(app)
        .post('/api/auth/register')
        .send({ name: 'B', email: 'taken@test.com', password: 'Password1' });

      expect(res.status).toBe(409);
    });
  });

  describe('POST /api/auth/login', () => {
    it('returns tokens for valid credentials', async () => {
      await createUser({ email: 'login@test.com' });
      // Note: createUser hashes the password via pre-save hook

      const res = await request(app)
        .post('/api/auth/login')
        .send({ email: 'login@test.com', password: 'Password1' });

      expect(res.status).toBe(200);
      expect(res.body.accessToken).toBeDefined();
    });

    it('returns 401 for wrong password', async () => {
      await createUser({ email: 'wrong@test.com' });

      const res = await request(app)
        .post('/api/auth/login')
        .send({ email: 'wrong@test.com', password: 'WrongPassword' });

      expect(res.status).toBe(401);
    });
  });

  describe('Protected route flow', () => {
    it('can access protected route with valid token', async () => {
      // Register → get token → access protected route
      const registerRes = await request(app)
        .post('/api/auth/register')
        .send({ name: 'Protected', email: 'prot@test.com', password: 'Password1' });

      const { accessToken } = registerRes.body;

      const res = await request(app)
        .get('/api/auth/me')
        .set('Authorization', `Bearer ${accessToken}`);

      expect(res.status).toBe(200);
      expect(res.body.email).toBe('prot@test.com');
    });

    it('rejects request with no token', async () => {
      const res = await request(app).get('/api/auth/me');
      expect(res.status).toBe(401);
    });

    it('rejects expired token', async () => {
      // Sign a token with very short expiry
      const jwt = require('jsonwebtoken');
      const expiredToken = jwt.sign(
        { sub: 1, type: 'access', jti: 'test-jti' },
        process.env.ACCESS_TOKEN_SECRET ?? 'test-secret',
        { expiresIn: '-1s' } // already expired
      );

      const res = await request(app)
        .get('/api/auth/me')
        .set('Authorization', `Bearer ${expiredToken}`);

      expect(res.status).toBe(401);
      expect(res.body.code).toBe('TOKEN_EXPIRED');
    });
  });
});
```

### 4. Integration Test: Posts with Auth

```js
// tests/integration/posts.test.js
const request = require('supertest');
const app = require('../../src/app');
const { createUser } = require('../factories/user.factory');
const { createPost } = require('../factories/post.factory');

let userToken, userId, adminToken;

// Create test users once for the suite
beforeAll(async () => {
  const user = await createUser({ email: 'user@posts.test' });
  userId = user._id;
  const loginRes = await request(app)
    .post('/api/auth/login')
    .send({ email: 'user@posts.test', password: 'Password1' });
  userToken = loginRes.body.accessToken;

  const admin = await createUser({ email: 'admin@posts.test', role: 'admin' });
  const adminRes = await request(app)
    .post('/api/auth/login')
    .send({ email: 'admin@posts.test', password: 'Password1' });
  adminToken = adminRes.body.accessToken;
});

describe('GET /api/posts', () => {
  it('returns paginated posts', async () => {
    await createPost(userId);
    await createPost(userId);
    await createPost(userId);

    const res = await request(app).get('/api/posts?limit=2&page=1');
    expect(res.status).toBe(200);
    expect(res.body.data.length).toBe(2);
    expect(res.body.meta.totalPages).toBeGreaterThanOrEqual(2);
  });
});

describe('POST /api/posts', () => {
  it('creates a post when authenticated', async () => {
    const res = await request(app)
      .post('/api/posts')
      .set('Authorization', `Bearer ${userToken}`)
      .send({ title: 'My Post', body: 'Content here', category: 'tech' });

    expect(res.status).toBe(201);
    expect(res.body.title).toBe('My Post');
  });

  it('returns 401 without authentication', async () => {
    const res = await request(app)
      .post('/api/posts')
      .send({ title: 'My Post', body: 'Content', category: 'tech' });

    expect(res.status).toBe(401);
  });
});

describe('Authorization', () => {
  it('user cannot delete another users post', async () => {
    const otherUser = await createUser({ email: 'other@test.com' });
    const post = await createPost(otherUser._id);

    const res = await request(app)
      .delete(`/api/posts/${post._id}`)
      .set('Authorization', `Bearer ${userToken}`);

    expect(res.status).toBe(403);
  });

  it('admin can delete any post', async () => {
    const post = await createPost(userId);

    const res = await request(app)
      .delete(`/api/posts/${post._id}`)
      .set('Authorization', `Bearer ${adminToken}`);

    expect(res.status).toBe(200);
  });
});
```

### 5. GitHub Actions CI

```yaml
# .github/workflows/test.yml
name: Test

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest

    services:
      redis:
        image: redis:7-alpine
        ports:
          - 6379:6379
        options: >-
          --health-cmd "redis-cli ping"
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Install dependencies
        run: npm ci

      - name: Run tests
        run: npm test -- --coverage --runInBand
        env:
          NODE_ENV: test
          ACCESS_TOKEN_SECRET: ci-test-access-secret
          REFRESH_TOKEN_SECRET: ci-test-refresh-secret
          REDIS_URL: redis://localhost:6379

      - name: Upload coverage
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: coverage-report
          path: coverage/
```
