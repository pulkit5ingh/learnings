## Goal

## Background

### Why Router Modules?
As an application grows, putting all routes in a single `app.js` file becomes unmanageable. Express provides `express.Router()` — a mini-application that supports middleware and routing. A Router instance is itself middleware; you mount it on the main app with `app.use('/prefix', router)`.

This separation means route files only care about their own resource. The routes file defines paths and HTTP methods; it delegates the actual logic to controller functions. This is the Router + Controller pattern — analogous to the Model-View-Controller separation used in frameworks like Rails or Django.

### The Controller Pattern
A controller is a plain function (or object of functions) that handles a specific route's logic. Controllers should:
- Extract data from `req` (params, body, query)
- Call service/model layer functions
- Send the response or call `next(err)`

Controllers should NOT directly interact with the database. That belongs in a service layer. For now, we'll keep it simple with inline logic, but we'll separate services properly in Phase 2.

### Nested Routers
You can mount routers inside other routers. This is useful for resources that live under a parent: `/users/:userId/posts/:postId/comments`. Express merges route params from parent routers if you use `{ mergeParams: true }` when creating the child router.

### 3. Users Controller

```js
// src/controllers/users.controller.js
const store = require('../data/store');

const getAllUsers = (req, res) => {
  res.json({ data: store.users });
};

const getUserById = (req, res) => {
  const id = parseInt(req.params.id, 10);
  const user = store.users.find(u => u.id === id);

  if (!user) {
    return res.status(404).json({ error: `User ${id} not found` });
  }

  res.json(user);
};

const createUser = (req, res) => {
  const { name, email } = req.body;

  if (!name || !email) {
    return res.status(400).json({ error: 'name and email are required' });
  }

  const user = { id: store.nextUserId++, name, email };
  store.users.push(user);

  res.status(201).json(user);
};

const updateUser = (req, res) => {
  const id = parseInt(req.params.id, 10);
  const index = store.users.findIndex(u => u.id === id);

  if (index === -1) {
    return res.status(404).json({ error: `User ${id} not found` });
  }

  store.users[index] = { ...store.users[index], ...req.body, id };
  res.json(store.users[index]);
};

const deleteUser = (req, res) => {
  const id = parseInt(req.params.id, 10);
  const index = store.users.findIndex(u => u.id === id);

  if (index === -1) {
    return res.status(404).json({ error: `User ${id} not found` });
  }

  const [deleted] = store.users.splice(index, 1);
  res.json({ message: 'Deleted', user: deleted });
};

module.exports = { getAllUsers, getUserById, createUser, updateUser, deleteUser };
```

### 4. Posts Controller (with userId param)

```js
// src/controllers/posts.controller.js
const store = require('../data/store');

// GET /users/:userId/posts
const getUserPosts = (req, res) => {
  const userId = parseInt(req.params.userId, 10);
  const user = store.users.find(u => u.id === userId);

  if (!user) {
    return res.status(404).json({ error: `User ${userId} not found` });
  }

  const posts = store.posts.filter(p => p.userId === userId);
  res.json({ data: posts, user });
};

// GET /users/:userId/posts/:postId
const getPost = (req, res) => {
  const userId = parseInt(req.params.userId, 10);
  const postId = parseInt(req.params.postId, 10);

  const post = store.posts.find(p => p.id === postId && p.userId === userId);

  if (!post) {
    return res.status(404).json({ error: 'Post not found' });
  }

  res.json(post);
};

// POST /users/:userId/posts
const createPost = (req, res) => {
  const userId = parseInt(req.params.userId, 10);
  const user = store.users.find(u => u.id === userId);

  if (!user) {
    return res.status(404).json({ error: `User ${userId} not found` });
  }

  const { title, body } = req.body;
  if (!title || !body) {
    return res.status(400).json({ error: 'title and body are required' });
  }

  const post = { id: store.nextPostId++, userId, title, body };
  store.posts.push(post);

  res.status(201).json(post);
};

module.exports = { getUserPosts, getPost, createPost };
```

### 5. Route Files

```js
// src/routes/users.js
const router = require('express').Router();
const postsRouter = require('./posts');
const c = require('../controllers/users.controller');

// Route-level middleware example: log user routes
router.use((req, res, next) => {
  console.log(`[Users] ${req.method} ${req.path}`);
  next();
});

router.route('/')
  .get(c.getAllUsers)
  .post(c.createUser);

router.route('/:id')
  .get(c.getUserById)
  .patch(c.updateUser)
  .delete(c.deleteUser);

// Nest posts router under /users/:userId/posts
// mergeParams: true lets posts router access req.params.userId
router.use('/:userId/posts', postsRouter);

module.exports = router;
```

```js
// src/routes/posts.js
const router = require('express').Router({ mergeParams: true });
const c = require('../controllers/posts.controller');

router.route('/')
  .get(c.getUserPosts)
  .post(c.createPost);

router.route('/:postId')
  .get(c.getPost);

module.exports = router;
```
