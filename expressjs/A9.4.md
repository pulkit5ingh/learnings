# A9.4 — API Versioning & Swagger Documentation

**Phase:** 9 — Advanced Patterns  
**Topic:** swagger-jsdoc, swagger-ui-express, versioned routes, deprecation headers  
**Type:** Assignment

---

## Goal

Document a REST API with OpenAPI 3.0 using swagger-jsdoc JSDoc comments, serve interactive docs via swagger-ui-express, implement versioned routes (`/api/v1`, `/api/v2`), and communicate deprecation to clients.

## Background

- **OpenAPI 3.0** is the spec; **Swagger UI** renders it as an interactive browser page.
- **swagger-jsdoc** scans your source files for `@swagger` / `@openapi` JSDoc blocks and generates the JSON spec automatically — no separate YAML file to maintain.
- API versioning lets you ship breaking changes without forcing all clients to upgrade at once. The deprecation header (`Deprecation`, `Sunset`) gives clients advance notice.

## Tasks

### 1. Install dependencies

```bash
npm install swagger-jsdoc swagger-ui-express
```

### 2. Configure swagger-jsdoc

```js
// src/docs/swaggerConfig.js
import swaggerJsdoc from 'swagger-jsdoc';

const options = {
  definition: {
    openapi: '3.0.3',
    info: {
      title: 'My Express API',
      version: '2.0.0',
      description: 'Production-grade REST API with versioning',
      contact: { name: 'API Support', email: 'api@example.com' },
      license: { name: 'MIT' },
    },
    servers: [
      { url: 'http://localhost:4000', description: 'Local dev' },
      { url: 'https://api.example.com', description: 'Production' },
    ],
    components: {
      securitySchemes: {
        bearerAuth: {
          type: 'http',
          scheme: 'bearer',
          bearerFormat: 'JWT',
        },
      },
      schemas: {
        Error: {
          type: 'object',
          properties: {
            type:     { type: 'string', example: 'https://example.com/errors/not-found' },
            title:    { type: 'string', example: 'Not Found' },
            status:   { type: 'integer', example: 404 },
            detail:   { type: 'string' },
            instance: { type: 'string' },
          },
        },
      },
    },
    security: [{ bearerAuth: [] }],
  },
  // Glob of all files containing @swagger JSDoc blocks
  apis: ['./src/routes/**/*.js', './src/models/**/*.js'],
};

export const swaggerSpec = swaggerJsdoc(options);
```

### 3. Serve Swagger UI

```js
// src/server.js
import express from 'express';
import swaggerUi from 'swagger-ui-express';
import { swaggerSpec } from './docs/swaggerConfig.js';

const app = express();
app.use(express.json());

// Interactive docs at /api-docs
app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerSpec, {
  customSiteTitle: 'My API Docs',
  swaggerOptions: { persistAuthorization: true },
}));

// Raw OpenAPI JSON (useful for code gen)
app.get('/api-docs.json', (_req, res) => res.json(swaggerSpec));
```

### 4. JSDoc annotations on route files

```js
// src/routes/v1/users.js
import { Router } from 'express';
const router = Router();

/**
 * @swagger
 * /api/v1/users:
 *   get:
 *     summary: List all users (v1)
 *     tags: [Users]
 *     deprecated: true
 *     parameters:
 *       - in: query
 *         name: page
 *         schema: { type: integer, default: 1 }
 *       - in: query
 *         name: limit
 *         schema: { type: integer, default: 20 }
 *     responses:
 *       200:
 *         description: Paginated user list
 *         content:
 *           application/json:
 *             schema:
 *               type: object
 *               properties:
 *                 data:
 *                   type: array
 *                   items:
 *                     $ref: '#/components/schemas/User'
 *                 meta:
 *                   type: object
 *                   properties:
 *                     total: { type: integer }
 *                     page:  { type: integer }
 *       401:
 *         description: Unauthorized
 *         content:
 *           application/json:
 *             schema: { $ref: '#/components/schemas/Error' }
 */
router.get('/', async (req, res) => {
  const { page = 1, limit = 20 } = req.query;
  const users = await User.find()
    .skip((page - 1) * limit)
    .limit(Number(limit));
  res.json({ data: users, meta: { page: Number(page), total: await User.countDocuments() } });
});

/**
 * @swagger
 * /api/v1/users/{id}:
 *   get:
 *     summary: Get user by ID (v1)
 *     tags: [Users]
 *     deprecated: true
 *     parameters:
 *       - in: path
 *         name: id
 *         required: true
 *         schema: { type: string }
 *     responses:
 *       200:
 *         description: User object
 *       404:
 *         description: User not found
 */
router.get('/:id', async (req, res) => {
  const user = await User.findById(req.params.id);
  if (!user) return res.status(404).json({ title: 'Not Found', status: 404 });
  res.json(user);
});

export default router;
```

### 5. V2 route — same resource, breaking changes

```js
// src/routes/v2/users.js — v2 uses cursor pagination and returns camelCase
/**
 * @swagger
 * /api/v2/users:
 *   get:
 *     summary: List all users (v2 — cursor pagination)
 *     tags: [Users]
 *     parameters:
 *       - in: query
 *         name: cursor
 *         schema: { type: string }
 *         description: Opaque cursor from previous response
 *       - in: query
 *         name: limit
 *         schema: { type: integer, default: 20 }
 *     responses:
 *       200:
 *         description: Users with cursor pagination
 */
router.get('/', async (req, res) => {
  const { cursor, limit = 20 } = req.query;
  const filter = cursor ? { _id: { $gt: cursor } } : {};
  const users = await User.find(filter).limit(Number(limit) + 1).sort({ _id: 1 });

  const hasMore = users.length > limit;
  const data = hasMore ? users.slice(0, -1) : users;
  const nextCursor = hasMore ? data.at(-1)._id : null;

  res.json({ data, pagination: { nextCursor, hasMore } });
});
```

### 6. Mount versioned routers

```js
// src/server.js
import v1UsersRouter from './routes/v1/users.js';
import v2UsersRouter from './routes/v2/users.js';

// v1 — deprecated
app.use('/api/v1/users', v1UsersRouter);
// v2 — current
app.use('/api/v2/users', v2UsersRouter);
```

### 7. Deprecation headers middleware

```js
// src/middleware/deprecation.js
/**
 * Attach RFC 8594 Deprecation and Sunset headers to all v1 responses.
 * Clients can detect these headers and display warnings.
 */
export const deprecateV1 = (_req, res, next) => {
  // Deprecation: HTTP date format
  res.set('Deprecation', 'Sat, 01 Jan 2026 00:00:00 GMT');
  // Sunset: the date v1 will be removed
  res.set('Sunset', 'Sun, 01 Jul 2026 00:00:00 GMT');
  // Link to migration guide
  res.set('Link', '</api/v2>; rel="successor-version"');
  next();
};

// Apply only to v1 routes
app.use('/api/v1', deprecateV1);
```

### 8. Changelog pattern — versioned changelog endpoint

```js
/**
 * @swagger
 * /api/changelog:
 *   get:
 *     summary: API version changelog
 *     tags: [Meta]
 *     security: []
 *     responses:
 *       200:
 *         description: List of versions and changes
 */
app.get('/api/changelog', (_req, res) => {
  res.json({
    versions: [
      {
        version: 'v2',
        released: '2025-01-01',
        status: 'current',
        changes: [
          'Cursor-based pagination replaces offset pagination',
          'Response fields renamed to camelCase',
          'Removed deprecated `username` field — use `handle` instead',
        ],
      },
      {
        version: 'v1',
        released: '2024-01-01',
        status: 'deprecated',
        sunset: '2026-07-01',
        changes: ['Initial release'],
      },
    ],
  });
});
```
