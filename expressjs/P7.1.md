## Goal

Build a contacts REST API with 85%+ test coverage using a three-layer test suite: unit tests (controllers/services), integration tests (Supertest + in-memory MongoDB), and e2e tests (real DB in CI).

## What to Build

- Contacts CRUD (name, email, phone, tags, company)
- Three test layers: unit, integration, e2e
- MSW (Mock Service Worker) to mock external HTTP calls
- GitHub Actions CI that runs tests on every PR
- Coverage report (lcov + HTML) published as CI artifact

// tests/setup.js
import { MongoMemoryServer } from 'mongodb-memory-server';
import mongoose from 'mongoose';

let mongoServer;

beforeAll(async () => {
  mongoServer = await MongoMemoryServer.create();
  await mongoose.connect(mongoServer.getUri());
});

afterAll(async () => {
  await mongoose.disconnect();
  await mongoServer.stop();
});

afterEach(async () => {
  // Clean all collections between tests
  const collections = mongoose.connection.collections;
  for (const col of Object.values(collections)) {
    await col.deleteMany({});
  }
});

// tests/unit/contactsService.test.js
import { jest } from '@jest/globals';

// Mock the model before importing the service
const mockSave = jest.fn();
const mockFind = jest.fn();
const mockFindById = jest.fn();
const mockFindByIdAndUpdate = jest.fn();
const mockFindByIdAndDelete = jest.fn();

jest.unstable_mockModule('../../src/models/Contact.js', () => ({
  default: {
    find: mockFind,
    findById: mockFindById,
    findByIdAndUpdate: mockFindByIdAndUpdate,
    findByIdAndDelete: mockFindByIdAndDelete,
    prototype: { save: mockSave },
  },
}));

const { createContact, getContactById } = await import('../../src/services/contactsService.js');

describe('contactsService', () => {
  beforeEach(() => jest.clearAllMocks());

  describe('createContact', () => {
    it('should create and return a contact', async () => {
      const input = { name: 'Alice', email: 'alice@example.com' };
      const saved  = { _id: '123', ...input };
      mockSave.mockResolvedValueOnce(saved);

      const result = await createContact(input);
      expect(result).toEqual(saved);
      expect(mockSave).toHaveBeenCalledTimes(1);
    });

    it('should throw if email is missing', async () => {
      await expect(createContact({ name: 'Bob' })).rejects.toThrow('email is required');
    });
  });
});

```js
// tests/integration/contacts.test.js
import request from 'supertest';
import app from '../../src/app.js';
import Contact from '../../src/models/Contact.js';

describe('Contacts API — integration', () => {
  describe('POST /api/contacts', () => {
    it('creates a contact and returns 201', async () => {
      const res = await request(app)
        .post('/api/contacts')
        .send({ name: 'Carol', email: 'carol@example.com', phone: '+15550001234' })
        .expect(201);

      expect(res.body).toMatchObject({ name: 'Carol', email: 'carol@example.com' });
      expect(res.body._id).toBeDefined();

      const inDB = await Contact.findById(res.body._id);
      expect(inDB).not.toBeNull();
    });

    it('returns 422 when email is missing', async () => {
      const res = await request(app)
        .post('/api/contacts')
        .send({ name: 'No Email' })
        .expect(422);

      expect(res.body.error).toMatch(/email/i);
    });
  });

  describe('GET /api/contacts', () => {
    it('returns paginated contacts', async () => {
      await Contact.insertMany([
        { name: 'A', email: 'a@a.com' },
        { name: 'B', email: 'b@b.com' },
      ]);

      const res = await request(app).get('/api/contacts?limit=1').expect(200);
      expect(res.body.data).toHaveLength(1);
      expect(res.body.meta.total).toBe(2);
    });
  });

  describe('DELETE /api/contacts/:id', () => {
    it('deletes existing contact', async () => {
      const { _id } = await Contact.create({ name: 'D', email: 'd@d.com' });
      await request(app).delete(`/api/contacts/${_id}`).expect(204);
      expect(await Contact.findById(_id)).toBeNull();
    });

    it('returns 404 for unknown ID', async () => {
      await request(app).delete('/api/contacts/000000000000000000000000').expect(404);
    });
  });
});

// tests/integration/contacts.msw.test.js — mock external enrichment API
import { setupServer } from 'msw/node';
import { http, HttpResponse } from 'msw';
import request from 'supertest';
import app from '../../src/app.js';

const server = setupServer(
  http.get('https://api.clearbit.com/v2/people/find', () =>
    HttpResponse.json({ name: { fullName: 'Alice Smith' }, company: { name: 'Acme' } }),
  ),
);

beforeAll(() => server.listen({ onUnhandledRequest: 'error' }));
afterAll(() => server.close());
afterEach(() => server.resetHandlers());

it('enriches contact data from Clearbit', async () => {
  const res = await request(app)
    .post('/api/contacts/enrich')
    .send({ email: 'alice@example.com' })
    .expect(200);

  expect(res.body.company).toBe('Acme');
});

# .github/workflows/test.yml
name: Tests

on:
  pull_request:
  push:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - uses: actions/setup-node@v4
        with:
          node-version: '22'
          cache: 'npm'

      - run: npm ci

      - run: npm test -- --coverage --ci
        env:
          NODE_ENV: test

      - name: Upload coverage report
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: coverage-report
          path: coverage/

      - name: Check coverage thresholds
        run: npm test -- --coverage --coverageThreshold='{"global":{"lines":85}}'
```
