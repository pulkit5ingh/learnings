## Goal
Consume external HTTP APIs reliably using axios with interceptors, retry logic, and handle incoming webhooks securely.

## Background

### axios vs fetch
Node.js 18+ includes the native `fetch` API. `axios` adds: request/response interceptors, automatic JSON parsing, timeout configuration, request cancellation, and a simpler error model (HTTP errors throw, not just network errors). For simple cases, `fetch` is fine. For production API clients with retries, auth headers, and logging, `axios` is worth the dependency.

### Interceptors
Axios interceptors run before every request (request interceptor) or after every response (response interceptor). Use them to:
- **Request**: inject auth headers, add correlation ID, log outgoing requests
- **Response**: log responses, transform error formats, trigger token refresh on 401

### Retry Logic
Network failures and 5xx errors are transient. Implement retry with exponential backoff: wait 1s, then 2s, then 4s before giving up. Only retry idempotent operations (GET, PUT, DELETE) — never blindly retry POST requests as they may create duplicates. `axios-retry` simplifies this.

### Webhook Handling
Webhooks are HTTP POST requests from external services (Stripe, GitHub, Twilio) to your server. Always verify webhook signatures — an HMAC-SHA256 hash of the payload using a shared secret. Without verification, anyone could send fake webhook events to your server.

### 2. Axios HTTP Client with Interceptors

```js
// src/config/httpClient.js
const axios = require('axios');
const axiosRetry = require('axios-retry').default;

const createHttpClient = ({
  baseURL,
  apiKey,
  timeout = 10_000,
  maxRetries = 3,
  headers = {},
}) => {
  const client = axios.create({
    baseURL,
    timeout,
    headers: {
      'Content-Type': 'application/json',
      'Accept': 'application/json',
      ...headers,
    },
  });

  // Retry on network errors and 5xx (not 4xx — those are client errors)
  axiosRetry(client, {
    retries: maxRetries,
    retryDelay: (retryCount) => {
      const delay = axiosRetry.exponentialDelay(retryCount);
      console.log(`Retry attempt ${retryCount}, waiting ${delay}ms...`);
      return delay;
    },
    retryCondition: (error) => {
      return axiosRetry.isNetworkOrIdempotentRequestError(error)
        || (error.response?.status >= 500 && error.response?.status < 600);
    },
    onRetry: (retryCount, error, requestConfig) => {
      console.warn(`Retrying ${requestConfig.method?.toUpperCase()} ${requestConfig.url} (attempt ${retryCount}): ${error.message}`);
    },
  });

  // Request interceptor: add auth header and request ID
  client.interceptors.request.use((config) => {
    if (apiKey) {
      config.headers['Authorization'] = `Bearer ${apiKey}`;
    }
    config.headers['X-Request-Id'] = require('node:crypto').randomUUID();
    config.metadata = { startTime: Date.now() };

    console.log(`[HTTP OUT] ${config.method?.toUpperCase()} ${config.baseURL}${config.url}`);
    return config;
  });

  // Response interceptor: log timing
  client.interceptors.response.use(
    (response) => {
      const duration = Date.now() - response.config.metadata.startTime;
      console.log(`[HTTP IN] ${response.status} ${response.config.url} (${duration}ms)`);
      return response;
    },
    (error) => {
      const duration = error.config?.metadata
        ? Date.now() - error.config.metadata.startTime
        : 0;

      if (error.response) {
        // HTTP error response (4xx, 5xx)
        console.error(`[HTTP ERR] ${error.response.status} ${error.config?.url} (${duration}ms)`);
      } else if (error.code === 'ECONNABORTED') {
        console.error(`[HTTP TIMEOUT] ${error.config?.url}`);
      } else {
        console.error(`[HTTP FAIL] ${error.message}`);
      }

      return Promise.reject(error);
    }
  );

  return client;
};

module.exports = { createHttpClient };
```

### 3. Weather API Wrapper

```js
// src/services/weather.service.js
const { createHttpClient } = require('../config/httpClient');

// Using OpenWeatherMap API (free tier)
const weatherClient = createHttpClient({
  baseURL: 'https://api.openweathermap.org/data/2.5',
  timeout: 5000,
  headers: {},
});

const getWeather = async (city) => {
  try {
    const { data } = await weatherClient.get('/weather', {
      params: {
        q: city,
        appid: process.env.OPENWEATHER_API_KEY,
        units: 'metric',
      },
    });

    return {
      city: data.name,
      country: data.sys.country,
      temperature: data.main.temp,
      feelsLike: data.main.feels_like,
      humidity: data.main.humidity,
      description: data.weather[0].description,
      icon: `https://openweathermap.org/img/wn/${data.weather[0].icon}@2x.png`,
    };
  } catch (err) {
    if (err.response?.status === 404) {
      throw Object.assign(new Error(`City '${city}' not found`), { statusCode: 404 });
    }
    if (err.response?.status === 401) {
      throw new Error('Invalid OpenWeather API key');
    }
    throw err;
  }
};

module.exports = { getWeather };
```

### 4. GitHub API Wrapper

```js
// src/services/github.service.js
const { createHttpClient } = require('../config/httpClient');

const githubClient = createHttpClient({
  baseURL: 'https://api.github.com',
  apiKey: process.env.GITHUB_TOKEN, // Personal access token
  timeout: 8000,
  headers: {
    'Accept': 'application/vnd.github+json',
    'X-GitHub-Api-Version': '2022-11-28',
  },
});

const getUserRepos = async (username) => {
  const { data } = await githubClient.get(`/users/${username}/repos`, {
    params: { sort: 'updated', per_page: 10 },
  });

  return data.map(repo => ({
    name: repo.name,
    description: repo.description,
    stars: repo.stargazers_count,
    language: repo.language,
    url: repo.html_url,
    updatedAt: repo.updated_at,
  }));
};

const getRepo = async (owner, repo) => {
  const { data } = await githubClient.get(`/repos/${owner}/${repo}`);
  return data;
};

module.exports = { getUserRepos, getRepo };
```

### 5. Webhook Handler

```js
// src/routes/webhooks.js
const router = require('express').Router();
const crypto = require('node:crypto');

// Stripe webhook signature verification
const verifyStripeSignature = (payload, signature, secret) => {
  const timestamp = signature.split(',')[0].replace('t=', '');
  const sigHash = signature.split(',').find(s => s.startsWith('v1=')).replace('v1=', '');

  const expectedSig = crypto
    .createHmac('sha256', secret)
    .update(`${timestamp}.${payload}`)
    .digest('hex');

  // Constant-time comparison prevents timing attacks
  const expectedBuf = Buffer.from(expectedSig, 'hex');
  const receivedBuf = Buffer.from(sigHash, 'hex');

  if (expectedBuf.length !== receivedBuf.length) return false;
  return crypto.timingSafeEqual(expectedBuf, receivedBuf);
};

// IMPORTANT: use express.raw() for Stripe — needs raw body
router.post('/stripe',
  express.raw({ type: 'application/json' }),
  (req, res) => {
    const signature = req.headers['stripe-signature'];

    if (!verifyStripeSignature(
      req.body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET
    )) {
      return res.status(400).json({ error: 'Invalid webhook signature' });
    }

    const event = JSON.parse(req.body);
    console.log(`Stripe webhook: ${event.type}`);

    switch (event.type) {
      case 'payment_intent.succeeded':
        handlePaymentSuccess(event.data.object);
        break;
      case 'payment_intent.payment_failed':
        handlePaymentFailed(event.data.object);
        break;
      case 'customer.subscription.deleted':
        handleSubscriptionCancelled(event.data.object);
        break;
      default:
        console.log(`Unhandled webhook event: ${event.type}`);
    }

    res.json({ received: true }); // always respond 200 quickly
  }
);

const handlePaymentSuccess = (paymentIntent) => {
  console.log(`Payment succeeded: ${paymentIntent.id}`);
  // Update order status, send confirmation email, etc.
};

const handlePaymentFailed = (paymentIntent) => {
  console.log(`Payment failed: ${paymentIntent.id}`);
  // Notify user, log failure
};

const handleSubscriptionCancelled = (subscription) => {
  console.log(`Subscription cancelled: ${subscription.id}`);
  // Downgrade user account
};

module.exports = router;
```

### 6. Routes

```js
// src/routes/external.js
const router = require('express').Router();
const { getWeather } = require('../services/weather.service');
const { getUserRepos } = require('../services/github.service');

router.get('/weather/:city', async (req, res, next) => {
  try {
    const data = await getWeather(req.params.city);
    res.json(data);
  } catch (err) {
    if (err.statusCode) return res.status(err.statusCode).json({ error: err.message });
    next(err);
  }
});

router.get('/github/:username/repos', async (req, res, next) => {
  try {
    const repos = await getUserRepos(req.params.username);
    res.json({ data: repos, count: repos.length });
  } catch (err) { next(err); }
});

module.exports = router;
```
