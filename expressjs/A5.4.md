## Background

### Why Queries Get Slow
Without an index, MongoDB performs a "collection scan" — it reads every document to find matches. On a collection with 1 million documents, a query without an index takes seconds. With the right index, the same query takes milliseconds. An index is a data structure (B-tree) that stores a sorted subset of document fields, enabling binary search.

MongoDB indexes are per-collection. Each query should use an index — if the explain plan shows `COLLSCAN` (collection scan) instead of `IXSCAN` (index scan), you need an index. However, indexes have costs: they slow down writes (each insert/update must update the index) and consume memory.

### explain() Plans
`collection.find(filter).explain('executionStats')` returns detailed information about how MongoDB executed the query:
- `winningPlan.stage`: `COLLSCAN` (slow) or `IXSCAN` (fast) or `FETCH`
- `executionStats.totalKeysExamined`: how many index keys were scanned
- `executionStats.totalDocsExamined`: how many documents were loaded
- `executionStats.executionTimeMillis`: total query time
- `executionStats.nReturned`: number of documents returned

Ideal ratio: `totalDocsExamined / nReturned` close to 1. If you examined 10,000 documents to return 5, the query is inefficient.

### Index Types
- **Single field**: `{ email: 1 }` — ascending index on email
- **Compound**: `{ category: 1, createdAt: -1 }` — must match query sort order
- **Text**: `{ title: 'text', body: 'text' }` — full-text search
- **Partial**: `{ status: 1 }, { partialFilterExpression: { active: true } }` — index only active documents

The **ESR rule** for compound indexes: Equality fields first, Sort fields second, Range fields last.

```js
// src/models/Order.js
const mongoose = require('mongoose');

const orderSchema = new mongoose.Schema({
  userId: { type: mongoose.Schema.Types.ObjectId, required: true },
  status: {
    type: String,
    enum: ['pending', 'processing', 'shipped', 'delivered', 'cancelled'],
    required: true,
  },
  total: { type: Number, required: true },
  items: [{
    productId: mongoose.Schema.Types.ObjectId,
    name: String,
    quantity: Number,
    price: Number,
  }],
  shippedAt: Date,
  deliveredAt: Date,
  cancelledAt: Date,
  region: String,
}, { timestamps: true });

// --- Indexes ---

// Equality first, sort second (ESR rule)
// Common query: find orders for a user, sorted by date
orderSchema.index({ userId: 1, createdAt: -1 });

// Equality first: status filter + date sort
orderSchema.index({ status: 1, createdAt: -1 });

// Compound: region + status + date (used in aggregations)
orderSchema.index({ region: 1, status: 1, createdAt: -1 });

// Partial index: only index non-cancelled orders (reduces index size)
orderSchema.index(
  { userId: 1, status: 1 },
  { partialFilterExpression: { status: { $ne: 'cancelled' } } }
);

module.exports = mongoose.model('Order', orderSchema);
```

### 3. Query Analysis

```js
// src/utils/queryAnalyzer.js
const analyzeQuery = async (queryBuilder, label = 'Query') => {
  const explained = await queryBuilder.explain('executionStats');
  const stats = explained.executionStats;
  const plan = explained.queryPlanner.winningPlan;

  const result = {
    label,
    stage: plan.stage,
    indexName: plan.inputStage?.indexName ?? 'none',
    executionTimeMs: stats.executionTimeMillis,
    totalDocsExamined: stats.totalDocsExamined,
    totalKeysExamined: stats.totalKeysExamined,
    nReturned: stats.nReturned,
    efficiency: stats.nReturned > 0
      ? (stats.nReturned / stats.totalDocsExamined).toFixed(2)
      : 0,
    isCollScan: plan.stage === 'COLLSCAN',
  };

  console.table([result]);

  if (result.isCollScan) {
    console.warn(`[SLOW QUERY] ${label} uses COLLSCAN — add an index!`);
  }

  return result;
};

module.exports = { analyzeQuery };
```

### 4. Routes with Profiling

```js
// src/routes/orders.js
const router = require('express').Router();
const Order = require('../models/Order');
const { analyzeQuery } = require('../utils/queryAnalyzer');

// Efficient: uses { userId, createdAt } compound index
router.get('/user/:userId', async (req, res, next) => {
  try {
    const { userId } = req.params;
    const { status, page = 1, limit = 20 } = req.query;

    const filter = { userId };
    if (status) filter.status = status;

    // Analyze the query
    if (process.env.EXPLAIN_QUERIES === 'true') {
      await analyzeQuery(
        Order.find(filter).sort({ createdAt: -1 }),
        `Orders for user ${userId}`
      );
    }

    const orders = await Order.find(filter)
      .sort({ createdAt: -1 })
      .skip((parseInt(page) - 1) * parseInt(limit))
      .limit(parseInt(limit))
      .lean();

    res.json({ data: orders });
  } catch (err) { next(err); }
});

// Analytics aggregation: revenue by region and status
router.get('/analytics/revenue', async (req, res, next) => {
  try {
    const { startDate, endDate } = req.query;

    const matchStage = { status: { $ne: 'cancelled' } };
    if (startDate) matchStage.createdAt = { $gte: new Date(startDate) };
    if (endDate) matchStage.createdAt = { ...(matchStage.createdAt ?? {}), $lte: new Date(endDate) };

    const revenue = await Order.aggregate([
      // Stage 1: filter (uses index)
      { $match: matchStage },

      // Stage 2: group by region + status
      {
        $group: {
          _id: { region: '$region', status: '$status' },
          orderCount: { $sum: 1 },
          totalRevenue: { $sum: '$total' },
          avgOrderValue: { $avg: '$total' },
          minOrder: { $min: '$total' },
          maxOrder: { $max: '$total' },
        },
      },

      // Stage 3: reshape
      {
        $project: {
          region: '$_id.region',
          status: '$_id.status',
          orderCount: 1,
          totalRevenue: { $round: ['$totalRevenue', 2] },
          avgOrderValue: { $round: ['$avgOrderValue', 2] },
          minOrder: 1,
          maxOrder: 1,
          _id: 0,
        },
      },

      // Stage 4: sort by revenue descending
      { $sort: { totalRevenue: -1 } },
    ]);

    res.json({ data: revenue });
  } catch (err) { next(err); }
});

// Top customers using $lookup (JOIN equivalent)
router.get('/analytics/top-customers', async (req, res, next) => {
  try {
    const topCustomers = await Order.aggregate([
      { $match: { status: 'delivered' } },
      {
        $group: {
          _id: '$userId',
          orderCount: { $sum: 1 },
          totalSpent: { $sum: '$total' },
          lastOrderDate: { $max: '$createdAt' },
        },
      },
      { $sort: { totalSpent: -1 } },
      { $limit: 10 },
      // $lookup to join with users (if they're in the same DB)
      {
        $lookup: {
          from: 'users',
          localField: '_id',
          foreignField: '_id',
          as: 'userInfo',
          pipeline: [{ $project: { name: 1, email: 1 } }],
        },
      },
      { $unwind: { path: '$userInfo', preserveNullAndEmpty: true } },
      {
        $project: {
          userId: '$_id',
          name: { $ifNull: ['$userInfo.name', 'Unknown'] },
          email: { $ifNull: ['$userInfo.email', 'Unknown'] },
          orderCount: 1,
          totalSpent: { $round: ['$totalSpent', 2] },
          lastOrderDate: 1,
          _id: 0,
        },
      },
    ]);

    res.json({ data: topCustomers });
  } catch (err) { next(err); }
});

module.exports = router;
```

```js
const mongoose = require('mongoose');
const Order = require('../src/models/Order');
```
