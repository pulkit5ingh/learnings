## Goal
Implement health check and metrics endpoints, configure Prometheus scraping, and handle OS signals for graceful shutdown.

## Background

### Health Checks
A health check endpoint (`GET /health`) tells load balancers, orchestrators (Kubernetes), and monitoring systems whether the application is healthy. Return 200 if healthy, 503 if degraded.

Two levels:
- **Liveness**: is the process running? (just return 200)
- **Readiness**: is the app ready to serve traffic? (check DB, Redis, external dependencies)

Kubernetes calls liveness probes to know if it should restart a pod. It calls readiness probes to know if a pod should receive traffic. Use separate endpoints for each.

### Prometheus Metrics
Prometheus is an open-source monitoring system that scrapes metrics from HTTP endpoints (`/metrics`). Metrics are time-series data: request rate, error rate, response latency, memory usage.

`prom-client` is the Node.js Prometheus client library. It provides default Node.js metrics (memory, CPU, event loop lag) and lets you define custom metrics (counters, gauges, histograms).

A counter counts things that only go up (total requests). A gauge measures a current value (active connections). A histogram measures distribution (response time buckets).

### Graceful Shutdown
When a process receives `SIGTERM` (from `docker stop`, `kubectl delete pod`, PM2 reload), it should:
1. Stop accepting new connections (`server.close()`)
2. Wait for in-flight requests to complete
3. Close database connections
4. Exit cleanly

Without graceful shutdown: in-flight requests are dropped mid-response, database writes may be incomplete.

### 2. Health Check Endpoint

```js
// src/routes/health.js
const router = require('express').Router();
const mongoose = require('mongoose');
const { redis } = require('../config/cache');

// Lightweight liveness check — just confirms process is running
router.get('/live', (req, res) => {
  res.status(200).json({ status: 'alive' });
});

// Readiness check — verifies all dependencies are reachable
router.get('/ready', async (req, res) => {
  const checks = {};
  let allHealthy = true;

  // MongoDB check
  try {
    const state = mongoose.connection.readyState;
    // 0=disconnected, 1=connected, 2=connecting, 3=disconnecting
    checks.mongodb = { status: state === 1 ? 'up' : 'down', state };
    if (state !== 1) allHealthy = false;
  } catch (err) {
    checks.mongodb = { status: 'down', error: err.message };
    allHealthy = false;
  }

  // Redis check
  try {
    await redis.ping();
    checks.redis = { status: 'up' };
  } catch (err) {
    checks.redis = { status: 'down', error: err.message };
    allHealthy = false;
  }

  const statusCode = allHealthy ? 200 : 503;

  res.status(statusCode).json({
    status: allHealthy ? 'ready' : 'not_ready',
    timestamp: new Date().toISOString(),
    uptime: process.uptime(),
    checks,
  });
});

// Combined health endpoint (used by simple health checkers)
router.get('/', async (req, res) => {
  res.json({
    status: 'ok',
    pid: process.pid,
    version: process.env.npm_package_version ?? '0.0.0',
    env: process.env.NODE_ENV,
    uptime: process.uptime(),
    memory: {
      rss: Math.round(process.memoryUsage().rss / 1024 / 1024) + 'MB',
      heapUsed: Math.round(process.memoryUsage().heapUsed / 1024 / 1024) + 'MB',
    },
  });
});

module.exports = router;
```

### 3. Prometheus Metrics

```js
// src/config/metrics.js
const client = require('prom-client');

// Collect default Node.js metrics (memory, CPU, event loop, GC)
const register = new client.Registry();
client.collectDefaultMetrics({ register, prefix: 'node_' });

// Custom metrics
const httpRequestsTotal = new client.Counter({
  name: 'http_requests_total',
  help: 'Total number of HTTP requests',
  labelNames: ['method', 'route', 'status'],
  registers: [register],
});

const httpRequestDuration = new client.Histogram({
  name: 'http_request_duration_seconds',
  help: 'Duration of HTTP requests in seconds',
  labelNames: ['method', 'route', 'status'],
  buckets: [0.001, 0.005, 0.01, 0.05, 0.1, 0.5, 1, 2, 5],
  registers: [register],
});

const activeConnections = new client.Gauge({
  name: 'http_active_connections',
  help: 'Number of active HTTP connections',
  registers: [register],
});

const dbQueryDuration = new client.Histogram({
  name: 'db_query_duration_seconds',
  help: 'Duration of database queries',
  labelNames: ['operation', 'collection'],
  buckets: [0.001, 0.01, 0.05, 0.1, 0.5, 1],
  registers: [register],
});

module.exports = {
  register,
  httpRequestsTotal,
  httpRequestDuration,
  activeConnections,
  dbQueryDuration,
};
```

### 4. Metrics Middleware and Endpoint

```js
// src/middleware/metricsMiddleware.js
const { httpRequestsTotal, httpRequestDuration, activeConnections } = require('../config/metrics');

let connections = 0;

const metricsMiddleware = (req, res, next) => {
  connections++;
  activeConnections.set(connections);

  const start = Date.now();

  res.on('finish', () => {
    connections--;
    activeConnections.set(connections);

    const duration = (Date.now() - start) / 1000;
    // Normalize route to avoid label cardinality explosion
    // /users/123 → /users/:id
    const route = req.route?.path ?? req.path ?? 'unknown';

    const labels = {
      method: req.method,
      route,
      status: res.statusCode.toString(),
    };

    httpRequestsTotal.inc(labels);
    httpRequestDuration.observe(labels, duration);
  });

  next();
};

module.exports = metricsMiddleware;
```

```js
// Add metrics endpoint to app.js
const { register } = require('./config/metrics');

// Metrics endpoint — restrict to internal access only in production
app.get('/metrics', async (req, res) => {
  // In production: verify request is from Prometheus server's IP
  if (process.env.NODE_ENV === 'production') {
    const allowedIPs = (process.env.METRICS_ALLOWED_IPS ?? '127.0.0.1').split(',');
    if (!allowedIPs.includes(req.ip)) {
      return res.status(403).json({ error: 'Forbidden' });
    }
  }

  res.set('Content-Type', register.contentType);
  res.end(await register.metrics());
});
```

### 5. Graceful Shutdown

```js
// src/server.js
require('dotenv').config();
const mongoose = require('mongoose');
const { redis } = require('./config/cache');
const app = require('./app');
const logger = require('./config/logger');

const PORT = process.env.PORT ?? 3000;

const server = app.listen(PORT, () => {
  logger.info(`Server started on port ${PORT}`);
});

// Track active connections for graceful shutdown
const connections = new Set();
server.on('connection', (conn) => {
  connections.add(conn);
  conn.on('close', () => connections.delete(conn));
});

const shutdown = async (signal) => {
  logger.info(`Received ${signal}, starting graceful shutdown...`);

  // Stop accepting new connections
  server.close(async () => {
    logger.info('HTTP server closed');
  });

  // Force-close keep-alive connections after 10s
  setTimeout(() => {
    logger.warn('Forcing connection closure after timeout');
    for (const conn of connections) conn.destroy();
  }, 10_000);

  try {
    // Close database connections
    await mongoose.disconnect();
    logger.info('MongoDB disconnected');

    await redis.quit();
    logger.info('Redis disconnected');

    logger.info('Graceful shutdown complete');
    process.exit(0);
  } catch (err) {
    logger.error('Error during shutdown:', err);
    process.exit(1);
  }
};

// Handle OS signals
process.on('SIGTERM', () => shutdown('SIGTERM')); // docker stop, k8s
process.on('SIGINT', () => shutdown('SIGINT'));   // Ctrl+C

// Catch unhandled errors
process.on('unhandledRejection', (err) => {
  logger.error('Unhandled rejection', { error: err.message });
  shutdown('unhandledRejection');
});

process.on('uncaughtException', (err) => {
  logger.error('Uncaught exception', { error: err.message, stack: err.stack });
  process.exit(1); // must exit immediately — state may be corrupted
});
```
