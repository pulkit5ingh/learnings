## Background

### The Event Loop and libuv
Node.js is not multi-threaded in the traditional sense. It runs JavaScript on a single thread using V8, but delegates I/O operations (file reads, network requests, timers) to libuv — a C library that manages a thread pool and OS-level async I/O. When an I/O operation completes, libuv places a callback in the event queue. The event loop checks this queue after each "tick" and executes waiting callbacks.

The event loop has distinct phases: timers (`setTimeout`/`setInterval`), I/O callbacks, idle/prepare, poll (waiting for I/O), check (`setImmediate`), and close callbacks. Understanding this order explains why `setImmediate` runs before `setTimeout(fn, 0)` in some contexts and not others.

`process.nextTick()` is not part of the event loop — it runs at the end of the current operation, before the event loop continues. This makes it useful for deferring callbacks while still running before any I/O. `Promise` callbacks (microtasks) also run before the next event loop tick.

### Streams
Streams are one of Node.js's most powerful abstractions. Instead of loading an entire file into memory, a Readable stream emits data in chunks. A Writable stream consumes chunks. A Transform stream reads, modifies, and writes. Streams implement the `EventEmitter` interface, emitting `data`, `end`, `error`, and `finish` events. The `.pipe()` method connects a Readable to a Writable, handling backpressure automatically. From Node.js 10+, you can `await pipeline(readable, transform, writable)` from `node:stream/promises` for cleaner async stream handling.

### Buffer and Encoding
`Buffer` represents binary data. When you read a file without specifying an encoding, you get a Buffer. Use `buffer.toString('utf8')` to decode it. Use `Buffer.from('hello', 'utf8')` to encode. Buffers are fixed-size allocations directly in C memory, outside V8's garbage collected heap. This matters for performance when processing binary protocols or large files.

### 2. EventEmitter

```js
// emitter.js
const { EventEmitter } = require('node:events');

class TaskQueue extends EventEmitter {
  #tasks = [];

  enqueue(task) {
    this.#tasks.push(task);
    this.emit('taskAdded', task);
  }

  async processAll() {
    for (const task of this.#tasks) {
      try {
        const result = await task();
        this.emit('taskComplete', result);
      } catch (err) {
        this.emit('taskError', err);
      }
    }
    this.#tasks = [];
    this.emit('drain');
  }
}

const queue = new TaskQueue();

queue.on('taskAdded', (task) => console.log('Task enqueued:', task.name));
queue.on('taskComplete', (result) => console.log('Complete:', result));
queue.on('taskError', (err) => console.error('Error:', err.message));
queue.on('drain', () => console.log('Queue drained'));

queue.enqueue(async function fetchUser() { return { id: 1, name: 'Alice' }; });
queue.enqueue(async function failingTask() { throw new Error('DB timeout'); });
queue.enqueue(async function computeSum() { return 1 + 1; });

queue.processAll();
```

### 3. Readable and Writable Streams

```js
// streams.js
const { Readable, Writable, Transform, pipeline } = require('node:stream');
const { pipeline: pipelineAsync } = require('node:stream/promises');

// Custom Readable: emits numbers 1 to N
class NumberStream extends Readable {
  constructor(max) {
    super({ objectMode: true });
    this.current = 1;
    this.max = max;
  }

  _read() {
    if (this.current <= this.max) {
      this.push(this.current++);
    } else {
      this.push(null); // signal end
    }
  }
}

// Custom Transform: square each number
class SquareTransform extends Transform {
  constructor() {
    super({ objectMode: true });
  }

  _transform(chunk, _encoding, callback) {
    this.push(chunk * chunk);
    callback();
  }
}

// Custom Writable: collect results
class CollectorWritable extends Writable {
  constructor() {
    super({ objectMode: true });
    this.results = [];
  }

  _write(chunk, _encoding, callback) {
    this.results.push(chunk);
    callback();
  }
}

async function main() {
  const source = new NumberStream(5);
  const square = new SquareTransform();
  const collector = new CollectorWritable();

  await pipelineAsync(source, square, collector);
  console.log('Results:', collector.results); // [1, 4, 9, 16, 25]
}

main().catch(console.error);
```

### 4. File System and Path Modules

```js
// fs-ops.js
const fs = require('node:fs/promises');
const path = require('node:path');
const os = require('node:os');

async function main() {
  // os module
  console.log('Platform:', os.platform());
  console.log('CPUs:', os.cpus().length);
  console.log('Free memory (MB):', Math.round(os.freemem() / 1024 / 1024));
  console.log('Temp dir:', os.tmpdir());

  // path module
  const filePath = path.join(os.tmpdir(), 'test-assignment', 'data.txt');
  console.log('Target path:', filePath);
  console.log('Dir:', path.dirname(filePath));
  console.log('Base:', path.basename(filePath));
  console.log('Ext:', path.extname(filePath));

  // fs operations
  await fs.mkdir(path.dirname(filePath), { recursive: true });
  await fs.writeFile(filePath, 'Hello, Node.js!\nLine two.\n', 'utf8');

  const content = await fs.readFile(filePath, 'utf8');
  console.log('File content:', content);

  const stats = await fs.stat(filePath);
  console.log('File size (bytes):', stats.size);
  console.log('Is file:', stats.isFile());

  // List directory
  const files = await fs.readdir(path.dirname(filePath));
  console.log('Directory contents:', files);

  await fs.rm(path.dirname(filePath), { recursive: true });
  console.log('Cleaned up.');
}

main().catch(console.error);
```

### 5. Buffer Manipulation

```js
// buffers.js

// Creating buffers
const buf1 = Buffer.from('Hello, World!', 'utf8');
const buf2 = Buffer.alloc(4); // zero-filled

// Writing to buffer
buf2.writeUInt32BE(0xDEADBEEF, 0);
console.log('Hex:', buf2.toString('hex')); // deadbeef
console.log('Base64:', buf1.toString('base64'));
console.log('Back to string:', Buffer.from(buf1.toString('base64'), 'base64').toString('utf8'));

// Slicing (shares memory)
const slice = buf1.slice(0, 5);
console.log('Slice:', slice.toString()); // Hello

// Concat
const combined = Buffer.concat([buf1, Buffer.from(' Extra')]);
console.log('Combined:', combined.toString());

// process.env and process.argv
console.log('\nprocess.argv:', process.argv);
console.log('NODE_ENV:', process.env.NODE_ENV ?? 'not set');

// process events
process.on('exit', (code) => {
  // synchronous only — no async here
  console.log(`Exiting with code: ${code}`);
});

process.on('uncaughtException', (err) => {
  console.error('Uncaught:', err.message);
  process.exit(1);
});
```
