## Goal

## Background

### Why Prisma Over Sequelize?
Prisma takes a schema-first approach: you define your data model in `schema.prisma`, then run `prisma generate` to produce a fully typed client. This means TypeScript (and even JavaScript with JSDoc) gets autocomplete for every model field, every query option, every filter. Sequelize's TypeScript support requires manual typings or decorators.

Prisma's query API is more explicit and readable than Sequelize. It uses plain objects rather than chained methods, making queries easier to construct dynamically. Prisma also has Prisma Studio — a visual database browser that runs in your browser.

### Prisma Migrate vs prisma db push
`prisma migrate dev` is for development: it generates a SQL migration file and applies it. Migration files are checked into source control and run in order. `prisma db push` directly syncs your schema to the database without creating migration files — useful for quick prototyping but not for production (changes aren't tracked).

### The Prisma Client
`PrismaClient` is auto-generated in `node_modules/.prisma/client`. You import it and call methods like `prisma.user.findMany()`, `prisma.post.create()`, etc. In a Node.js app, create one shared instance:

```js
const { PrismaClient } = require('@prisma/client');
const prisma = new PrismaClient({ log: ['query'] });
```

Prisma handles connection pooling internally. In serverless environments, use the `@prisma/adapter-*` packages to configure connection pooling correctly.

### 2. schema.prisma

```prisma
// prisma/schema.prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id        Int      @id @default(autoincrement())
  name      String   @db.VarChar(100)
  email     String   @unique
  bio       String?
  role      Role     @default(USER)
  posts     Post[]
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@index([email])
  @@map("users")
}

model Post {
  id          Int       @id @default(autoincrement())
  title       String
  body        String    @db.Text
  published   Boolean   @default(false)
  author      User      @relation(fields: [authorId], references: [id], onDelete: Cascade)
  authorId    Int
  tags        TagsOnPosts[]
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@index([authorId])
  @@map("posts")
}

model Tag {
  id    Int           @id @default(autoincrement())
  name  String        @unique
  posts TagsOnPosts[]

  @@map("tags")
}

// Explicit junction table for many-to-many
model TagsOnPosts {
  post      Post     @relation(fields: [postId], references: [id], onDelete: Cascade)
  postId    Int
  tag       Tag      @relation(fields: [tagId], references: [id], onDelete: Cascade)
  tagId     Int
  assignedAt DateTime @default(now())

  @@id([postId, tagId])
  @@map("tags_on_posts")
}

enum Role {
  USER
  MODERATOR
  ADMIN
}
```

```bash
# Create and apply migration

# Open Prisma Studio (browser GUI)

# Regenerate client after schema changes
npx prisma generate
```

### 4. User Queries

```js
// src/services/user.service.js
const prisma = require('../config/prisma');

const findAll = ({ skip = 0, take = 20, role } = {}) => {
  return prisma.user.findMany({
    skip,
    take,
    where: role ? { role } : undefined,
    select: {
      id: true,
      name: true,
      email: true,
      role: true,
      createdAt: true,
      _count: { select: { posts: true } },
    },
    orderBy: { createdAt: 'desc' },
  });
};

const findById = (id) => {
  return prisma.user.findUnique({
    where: { id },
    include: {
      posts: {
        where: { published: true },
        orderBy: { createdAt: 'desc' },
        take: 5,
      },
    },
  });
};

const create = ({ name, email, bio }) => {
  return prisma.user.create({
    data: { name, email, bio },
  });
};

const update = (id, data) => {
  return prisma.user.update({
    where: { id },
    data,
  });
};

const remove = (id) => {
  return prisma.user.delete({ where: { id } });
};

module.exports = { findAll, findById, create, update, remove };
```

### 5. Post Queries with Tags (Many-to-Many)

```js
// src/services/post.service.js
const prisma = require('../config/prisma');

const createWithTags = async ({ title, body, authorId, tagNames = [] }) => {
  // Upsert tags and connect them to the new post
  return prisma.post.create({
    data: {
      title,
      body,
      authorId,
      tags: {
        create: tagNames.map(name => ({
          tag: {
            connectOrCreate: {
              where: { name },
              create: { name },
            },
          },
        })),
      },
    },
    include: {
      author: { select: { id: true, name: true } },
      tags: { include: { tag: true } },
    },
  });
};

const findPublished = ({ skip = 0, take = 10, tagName } = {}) => {
  return prisma.post.findMany({
    skip,
    take,
    where: {
      published: true,
      ...(tagName && {
        tags: { some: { tag: { name: tagName } } },
      }),
    },
    include: {
      author: { select: { id: true, name: true } },
      tags: { include: { tag: { select: { name: true } } } },
    },
    orderBy: { createdAt: 'desc' },
  });
};

// Raw query example
const getPostStats = async () => {
  const result = await prisma.$queryRaw`
    SELECT 
      u.name AS author,
      COUNT(p.id)::int AS total_posts,
      COUNT(p.id) FILTER (WHERE p.published) ::int AS published_posts
    FROM users u
    LEFT JOIN posts p ON p."authorId" = u.id
    GROUP BY u.id, u.name
    ORDER BY total_posts DESC
  `;
  return result;
};

module.exports = { createWithTags, findPublished, getPostStats };
```

### 6. Routes

```js
// src/routes/posts.js
const router = require('express').Router();
const postService = require('../services/post.service');
const { Prisma } = require('@prisma/client');

router.get('/', async (req, res, next) => {
  try {
    const { page = 1, limit = 10, tag } = req.query;
    const posts = await postService.findPublished({
      skip: (parseInt(page) - 1) * parseInt(limit),
      take: parseInt(limit),
      tagName: tag,
    });
    res.json({ data: posts });
  } catch (err) { next(err); }
});

router.post('/', async (req, res, next) => {
  try {
    const { title, body, authorId, tags } = req.body;
    const post = await postService.createWithTags({ title, body, authorId, tagNames: tags ?? [] });
    res.status(201).json(post);
  } catch (err) {
    if (err instanceof Prisma.PrismaClientKnownRequestError) {
      if (err.code === 'P2003') {
        return res.status(400).json({ error: 'Author not found (foreign key violation)' });
      }
    }
    next(err);
  }
});

router.get('/stats', async (req, res, next) => {
  try {
    const stats = await postService.getPostStats();
    res.json({ data: stats });
  } catch (err) { next(err); }
});

module.exports = router;
```
