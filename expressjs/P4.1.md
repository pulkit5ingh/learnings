# P4.1 — Validated CRUD API

**Phase:** 4 — Validation & Error Handling  
**Topic:** Joi validation, RFC 7807 error responses, structured logging  
**Type:** Project

---

## Goal

Build a CRUD API where every endpoint is validated with Joi schemas, errors follow RFC 7807 Problem Details format, and all requests are logged in structured JSON.

## What to Build

- Product CRUD with strict Joi validation on all inputs
- Centralized validation middleware factory (`validate(schema)`)
- RFC 7807 `application/problem+json` error responses on every error
- Structured JSON request logging with winston (request ID, method, URL, status, duration, user)
- Consistent `AppError` class for operational errors

// src/middleware/requestContext.js
import { randomUUID } from 'node:crypto';
import { logger } from '../lib/logger.js';

export const requestContext = (req, res, next) => {
  req.id = req.headers['x-request-id'] ?? randomUUID();
  req.startTime = Date.now();
  res.set('x-request-id', req.id);

  res.on('finish', () => {
    logger.info('request', {
      requestId: req.id,
      method:    req.method,
      url:       req.originalUrl,
      status:    res.statusCode,
      ms:        Date.now() - req.startTime,
      userId:    req.user?._id,
      ip:        req.ip,
    });
  });

  next();
};

```js
// src/middleware/validate.js
import Joi from 'joi';
import { AppError } from '../lib/AppError.js';

/**
 * validate({ body: schema, params: schema, query: schema })
 * Runs all provided schemas and collects all errors before rejecting.
 */
export const validate = (schemas) => (req, res, next) => {
  const errors = [];

  for (const [target, schema] of Object.entries(schemas)) {
    const { error, value } = schema.validate(req[target], {
      abortEarly: false,
      allowUnknown: false,
      stripUnknown: true,
    });

    if (error) {
      errors.push(...error.details.map((d) => ({ field: d.path.join('.'), message: d.message })));
    } else {
      req[target] = value;   // replace with sanitized value
    }
  }

  if (errors.length) {
    return next(
      new AppError('Validation Failed', 422, JSON.stringify(errors), 'https://api.example.com/errors/validation'),
    );
  }
  next();
};
```

// src/schemas/productSchemas.js
import Joi from 'joi';

export const createProductSchema = {
  body: Joi.object({
    name:        Joi.string().min(2).max(200).required(),
    description: Joi.string().max(2000).default(''),
    price:       Joi.number().positive().precision(2).required(),
    sku:         Joi.string().uppercase().pattern(/^[A-Z0-9-]+$/).required()
                    .messages({ 'string.pattern.base': 'SKU must be uppercase alphanumeric with dashes' }),
    category:    Joi.string().valid('electronics', 'clothing', 'books', 'home').required(),
    tags:        Joi.array().items(Joi.string().lowercase().max(30)).max(10).default([]),
    inStock:     Joi.boolean().default(true),
  }),
};

export const updateProductSchema = {
  params: Joi.object({ id: Joi.string().length(24).hex().required() }),
  body: Joi.object({
    name:        Joi.string().min(2).max(200),
    description: Joi.string().max(2000),
    price:       Joi.number().positive().precision(2),
    category:    Joi.string().valid('electronics', 'clothing', 'books', 'home'),
    tags:        Joi.array().items(Joi.string().lowercase().max(30)).max(10),
    inStock:     Joi.boolean(),
  }).min(1).messages({ 'object.min': 'At least one field must be provided' }),
};

export const listProductsSchema = {
  query: Joi.object({
    page:     Joi.number().integer().min(1).default(1),
    limit:    Joi.number().integer().min(1).max(100).default(20),
    category: Joi.string().valid('electronics', 'clothing', 'books', 'home'),
    minPrice: Joi.number().positive(),
    maxPrice: Joi.number().positive().when('minPrice', {
      is: Joi.exist(), then: Joi.number().greater(Joi.ref('minPrice')),
    }),
    search:   Joi.string().max(100),
    inStock:  Joi.boolean(),
  }),
};

```js
// src/middleware/errorHandler.js
import { logger } from '../lib/logger.js';

export const errorHandler = (err, req, res, _next) => {
  const status = err.status ?? 500;
  const isOperational = err.isOperational === true;

  if (!isOperational) {
    logger.error('Unexpected error', { err, requestId: req.id });
  } else {
    logger.warn('Operational error', { title: err.title, status, requestId: req.id });
  }

  // RFC 7807 Problem Details
  res
    .status(status)
    .type('application/problem+json')
    .json({
      type:     err.type     ?? `https://api.example.com/errors/${status}`,
      title:    err.title    ?? err.message ?? 'Internal Server Error',
      status,
      detail:   err.detail   ?? undefined,
      instance: req.originalUrl,
      requestId: req.id,
      ...(err.detail && err.status === 422 ? { errors: JSON.parse(err.detail) } : {}),
    });
};
```

## Deliverables

- [ ] `AppError` class with `isOperational`, `title`, `status`, `detail`, `type`
- [ ] `validate()` middleware strips unknown fields, collects all Joi errors before rejecting
- [ ] All 5 product endpoints guarded by appropriate schemas
- [ ] All error responses use `application/problem+json` content type and RFC 7807 shape
- [ ] Validation errors return 422 with an `errors` array (field + message per item)
- [ ] `requestContext` logs structured JSON per request (method, URL, status, ms, requestId)
- [ ] Winston logs errors to `logs/error.log` and all levels to `logs/combined.log`
- [ ] `POST /products` with invalid SKU format returns the custom message from the Joi schema
