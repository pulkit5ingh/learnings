# P9.3 — Full-featured REST API

**Phase:** 9 — Advanced Patterns  
**Topic:** Versioning, OpenAPI docs, security hardening, monitoring, Docker, CI/CD  
**Type:** Project

---

## Goal

Build a production-ready REST API that integrates every major concept from the curriculum: versioned routes, Swagger docs, security hardening, Prometheus metrics, graceful shutdown, Docker, and a full CI/CD pipeline.

## What to Build

- `/api/v1` and `/api/v2` versioned resource routes (Tasks domain)
- OpenAPI 3.0 docs via swagger-jsdoc + swagger-ui-express
- Helmet, CORS, rate limiting (Redis-backed), CSRF on state-changing routes
- Prometheus metrics endpoint
- Graceful shutdown + health probes
- Multi-stage Dockerfile + docker-compose
- GitHub Actions: lint → test → build/push → deploy

```js
// src/middleware/security.js
import helmet from 'helmet';
import cors from 'cors';
import hpp from 'hpp';

const ALLOWED_ORIGINS = (process.env.CORS_ORIGINS ?? 'http://localhost:3000').split(',');

export const applySecurityMiddleware = (app) => {
  // Helmet sets 11 security headers (CSP, HSTS, X-Frame-Options, etc.)
  app.use(helmet({
    contentSecurityPolicy: {
      directives: {
        defaultSrc: ["'self'"],
        scriptSrc:  ["'self'"],
        styleSrc:   ["'self'", 'https://unpkg.com'],   // for Swagger UI
        imgSrc:     ["'self'", 'data:'],
      },
    },
  }));

  app.use(cors({
    origin: (origin, cb) => {
      if (!origin || ALLOWED_ORIGINS.includes(origin)) return cb(null, true);
      cb(new Error(`CORS: origin ${origin} not allowed`));
    },
    methods: ['GET', 'POST', 'PUT', 'PATCH', 'DELETE', 'OPTIONS'],
    allowedHeaders: ['Content-Type', 'Authorization', 'x-request-id'],
    credentials: true,
  }));

  // Prevent HTTP Parameter Pollution
  app.use(hpp());
};
```

```js
// src/routes/v2/tasks.js
import { Router } from 'express';
import { authenticate } from '../../middleware/authenticate.js';
import { validate } from '../../middleware/validate.js';
import { createTaskSchema, updateTaskSchema, listTasksSchema } from '../../schemas/taskSchemas.js';
import Task from '../../models/Task.js';

const router = Router();

/**
 * @swagger
 * /api/v2/tasks:
 *   get:
 *     summary: List tasks with cursor pagination (v2)
 *     tags: [Tasks]
 *     parameters:
 *       - in: query
 *         name: cursor
 *         schema: { type: string }
 *       - in: query
 *         name: limit
 *         schema: { type: integer, default: 20 }
 *       - in: query
 *         name: status
 *         schema: { type: string, enum: [todo, in_progress, done] }
 *     responses:
 *       200:
 *         description: Paginated task list
 */
router.get('/', authenticate, validate({ query: listTasksSchema }), async (req, res) => {
  const { cursor, limit = 20, status, priority, tag } = req.query;
  const filter = { user: req.user._id };
  if (status)   filter.status = status;
  if (priority) filter.priority = priority;
  if (tag)      filter.tags = tag;
  if (cursor)   filter._id = { $lt: cursor };

  const tasks = await Task.find(filter).sort({ _id: -1 }).limit(Number(limit) + 1).lean();
  const hasMore = tasks.length > limit;
  const data = hasMore ? tasks.slice(0, -1) : tasks;
  res.json({ data, pagination: { nextCursor: hasMore ? data.at(-1)._id : null, hasMore } });
});

/**
 * @swagger
 * /api/v2/tasks:
 *   post:
 *     summary: Create a task (v2)
 *     tags: [Tasks]
 *     requestBody:
 *       required: true
 *       content:
 *         application/json:
 *           schema:
 *             type: object
 *             required: [title]
 *             properties:
 *               title:       { type: string }
 *               description: { type: string }
 *               priority:    { type: string, enum: [low, medium, high] }
 *               dueDate:     { type: string, format: date }
 *               tags:        { type: array, items: { type: string } }
 *     responses:
 *       201:
 *         description: Created task
 */
router.post('/', authenticate, validate({ body: createTaskSchema }), async (req, res) => {
  const task = await Task.create({ ...req.body, user: req.user._id });
  res.status(201).json(task);
});

router.get('/:id', authenticate, async (req, res) => {
  const task = await Task.findOne({ _id: req.params.id, user: req.user._id });
  if (!task) return res.status(404).json({ title: 'Not Found', status: 404 });
  res.json(task);
});

router.patch('/:id', authenticate, validate({ body: updateTaskSchema }), async (req, res) => {
  const task = await Task.findOneAndUpdate({ _id: req.params.id, user: req.user._id }, req.body, { new: true });
  if (!task) return res.status(404).json({ title: 'Not Found', status: 404 });
  res.json(task);
});

router.delete('/:id', authenticate, async (req, res) => {
  const task = await Task.findOneAndDelete({ _id: req.params.id, user: req.user._id });
  if (!task) return res.status(404).json({ title: 'Not Found', status: 404 });
  res.status(204).send();
});

export default router;
```

```js
// src/app.js
import express from 'express';
import swaggerUi from 'swagger-ui-express';
import { applySecurityMiddleware } from './middleware/security.js';
import { rateLimiterMiddleware } from './middleware/rateLimiter.js';
import { requestLogger } from './middleware/requestLogger.js';
import { deprecateV1 } from './middleware/deprecateV1.js';
import { errorHandler } from './middleware/errorHandler.js';
import { swaggerSpec } from './docs/swaggerConfig.js';
import healthRouter from './routes/health.js';
import metricsRouter from './routes/metricsRoute.js';
import v1TasksRouter from './routes/v1/tasks.js';
import v2TasksRouter from './routes/v2/tasks.js';

const app = express();

applySecurityMiddleware(app);
app.use(express.json({ limit: '1mb' }));
app.use(requestLogger);
app.use(rateLimiterMiddleware);

// Docs
app.use('/api-docs', swaggerUi.serve, swaggerUi.setup(swaggerSpec, { swaggerOptions: { persistAuthorization: true } }));
app.get('/api-docs.json', (_req, res) => res.json(swaggerSpec));

// Routes
app.use(healthRouter);
app.use(metricsRouter);
app.use('/api/v1/tasks', deprecateV1, v1TasksRouter);
app.use('/api/v2/tasks', v2TasksRouter);

// Changelog
app.get('/api/changelog', (_req, res) => res.json({
  versions: [
    { version: 'v2', status: 'current', changes: ['Cursor pagination', 'Added priority and dueDate fields'] },
    { version: 'v1', status: 'deprecated', sunset: '2026-07-01', changes: ['Initial release'] },
  ],
}));

app.use((_req, res) => res.status(404).json({ title: 'Not Found', status: 404 }));
app.use(errorHandler);

export default app;
```

### Security checklist for this project

```
[x] helmet() sets 11 security headers
[x] CORS whitelist (not *)
[x] hpp() prevents parameter pollution
[x] Rate limiting per IP via Redis
[x] JWT required on all task routes
[x] Users can only access their own tasks (scoped by req.user._id)
[x] Input sanitized / stripped via Joi (validate middleware)
[x] Errors never leak stack traces in production
[x] Secrets in env vars, not committed
[x] Non-root user in Docker image
[x] Docker image built from node:22-alpine (small attack surface)
```
