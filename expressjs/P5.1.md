# P5.1 — High-traffic News API

**Phase:** 5 — Performance & Caching  
**Topic:** Redis caching, pagination, index optimization  
**Type:** Project

---

## Goal

Build a news article API optimized for read-heavy traffic using Redis cache-aside, cursor pagination, and database index tuning.

## What to Build

- Articles: CRUD with category, author, tags, publishedAt
- `GET /api/articles` — paginated, sortable, filterable (cursor-based)
- `GET /api/articles/:slug` — single article, cached in Redis
- `GET /api/articles/trending` — top articles by views in last 24 h (cached 5 min)
- Cache invalidation on article update/delete
- Cache middleware factory reusable across routes
- MongoDB explain() output compared before/after indexes

## Implementation

```js
// src/cache.js
import Redis from 'ioredis';

export const redis = new Redis({
  host: process.env.REDIS_HOST ?? 'localhost',
  port: 6379,
  enableReadyCheck: true,
  maxRetriesPerRequest: 3,
  retryStrategy: (times) => Math.min(times * 100, 3000),
});

redis.on('error', (err) => console.error('[Redis]', err.message));

export const getOrSet = async (key, ttlSeconds, fetchFn) => {
  const cached = await redis.get(key);
  if (cached) {
    return { data: JSON.parse(cached), fromCache: true };
  }
  const data = await fetchFn();
  await redis.setex(key, ttlSeconds, JSON.stringify(data));
  return { data, fromCache: false };
};

export const invalidate = (...keys) => redis.del(...keys);
export const invalidatePattern = async (pattern) => {
  const keys = await redis.keys(pattern);
  if (keys.length) await redis.del(...keys);
};
```

```js
import mongoose from 'mongoose';

const articleSchema = new mongoose.Schema(
  {
    title:       { type: String, required: true },
    slug:        { type: String, required: true, unique: true },
    body:        { type: String, required: true },
    excerpt:     { type: String },
    author:      { type: mongoose.Schema.Types.ObjectId, ref: 'Author', required: true },
    category:    { type: String, required: true, index: true },
    tags:        [{ type: String, lowercase: true }],
    published:   { type: Boolean, default: false },
    publishedAt: { type: Date, index: true },
    viewCount:   { type: Number, default: 0 },
    featured:    { type: Boolean, default: false },
  },
  { timestamps: true },
);

// Compound indexes for common query patterns
articleSchema.index({ published: 1, publishedAt: -1 });    // main listing
articleSchema.index({ published: 1, category: 1, publishedAt: -1 }); // category filter
articleSchema.index({ tags: 1, published: 1 });            // tag filter
articleSchema.index({ viewCount: -1, publishedAt: -1 });   // trending
articleSchema.index({ title: 'text', body: 'text' });      // full-text search

export default mongoose.model('Article', articleSchema);
```

```js
// src/middleware/cacheMiddleware.js
import { redis } from '../cache.js';

/**
 * Cache GET responses. Key is built from req.originalUrl.
 *
 * Usage: router.get('/trending', cache(300), ctrl.getTrending)
 */
export const cache = (ttlSeconds) => async (req, res, next) => {
  if (req.method !== 'GET') return next();

  const key = `route:${req.originalUrl}`;
  const cached = await redis.get(key);

  if (cached) {
    res.set('X-Cache', 'HIT');
    return res.json(JSON.parse(cached));
  }

  res.set('X-Cache', 'MISS');

  // Intercept res.json to capture and store the response
  const originalJson = res.json.bind(res);
  res.json = (body) => {
    if (res.statusCode < 400) {
      redis.setex(key, ttlSeconds, JSON.stringify(body)).catch(console.error);
    }
    return originalJson(body);
  };

  next();
};
```

```js
// src/controllers/articlesController.js
import Article from '../models/Article.js';
import { getOrSet, invalidate, invalidatePattern } from '../cache.js';

export const listArticles = async (req, res) => {
  const { cursor, limit = 20, category, tag, search } = req.query;

  const filter = { published: true };
  if (category) filter.category = category;
  if (tag) filter.tags = tag;
  if (search) filter.$text = { $search: search };
  if (cursor) filter.publishedAt = { $lt: new Date(cursor) };

  const articles = await Article.find(filter, search ? { score: { $meta: 'textScore' } } : undefined)
    .populate('author', 'name avatar')
    .sort(search ? { score: { $meta: 'textScore' } } : { publishedAt: -1 })
    .limit(Number(limit) + 1)
    .lean();

  const hasMore = articles.length > limit;
  const data = hasMore ? articles.slice(0, -1) : articles;
  const nextCursor = hasMore ? data.at(-1).publishedAt?.toISOString() : null;

  res.json({ data, pagination: { nextCursor, hasMore } });
};

export const getArticle = async (req, res) => {
  const { data } = await getOrSet(`article:${req.params.slug}`, 300, async () => {
    const article = await Article.findOne({ slug: req.params.slug, published: true })
      .populate('author', 'name avatar bio')
      .lean();
    if (!article) return null;
    return article;
  });

  if (!data) return res.status(404).json({ error: 'Article not found' });

  // Increment view count in background — don't await
  Article.findOneAndUpdate({ slug: req.params.slug }, { $inc: { viewCount: 1 } }).exec();

  res.json(data);
};

export const getTrending = async (req, res) => {
  const { data } = await getOrSet('articles:trending', 300, async () => {
    const since = new Date(Date.now() - 24 * 60 * 60 * 1000);
    return Article.find({ published: true, publishedAt: { $gte: since } })
      .sort({ viewCount: -1 })
      .limit(10)
      .select('title slug excerpt viewCount publishedAt category')
      .lean();
  });
  res.json(data);
};

export const updateArticle = async (req, res) => {
  const article = await Article.findByIdAndUpdate(req.params.id, req.body, {
    new: true, runValidators: true,
  });
  if (!article) return res.status(404).json({ error: 'Article not found' });

  // Invalidate cached article and trending list
  await Promise.all([
    invalidate(`article:${article.slug}`),
    invalidate('articles:trending'),
    invalidatePattern('route:/api/articles*'),
  ]);

  res.json(article);
};
```
