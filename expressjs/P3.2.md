# P3.2 — Multi-role Application

**Phase:** 3 — Authentication & Security  
**Topic:** RBAC, protected routes per role, audit log  
**Type:** Project

---

## Goal

Build a multi-role API with admin, moderator, and user roles using middleware guards, resource ownership checks, and an immutable audit log.

## What to Build

- Three roles: `admin`, `moderator`, `user`
- Role guards on route level (`requireRole('admin')`)
- Resource ownership check (`requireOwnerOrAdmin`)
- Audit log: every write operation logged (who, what, when, before/after)
- `GET /api/admin/audit-log` — admin-only, paginated
- User management endpoints (admin only)
- Content moderation endpoints (moderator + admin)

// src/models/AuditLog.js
import mongoose from 'mongoose';

const auditLogSchema = new mongoose.Schema(
  {
    actor:      { type: mongoose.Schema.Types.ObjectId, ref: 'User', required: true },
    actorEmail: { type: String, required: true },   // denormalized for fast display
    action:     { type: String, required: true },   // e.g. 'user.ban', 'post.delete'
    resource:   { type: String, required: true },   // e.g. 'User', 'Post'
    resourceId: { type: String, required: true },
    before:     mongoose.Schema.Types.Mixed,
    after:      mongoose.Schema.Types.Mixed,
    ip:         String,
    userAgent:  String,
  },
  {
    timestamps: true,
    // Audit logs must never be mutated
    strict: true,
  },
);

// TTL: retain logs for 1 year
auditLogSchema.index({ createdAt: 1 }, { expireAfterSeconds: 365 * 24 * 60 * 60 });
auditLogSchema.index({ actor: 1 });
auditLogSchema.index({ resource: 1, resourceId: 1 });

export default mongoose.model('AuditLog', auditLogSchema);
```

// src/middleware/authorize.js
export const requireRole = (...roles) => (req, res, next) => {
  if (!req.user) return res.status(401).json({ error: 'Not authenticated' });
  if (!roles.includes(req.user.role)) {
    return res.status(403).json({ error: `Requires role: ${roles.join(' or ')}` });
  }
  next();
};

// Resource must have an `author` or `userId` field
export const requireOwnerOrAdmin = (getResourceId) => async (req, res, next) => {
  const resourceUserId = typeof getResourceId === 'function'
    ? getResourceId(req)
    : req.params[getResourceId];

  if (req.user.role === 'admin' || req.user._id.toString() === resourceUserId) {
    return next();
  }
  res.status(403).json({ error: 'You do not own this resource' });
};
```

```js
// src/middleware/auditLogger.js
import AuditLog from '../models/AuditLog.js';

/**
 * Factory — wrap a controller to log audit entries automatically.
 *
 * Usage:
 *   router.delete('/:id', authenticate, requireRole('admin'),
 *     withAudit('user.delete', 'User', (req) => req.params.id)(usersController.deleteUser));
 */
export const withAudit = (action, resource, getResourceId) =>
  (handler) =>
  async (req, res, next) => {
    // Capture original json() to intercept response
    const originalJson = res.json.bind(res);
    let responseBody;
    res.json = (body) => {
      responseBody = body;
      return originalJson(body);
    };

    await handler(req, res, next);

    if (res.statusCode < 400 && req.user) {
      try {
        await AuditLog.create({
          actor:      req.user._id,
          actorEmail: req.user.email,
          action,
          resource,
          resourceId: typeof getResourceId === 'function' ? getResourceId(req) : req.params.id,
          after:      responseBody,
          ip:         req.ip,
          userAgent:  req.headers['user-agent'],
        });
      } catch (err) {
        console.error('Audit log write failed:', err.message);
        // Never fail the request because of audit log failure
      }
    }
  };
```

```js
// src/controllers/adminController.js
import User from '../models/User.js';
import AuditLog from '../models/AuditLog.js';

export const listUsers = async (req, res) => {
  const { page = 1, limit = 20, role, banned } = req.query;
  const filter = {};
  if (role) filter.role = role;
  if (banned !== undefined) filter.isBanned = banned === 'true';

  const [users, total] = await Promise.all([
    User.find(filter).select('-passwordHash').skip((page - 1) * limit).limit(Number(limit)),
    User.countDocuments(filter),
  ]);
  res.json({ data: users, meta: { total, page: Number(page) } });
};

export const banUser = async (req, res) => {
  const user = await User.findByIdAndUpdate(
    req.params.id,
    { isBanned: true, bannedAt: new Date(), bannedBy: req.user._id },
    { new: true },
  ).select('-passwordHash');
  if (!user) return res.status(404).json({ error: 'User not found' });
  res.json(user);
};

export const setRole = async (req, res) => {
  const { role } = req.body;
  const allowed = ['user', 'moderator', 'admin'];
  if (!allowed.includes(role)) return res.status(422).json({ error: 'Invalid role' });

  const user = await User.findByIdAndUpdate(req.params.id, { role }, { new: true }).select('-passwordHash');
  if (!user) return res.status(404).json({ error: 'User not found' });
  res.json(user);
};

export const getAuditLog = async (req, res) => {
  const { page = 1, limit = 50, actor, resource } = req.query;
  const filter = {};
  if (actor) filter.actor = actor;
  if (resource) filter.resource = resource;

  const [logs, total] = await Promise.all([
    AuditLog.find(filter)
      .populate('actor', 'name email')
      .sort({ createdAt: -1 })
      .skip((page - 1) * limit)
      .limit(Number(limit)),
    AuditLog.countDocuments(filter),
  ]);
  res.json({ data: logs, meta: { total, page: Number(page) } });
};
```

```js
// src/routes/admin.js
import { Router } from 'express';
import { authenticate } from '../middleware/authenticate.js';
import { requireRole } from '../middleware/authorize.js';
import { withAudit } from '../middleware/auditLogger.js';
import * as ctrl from '../controllers/adminController.js';

const router = Router();
router.use(authenticate, requireRole('admin'));

router.get('/users',              ctrl.listUsers);
router.patch('/users/:id/ban',    withAudit('user.ban',     'User', (r) => r.params.id)(ctrl.banUser));
router.patch('/users/:id/role',   withAudit('user.setRole', 'User', (r) => r.params.id)(ctrl.setRole));
router.get('/audit-log',          ctrl.getAuditLog);

export default router;
```

## Deliverables

- [ ] `User.role` field with `enum: ['user', 'moderator', 'admin']`
- [ ] `requireRole('admin')` middleware returns 403 for lower roles
- [ ] `requireOwnerOrAdmin` allows owner or admin, 403 otherwise
- [ ] `AuditLog` model with TTL index (1 year)
- [ ] `withAudit` wrapper logs action, actor, resource, IP after every successful write
- [ ] `PATCH /admin/users/:id/ban` only accessible to admin; writes audit entry
- [ ] `PATCH /admin/users/:id/role` only accessible to admin; writes audit entry
- [ ] `GET /admin/audit-log` filterable by `actor` and `resource`, paginated
- [ ] Audit log write failure does NOT cause the request to fail
