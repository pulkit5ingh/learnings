# P6.2 — E-commerce with Payments

**Phase:** 6 — File Handling & External Services  
**Topic:** Stripe checkout, webhook, order email, idempotency  
**Type:** Project

---

## Goal

Build an e-commerce API with a product catalog, shopping cart, Stripe Checkout session, webhook order fulfillment, and transactional email with Bull queue.

## What to Build

- Products: browse catalog
- Cart: add/remove/clear items (stored in Redis)
- `POST /api/orders/checkout` — create Stripe Checkout Session, return URL
- `POST /api/webhooks/stripe` — handle `checkout.session.completed` → create order, queue email
- `GET /api/orders/:id` — order details (authenticated)
- Email: order confirmation via Bull + Nodemailer

## Implementation

```js
// src/services/cartService.js
import { redis } from '../cache.js';

const KEY = (userId) => `cart:${userId}`;
const TTL = 7 * 24 * 60 * 60; // 7 days

export const getCart = async (userId) => {
  const raw = await redis.get(KEY(userId));
  return raw ? JSON.parse(raw) : { items: [] };
};

export const addItem = async (userId, { productId, name, price, quantity = 1 }) => {
  const cart = await getCart(userId);
  const existing = cart.items.find((i) => i.productId === productId);

  if (existing) {
    existing.quantity += quantity;
  } else {
    cart.items.push({ productId, name, price, quantity });
  }

  await redis.setex(KEY(userId), TTL, JSON.stringify(cart));
  return cart;
};

export const removeItem = async (userId, productId) => {
  const cart = await getCart(userId);
  cart.items = cart.items.filter((i) => i.productId !== productId);
  await redis.setex(KEY(userId), TTL, JSON.stringify(cart));
  return cart;
};

export const clearCart = async (userId) => redis.del(KEY(userId));

export const getTotal = (cart) =>
  cart.items.reduce((sum, item) => sum + item.price * item.quantity, 0);
```

```js
// src/services/stripeService.js
import Stripe from 'stripe';

const stripe = new Stripe(process.env.STRIPE_SECRET_KEY);

export const createCheckoutSession = async ({ userId, cart, successUrl, cancelUrl }) => {
  const lineItems = cart.items.map((item) => ({
    price_data: {
      currency: 'usd',
      product_data: { name: item.name },
      unit_amount: Math.round(item.price * 100), // cents
    },
    quantity: item.quantity,
  }));

  return stripe.checkout.sessions.create({
    mode: 'payment',
    line_items: lineItems,
    success_url: successUrl,
    cancel_url:  cancelUrl,
    metadata: { userId },    // passed back in webhook
    payment_intent_data: {
      // Idempotency: use userId + timestamp bucket so duplicate clicks don't double-charge
      metadata: { userId },
    },
  });
};

export const constructWebhookEvent = (rawBody, signature) =>
  stripe.webhooks.constructEvent(rawBody, signature, process.env.STRIPE_WEBHOOK_SECRET);
```

```js
// src/controllers/webhooksController.js
import { constructWebhookEvent } from '../services/stripeService.js';
import { clearCart } from '../services/cartService.js';
import Order from '../models/Order.js';
import { enqueueOrderConfirmation } from '../queues/emailQueue.js';

// IMPORTANT: use express.raw() on this route — NOT express.json()
export const stripeWebhook = async (req, res) => {
  const sig = req.headers['stripe-signature'];

  let event;
  try {
    event = constructWebhookEvent(req.body, sig);
  } catch (err) {
    console.error('Webhook signature verification failed:', err.message);
    return res.status(400).send(`Webhook Error: ${err.message}`);
  }

  if (event.type === 'checkout.session.completed') {
    const session = event.data.object;
    const { userId } = session.metadata;

    // Idempotent: skip if order already created for this session
    const existing = await Order.findOne({ stripeSessionId: session.id });
    if (!existing) {
      const order = await Order.create({
        user: userId,
        stripeSessionId: session.id,
        amountTotal: session.amount_total / 100,
        currency: session.currency,
        customerEmail: session.customer_details?.email,
        status: 'paid',
        items: session.line_items ?? [],
      });

      await clearCart(userId);
      await enqueueOrderConfirmation({ orderId: order._id, email: order.customerEmail });
    }
  }

  // Always return 200 — Stripe retries on non-2xx
  res.json({ received: true });
};
```

```js
// src/queues/emailQueue.js
import Bull from 'bull';
import { sendOrderConfirmationEmail } from '../services/emailService.js';

const emailQueue = new Bull('email', {
  redis: { host: process.env.REDIS_HOST ?? 'localhost', port: 6379 },
  defaultJobOptions: { attempts: 3, backoff: { type: 'exponential', delay: 5000 } },
});

export const enqueueOrderConfirmation = (data) =>
  emailQueue.add('order-confirmation', data);

emailQueue.process('order-confirmation', async (job) => {
  await sendOrderConfirmationEmail(job.data);
});

emailQueue.on('failed', (job, err) => {
  console.error(`Email job ${job.id} failed after ${job.attemptsMade} attempts:`, err.message);
});
```

// src/services/emailService.js
import nodemailer from 'nodemailer';
import handlebars from 'handlebars';
import { readFileSync } from 'node:fs';
import { join, dirname } from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = dirname(fileURLToPath(import.meta.url));

const transporter = nodemailer.createTransport({
  host: process.env.SMTP_HOST ?? 'smtp.sendgrid.net',
  port: 587,
  auth: { user: 'apikey', pass: process.env.SENDGRID_API_KEY },
});

const compileTemplate = (name, context) => {
  const source = readFileSync(join(__dirname, `../templates/${name}.hbs`), 'utf8');
  return handlebars.compile(source)(context);
};

export const sendOrderConfirmationEmail = async ({ orderId, email }) => {
  const html = compileTemplate('order-confirmation', { orderId });
  await transporter.sendMail({
    from:    process.env.EMAIL_FROM ?? 'shop@example.com',
    to:      email,
    subject: `Order #${orderId} confirmed`,
    html,
  });
};

## Deliverables

- [ ] Cart stored in Redis with 7-day TTL; `addItem` increments quantity if product already in cart
- [ ] `createCheckoutSession` builds `line_items` from cart and stores `userId` in metadata
- [ ] Webhook handler verifies Stripe signature with `constructWebhookEvent`
- [ ] `stripeWebhook` is idempotent: duplicate `checkout.session.completed` events do not create duplicate orders
- [ ] Order created and cart cleared inside the webhook handler
- [ ] `enqueueOrderConfirmation` queues email via Bull after order creation
- [ ] Email job retries 3 times with exponential backoff on failure
- [ ] `/webhooks/stripe` uses `express.raw()` not `express.json()`
