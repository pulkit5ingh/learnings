## Goal
Use Joi to define reusable validation schemas that validate request data before it reaches route handlers.

## Background

### Declarative Validation
Joi is a schema description language and validator. You define what valid data looks like — `Joi.string().email().required()` — and Joi validates values against the schema and produces structured error messages. This declarative approach separates validation logic from business logic.

By centralizing validation in schemas, you get consistent error messages across the API, no duplicate `if (!email)` checks in handlers, and schemas that double as documentation of what the API accepts.

### Validation Middleware Pattern
Rather than calling `schema.validate(req.body)` in every route, create a `validate` middleware factory. It takes a Joi schema (or an object of schemas for body/params/query) and returns middleware that validates and either calls `next()` or returns 400.

```js
const validate = (schema) => (req, res, next) => {
  const { error } = schema.validate(req.body, { abortEarly: false });
  if (error) return res.status(400).json({ errors: error.details });
  next();
};
```

`abortEarly: false` collects all validation errors, not just the first one. This is important for user-facing forms where you want to show all problems at once.

### Conditional Validation
Joi supports conditional schemas with `.when()`. For example, a `shippingAddress` field that's required only when `requiresShipping` is true. `.alternatives().conditional()` handles more complex cases where the schema type itself depends on another field.

### 2. Validation Middleware Factory

```js
// src/middleware/validate.js
const Joi = require('joi');

// Validates req.body, req.params, or req.query
const validate = (schemas) => (req, res, next) => {
  const targets = {
    body: req.body,
    params: req.params,
    query: req.query,
  };

  const errors = [];

  for (const [key, schema] of Object.entries(schemas)) {
    if (!schema) continue;

    const { error, value } = schema.validate(targets[key], {
      abortEarly: false,       // collect all errors
      stripUnknown: true,      // remove fields not in schema
      convert: true,           // coerce types (string '3' → number 3)
    });

    if (error) {
      errors.push(...error.details.map(d => ({
        field: d.path.join('.'),
        message: d.message.replace(/['"]/g, ''),
        type: d.type,
      })));
    } else {
      // Replace with validated/coerced values
      req[key] = value;
    }
  }

  if (errors.length > 0) {
    return res.status(400).json({
      error: 'Validation failed',
      details: errors,
    });
  }

  next();
};

module.exports = validate;
```

### 3. Define Schemas

```js
// src/schemas/user.schemas.js
const Joi = require('joi');

// Reusable sub-schemas
const passwordSchema = Joi.string()
  .min(8)
  .max(72)
  .pattern(/^(?=.*[a-z])(?=.*[A-Z])(?=.*\d)/)
  .required()
  .messages({
    'string.pattern.base': 'Password must contain at least one uppercase letter, one lowercase, and one digit',
    'string.min': 'Password must be at least 8 characters',
  });

const addressSchema = Joi.object({
  street: Joi.string().max(100).required(),
  city: Joi.string().max(50).required(),
  country: Joi.string().length(2).uppercase().required()
    .messages({ 'string.length': 'Country must be a 2-letter ISO code (e.g. US, GB)' }),
  postalCode: Joi.string().max(20),
});

// Register
const registerSchema = Joi.object({
  name: Joi.string().trim().min(2).max(50).required()
    .messages({ 'string.min': 'Name must be at least 2 characters' }),

  email: Joi.string().email({ tlds: { allow: false } }).lowercase().required()
    .messages({ 'string.email': 'Please provide a valid email address' }),

  password: passwordSchema,

  role: Joi.string()
    .valid('user', 'moderator')
    .default('user'),

  // Optional address object
  address: addressSchema.optional(),

  // Array of tags
  tags: Joi.array()
    .items(Joi.string().trim().lowercase().max(30))
    .max(10)
    .default([])
    .messages({ 'array.max': 'Maximum 10 tags allowed' }),
});

// Update (all fields optional except those explicitly marked)
const updateSchema = Joi.object({
  name: Joi.string().trim().min(2).max(50),
  address: addressSchema,
  tags: Joi.array().items(Joi.string().trim().lowercase().max(30)).max(10),
}).min(1) // at least one field required
  .messages({ 'object.min': 'Provide at least one field to update' });

// ID param
const idParamSchema = Joi.object({
  id: Joi.number().integer().positive().required()
    .messages({ 'number.base': 'ID must be a positive integer' }),
});

// Query string
const listQuerySchema = Joi.object({
  page: Joi.number().integer().min(1).default(1),
  limit: Joi.number().integer().min(1).max(100).default(20),
  sort: Joi.string().valid('name', 'email', 'createdAt', '-name', '-email', '-createdAt').default('-createdAt'),
  role: Joi.string().valid('user', 'moderator', 'admin'),
  search: Joi.string().trim().max(100),
});

module.exports = { registerSchema, updateSchema, idParamSchema, listQuerySchema };
```

### 4. Conditional Validation

```js
// src/schemas/order.schemas.js
const Joi = require('joi');

const createOrderSchema = Joi.object({
  items: Joi.array()
    .items(
      Joi.object({
        productId: Joi.number().integer().positive().required(),
        quantity: Joi.number().integer().min(1).max(99).required(),
      })
    )
    .min(1)
    .required()
    .messages({ 'array.min': 'Order must have at least one item' }),

  requiresShipping: Joi.boolean().default(false),

  // shippingAddress is required ONLY if requiresShipping is true
  shippingAddress: Joi.when('requiresShipping', {
    is: true,
    then: Joi.object({
      street: Joi.string().required(),
      city: Joi.string().required(),
      country: Joi.string().length(2).required(),
    }).required(),
    otherwise: Joi.forbidden(),
  }),

  // discountCode is optional string
  discountCode: Joi.string().uppercase().alphanum().length(10).optional(),

  // paymentMethod with conditional sub-validation
  paymentMethod: Joi.string().valid('card', 'paypal', 'crypto').required(),

  // card details only if paymentMethod is 'card'
  cardDetails: Joi.when('paymentMethod', {
    is: 'card',
    then: Joi.object({
      last4: Joi.string().length(4).pattern(/^\d+$/).required(),
      brand: Joi.string().valid('visa', 'mastercard', 'amex').required(),
    }).required(),
    otherwise: Joi.forbidden(),
  }),
});

module.exports = { createOrderSchema };
```

### 5. Routes Using Validation

```js
// src/routes/users.js
const router = require('express').Router();
const validate = require('../middleware/validate');
const { registerSchema, updateSchema, idParamSchema, listQuerySchema } = require('../schemas/user.schemas');
const { createOrderSchema } = require('../schemas/order.schemas');

// Fake store
const users = [];
let nextId = 1;

router.get('/',
  validate({ query: listQuerySchema }),
  (req, res) => {
    // req.query is now validated and has defaults applied
    const { page, limit, sort, role, search } = req.query;
    res.json({ page, limit, sort, role, search, data: users });
  }
);

router.get('/:id',
  validate({ params: idParamSchema }),
  (req, res) => {
    const user = users.find(u => u.id === req.params.id); // already a number
    if (!user) return res.status(404).json({ error: 'Not found' });
    res.json(user);
  }
);

router.post('/',
  validate({ body: registerSchema }),
  (req, res) => {
    const user = { id: nextId++, ...req.body };
    delete user.password; // never return password
    users.push({ id: user.id, ...req.body });
    res.status(201).json(user);
  }
);

router.patch('/:id',
  validate({ params: idParamSchema, body: updateSchema }),
  (req, res) => {
    const index = users.findIndex(u => u.id === req.params.id);
    if (index === -1) return res.status(404).json({ error: 'Not found' });
    Object.assign(users[index], req.body);
    res.json(users[index]);
  }
);

router.post('/orders',
  validate({ body: createOrderSchema }),
  (req, res) => {
    res.status(201).json({ message: 'Order received', order: req.body });
  }
);

module.exports = router;
```
