# P3.1 — Auth API

**Phase:** 3 — Authentication & Security  
**Topic:** register/login/logout/refresh, bcrypt, JWT, Redis blacklist, email verification  
**Type:** Project

---

## Goal

Build a complete authentication API with access + refresh tokens, Redis-based token blacklisting, and email verification flow.

## What to Build

- `POST /api/auth/register` — create user, send verification email
- `POST /api/auth/verify-email` — verify email with token
- `POST /api/auth/login` — return access + refresh tokens
- `POST /api/auth/refresh` — rotate refresh token, return new access token
- `POST /api/auth/logout` — blacklist refresh token in Redis
- `POST /api/auth/forgot-password` — email reset link
- `POST /api/auth/reset-password` — apply new password
- `GET /api/auth/me` — protected route, returns current user

// src/models/User.js
import mongoose from 'mongoose';
import bcrypt from 'bcrypt';

const userSchema = new mongoose.Schema(
  {
    email:          { type: String, required: true, unique: true, lowercase: true },
    passwordHash:   { type: String, required: true },
    name:           { type: String, required: true },
    emailVerified:  { type: Boolean, default: false },
    verifyToken:    String,
    resetToken:     String,
    resetTokenExpiresAt: Date,
  },
  { timestamps: true },
);

userSchema.methods.comparePassword = async function (plain) {
  return bcrypt.compare(plain, this.passwordHash);
};

userSchema.methods.toJSON = function () {
  const obj = this.toObject();
  delete obj.passwordHash;
  delete obj.verifyToken;
  delete obj.resetToken;
  delete obj.__v;
  return obj;
};

export default mongoose.model('User', userSchema);

```js
// src/services/tokenService.js
import jwt from 'jsonwebtoken';
import { createClient } from 'ioredis';
import { randomUUID } from 'node:crypto';

const redis = new createClient({ host: process.env.REDIS_HOST ?? 'localhost' });

const ACCESS_TTL  = 15 * 60;          // 15 minutes
const REFRESH_TTL = 7 * 24 * 60 * 60; // 7 days

export const signAccessToken = (userId) =>
  jwt.sign({ sub: userId, type: 'access' }, process.env.JWT_ACCESS_SECRET, { expiresIn: ACCESS_TTL });

export const signRefreshToken = (userId) => {
  const jti = randomUUID();
  const token = jwt.sign(
    { sub: userId, jti, type: 'refresh' },
    process.env.JWT_REFRESH_SECRET,
    { expiresIn: REFRESH_TTL },
  );
  return { token, jti };
};

export const verifyAccessToken = (token) =>
  jwt.verify(token, process.env.JWT_ACCESS_SECRET);

export const verifyRefreshToken = async (token) => {
  const payload = jwt.verify(token, process.env.JWT_REFRESH_SECRET);
  const blacklisted = await redis.get(`bl:${payload.jti}`);
  if (blacklisted) throw new Error('Token revoked');
  return payload;
};

export const blacklistRefreshToken = async (jti, expiresIn) => {
  await redis.setex(`bl:${jti}`, expiresIn, '1');
};
```

```js
// src/services/authService.js
import bcrypt from 'bcrypt';
import { randomBytes } from 'node:crypto';
import User from '../models/User.js';
import * as tokenService from './tokenService.js';
import { sendVerificationEmail, sendPasswordResetEmail } from './emailService.js';

const SALT_ROUNDS = 12;

export const register = async ({ email, password, name }) => {
  const existing = await User.findOne({ email });
  if (existing) {
    const err = new Error('Email already in use'); err.status = 409; throw err;
  }

  const passwordHash = await bcrypt.hash(password, SALT_ROUNDS);
  const verifyToken  = randomBytes(32).toString('hex');

  const user = await User.create({ email, passwordHash, name, verifyToken });
  await sendVerificationEmail(user.email, verifyToken);
  return user;
};

export const verifyEmail = async (token) => {
  const user = await User.findOneAndUpdate(
    { verifyToken: token },
    { emailVerified: true, verifyToken: undefined },
    { new: true },
  );
  if (!user) { const err = new Error('Invalid token'); err.status = 400; throw err; }
  return user;
};

export const login = async ({ email, password }) => {
  const user = await User.findOne({ email });
  if (!user || !(await user.comparePassword(password))) {
    const err = new Error('Invalid credentials'); err.status = 401; throw err;
  }
  if (!user.emailVerified) {
    const err = new Error('Email not verified'); err.status = 403; throw err;
  }

  const accessToken = tokenService.signAccessToken(user._id);
  const { token: refreshToken } = tokenService.signRefreshToken(user._id);
  return { user, accessToken, refreshToken };
};

export const refresh = async (refreshToken) => {
  const payload = await tokenService.verifyRefreshToken(refreshToken);

  // Blacklist old refresh token immediately (rotation)
  const ttl = payload.exp - Math.floor(Date.now() / 1000);
  await tokenService.blacklistRefreshToken(payload.jti, ttl);

  const newAccessToken = tokenService.signAccessToken(payload.sub);
  const { token: newRefreshToken } = tokenService.signRefreshToken(payload.sub);
  return { accessToken: newAccessToken, refreshToken: newRefreshToken };
};

export const logout = async (refreshToken) => {
  const payload = await tokenService.verifyRefreshToken(refreshToken);
  const ttl = payload.exp - Math.floor(Date.now() / 1000);
  await tokenService.blacklistRefreshToken(payload.jti, Math.max(ttl, 1));
};

export const forgotPassword = async (email) => {
  const user = await User.findOne({ email });
  if (!user) return; // silently succeed to prevent email enumeration
  const resetToken = randomBytes(32).toString('hex');
  user.resetToken = resetToken;
  user.resetTokenExpiresAt = new Date(Date.now() + 60 * 60 * 1000); // 1 hour
  await user.save();
  await sendPasswordResetEmail(email, resetToken);
};

export const resetPassword = async (token, newPassword) => {
  const user = await User.findOne({
    resetToken: token,
    resetTokenExpiresAt: { $gt: new Date() },
  });
  if (!user) { const err = new Error('Invalid or expired token'); err.status = 400; throw err; }

  user.passwordHash = await bcrypt.hash(newPassword, SALT_ROUNDS);
  user.resetToken = undefined;
  user.resetTokenExpiresAt = undefined;
  await user.save();
};
```

```js
// src/middleware/authenticate.js
import * as tokenService from '../services/tokenService.js';
import User from '../models/User.js';

export const authenticate = async (req, res, next) => {
  const header = req.headers.authorization ?? '';
  if (!header.startsWith('Bearer ')) return res.status(401).json({ error: 'No token' });

  try {
    const payload = tokenService.verifyAccessToken(header.slice(7));
    req.user = await User.findById(payload.sub);
    if (!req.user) return res.status(401).json({ error: 'User not found' });
    next();
  } catch {
    res.status(401).json({ error: 'Invalid or expired token' });
  }
};
```

## Deliverables

- [ ] `POST /auth/register` hashes password with bcrypt (12 rounds), sends verification email
- [ ] `POST /auth/verify-email?token=` marks user verified; invalid token returns 400
- [ ] `POST /auth/login` returns `{ accessToken, refreshToken, user }` (no `passwordHash` in user)
- [ ] `POST /auth/refresh` rotates both tokens (old refresh blacklisted immediately)
- [ ] `POST /auth/logout` blacklists the refresh token in Redis
- [ ] `POST /auth/forgot-password` silent success to prevent email enumeration
- [ ] `POST /auth/reset-password` rejects expired or used tokens
- [ ] `GET /auth/me` protected by `authenticate` middleware
- [ ] Login on unverified account returns 403
