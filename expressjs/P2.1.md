# P2.1 — Blog API with MongoDB

**Phase:** 2 — Database Integration  
**Topic:** Mongoose models, relationships, pagination, population  
**Type:** Project

---

## Goal

Build a blog REST API backed by MongoDB with Mongoose: posts, comments, tags, and authors with pagination and population.

## What to Build

- Authors: register, get profile
- Posts: CRUD, publish/unpublish, list with author populated, paginated
- Comments: add/delete comments on a post
- Tags: filter posts by tag; list all tags with post counts
- Cursor-based and offset pagination for posts

```js
// src/models/Author.js
import mongoose from 'mongoose';

const authorSchema = new mongoose.Schema(
  {
    name:  { type: String, required: true, trim: true },
    email: { type: String, required: true, unique: true, lowercase: true },
    bio:   { type: String, default: '' },
    avatar: String,
  },
  { timestamps: true },
);

authorSchema.virtual('posts', {
  ref: 'Post', localField: '_id', foreignField: 'author',
});

authorSchema.methods.toJSON = function () {
  const obj = this.toObject();
  delete obj.__v;
  return obj;
};

export default mongoose.model('Author', authorSchema);
```

```js
// src/models/Post.js
import mongoose from 'mongoose';

const postSchema = new mongoose.Schema(
  {
    title:     { type: String, required: true, trim: true },
    slug:      { type: String, required: true, unique: true },
    body:      { type: String, required: true },
    excerpt:   { type: String, default: '' },
    author:    { type: mongoose.Schema.Types.ObjectId, ref: 'Author', required: true },
    tags:      [{ type: String, lowercase: true, trim: true }],
    published: { type: Boolean, default: false },
    publishedAt: Date,
    viewCount: { type: Number, default: 0 },
  },
  { timestamps: true },
);

postSchema.index({ slug: 1 }, { unique: true });
postSchema.index({ tags: 1 });
postSchema.index({ published: 1, createdAt: -1 });

// Auto-generate slug
postSchema.pre('validate', function () {
  if (!this.slug && this.title) {
    this.slug = this.title.toLowerCase().replace(/[^a-z0-9]+/g, '-').replace(/(^-|-$)/g, '');
  }
});

export default mongoose.model('Post', postSchema);
```

```js
// src/controllers/postsController.js
import Post from '../models/Post.js';
import Comment from '../models/Comment.js';

export const listPosts = async (req, res) => {
  const { tag, page = 1, limit = 10, cursor } = req.query;

  // Offset pagination (default)
  if (!cursor) {
    const filter = { published: true, ...(tag && { tags: tag }) };
    const [posts, total] = await Promise.all([
      Post.find(filter)
        .populate('author', 'name email avatar')
        .sort({ createdAt: -1 })
        .skip((page - 1) * limit)
        .limit(Number(limit)),
      Post.countDocuments(filter),
    ]);

    return res.json({
      data: posts,
      meta: { total, page: Number(page), limit: Number(limit), pages: Math.ceil(total / limit) },
    });
  }

  // Cursor pagination
  const filter = { published: true, _id: { $lt: cursor }, ...(tag && { tags: tag }) };
  const posts = await Post.find(filter)
    .populate('author', 'name email avatar')
    .sort({ _id: -1 })
    .limit(Number(limit) + 1);

  const hasMore = posts.length > limit;
  const data = hasMore ? posts.slice(0, -1) : posts;
  res.json({ data, pagination: { nextCursor: hasMore ? data.at(-1)._id : null, hasMore } });
};

export const getPost = async (req, res) => {
  const post = await Post.findOne({ slug: req.params.slug })
    .populate('author', 'name email avatar bio');
  if (!post) return res.status(404).json({ error: 'Post not found' });

  await Post.findByIdAndUpdate(post._id, { $inc: { viewCount: 1 } });
  res.json(post);
};

export const createPost = async (req, res) => {
  const post = await Post.create(req.body);
  res.status(201).json(post);
};

export const updatePost = async (req, res) => {
  const post = await Post.findByIdAndUpdate(req.params.id, req.body, { new: true, runValidators: true });
  if (!post) return res.status(404).json({ error: 'Post not found' });
  res.json(post);
};

export const publishPost = async (req, res) => {
  const post = await Post.findByIdAndUpdate(
    req.params.id,
    { published: true, publishedAt: new Date() },
    { new: true },
  );
  if (!post) return res.status(404).json({ error: 'Post not found' });
  res.json(post);
};

export const deletePost = async (req, res) => {
  const post = await Post.findByIdAndDelete(req.params.id);
  if (!post) return res.status(404).json({ error: 'Post not found' });
  await Comment.deleteMany({ post: req.params.id });
  res.status(204).send();
};

export const listTags = async (_req, res) => {
  const tags = await Post.aggregate([
    { $match: { published: true } },
    { $unwind: '$tags' },
    { $group: { _id: '$tags', count: { $sum: 1 } } },
    { $sort: { count: -1 } },
    { $project: { tag: '$_id', count: 1, _id: 0 } },
  ]);
  res.json(tags);
};
```

## Deliverables

- [ ] `Author`, `Post`, and `Comment` Mongoose models with proper indexes
- [ ] Slug auto-generated from title via `pre('validate')` hook
- [ ] `GET /api/posts` supports offset and cursor pagination
- [ ] `GET /api/posts/:slug` populates author and increments `viewCount`
- [ ] `PATCH /api/posts/:id/publish` sets `published: true` and `publishedAt`
- [ ] `DELETE /api/posts/:id` also deletes associated comments
- [ ] `GET /api/posts/tags` returns tags with post counts via aggregation pipeline
- [ ] MongoDB connected with event-based logging
