# A6.1 — File Uploads

**Phase:** 6 — File Handling & External Services

**Type:** Assignment

## Goal
Accept file uploads, validate file types, generate image thumbnails with sharp, and store files in AWS S3 (or compatible storage).

## Background

### Multer
Multer is Express middleware for handling `multipart/form-data` requests (file uploads). It can store files on disk (`multer.diskStorage`) or in memory as a Buffer (`multer.memoryStorage`). Use memory storage when you need to process the buffer (resize with sharp, upload to S3) before writing to disk. Use disk storage when you want to write the file directly and don't need to transform it.

Multer does not validate file types — it passes whatever the client sends. Always validate MIME type on the server side by checking the actual file signature (magic bytes), not just the `Content-Type` header which the client controls.

### sharp
`sharp` is a Node.js image processing library built on libvips. It's fast (uses native code, not JavaScript) and supports resize, crop, rotate, convert formats, and compression. Key API: `sharp(buffer).resize(width, height).jpeg({ quality: 80 }).toBuffer()`.

### S3
AWS S3 (Simple Storage Service) stores objects (files) in buckets. The `@aws-sdk/client-s3` SDK provides `PutObjectCommand` to upload and `GetObjectCommand` to download. For browser-accessible uploads, generate presigned URLs that allow direct browser-to-S3 upload without going through your server.

Use environment variables for credentials — never hardcode AWS keys.

### 2. Multer Configuration

```js
// src/config/upload.js
const multer = require('multer');
const path = require('node:path');

const MAX_SIZE_BYTES = parseInt(process.env.MAX_FILE_SIZE_MB ?? '5') * 1024 * 1024;

// Allowed MIME types with their magic bytes
const ALLOWED_TYPES = {
  'image/jpeg': [Buffer.from([0xFF, 0xD8, 0xFF])],
  'image/png': [Buffer.from([0x89, 0x50, 0x4E, 0x47])],
  'image/webp': [Buffer.from([0x52, 0x49, 0x46, 0x46])],
  'application/pdf': [Buffer.from([0x25, 0x50, 0x44, 0x46])],
};

const validateMagicBytes = (buffer, mimeType) => {
  const signatures = ALLOWED_TYPES[mimeType];
  if (!signatures) return false;

  return signatures.some(sig => {
    const slice = buffer.slice(0, sig.length);
    return slice.equals(sig);
  });
};

// Memory storage for processing before S3 upload
const memoryStorage = multer.memoryStorage();

// Disk storage for local uploads (development)
const diskStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, 'uploads/');
  },
  filename: (req, file, cb) => {
    const ext = path.extname(file.originalname).toLowerCase();
    const uniqueName = `${Date.now()}-${Math.round(Math.random() * 1e9)}${ext}`;
    cb(null, uniqueName);
  },
});

const fileFilter = (req, file, cb) => {
  if (!Object.keys(ALLOWED_TYPES).includes(file.mimetype)) {
    return cb(new Error(`File type ${file.mimetype} is not allowed`), false);
  }
  cb(null, true);
};

const uploadMemory = multer({
  storage: memoryStorage,
  limits: { fileSize: MAX_SIZE_BYTES, files: 5 },
  fileFilter,
});

const uploadDisk = multer({
  storage: diskStorage,
  limits: { fileSize: MAX_SIZE_BYTES, files: 5 },
  fileFilter,
});

module.exports = { uploadMemory, uploadDisk, validateMagicBytes, ALLOWED_TYPES };
```

### 3. S3 Client

```js
// src/config/s3.js
const { S3Client, PutObjectCommand, DeleteObjectCommand, GetObjectCommand } = require('@aws-sdk/client-s3');
const { getSignedUrl } = require('@aws-sdk/s3-request-presigner');

const s3 = new S3Client({
  region: process.env.AWS_REGION ?? 'us-east-1',
  credentials: {
    accessKeyId: process.env.AWS_ACCESS_KEY_ID,
    secretAccessKey: process.env.AWS_SECRET_ACCESS_KEY,
  },
});

const BUCKET = process.env.S3_BUCKET;

const uploadToS3 = async ({ key, buffer, contentType, metadata = {} }) => {
  await s3.send(new PutObjectCommand({
    Bucket: BUCKET,
    Key: key,
    Body: buffer,
    ContentType: contentType,
    Metadata: metadata,
  }));

  return `https://${BUCKET}.s3.${process.env.AWS_REGION}.amazonaws.com/${key}`;
};

const deleteFromS3 = async (key) => {
  await s3.send(new DeleteObjectCommand({ Bucket: BUCKET, Key: key }));
};

const getPresignedUrl = async (key, expiresInSeconds = 3600) => {
  const command = new GetObjectCommand({ Bucket: BUCKET, Key: key });
  return getSignedUrl(s3, command, { expiresIn: expiresInSeconds });
};

const getUploadPresignedUrl = async (key, contentType, expiresInSeconds = 300) => {
  const command = new PutObjectCommand({
    Bucket: BUCKET,
    Key: key,
    ContentType: contentType,
  });
  return getSignedUrl(s3, command, { expiresIn: expiresInSeconds });
};

module.exports = { uploadToS3, deleteFromS3, getPresignedUrl, getUploadPresignedUrl };
```

### 4. Image Processing

```js
// src/services/imageProcessor.js
const sharp = require('sharp');
const { v4: uuidv4 } = require('uuid');
const { uploadToS3 } = require('../config/s3');
const { validateMagicBytes } = require('../config/upload');

const THUMBNAIL_SIZES = {
  thumb: { width: 150, height: 150 },
  medium: { width: 600, height: 600 },
  large: { width: 1200, height: 1200 },
};

const processAndUploadImage = async (file, userId) => {
  const { buffer, mimetype, originalname } = file;

  // Validate magic bytes (not just Content-Type header)
  if (!validateMagicBytes(buffer, mimetype)) {
    throw Object.assign(new Error('File content does not match declared MIME type'), { statusCode: 400 });
  }

  // Get image metadata
  const metadata = await sharp(buffer).metadata();
  const { width, height, format } = metadata;

  const fileId = uuidv4();
  const baseKey = `uploads/${userId}/${fileId}`;
  const results = {};

  // Upload original
  const originalKey = `${baseKey}/original.${format}`;
  const originalUrl = await uploadToS3({
    key: originalKey,
    buffer,
    contentType: mimetype,
    metadata: { userId, originalName: originalname },
  });

  results.original = { url: originalUrl, width, height, size: buffer.length };

  // Generate and upload thumbnails
  for (const [sizeName, dimensions] of Object.entries(THUMBNAIL_SIZES)) {
    if (width > dimensions.width || height > dimensions.height) {
      const thumbBuffer = await sharp(buffer)
        .resize(dimensions.width, dimensions.height, {
          fit: 'inside',           // maintain aspect ratio
          withoutEnlargement: true, // don't upscale small images
        })
        .jpeg({ quality: 80, progressive: true })
        .toBuffer();

      const thumbKey = `${baseKey}/${sizeName}.jpg`;
      const thumbUrl = await uploadToS3({
        key: thumbKey,
        buffer: thumbBuffer,
        contentType: 'image/jpeg',
      });

      const thumbMeta = await sharp(thumbBuffer).metadata();
      results[sizeName] = {
        url: thumbUrl,
        width: thumbMeta.width,
        height: thumbMeta.height,
        size: thumbBuffer.length,
      };
    } else {
      results[sizeName] = results.original; // image smaller than thumbnail size
    }
  }

  return { fileId, baseKey, variants: results };
};

module.exports = { processAndUploadImage };
```

### 5. Upload Routes

```js
// src/routes/upload.js
const router = require('express').Router();
const { uploadMemory } = require('../config/upload');
const { processAndUploadImage } = require('../services/imageProcessor');
const { getUploadPresignedUrl } = require('../config/s3');
const { v4: uuidv4 } = require('uuid');

// Server-side upload (server receives file, processes, uploads to S3)
router.post('/image',
  uploadMemory.single('image'), // 'image' is the form field name
  async (req, res, next) => {
    try {
      if (!req.file) {
        return res.status(400).json({ error: 'No file uploaded' });
      }

      const userId = req.user?.id ?? 'anonymous';
      const result = await processAndUploadImage(req.file, userId);

      res.status(201).json({
        message: 'Image uploaded successfully',
        fileId: result.fileId,
        variants: result.variants,
      });
    } catch (err) {
      if (err.statusCode) return res.status(err.statusCode).json({ error: err.message });
      next(err);
    }
  }
);

// Multiple files
router.post('/images',
  uploadMemory.array('images', 5),
  async (req, res, next) => {
    try {
      if (!req.files?.length) {
        return res.status(400).json({ error: 'No files uploaded' });
      }

      const userId = req.user?.id ?? 'anonymous';
      const results = await Promise.all(
        req.files.map(f => processAndUploadImage(f, userId))
      );

      res.status(201).json({ files: results.map(r => r.variants) });
    } catch (err) { next(err); }
  }
);

// Client-side upload — get presigned URL, then client uploads directly to S3
router.post('/presigned-url', async (req, res, next) => {
  try {
    const { contentType, filename } = req.body;

    const allowedTypes = ['image/jpeg', 'image/png', 'image/webp'];
    if (!allowedTypes.includes(contentType)) {
      return res.status(400).json({ error: 'Content type not allowed' });
    }

    const fileId = uuidv4();
    const key = `uploads/${fileId}/${filename}`;

    const uploadUrl = await getUploadPresignedUrl(key, contentType, 300);

    res.json({
      uploadUrl,
      key,
      fileId,
      expiresIn: 300,
      instructions: 'PUT the file directly to uploadUrl',
    });
  } catch (err) { next(err); }
});

// Multer error handling
router.use((err, req, res, next) => {
  if (err.code === 'LIMIT_FILE_SIZE') {
    return res.status(413).json({
      error: `File too large. Maximum size: ${process.env.MAX_FILE_SIZE_MB ?? 5}MB`,
    });
  }
  if (err.code === 'LIMIT_FILE_COUNT') {
    return res.status(400).json({ error: 'Too many files' });
  }
  if (err.message?.includes('File type')) {
    return res.status(400).json({ error: err.message });
  }
  next(err);
});

module.exports = router;
```
