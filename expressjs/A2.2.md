## Goal

## Background

### Sequelize and SQL ORMs
Sequelize is a promise-based ORM for Node.js that supports PostgreSQL, MySQL, SQLite, and MSSQL. It abstracts SQL queries into JavaScript method calls: `User.findAll()` generates `SELECT * FROM "Users"`. Unlike Mongoose (schema-first), Sequelize can sync with an existing database or manage schema changes via migrations.

The Sequelize instance is a database connection. Models are registered on it with `sequelize.define()` or by extending `Model`. Once defined, models expose class methods for queries (`findOne`, `findAll`, `create`, `update`, `destroy`) and instance methods for document-level operations.

### Associations
Sequelize associations define foreign keys and enable JOIN queries:
- `User.hasMany(Post)` → adds `userId` to Posts table, gives User instance `getPosts()`
- `Post.belongsTo(User)` → gives Post instance `getUser()`
- `Post.belongsToMany(Tag, { through: 'PostTags' })` → many-to-many via junction table

Both sides of an association must be declared for it to work fully. After declaring associations, use `include` in queries to JOIN: `User.findOne({ include: Post })`.

### Migrations
Rather than calling `sequelize.sync()` in production (which can destroy data on schema changes), use migrations. Each migration file has an `up` function (apply change) and `down` function (revert). Sequelize CLI (`sequelize-cli`) manages a `SequelizeMeta` table tracking which migrations have run.

### 4. Models with Associations

```js
// src/models/User.js
const { Model, DataTypes } = require('sequelize');

class User extends Model {
  static associate(models) {
    User.hasMany(models.Post, { foreignKey: 'userId', as: 'posts' });
  }
}

module.exports = (sequelize) => {
  User.init(
    {
      name: {
        type: DataTypes.STRING(100),
        allowNull: false,
        validate: { notEmpty: { msg: 'Name cannot be empty' } },
      },
      email: {
        type: DataTypes.STRING,
        allowNull: false,
        unique: true,
        validate: { isEmail: { msg: 'Must be a valid email' } },
      },
      bio: { type: DataTypes.TEXT },
    },
    { sequelize, modelName: 'User' }
  );

  return User;
};
```

```js
// src/models/Post.js
const { Model, DataTypes } = require('sequelize');

class Post extends Model {
  static associate(models) {
    Post.belongsTo(models.User, { foreignKey: 'userId', as: 'author' });
    Post.belongsToMany(models.Tag, { through: 'PostTags', as: 'tags' });
  }
}

module.exports = (sequelize) => {
  Post.init(
    {
      title: { type: DataTypes.STRING, allowNull: false },
      body: { type: DataTypes.TEXT, allowNull: false },
      published: { type: DataTypes.BOOLEAN, defaultValue: false },
      userId: { type: DataTypes.INTEGER },
    },
    { sequelize, modelName: 'Post' }
  );

  return Post;
};
```

```js
// src/models/index.js
const { sequelize } = require('../config/database');
const UserModel = require('./User');
const PostModel = require('./Post');

const User = UserModel(sequelize);
const Post = PostModel(sequelize);

// Run associations
const models = { User, Post };
Object.values(models).forEach(model => {
  if (model.associate) model.associate(models);
});

module.exports = { User, Post, sequelize };
```

### 5. Routes with Eager Loading

```js
// src/routes/posts.js
const router = require('express').Router();
const { Post, User } = require('../models');

router.get('/', async (req, res, next) => {
  try {
    const posts = await Post.findAll({
      where: { published: true },
      include: [{ model: User, as: 'author', attributes: ['id', 'name'] }],
      order: [['createdAt', 'DESC']],
    });
    res.json({ data: posts });
  } catch (err) { next(err); }
});

router.post('/', async (req, res, next) => {
  try {
    const post = await Post.create(req.body);
    res.status(201).json(post);
  } catch (err) {
    if (err.name === 'SequelizeValidationError') {
      return res.status(400).json({
        error: err.errors.map(e => e.message),
      });
    }
    next(err);
  }
});

module.exports = router;
```
