## Goal

## Background

### Passport.js Architecture
Passport.js is authentication middleware for Express. Rather than implementing auth logic inline, you configure "strategies" — pluggable modules for different authentication methods. Passport normalizes the result: on success, it calls `req.logIn(user)` (session-based) or passes the user to `req.user` (stateless).

Passport has 500+ strategies on npm: local (username/password), JWT, OAuth providers (Google, GitHub, Facebook), SAML, LDAP, etc. This pluggable architecture lets you add or swap auth methods without restructuring the app.

### Sessions vs Stateless JWT
**Session-based auth:** On login, server creates a session object (stored in memory, Redis, or a database) and sends a `session_id` cookie. On each request, the cookie is sent, the server looks up the session, and loads the user. State lives on the server.

**Stateless JWT:** On login, server issues a signed JWT. On each request, the JWT is sent in the Authorization header. The server verifies the signature — no database lookup needed. State lives in the token.

Use sessions for traditional web apps where you control the client. Use JWT for APIs consumed by mobile apps, SPAs, or third-party clients. Sessions are easier to invalidate (delete the session record); JWTs require a blacklist for immediate invalidation.

### OAuth2 Flow
OAuth2 allows users to authenticate via a third-party provider (Google) without sharing credentials with your app. The flow:
1. User clicks "Login with Google"
2. User is redirected to Google's consent screen
3. Google redirects back to your callback URL with an authorization code
4. Your server exchanges the code for an access token
5. Your server fetches user profile from Google using the access token
6. Your server creates/updates a local user record and issues a session or JWT

### 2. Local Strategy (username/password)

```js
// src/strategies/local.strategy.js
const LocalStrategy = require('passport-local').Strategy;
const bcrypt = require('bcryptjs');
const userStore = require('../data/users');

const localStrategy = new LocalStrategy(
  {
    usernameField: 'email',   // field name in req.body
    passwordField: 'password',
  },
  async (email, password, done) => {
    try {
      const user = userStore.findByEmail(email);

      if (!user) {
        return done(null, false, { message: 'Invalid credentials' });
      }

      const match = await bcrypt.compare(password, user.password);
      if (!match) {
        return done(null, false, { message: 'Invalid credentials' });
      }

      return done(null, user); // success — user is passed to req.user
    } catch (err) {
      return done(err); // server error
    }
  }
);

module.exports = localStrategy;
```

### 3. JWT Strategy

```js
// src/strategies/jwt.strategy.js
const { Strategy: JwtStrategy, ExtractJwt } = require('passport-jwt');
const userStore = require('../data/users');

const jwtStrategy = new JwtStrategy(
  {
    jwtFromRequest: ExtractJwt.fromAuthHeaderAsBearerToken(),
    secretOrKey: process.env.JWT_SECRET ?? 'dev-secret',
    // Could also use secretOrKeyProvider for per-request key selection
  },
  async (payload, done) => {
    try {
      const user = userStore.findById(payload.sub);
      if (!user) return done(null, false);
      return done(null, user);
    } catch (err) {
      return done(err);
    }
  }
);

module.exports = jwtStrategy;
```

### 4. Google OAuth2 Strategy

```js
// src/strategies/google.strategy.js
const GoogleStrategy = require('passport-google-oauth20').Strategy;
const userStore = require('../data/users');

const googleStrategy = new GoogleStrategy(
  {
    clientID: process.env.GOOGLE_CLIENT_ID,
    clientSecret: process.env.GOOGLE_CLIENT_SECRET,
    callbackURL: process.env.GOOGLE_CALLBACK_URL ?? 'http://localhost:3000/auth/google/callback',
    scope: ['profile', 'email'],
  },
  async (accessToken, refreshToken, profile, done) => {
    try {
      const email = profile.emails?.[0]?.value;
      if (!email) return done(new Error('No email from Google profile'));

      // Find or create user
      let user = userStore.findByEmail(email);

      if (!user) {
        user = await userStore.createOAuth({
          name: profile.displayName,
          email,
          googleId: profile.id,
          avatar: profile.photos?.[0]?.value,
        });
      }

      return done(null, user);
    } catch (err) {
      return done(err);
    }
  }
);

module.exports = googleStrategy;
```

### 5. Passport Configuration

```js
// src/config/passport.js
const passport = require('passport');
const localStrategy = require('../strategies/local.strategy');
const jwtStrategy = require('../strategies/jwt.strategy');
const googleStrategy = require('../strategies/google.strategy');
const userStore = require('../data/users');

passport.use('local', localStrategy);
passport.use('jwt', jwtStrategy);
passport.use('google', googleStrategy);

// Session serialization (only needed for session-based auth)
passport.serializeUser((user, done) => {
  done(null, user.id); // store user.id in session
});

passport.deserializeUser((id, done) => {
  const user = userStore.findById(id);
  done(null, user ?? false);
});

module.exports = passport;
```

### 6. Auth Routes

```js
// src/routes/auth.js
const router = require('express').Router();
const passport = require('../config/passport');
const jwt = require('jsonwebtoken');

// --- Local / Session-based login ---
router.post('/login/session', (req, res, next) => {
  passport.authenticate('local', (err, user, info) => {
    if (err) return next(err);
    if (!user) return res.status(401).json({ error: info?.message ?? 'Invalid credentials' });

    req.logIn(user, (err) => {
      if (err) return next(err);
      res.json({ message: 'Logged in', user: { id: user.id, name: user.name } });
    });
  })(req, res, next); // IIFE pattern — authenticate returns a middleware
});

router.post('/logout/session', (req, res) => {
  req.logout((err) => {
    if (err) return res.status(500).json({ error: 'Logout failed' });
    res.json({ message: 'Logged out' });
  });
});

// --- Local / JWT-based login ---
router.post('/login', (req, res, next) => {
  passport.authenticate('local', { session: false }, (err, user, info) => {
    if (err) return next(err);
    if (!user) return res.status(401).json({ error: info?.message ?? 'Invalid credentials' });

    const token = jwt.sign(
      { sub: user.id, name: user.name },
      process.env.JWT_SECRET ?? 'dev-secret',
      { expiresIn: '15m' }
    );

    res.json({ token, user: { id: user.id, name: user.name, email: user.email } });
  })(req, res, next);
});

// --- Protected with JWT ---
router.get('/profile',
  passport.authenticate('jwt', { session: false }),
  (req, res) => {
    res.json({ user: req.user });
  }
);

// --- Google OAuth2 ---
router.get('/google',
  passport.authenticate('google')
  // Redirects to Google — no response here
);

router.get('/google/callback',
  passport.authenticate('google', { session: false, failureRedirect: '/login' }),
  (req, res) => {
    // Issue JWT after Google auth
    const token = jwt.sign(
      { sub: req.user.id },
      process.env.JWT_SECRET ?? 'dev-secret',
      { expiresIn: '15m' }
    );

    // In a real app, redirect to frontend with token as query param or cookie
    res.json({ message: 'Google auth successful', token });
  }
);

module.exports = router;
```

### 7. App Assembly

```js
// src/app.js
const express = require('express');
const session = require('express-session');
const passport = require('./config/passport');
const authRoutes = require('./routes/auth');

const app = express();
app.use(express.json());

// Session middleware (needed for session-based auth only)
app.use(session({
  secret: process.env.SESSION_SECRET ?? 'keyboard cat',
  resave: false,
  saveUninitialized: false,
  cookie: {
    httpOnly: true,
    secure: process.env.NODE_ENV === 'production',
    maxAge: 24 * 60 * 60 * 1000, // 1 day
  },
}));

app.use(passport.initialize());
app.use(passport.session()); // only needed for session-based auth

app.use('/auth', authRoutes);

module.exports = app;
```
