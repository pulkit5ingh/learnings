## Goal
Implement robust pagination and filtering for list endpoints, understanding the tradeoffs between offset and cursor-based approaches.

## Background

### Offset vs Cursor Pagination
**Offset pagination** uses `SKIP` and `LIMIT`. Page 3 with limit 10 = `SKIP 20, LIMIT 10`. Simple to implement and allows jumping to arbitrary pages. Problem: if new items are inserted or deleted while paginating, items can appear twice or be skipped (the "page drift" problem). Also, large `SKIP` values force the database to scan through all skipped rows — inefficient on large datasets.

**Cursor pagination** uses a stable pointer (usually the ID or a timestamp of the last seen item). Instead of "give me page 3," you say "give me 10 items after ID 42." No page drift because the cursor anchors to a specific record. Cannot jump to arbitrary pages, but that's rarely needed in infinite-scroll UIs. Efficient because the database can use indexes to find the cursor position.

### Sorting and Filtering
Expose sort and filter as query parameters: `?sort=-createdAt&minPrice=10&category=electronics`. Always whitelist allowed sort fields — never pass user sort values directly to the database. Convert filter parameters to database-appropriate types (strings to numbers, handle missing values).

For text search, use database-native full-text search (MongoDB's `$text`, PostgreSQL's `tsvector`) rather than regex, which can't use indexes.

```js
// src/utils/pagination.js

// Parse pagination params from query string
const parsePagination = (query, defaults = {}) => {
  const page = Math.max(1, parseInt(query.page ?? defaults.page ?? 1));
  const limit = Math.min(
    parseInt(query.limit ?? defaults.limit ?? 20),
    defaults.maxLimit ?? 100
  );
  const skip = (page - 1) * limit;

  return { page, limit, skip };
};

// Build pagination metadata
const buildMeta = (total, page, limit) => {
  const totalPages = Math.ceil(total / limit);

  return {
    total,
    page,
    limit,
    totalPages,
    hasNextPage: page < totalPages,
    hasPrevPage: page > 1,
    nextPage: page < totalPages ? page + 1 : null,
    prevPage: page > 1 ? page - 1 : null,
  };
};

// Parse sort string: "-createdAt,name" → { createdAt: -1, name: 1 }
const parseSort = (sortString, allowedFields = []) => {
  if (!sortString) return { createdAt: -1 };

  const sortObj = {};

  sortString.split(',').forEach(field => {
    const direction = field.startsWith('-') ? -1 : 1;
    const name = field.replace(/^-/, '');

    // Only allow whitelisted sort fields
    if (allowedFields.length === 0 || allowedFields.includes(name)) {
      sortObj[name] = direction;
    }
  });

  return Object.keys(sortObj).length > 0 ? sortObj : { createdAt: -1 };
};

// Parse filter from query (returns Mongoose-compatible filter object)
const parseFilter = (query, fieldMap) => {
  const filter = {};

  for (const [queryKey, { field, type, operator }] of Object.entries(fieldMap)) {
    const value = query[queryKey];
    if (value === undefined || value === '') continue;

    const castValue = type === 'number' ? parseFloat(value) : value;
    const op = operator ?? '$eq';

    if (op === '$eq') {
      filter[field] = castValue;
    } else {
      filter[field] = { ...(filter[field] ?? {}), [op]: castValue };
    }
  }

  return filter;
};

module.exports = { parsePagination, buildMeta, parseSort, parseFilter };
```

### 3. Cursor Pagination

```js
// src/utils/cursorPagination.js
const { Types } = require('mongoose');

// Encode/decode cursor (base64 of JSON)
const encodeCursor = (obj) =>
  Buffer.from(JSON.stringify(obj)).toString('base64url');

const decodeCursor = (cursor) => {
  try {
    return JSON.parse(Buffer.from(cursor, 'base64url').toString('utf8'));
  } catch {
    return null;
  }
};

// Build Mongoose query for cursor-based pagination
// Assumes sorting by createdAt DESC + _id DESC (stable sort)
const buildCursorQuery = (after, before) => {
  if (after) {
    const decoded = decodeCursor(after);
    if (!decoded) return {};

    return {
      $or: [
        { createdAt: { $lt: new Date(decoded.createdAt) } },
        {
          createdAt: new Date(decoded.createdAt),
          _id: { $lt: new Types.ObjectId(decoded.id) },
        },
      ],
    };
  }

  if (before) {
    const decoded = decodeCursor(before);
    if (!decoded) return {};

    return {
      $or: [
        { createdAt: { $gt: new Date(decoded.createdAt) } },
        {
          createdAt: new Date(decoded.createdAt),
          _id: { $gt: new Types.ObjectId(decoded.id) },
        },
      ],
    };
  }

  return {};
};

const buildCursorResponse = (items, limit) => {
  const hasMore = items.length > limit;
  const results = hasMore ? items.slice(0, limit) : items;

  const startCursor = results.length > 0
    ? encodeCursor({ id: results[0]._id, createdAt: results[0].createdAt })
    : null;

  const endCursor = results.length > 0
    ? encodeCursor({ id: results[results.length - 1]._id, createdAt: results[results.length - 1].createdAt })
    : null;

  return {
    data: results,
    pageInfo: {
      hasNextPage: hasMore,
      hasPreviousPage: false, // simplified
      startCursor,
      endCursor,
      count: results.length,
    },
  };
};

module.exports = { encodeCursor, decodeCursor, buildCursorQuery, buildCursorResponse };
```

const mongoose = require('mongoose');

const postSchema = new mongoose.Schema({
  title: { type: String, required: true },
  body: { type: String, required: true },
  category: { type: String, enum: ['tech', 'lifestyle', 'news'], required: true },
  tags: [String],
  views: { type: Number, default: 0 },
  price: Number,
  authorId: { type: Number, required: true },
  published: { type: Boolean, default: true },
}, { timestamps: true });

// Text index for search
postSchema.index({ title: 'text', body: 'text' });

// Compound index for cursor pagination
postSchema.index({ createdAt: -1, _id: -1 });

// Index for common filter + sort combinations
postSchema.index({ category: 1, createdAt: -1 });
postSchema.index({ authorId: 1, createdAt: -1 });

module.exports = mongoose.model('Post', postSchema);

### 5. Routes with Pagination

```js
// src/routes/posts.js
const router = require('express').Router();
const Post = require('../models/Post');
const { parsePagination, buildMeta, parseSort, parseFilter } = require('../utils/pagination');
const { buildCursorQuery, buildCursorResponse } = require('../utils/cursorPagination');

const ALLOWED_SORT_FIELDS = ['title', 'views', 'createdAt', 'price'];

// Offset pagination
router.get('/', async (req, res, next) => {
  try {
    const { page, limit, skip } = parsePagination(req.query);
    const sort = parseSort(req.query.sort, ALLOWED_SORT_FIELDS);

    const filter = parseFilter(req.query, {
      category: { field: 'category', type: 'string' },
      authorId: { field: 'authorId', type: 'number' },
      minPrice: { field: 'price', type: 'number', operator: '$gte' },
      maxPrice: { field: 'price', type: 'number', operator: '$lte' },
      minViews: { field: 'views', type: 'number', operator: '$gte' },
    });

    filter.published = true;

    // Full-text search
    if (req.query.search) {
      filter.$text = { $search: req.query.search };
    }

    const [posts, total] = await Promise.all([
      Post.find(filter)
        .sort(sort)
        .skip(skip)
        .limit(limit)
        .select('-body') // omit body for list view
        .lean(),
      Post.countDocuments(filter),
    ]);

    res.json({
      data: posts,
      meta: buildMeta(total, page, limit),
    });
  } catch (err) { next(err); }
});

// Cursor pagination
router.get('/feed', async (req, res, next) => {
  try {
    const limit = Math.min(parseInt(req.query.limit ?? '20'), 50);
    const { after, before, category } = req.query;

    const filter = {
      published: true,
      ...buildCursorQuery(after, before),
      ...(category && { category }),
    };

    // Fetch limit+1 to determine if there's a next page
    const posts = await Post.find(filter)
      .sort({ createdAt: -1, _id: -1 })
      .limit(limit + 1)
      .select('-body')
      .lean();

    res.json(buildCursorResponse(posts, limit));
  } catch (err) { next(err); }
});

module.exports = router;
```

### 6. Aggregation Pipeline Example

```js
// Category stats with aggregation
router.get('/stats', async (req, res, next) => {
  try {
    const stats = await Post.aggregate([
      { $match: { published: true } },
      {
        $group: {
          _id: '$category',
          count: { $sum: 1 },
          totalViews: { $sum: '$views' },
          avgViews: { $avg: '$views' },
          maxViews: { $max: '$views' },
        },
      },
      { $sort: { totalViews: -1 } },
      {
        $project: {
          category: '$_id',
          count: 1,
          totalViews: 1,
          avgViews: { $round: ['$avgViews', 1] },
          maxViews: 1,
          _id: 0,
        },
      },
    ]);

    res.json({ data: stats });
  } catch (err) { next(err); }
});
```
