## Background

### Express as a Minimal Framework
Express is essentially a thin layer over Node.js's built-in `http` module. It adds a routing system, middleware pipeline, and convenience methods on the `req` and `res` objects. When you call `express()`, you get an `Application` object whose core job is matching incoming HTTP requests to handler functions.

Every request-response pair goes through the same cycle: a request arrives, Express matches it against registered routes in registration order, the matching route's handler receives `(req, res, next)`, the handler sends a response using `res.json()`, `res.send()`, `res.status().json()`, etc. If no route matches, Express falls through to a 404.

### The req and res Objects
`req` is an enhanced version of Node's `http.IncomingMessage`. Key properties:
- `req.params` — named URL segments: `/users/:id` → `req.params.id`
- `req.query` — parsed query string: `?sort=desc&page=2` → `{ sort: 'desc', page: '2' }`
- `req.body` — parsed request body (requires `express.json()` or `express.urlencoded()` middleware)
- `req.headers` — HTTP headers as an object
- `req.method` — HTTP verb as string
- `req.path` — URL path without query string
- `req.ip` — client IP address

`res` is an enhanced version of Node's `http.ServerResponse`. Key methods:
- `res.json(obj)` — sets Content-Type to application/json and serializes obj
- `res.send(str)` — sends a string body, infers Content-Type
- `res.status(code)` — sets HTTP status code (chainable: `res.status(404).json({})`)
- `res.set(header, value)` — sets a response header
- `res.redirect(url)` — sends a 302 redirect
- `res.sendFile(path)` — sends a file with appropriate Content-Type

### HTTP Methods and REST Conventions
REST uses HTTP verbs semantically:
- `GET /resources` — list all
- `GET /resources/:id` — get one
- `POST /resources` — create
- `PUT /resources/:id` — full replace
- `PATCH /resources/:id` — partial update
- `DELETE /resources/:id` — delete

Always return appropriate status codes: 200 OK, 201 Created, 204 No Content, 400 Bad Request, 404 Not Found, 500 Internal Server Error.

```js
// server.js
const express = require('express');

const app = express();
const PORT = process.env.PORT ?? 3000;

// Parse JSON bodies
app.use(express.json());

// Parse URL-encoded bodies (HTML forms)
app.use(express.urlencoded({ extended: true }));

// Serve static files from /public
app.use(express.static('public'));

app.listen(PORT, () => {
  console.log(`Server running on http://localhost:${PORT}`);
});
```

### 2. All HTTP Methods

```js
// routes/items.js — add these to server.js for now
const items = [
  { id: 1, name: 'Apple', price: 1.5 },
  { id: 2, name: 'Banana', price: 0.5 },
];
let nextId = 3;

// GET all with optional query filters
app.get('/items', (req, res) => {
  const { sort, minPrice } = req.query;

  let result = [...items];

  if (minPrice) {
    result = result.filter(i => i.price >= parseFloat(minPrice));
  }

  if (sort === 'name') {
    result.sort((a, b) => a.name.localeCompare(b.name));
  } else if (sort === 'price') {
    result.sort((a, b) => a.price - b.price);
  }

  res.json({ data: result, count: result.length });
});

// GET one by ID
app.get('/items/:id', (req, res) => {
  const id = parseInt(req.params.id, 10);
  const item = items.find(i => i.id === id);

  if (!item) {
    return res.status(404).json({ error: 'Item not found', id });
  }

  res.json(item);
});

// POST — create
app.post('/items', (req, res) => {
  const { name, price } = req.body;

  if (!name || price === undefined) {
    return res.status(400).json({ error: 'name and price are required' });
  }

  const item = { id: nextId++, name, price: parseFloat(price) };
  items.push(item);

  res.status(201).json(item);
});

// PUT — full replace
app.put('/items/:id', (req, res) => {
  const id = parseInt(req.params.id, 10);
  const index = items.findIndex(i => i.id === id);

  if (index === -1) {
    return res.status(404).json({ error: 'Item not found' });
  }

  const { name, price } = req.body;
  if (!name || price === undefined) {
    return res.status(400).json({ error: 'name and price are required for full replace' });
  }

  items[index] = { id, name, price: parseFloat(price) };
  res.json(items[index]);
});

// PATCH — partial update
app.patch('/items/:id', (req, res) => {
  const id = parseInt(req.params.id, 10);
  const index = items.findIndex(i => i.id === id);

  if (index === -1) {
    return res.status(404).json({ error: 'Item not found' });
  }

  items[index] = { ...items[index], ...req.body, id };
  res.json(items[index]);
});

// DELETE
app.delete('/items/:id', (req, res) => {
  const id = parseInt(req.params.id, 10);
  const index = items.findIndex(i => i.id === id);

  if (index === -1) {
    return res.status(404).json({ error: 'Item not found' });
  }

  const deleted = items.splice(index, 1)[0];
  res.status(200).json({ message: 'Deleted', item: deleted });
});
```

### 3. Route Parameters and Query Strings

```js
// Nested params example
app.get('/users/:userId/posts/:postId/comments', (req, res) => {
  const { userId, postId } = req.params;
  const { page = '1', limit = '10', sort = 'createdAt' } = req.query;

  res.json({
    userId: parseInt(userId),
    postId: parseInt(postId),
    pagination: {
      page: parseInt(page),
      limit: parseInt(limit),
      sort,
    },
    // In a real app: fetch from DB here
    comments: [],
  });
});

// Wildcard route — must come after specific routes
app.get('/files/*', (req, res) => {
  const filePath = req.params[0]; // everything after /files/
  res.json({ requestedPath: filePath });
});
```
