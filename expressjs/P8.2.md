# P8.2 — Production-grade API

**Phase:** 8 — Deployment & DevOps  
**Topic:** Graceful shutdown, health checks, Prometheus metrics, rate limiting  
**Type:** Project

---

## Goal

Harden an Express API for production: graceful shutdown, structured logging, `/health` readiness probes, Prometheus metrics endpoint, and rate limiting with Redis.

## What to Build

- Graceful shutdown on `SIGTERM`/`SIGINT` with configurable timeout
- `/health/live` and `/health/ready` probes
- `/metrics` in Prometheus exposition format using `prom-client`
- Rate limiting per IP backed by Redis (`rate-limiter-flexible`)
- Request duration histogram, active connections gauge, error counter
- Structured JSON logs with winston (no colorization in production)

```js
import { Registry, collectDefaultMetrics, Counter, Histogram, Gauge } from 'prom-client';

export const register = new Registry();
register.setDefaultLabels({ app: 'express-api', env: process.env.NODE_ENV });

// Collect default Node.js metrics (heap, GC, event loop lag)
collectDefaultMetrics({ register });

export const httpRequestDuration = new Histogram({
  name: 'http_request_duration_seconds',
  help: 'HTTP request duration in seconds',
  labelNames: ['method', 'route', 'status_code'],
  buckets: [0.005, 0.01, 0.025, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5],
  registers: [register],
});

export const httpRequestTotal = new Counter({
  name: 'http_requests_total',
  help: 'Total HTTP request count',
  labelNames: ['method', 'route', 'status_code'],
  registers: [register],
});

export const httpErrorTotal = new Counter({
  name: 'http_errors_total',
  help: 'Total HTTP errors (4xx and 5xx)',
  labelNames: ['status_code'],
  registers: [register],
});

export const activeConnections = new Gauge({
  name: 'http_active_connections',
  help: 'Number of active HTTP connections',
  registers: [register],
});
```

```js
// src/middleware/requestLogger.js
import { httpRequestDuration, httpRequestTotal, httpErrorTotal } from '../lib/metrics.js';
import { logger } from '../lib/logger.js';
import { randomUUID } from 'node:crypto';

export const requestLogger = (req, res, next) => {
  req.id = req.headers['x-request-id'] ?? randomUUID();
  res.set('x-request-id', req.id);

  const start = process.hrtime.bigint();

  res.on('finish', () => {
    const durationNs = process.hrtime.bigint() - start;
    const durationSec = Number(durationNs) / 1e9;

    // Normalize route for metric labels (avoid high cardinality)
    const route = req.route?.path ?? req.path;

    httpRequestDuration.observe({ method: req.method, route, status_code: res.statusCode }, durationSec);
    httpRequestTotal.inc({ method: req.method, route, status_code: res.statusCode });

    if (res.statusCode >= 400) {
      httpErrorTotal.inc({ status_code: res.statusCode });
    }

    logger.info({
      type:      'request',
      requestId: req.id,
      method:    req.method,
      url:       req.originalUrl,
      status:    res.statusCode,
      ms:        (durationSec * 1000).toFixed(2),
      ip:        req.ip,
      userAgent: req.headers['user-agent'],
    });
  });

  next();
};
```

```js
// src/middleware/rateLimiter.js
import { RateLimiterRedis } from 'rate-limiter-flexible';
import Redis from 'ioredis';

const redisClient = new Redis({ host: process.env.REDIS_HOST ?? 'localhost', enableOfflineQueue: false });

const rateLimiter = new RateLimiterRedis({
  storeClient: redisClient,
  keyPrefix:   'rl',
  points:      100,        // requests
  duration:    15 * 60,    // per 15 minutes
  blockDuration: 60,       // block for 60 s after limit hit
});

export const rateLimiterMiddleware = async (req, res, next) => {
  try {
    const result = await rateLimiter.consume(req.ip);
    res.set('X-RateLimit-Limit',     '100');
    res.set('X-RateLimit-Remaining', String(result.remainingPoints));
    res.set('X-RateLimit-Reset',     String(Math.ceil(Date.now() / 1000 + result.msBeforeNext / 1000)));
    next();
  } catch (rejRes) {
    res.set('Retry-After', String(Math.ceil(rejRes.msBeforeNext / 1000)));
    res.status(429).json({ error: 'Too many requests', retryAfter: Math.ceil(rejRes.msBeforeNext / 1000) });
  }
};
```

```js
// src/routes/health.js
import { Router } from 'express';
import mongoose from 'mongoose';
import { redis } from '../cache.js';

const router = Router();

router.get('/health/live', (_req, res) => {
  res.json({ status: 'ok', pid: process.pid, uptime: process.uptime() });
});

router.get('/health/ready', async (_req, res) => {
  const checks = { mongodb: 'unknown', redis: 'unknown' };
  let ok = true;

  try {
    await mongoose.connection.db.admin().ping();
    checks.mongodb = 'ok';
  } catch (err) {
    checks.mongodb = err.message;
    ok = false;
  }

  try {
    await redis.ping();
    checks.redis = 'ok';
  } catch (err) {
    checks.redis = err.message;
    ok = false;
  }

  res.status(ok ? 200 : 503).json({ status: ok ? 'ok' : 'degraded', checks });
});

export default router;
```

```js
// src/lib/shutdown.js
import { logger } from './logger.js';

export const createGracefulShutdown = (server, { timeoutMs = 10_000, onClose = async () => {} } = {}) => {
  const shutdown = async (signal) => {
    logger.info(`${signal} received — graceful shutdown started`);

    const forceExit = setTimeout(() => {
      logger.error('Graceful shutdown timeout — forcing exit');
      process.exit(1);
    }, timeoutMs);
    forceExit.unref(); // don't hold the event loop open

    server.close(async () => {
      logger.info('HTTP server closed (no new connections)');
      try {
        await onClose();
        logger.info('All connections closed — exiting cleanly');
        process.exit(0);
      } catch (err) {
        logger.error('Error during cleanup', { err });
        process.exit(1);
      }
    });
  };

  process.on('SIGTERM', () => shutdown('SIGTERM'));
  process.on('SIGINT',  () => shutdown('SIGINT'));
};
```

```js
// src/server.js
import app from './app.js';
import mongoose from 'mongoose';
import { redis } from './cache.js';
import { logger } from './lib/logger.js';
import { createGracefulShutdown } from './lib/shutdown.js';

await mongoose.connect(process.env.MONGODB_URI);

const server = app.listen(process.env.PORT ?? 4000, () => {
  logger.info(`Server listening on :${process.env.PORT ?? 4000}`);
  if (process.send) process.send('ready');
});

createGracefulShutdown(server, {
  timeoutMs: 10_000,
  onClose: async () => {
    await Promise.all([
      mongoose.connection.close(),
      redis.quit(),
    ]);
  },
});
```

## Deliverables

- [ ] `prom-client` registers default Node.js metrics + custom `httpRequestDuration` histogram
- [ ] `requestLogger` middleware records duration using `process.hrtime.bigint()` (nanosecond precision)
- [ ] `GET /metrics` returns Prometheus exposition format; `Content-Type` set correctly
- [ ] `rateLimiterMiddleware` uses `rate-limiter-flexible` backed by Redis; returns 429 with `Retry-After`
- [ ] `GET /health/live` always returns 200 (liveness)
- [ ] `GET /health/ready` returns 503 when MongoDB or Redis is down (readiness)
- [ ] `createGracefulShutdown` stops accepting connections, runs `onClose`, exits 0
- [ ] Forced exit fires after 10 s via `setTimeout(...).unref()`
- [ ] Structured JSON logs in production (no `printf`-style formatting)
