# P4.2 — Error Monitoring Setup

**Phase:** 4 — Validation & Error Handling  
**Topic:** Winston + Sentry, error tracking, request ID propagation  
**Type:** Project

---

## Goal

Integrate Sentry for error tracking, configure Winston for structured production logging, propagate request IDs end-to-end, and alert on 5xx errors.

## What to Build

- Sentry SDK initialized before any route handler
- Winston transports: Console (dev), File (prod), Sentry (errors only)
- AsyncLocalStorage for request-scoped context (no need to pass `req` everywhere)
- Request ID generated and stored in ALS; all log entries include it
- Custom Sentry breadcrumbs for key business events
- `GET /health` and `GET /metrics` (basic counters)

```js
// src/lib/sentry.js  — must be imported FIRST in server.js
import * as Sentry from '@sentry/node';
import { nodeProfilingIntegration } from '@sentry/profiling-node';

export const initSentry = () => {
  Sentry.init({
    dsn: process.env.SENTRY_DSN,
    environment: process.env.NODE_ENV ?? 'development',
    release: process.env.GIT_SHA,
    integrations: [
      Sentry.httpIntegration(),
      Sentry.expressIntegration(),
      nodeProfilingIntegration(),
    ],
    tracesSampleRate: process.env.NODE_ENV === 'production' ? 0.1 : 1.0,
    profilesSampleRate: 0.1,
    beforeSend(event) {
      // Scrub sensitive fields
      if (event.request?.headers?.authorization) {
        event.request.headers.authorization = '[Filtered]';
      }
      return event;
    },
  });
};
```

```js
// src/lib/logger.js
import winston from 'winston';
import * as Sentry from '@sentry/node';
import { getContext } from './requestContext.js';

// Custom transport that forwards error-level logs to Sentry
class SentryTransport extends winston.Transport {
  log({ level, message, err, ...meta }, callback) {
    if (level === 'error') {
      const ctx = getContext();
      Sentry.withScope((scope) => {
        scope.setTag('requestId', ctx.requestId);
        scope.setExtras(meta);
        if (err instanceof Error) {
          Sentry.captureException(err);
        } else {
          Sentry.captureMessage(message, 'error');
        }
      });
    }
    callback();
  }
}

// Add requestId to every log entry automatically
const injectContext = winston.format((info) => {
  const ctx = getContext();
  if (ctx.requestId) info.requestId = ctx.requestId;
  if (ctx.userId)    info.userId    = ctx.userId;
  return info;
});

export const logger = winston.createLogger({
  level: process.env.LOG_LEVEL ?? 'info',
  format: winston.format.combine(
    injectContext(),
    winston.format.timestamp(),
    winston.format.errors({ stack: true }),
    winston.format.json(),
  ),
  transports: [
    ...(process.env.NODE_ENV !== 'test' ? [new winston.transports.Console({
      format: process.env.NODE_ENV === 'development'
        ? winston.format.combine(winston.format.colorize(), winston.format.simple())
        : winston.format.json(),
    })] : []),
    new winston.transports.File({ filename: 'logs/error.log', level: 'error' }),
    new winston.transports.File({ filename: 'logs/combined.log' }),
    new SentryTransport({ level: 'error' }),
  ],
});
```

```js
// src/middleware/requestId.js
import { randomUUID } from 'node:crypto';
import { runWithContext } from '../lib/requestContext.js';
import { logger } from '../lib/logger.js';

export const requestId = (req, res, next) => {
  const id = req.headers['x-request-id'] ?? randomUUID();
  res.set('x-request-id', id);
  req.requestId = id;

  const start = Date.now();

  // All code running inside runWithContext has access to req ID and userId
  runWithContext({ requestId: id }, () => {
    res.on('finish', () => {
      logger.info('http', {
        method: req.method,
        url:    req.originalUrl,
        status: res.statusCode,
        ms:     Date.now() - start,
        ip:     req.ip,
      });
    });
    next();
  });
};
```

```js
// src/lib/metrics.js  — in-memory counters (replace with prom-client for production)
const counters = {
  requestTotal:    0,
  request2xx:      0,
  request4xx:      0,
  request5xx:      0,
  errorCount:      0,
};

export const increment = (key) => { if (key in counters) counters[key]++; };
export const getMetrics = () => ({ ...counters, uptime: process.uptime() });

// Middleware that increments counters
export const metricsMiddleware = (_req, res, next) => {
  res.on('finish', () => {
    increment('requestTotal');
    if (res.statusCode < 400)       increment('request2xx');
    else if (res.statusCode < 500)  increment('request4xx');
    else { increment('request5xx'); increment('errorCount'); }
  });
  next();
};
```

```js
// src/middleware/errorHandler.js
import * as Sentry from '@sentry/node';
import { logger } from '../lib/logger.js';

export const errorHandler = (err, req, res, _next) => {
  const status = err.status ?? err.statusCode ?? 500;

  if (status >= 500) {
    logger.error('Server error', { err, url: req.originalUrl });
    // Sentry.expressErrorHandler() captures it automatically if registered
  } else {
    logger.warn('Client error', { message: err.message, status, url: req.originalUrl });
  }

  res.status(status).json({
    type:      `https://api.example.com/errors/${status}`,
    title:     err.message ?? 'Internal Server Error',
    status,
    requestId: req.requestId,
  });
};
```

```js
// src/server.js
import './lib/sentry.js';   // side-effect import — must be FIRST
import { initSentry } from './lib/sentry.js';
initSentry();

import app from './app.js';
import * as Sentry from '@sentry/node';

// Sentry error handler must be BEFORE custom error handler
app.use(Sentry.expressErrorHandler());
app.use(errorHandler);

const server = app.listen(process.env.PORT ?? 4000);

process.on('SIGTERM', () => {
  server.close(async () => {
    await Sentry.flush(2000);
    process.exit(0);
  });
});
```
