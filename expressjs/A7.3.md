# A7.3 — Unit Testing with Mocks

**Phase:** 7 — Testing
**Topic:** jest.mock(), jest.fn(), spyOn, mock modules, mock database calls
**Type:** Assignment

---

## Goal
Write unit tests for service and controller layers by mocking all external dependencies (database, email, HTTP clients) with Jest.

## Background

### What to Mock
Mock external dependencies that make tests slow, unreliable, or side-effectful:
- Database calls (would require a real DB or in-memory server)
- Email sending (would send real emails)
- HTTP requests to external APIs (would hit real APIs, cost money, fail in CI)
- File system writes (would write real files)
- Time functions (non-deterministic)

Don't mock the code you're testing — only mock its dependencies.

### jest.mock() vs jest.fn() vs jest.spyOn()
- `jest.mock('module')` — replaces an entire module's exports with auto-mocked versions. Best for replacing third-party packages.
- `jest.fn()` — creates a mock function. Use to pass as a dependency injection (e.g., mock database passed as argument).
- `jest.spyOn(object, 'method')` — wraps an existing method so you can assert on calls and control return values without fully replacing the object.

### Resetting Mocks
- `mockFn.mockReset()` — clears call history and return value implementations
- `mockFn.mockRestore()` — restores original implementation (only for `spyOn`)
- `jest.clearAllMocks()` in `beforeEach` — clears all mock history

### 2. Service Under Test

```js
// src/services/auth.service.js
const bcrypt = require('bcryptjs');
const userRepository = require('../repositories/user.repository');
const emailService = require('./email.service');
const tokenService = require('./token.service');

const register = async ({ name, email, password }) => {
  const existing = await userRepository.findByEmail(email);
  if (existing) throw Object.assign(new Error('Email already registered'), { code: 'DUPLICATE_EMAIL' });

  const hashedPassword = await bcrypt.hash(password, 10);
  const user = await userRepository.create({ name, email, password: hashedPassword });

  const token = tokenService.generateVerificationToken(user.id);
  await emailService.sendWelcome(user, token); // fire, don't wait

  return { user: { id: user.id, name, email }, verified: false };
};

const login = async ({ email, password }) => {
  const user = await userRepository.findByEmail(email);
  if (!user) throw Object.assign(new Error('Invalid credentials'), { code: 'INVALID_CREDENTIALS' });

  const match = await bcrypt.compare(password, user.password);
  if (!match) throw Object.assign(new Error('Invalid credentials'), { code: 'INVALID_CREDENTIALS' });

  return tokenService.generateTokens(user.id);
};

const changePassword = async (userId, { currentPassword, newPassword }) => {
  const user = await userRepository.findById(userId, true); // include password
  if (!user) throw new Error('User not found');

  const match = await bcrypt.compare(currentPassword, user.password);
  if (!match) throw Object.assign(new Error('Current password is incorrect'), { code: 'WRONG_PASSWORD' });

  const hashed = await bcrypt.hash(newPassword, 10);
  await userRepository.update(userId, { password: hashed });
  return { success: true };
};

module.exports = { register, login, changePassword };
```

### 3. Mock the Dependencies

```js
// tests/auth.service.test.js
const bcrypt = require('bcryptjs');

// Mock entire modules before importing the service
jest.mock('../src/repositories/user.repository');
jest.mock('../src/services/email.service');
jest.mock('../src/services/token.service');

const authService = require('../src/services/auth.service');
const userRepository = require('../src/repositories/user.repository');
const emailService = require('../src/services/email.service');
const tokenService = require('../src/services/token.service');

describe('authService.register', () => {
  beforeEach(() => {
    jest.clearAllMocks();

    // Default mock implementations
    userRepository.findByEmail.mockResolvedValue(null); // no existing user
    userRepository.create.mockResolvedValue({ id: 1, name: 'Alice', email: 'alice@test.com' });
    tokenService.generateVerificationToken.mockReturnValue('verify-token-123');
    emailService.sendWelcome.mockResolvedValue(undefined);
  });

  it('creates a user and sends welcome email', async () => {
    const result = await authService.register({
      name: 'Alice',
      email: 'alice@test.com',
      password: 'Password1',
    });

    expect(result.user).toEqual({ id: 1, name: 'Alice', email: 'alice@test.com' });
    expect(result.verified).toBe(false);

    // Verify email was sent with the correct user and token
    expect(emailService.sendWelcome).toHaveBeenCalledWith(
      { id: 1, name: 'Alice', email: 'alice@test.com' },
      'verify-token-123'
    );
  });

  it('hashes the password before storing', async () => {
    const spyHash = jest.spyOn(bcrypt, 'hash');

    await authService.register({
      name: 'Alice',
      email: 'alice@test.com',
      password: 'Password1',
    });

    expect(spyHash).toHaveBeenCalledWith('Password1', 10);
    // Ensure raw password was NOT passed to create
    const createCall = userRepository.create.mock.calls[0][0];
    expect(createCall.password).not.toBe('Password1');
    expect(createCall.password).toMatch(/^\$2[aby]\$/); // bcrypt hash prefix
  });

  it('throws DUPLICATE_EMAIL when email already exists', async () => {
    userRepository.findByEmail.mockResolvedValue({ id: 99, email: 'alice@test.com' });

    await expect(
      authService.register({ name: 'Alice', email: 'alice@test.com', password: 'Password1' })
    ).rejects.toMatchObject({ message: 'Email already registered', code: 'DUPLICATE_EMAIL' });

    expect(userRepository.create).not.toHaveBeenCalled();
    expect(emailService.sendWelcome).not.toHaveBeenCalled();
  });
});

describe('authService.login', () => {
  const userWithHash = {
    id: 1,
    email: 'alice@test.com',
    password: '$2b$10$abc', // mock hash
  };

  beforeEach(() => {
    jest.clearAllMocks();
    tokenService.generateTokens.mockReturnValue({
      accessToken: 'access-123',
      refreshToken: 'refresh-456',
    });
  });

  it('returns tokens on successful login', async () => {
    userRepository.findByEmail.mockResolvedValue(userWithHash);
    jest.spyOn(bcrypt, 'compare').mockResolvedValue(true);

    const tokens = await authService.login({ email: 'alice@test.com', password: 'Password1' });

    expect(tokens).toEqual({ accessToken: 'access-123', refreshToken: 'refresh-456' });
    expect(tokenService.generateTokens).toHaveBeenCalledWith(1);
  });

  it('throws INVALID_CREDENTIALS when user not found', async () => {
    userRepository.findByEmail.mockResolvedValue(null);

    await expect(
      authService.login({ email: 'nobody@test.com', password: 'Password1' })
    ).rejects.toMatchObject({ code: 'INVALID_CREDENTIALS' });

    expect(tokenService.generateTokens).not.toHaveBeenCalled();
  });

  it('throws INVALID_CREDENTIALS when password is wrong', async () => {
    userRepository.findByEmail.mockResolvedValue(userWithHash);
    jest.spyOn(bcrypt, 'compare').mockResolvedValue(false);

    await expect(
      authService.login({ email: 'alice@test.com', password: 'WrongPassword' })
    ).rejects.toMatchObject({ code: 'INVALID_CREDENTIALS' });
  });

  it('uses same error message for missing user and wrong password', async () => {
    userRepository.findByEmail.mockResolvedValue(null);
    const err1 = await authService.login({ email: 'x@y.com', password: 'P' }).catch(e => e);

    userRepository.findByEmail.mockResolvedValue(userWithHash);
    jest.spyOn(bcrypt, 'compare').mockResolvedValue(false);
    const err2 = await authService.login({ email: 'alice@test.com', password: 'W' }).catch(e => e);

    // Same message prevents user enumeration
    expect(err1.message).toBe(err2.message);
  });
});

describe('authService.changePassword', () => {
  beforeEach(() => {
    jest.clearAllMocks();
    userRepository.findById.mockResolvedValue({
      id: 1,
      password: '$2b$10$hash',
    });
    userRepository.update.mockResolvedValue(true);
  });

  it('updates password when current password is correct', async () => {
    jest.spyOn(bcrypt, 'compare').mockResolvedValue(true);
    jest.spyOn(bcrypt, 'hash').mockResolvedValue('$2b$10$newHash');

    const result = await authService.changePassword(1, {
      currentPassword: 'OldPassword1',
      newPassword: 'NewPassword1',
    });

    expect(result.success).toBe(true);
    expect(userRepository.update).toHaveBeenCalledWith(1, { password: '$2b$10$newHash' });
  });

  it('throws WRONG_PASSWORD when current password is wrong', async () => {
    jest.spyOn(bcrypt, 'compare').mockResolvedValue(false);

    await expect(
      authService.changePassword(1, { currentPassword: 'wrong', newPassword: 'New1' })
    ).rejects.toMatchObject({ code: 'WRONG_PASSWORD' });

    expect(userRepository.update).not.toHaveBeenCalled();
  });
});
```

### 5. Timer Mocking

```js
// src/utils/debounce.js
const debounce = (fn, delay) => {
  let timer;
  return (...args) => {
    clearTimeout(timer);
    timer = setTimeout(() => fn(...args), delay);
  };
};
module.exports = debounce;
```

```js
// tests/debounce.test.js
const debounce = require('../src/utils/debounce');

describe('debounce', () => {
  beforeEach(() => jest.useFakeTimers());
  afterEach(() => jest.useRealTimers());

  it('only calls function once after delay', () => {
    const fn = jest.fn();
    const debounced = debounce(fn, 500);

    debounced('a'); debounced('b'); debounced('c');

    expect(fn).not.toHaveBeenCalled();

    jest.advanceTimersByTime(500);

    expect(fn).toHaveBeenCalledTimes(1);
    expect(fn).toHaveBeenCalledWith('c');
  });

  it('resets timer on each call', () => {
    const fn = jest.fn();
    const debounced = debounce(fn, 500);

    debounced('first');
    jest.advanceTimersByTime(400);
    debounced('second');
    jest.advanceTimersByTime(400);

    expect(fn).not.toHaveBeenCalled();

    jest.advanceTimersByTime(100);
    expect(fn).toHaveBeenCalledTimes(1);
  });
});
```
