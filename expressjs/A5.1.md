## Goal
Implement cache-aside caching with Redis to dramatically reduce database load and response times for read-heavy endpoints.

## Background

### The Cache-Aside Pattern
In cache-aside (also called lazy loading), the application manages the cache directly. On a read:
1. Check cache. If hit: return cached value.
2. If miss: fetch from database, store in cache with TTL, return value.

On a write/delete: invalidate or update the cache entry. This keeps the cache coherent without pre-populating it — only data that's actually requested gets cached.

Alternatives: write-through (update cache on every write, always consistent but higher write cost), write-behind (write to cache first, flush to DB asynchronously). Cache-aside is most common for read-heavy workloads.

### Redis and ioredis
Redis is an in-memory key-value store that supports strings, hashes, lists, sets, sorted sets, and more. `ioredis` is the recommended Redis client for Node.js — it supports Promises, clustering, Sentinel, Lua scripts, and pipelining.

Key operations:
- `SET key value EX seconds` — store with expiry
- `GET key` — retrieve (null if not found or expired)
- `DEL key` — delete
- `MGET key1 key2` — get multiple keys in one round trip
- `EXPIRE key seconds` — reset TTL of existing key

### TTL Strategy
Choose TTL based on how frequently data changes and how much staleness is acceptable:
- User profiles: 5–15 minutes (change rarely)
- Product catalog: 1 hour (updated by admins)
- News articles: 15 minutes (updated frequently)
- Authentication tokens: match token expiry exactly
- Session data: match session timeout

Very short TTLs (< 30s) reduce cache benefit. Very long TTLs increase staleness risk. When in doubt: measure how often data actually changes.

### 2. Redis Client and Cache Helpers

```js
// src/config/cache.js
const Redis = require('ioredis');

const redis = new Redis({
  host: process.env.REDIS_HOST ?? 'localhost',
  port: parseInt(process.env.REDIS_PORT ?? '6379'),
  enableReadyCheck: true,
  maxRetriesPerRequest: 3,
  retryStrategy: (times) => {
    if (times > 5) return null; // stop retrying
    return Math.min(times * 100, 3000); // exponential backoff
  },
});

redis.on('connect', () => console.log('Redis connected'));
redis.on('error', (err) => console.error('Redis error:', err.message));

// Generic cache helpers
const cache = {
  async get(key) {
    try {
      const value = await redis.get(key);
      return value ? JSON.parse(value) : null;
    } catch (err) {
      console.error('Cache GET error:', err.message);
      return null; // fail open — don't crash if Redis is down
    }
  },

  async set(key, value, ttlSeconds = 300) {
    try {
      await redis.set(key, JSON.stringify(value), 'EX', ttlSeconds);
    } catch (err) {
      console.error('Cache SET error:', err.message);
    }
  },

  async del(...keys) {
    try {
      if (keys.length > 0) await redis.del(...keys);
    } catch (err) {
      console.error('Cache DEL error:', err.message);
    }
  },

  // Delete all keys matching a pattern (use carefully in production)
  async delPattern(pattern) {
    try {
      const keys = await redis.keys(pattern);
      if (keys.length > 0) await redis.del(...keys);
      return keys.length;
    } catch (err) {
      console.error('Cache DEL pattern error:', err.message);
      return 0;
    }
  },

  async exists(key) {
    try {
      return (await redis.exists(key)) === 1;
    } catch {
      return false;
    }
  },
};

module.exports = { redis, cache };
```

### 3. Cache Middleware Factory

```js
// src/middleware/cacheMiddleware.js
const { cache } = require('../config/cache');

// Route-level cache middleware
// Usage: router.get('/items', cacheRoute('items:list', 300), handler)
const cacheRoute = (keyOrFn, ttl = 300) => async (req, res, next) => {
  const key = typeof keyOrFn === 'function' ? keyOrFn(req) : keyOrFn;

  try {
    const cached = await cache.get(key);

    if (cached !== null) {
      res.setHeader('X-Cache', 'HIT');
      res.setHeader('X-Cache-Key', key);
      return res.json(cached);
    }

    // Cache miss — intercept res.json to cache the response
    res.setHeader('X-Cache', 'MISS');
    const originalJson = res.json.bind(res);

    res.json = async (body) => {
      if (res.statusCode < 400) {
        await cache.set(key, body, ttl);
      }
      return originalJson(body);
    };

    next();
  } catch (err) {
    next(); // fail open
  }
};

module.exports = { cacheRoute };
```

### 4. Article Service with Cache-Aside

```js
// src/services/article.service.js
const { cache } = require('../config/cache');

// Simulated DB (replace with real DB calls)
const articlesDB = [
  { id: 1, title: 'Redis Caching', body: 'Use Redis...', authorId: 1, views: 100 },
  { id: 2, title: 'Express Tips', body: 'Organize routes...', authorId: 2, views: 50 },
  { id: 3, title: 'Node.js Streams', body: 'Use streams...', authorId: 1, views: 200 },
];

const TTL = {
  ARTICLE: 300,       // 5 minutes
  ARTICLE_LIST: 60,   // 1 minute (list changes more often)
  AUTHOR_ARTICLES: 120,
};

const findAll = async (filter = {}) => {
  const cacheKey = `articles:list:${JSON.stringify(filter)}`;

  // 1. Check cache
  const cached = await cache.get(cacheKey);
  if (cached) {
    console.log('[Cache HIT] articles list');
    return cached;
  }

  // 2. Fetch from "database"
  console.log('[Cache MISS] fetching articles from DB');
  let articles = [...articlesDB];

  if (filter.authorId) {
    articles = articles.filter(a => a.authorId === parseInt(filter.authorId));
  }

  // 3. Store in cache
  await cache.set(cacheKey, articles, TTL.ARTICLE_LIST);
  return articles;
};

const findById = async (id) => {
  const cacheKey = `articles:${id}`;

  const cached = await cache.get(cacheKey);
  if (cached) return cached;

  const article = articlesDB.find(a => a.id === id);
  if (!article) return null;

  await cache.set(cacheKey, article, TTL.ARTICLE);
  return article;
};

const update = async (id, data) => {
  const index = articlesDB.findIndex(a => a.id === id);
  if (index === -1) return null;

  articlesDB[index] = { ...articlesDB[index], ...data };

  // Invalidate: both the individual and the list cache
  await cache.del(`articles:${id}`);
  await cache.delPattern('articles:list:*');

  return articlesDB[index];
};

const create = async (data) => {
  const article = { id: articlesDB.length + 1, ...data, views: 0 };
  articlesDB.push(article);

  // Invalidate list cache (new item exists)
  await cache.delPattern('articles:list:*');

  return article;
};

module.exports = { findAll, findById, update, create };
```

### 5. Routes

```js
// src/routes/articles.js
const router = require('express').Router();
const articleService = require('../services/article.service');
const { cacheRoute } = require('../middleware/cacheMiddleware');

// Cache the list with dynamic key based on query
router.get('/',
  cacheRoute((req) => `articles:list:${JSON.stringify(req.query)}`, 60),
  async (req, res, next) => {
    try {
      const articles = await articleService.findAll(req.query);
      res.json({ data: articles, count: articles.length });
    } catch (err) { next(err); }
  }
);

router.get('/:id',
  cacheRoute((req) => `articles:${req.params.id}`, 300),
  async (req, res, next) => {
    try {
      const article = await articleService.findById(parseInt(req.params.id));
      if (!article) return res.status(404).json({ error: 'Article not found' });
      res.json(article);
    } catch (err) { next(err); }
  }
);

router.post('/', async (req, res, next) => {
  try {
    const article = await articleService.create(req.body);
    res.status(201).json(article);
  } catch (err) { next(err); }
});

router.patch('/:id', async (req, res, next) => {
  try {
    const article = await articleService.update(parseInt(req.params.id), req.body);
    if (!article) return res.status(404).json({ error: 'Article not found' });
    res.json(article);
  } catch (err) { next(err); }
});

module.exports = router;
```
