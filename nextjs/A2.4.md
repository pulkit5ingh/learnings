## Goal
Build REST-style API endpoints using Route Handlers and understand when to use them versus Server Actions.

## Background

Route Handlers are the App Router's replacement for `pages/api/` files. You create a `route.ts` file inside any folder under `app/` (including inside `app/api/` by convention). The file exports named functions for each HTTP method: `GET`, `POST`, `PUT`, `PATCH`, `DELETE`, `HEAD`, `OPTIONS`.

Route Handlers run on the server and have access to the full Node.js environment. They receive a `Request` object (Web API) and return a `Response` object (also Web API). `NextResponse` is a Next.js extension of `Response` that adds convenience methods like `NextResponse.json()` and `NextResponse.redirect()`.

**When to use Route Handlers vs Server Actions:**
- Use **Server Actions** for form mutations (create, update, delete from a form)
- Use **Route Handlers** for: external webhooks, third-party API callbacks, public APIs consumed by mobile apps, file downloads, streaming responses, and cases where you need full HTTP control (status codes, custom headers)

Route Handlers can be cached like pages. A `GET` Route Handler with no dynamic functions is cached at build time. Adding `export const dynamic = "force-dynamic"` or reading `cookies()` makes it dynamic.

## Tasks

### 1. Basic GET — return JSON
```ts
// src/app/api/posts/route.ts
import { NextResponse } from "next/server";

const POSTS = [
  { id: 1, title: "Hello World", author: "Alice" },
  { id: 2, title: "Next.js Tips", author: "Bob" },
  { id: 3, title: "TypeScript Guide", author: "Alice" },
];

export async function GET() {
  return NextResponse.json(POSTS);
}
```
Visit `/api/posts` in the browser to see the JSON.

### 2. GET with query params and filtering
```ts
// src/app/api/posts/route.ts (updated)
import { NextRequest, NextResponse } from "next/server";

export async function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const author = searchParams.get("author");
  const limit = Number(searchParams.get("limit") ?? "10");

  let result = [...POSTS];

  if (author) {
    result = result.filter((p) => p.author.toLowerCase() === author.toLowerCase());
  }

  return NextResponse.json({
    data: result.slice(0, limit),
    total: result.length,
  });
}
```
`/api/posts?author=Alice&limit=1` → returns only Alice's first post.

### 3. POST — create a resource
```ts
// src/app/api/posts/route.ts (add POST)
export async function POST(request: NextRequest) {
  const body = await request.json();

  if (!body.title || !body.author) {
    return NextResponse.json(
      { error: "title and author are required" },
      { status: 400 }
    );
  }

  const newPost = {
    id: POSTS.length + 1,
    title: body.title as string,
    author: body.author as string,
  };

  POSTS.push(newPost);

  return NextResponse.json(newPost, { status: 201 });
}
```

### 4. Dynamic route handler — GET/PUT/DELETE by ID
```ts
// src/app/api/posts/[id]/route.ts
import { NextRequest, NextResponse } from "next/server";

const POSTS = [
  { id: 1, title: "Hello World", author: "Alice" },
  { id: 2, title: "Next.js Tips", author: "Bob" },
];

type Params = { params: { id: string } };

export async function GET(_req: NextRequest, { params }: Params) {
  const id = Number(params.id);
  const post = POSTS.find((p) => p.id === id);

  if (!post) {
    return NextResponse.json({ error: "Post not found" }, { status: 404 });
  }

  return NextResponse.json(post);
}

export async function PUT(request: NextRequest, { params }: Params) {
  const id = Number(params.id);
  const index = POSTS.findIndex((p) => p.id === id);

  if (index === -1) {
    return NextResponse.json({ error: "Post not found" }, { status: 404 });
  }

  const body = await request.json();
  POSTS[index] = { ...POSTS[index], ...body, id };

  return NextResponse.json(POSTS[index]);
}

export async function DELETE(_req: NextRequest, { params }: Params) {
  const id = Number(params.id);
  const index = POSTS.findIndex((p) => p.id === id);

  if (index === -1) {
    return NextResponse.json({ error: "Post not found" }, { status: 404 });
  }

  POSTS.splice(index, 1);

  return new Response(null, { status: 204 });
}
```

### 5. Working with cookies and headers
```ts
// src/app/api/auth/session/route.ts
import { cookies, headers } from "next/headers";
import { NextResponse } from "next/server";

export async function GET() {
  const cookieStore = cookies();
  const headersList = headers();

  const sessionToken = cookieStore.get("session-token")?.value;
  const userAgent = headersList.get("user-agent");

  return NextResponse.json({
    authenticated: !!sessionToken,
    userAgent,
  });
}

export async function POST() {
  // Set a cookie in a Route Handler response
  const response = NextResponse.json({ message: "Logged in" });

  response.cookies.set("session-token", "fake-jwt-token", {
    httpOnly: true,
    secure: process.env.NODE_ENV === "production",
    sameSite: "lax",
    maxAge: 60 * 60 * 24 * 7, // 7 days
    path: "/",
  });

  return response;
}
```

### 6. Streaming a large response
```ts
// src/app/api/stream/route.ts
import { NextResponse } from "next/server";

export const dynamic = "force-dynamic";

export async function GET() {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    async start(controller) {
      for (let i = 1; i <= 5; i++) {
        await new Promise((r) => setTimeout(r, 500)); // simulate slow data
        controller.enqueue(
          encoder.encode(`data: {"chunk": ${i}, "time": "${new Date().toISOString()}"}\n\n`)
        );
      }
      controller.close();
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      Connection: "keep-alive",
    },
  });
}
```

### 7. CORS headers for a public API
```ts
// src/app/api/public/route.ts
import { NextRequest, NextResponse } from "next/server";

const CORS_HEADERS = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Methods": "GET, POST, OPTIONS",
  "Access-Control-Allow-Headers": "Content-Type",
};

export async function OPTIONS() {
  return NextResponse.json(null, { headers: CORS_HEADERS });
}

export async function GET() {
  return NextResponse.json({ message: "Public API" }, { headers: CORS_HEADERS });
}
```
