## Background

Tailwind CSS is a utility-first CSS framework. Instead of writing CSS classes like `.card { padding: 1rem; border-radius: 0.5rem; }`, you compose utilities directly in HTML: `className="p-4 rounded-lg"`. This eliminates the need to invent class names, keeps styles co-located with markup, and makes dead code elimination trivial (only utilities you use ship to production).

**Responsive design** uses prefix breakpoints: `sm:` (640px+), `md:` (768px+), `lg:` (1024px+), `xl:` (1280px+), `2xl:` (1536px+). These are mobile-first: `sm:flex` means "apply flex on small screens and above". The base (unprefixed) style applies at all sizes.

**Dark mode** can be configured two ways. `media` strategy uses the OS preference via `prefers-color-scheme`. `class` strategy requires a `.dark` class on `<html>`. The class strategy is almost always preferable in apps that let users toggle themes manually. In both cases you use the `dark:` prefix: `dark:bg-gray-900`.

**Custom config** lives in `tailwind.config.ts`. You should always use `theme.extend` rather than `theme` to add custom values — using `theme` directly replaces the defaults. Custom colors, spacing, fonts, and keyframes are all defined here.

**@apply** lets you use Tailwind utilities inside a CSS file. Use it sparingly — only for truly reusable patterns (e.g., a `.btn` class used across hundreds of elements). Overusing `@apply` defeats the purpose of utility-first.

### 2. Core utility categories
```tsx
// src/app/utilities-demo/page.tsx
export default function UtilitiesDemo() {
  return (
    <div className="p-8 space-y-8">

      {/* Layout */}
      <section>
        <h2 className="text-lg font-semibold mb-3">Layout</h2>
        <div className="flex gap-4 items-center">
          <div className="flex-1 bg-blue-100 p-4 rounded">flex-1</div>
          <div className="w-24 bg-blue-200 p-4 rounded text-center">w-24</div>
          <div className="grow bg-blue-300 p-4 rounded">grow</div>
        </div>
        <div className="grid grid-cols-3 gap-4 mt-4">
          {[1,2,3,4,5,6].map(n => (
            <div key={n} className="bg-purple-100 p-4 rounded text-center">{n}</div>
          ))}
        </div>
      </section>

      {/* Typography */}
      <section>
        <h2 className="text-lg font-semibold mb-3">Typography</h2>
        <p className="text-xs text-gray-400">text-xs</p>
        <p className="text-sm text-gray-500">text-sm</p>
        <p className="text-base text-gray-700">text-base</p>
        <p className="text-xl font-semibold">text-xl font-semibold</p>
        <p className="text-3xl font-bold tracking-tight">text-3xl font-bold tracking-tight</p>
        <p className="text-lg leading-relaxed">
          text-lg leading-relaxed — Lorem ipsum dolor sit amet, consectetur adipiscing elit.
        </p>
      </section>

      {/* Spacing */}
      <section>
        <h2 className="text-lg font-semibold mb-3">Spacing (p, m, gap)</h2>

            <div key={cls} className={`${cls} bg-green-100 border border-green-300 text-xs`}>{cls}</div>

        </div>
      </section>

      {/* Colors */}
      <section>

          {["bg-red-500","bg-orange-500","bg-yellow-500","bg-green-500",
            "bg-blue-500","bg-indigo-500","bg-purple-500","bg-pink-500"].map(cls => (
            <div key={cls} className={`${cls} w-12 h-12 rounded`} title={cls} />
          ))}
        </div>
      </section>

      {/* Borders & Effects */}
      <section>
        <h2 className="text-lg font-semibold mb-3">Borders & Shadows</h2>
        <div className="flex gap-4">
          <div className="p-4 border rounded">border rounded</div>
          <div className="p-4 border-2 border-blue-500 rounded-xl">border-2 border-blue-500 rounded-xl</div>
          <div className="p-4 shadow-lg rounded-lg">shadow-lg</div>
          <div className="p-4 shadow-2xl rounded-2xl">shadow-2xl</div>
        </div>
      </section>
    </div>
  );
}
```

### 3. Responsive design
```tsx
// src/app/responsive/page.tsx
export default function ResponsivePage() {
  return (
    <div className="p-4">
      {/* Stack on mobile, grid on tablet+ */}
      <div className="grid grid-cols-1 sm:grid-cols-2 lg:grid-cols-4 gap-4">
        {["Card 1","Card 2","Card 3","Card 4"].map(card => (
          <div key={card} className="p-6 bg-white border rounded-lg shadow-sm">
            <h3 className="text-base sm:text-lg font-semibold">{card}</h3>
            <p className="mt-2 text-sm text-gray-500 hidden sm:block">
              Visible on sm+
            </p>
          </div>
        ))}
      </div>

      {/* Responsive text */}
      <h1 className="mt-8 text-2xl sm:text-3xl md:text-4xl lg:text-5xl font-bold">
        Responsive heading
      </h1>

      {/* Hide/show at breakpoints */}
      <p className="mt-4 block sm:hidden text-sm text-gray-500">Mobile only</p>
      <p className="mt-4 hidden sm:block text-sm text-gray-500">Tablet and above</p>
    </div>
  );
}
```

### 5. Custom theme extension
```ts
// tailwind.config.ts (full)
import type { Config } from "tailwindcss";

const config: Config = {
  darkMode: "class",
  content: ["./src/**/*.{js,ts,jsx,tsx,mdx}"],
  theme: {
    extend: {
      colors: {
        brand: {
          50: "#eff6ff",
          100: "#dbeafe",
          500: "#3b82f6",
          600: "#2563eb",
          700: "#1d4ed8",
          900: "#1e3a8a",
        },
      },
      fontFamily: {
        sans: ["var(--font-inter)", "sans-serif"],
        mono: ["var(--font-mono)", "monospace"],
      },
      spacing: {
        18: "4.5rem",
        128: "32rem",
      },
      borderRadius: {
        "4xl": "2rem",
      },
      keyframes: {
        "fade-in": {
          from: { opacity: "0", transform: "translateY(8px)" },
          to: { opacity: "1", transform: "translateY(0)" },
        },
      },
      animation: {
        "fade-in": "fade-in 0.3s ease-out",
      },
    },
  },
};
export default config;
```
Usage: `className="bg-brand-600 hover:bg-brand-700 animate-fade-in"`

### 6. @apply for reusable utility combinations
```css
/* src/app/globals.css */
@tailwind base;
@tailwind components;
@tailwind utilities;

@layer components {
  .btn {
    @apply inline-flex items-center justify-center px-4 py-2 rounded-md font-medium
           focus:outline-none focus:ring-2 focus:ring-offset-2 transition-colors;
  }
  .btn-primary {
    @apply btn bg-brand-600 text-white hover:bg-brand-700 focus:ring-brand-500;
  }
  .btn-secondary {
    @apply btn bg-gray-100 text-gray-900 hover:bg-gray-200 focus:ring-gray-400;
  }
  .card {
    @apply bg-white dark:bg-gray-800 rounded-lg border border-gray-200 dark:border-gray-700 p-4;
  }
}
```
