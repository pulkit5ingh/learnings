# P7.2 — E2E Tested User Flow

**Phase:** 7 — Testing  
**Topic:** Playwright signup → login → CRUD → logout end-to-end test suite  
**Type:** Project

---

## Goal
Write a complete Playwright end-to-end test suite covering the full user journey: sign up → email verification → sign in → create/read/update/delete a resource → sign out. Tests should be isolated, deterministic, and run in CI.

## What to Build
- Playwright project setup with TypeScript
- Page Object Models (POMs) for Sign-in, Sign-up, Dashboard pages
- Tests: signup, login, CRUD tasks (create, edit, delete), logout
- API route seeding/cleanup for test isolation
- CI GitHub Actions workflow
- Visual snapshots (optional)

## Architecture

```
e2e/
  pages/
    SignInPage.ts       ← Page Object Model
    SignUpPage.ts
    DashboardPage.ts
  fixtures/
    auth.ts             ← shared authenticated context fixture
  tests/
    auth.spec.ts        ← signup, login, logout flows
    tasks.spec.ts       ← CRUD operations on tasks
    navigation.spec.ts  ← routing, 404, redirects
  global-setup.ts       ← seed test DB before all tests
  global-teardown.ts    ← clean test DB after all tests
playwright.config.ts
```

```ts
// e2e/pages/SignInPage.ts
import { type Page, type Locator, expect } from "@playwright/test";

export class SignInPage {
  readonly page: Page;
  readonly emailInput: Locator;
  readonly passwordInput: Locator;
  readonly submitButton: Locator;
  readonly errorMessage: Locator;
  readonly googleButton: Locator;

  constructor(page: Page) {
    this.page = page;
    this.emailInput = page.getByLabel("Email");
    this.passwordInput = page.getByLabel("Password");
    this.submitButton = page.getByRole("button", { name: "Sign in" });
    this.errorMessage = page.getByRole("alert");
    this.googleButton = page.getByRole("button", { name: /google/i });
  }

  async goto() {
    await this.page.goto("/sign-in");
  }

  async signIn(email: string, password: string) {
    await this.emailInput.fill(email);
    await this.passwordInput.fill(password);
    await this.submitButton.click();
  }

  async expectError(message: string) {
    await expect(this.errorMessage).toContainText(message);
  }

  async expectRedirectToDashboard() {
    await expect(this.page).toHaveURL(/\/dashboard/);
  }
}
```

```ts
// e2e/pages/DashboardPage.ts
import { type Page, type Locator, expect } from "@playwright/test";

export class DashboardPage {
  readonly page: Page;
  readonly taskInput: Locator;
  readonly addButton: Locator;
  readonly signOutButton: Locator;

  constructor(page: Page) {
    this.page = page;
    this.taskInput = page.getByPlaceholder("New task title...");
    this.addButton = page.getByRole("button", { name: "Add Task" });
    this.signOutButton = page.getByRole("button", { name: "Sign out" });
  }

  async goto() {
    await this.page.goto("/dashboard");
  }

  async addTask(title: string) {
    await this.taskInput.fill(title);
    await this.addButton.click();
  }

  async getTask(title: string): Promise<Locator> {
    return this.page.getByText(title);
  }

  async toggleTask(title: string) {
    const task = this.page.getByText(title).locator("xpath=ancestor::li");
    await task.getByRole("button", { name: /mark/i }).click();
  }

  async deleteTask(title: string) {
    const task = this.page.getByText(title).locator("xpath=ancestor::li");
    await task.hover();
    await task.getByLabel("Delete task").click();
  }

  async signOut() {
    await this.signOutButton.click();
  }
}
```

```ts
// e2e/tests/auth.spec.ts
import { test, expect } from "../fixtures/auth";

test.describe("Authentication", () => {
  test("sign-in page is accessible to unauthenticated users", async ({ page }) => {
    await page.goto("/sign-in");
    await expect(page).toHaveTitle(/sign in/i);
    await expect(page.getByRole("button", { name: "Sign in" })).toBeVisible();
  });

  test("shows error for wrong password", async ({ signInPage }) => {
    await signInPage.goto();
    await signInPage.signIn("test@example.com", "wrongpassword");
    await signInPage.expectError("Invalid email or password");
  });

  test("shows error for non-existent email", async ({ signInPage }) => {
    await signInPage.goto();
    await signInPage.signIn("nobody@example.com", "TestPassword1!");
    await signInPage.expectError("Invalid email or password");
  });

  test("redirects to dashboard after successful sign-in", async ({ signInPage }) => {
    await signInPage.goto();
    await signInPage.signIn("test@example.com", "TestPassword1!");
    await signInPage.expectRedirectToDashboard();
  });

  test("redirects authenticated users away from sign-in page", async ({ page }) => {
    // Uses the authenticated storage state (from the "authenticated" project)
    await page.goto("/sign-in");
    await expect(page).toHaveURL(/\/dashboard/);
  });

  test("sign-out clears session and redirects to sign-in", async ({
    page,
    dashboardPage,
  }) => {
    await dashboardPage.goto();
    await dashboardPage.signOut();
    await expect(page).toHaveURL(/\/sign-in/);

    // Verify session is gone — navigating to dashboard should redirect
    await page.goto("/dashboard");
    await expect(page).toHaveURL(/\/sign-in/);
  });
});
```

```ts
// e2e/tests/tasks.spec.ts
import { test, expect } from "../fixtures/auth";

// These tests run in the "authenticated" project — session is pre-loaded
test.describe("Task CRUD", () => {
  test.beforeEach(async ({ dashboardPage }) => {
    await dashboardPage.goto();
  });

  test("shows empty state when no tasks", async ({ page }) => {
    await expect(page.getByText("No tasks yet")).toBeVisible();
  });

  test("creates a task and shows it in the list", async ({ dashboardPage, page }) => {
    await dashboardPage.addTask("E2E test task");
    await expect(page.getByText("E2E test task")).toBeVisible();
  });

  test("shows validation error for empty task title", async ({ page }) => {
    await page.getByRole("button", { name: "Add Task" }).click();
    // HTML5 required validation
    await expect(page.getByPlaceholder("New task title...")).toBeFocused();
  });

  test("toggles a task to completed", async ({ dashboardPage, page }) => {
    await dashboardPage.addTask("Task to complete");
    await dashboardPage.toggleTask("Task to complete");

    // Task should have strikethrough styling
    const task = page.getByText("Task to complete");
    await expect(task).toHaveClass(/line-through/);
  });

  test("deletes a task", async ({ dashboardPage, page }) => {
    await dashboardPage.addTask("Task to delete");
    await expect(page.getByText("Task to delete")).toBeVisible();

    await dashboardPage.deleteTask("Task to delete");
    await expect(page.getByText("Task to delete")).not.toBeVisible();
  });

  test("persists tasks after page reload", async ({ dashboardPage, page }) => {
    await dashboardPage.addTask("Persistent task");
    await page.reload();
    await expect(page.getByText("Persistent task")).toBeVisible();
  });
});
```

```ts
// e2e/tests/navigation.spec.ts
import { test, expect } from "@playwright/test";

test.describe("Navigation", () => {
  test("home page loads", async ({ page }) => {
    await page.goto("/");
    await expect(page).toHaveTitle(/.+/);
    await expect(page.getByRole("main")).toBeVisible();
  });

  test("visiting /dashboard unauthenticated redirects to sign-in", async ({ page }) => {
    await page.goto("/dashboard");
    await expect(page).toHaveURL(/\/sign-in/);
    expect(new URL(page.url()).searchParams.get("callbackUrl")).toBe("/dashboard");
  });

  test("404 page shown for unknown routes", async ({ page }) => {
    await page.goto("/this-page-does-not-exist");
    await expect(page.getByText("404")).toBeVisible();
    await expect(page.getByRole("link", { name: /go home/i })).toBeVisible();
  });

  test("404 Go home link navigates to home", async ({ page }) => {
    await page.goto("/not-a-real-page");
    await page.getByRole("link", { name: /go home/i }).click();
    await expect(page).toHaveURL("/");
  });
});
```

```yaml
# .github/workflows/e2e.yml
name: E2E Tests

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  e2e:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - uses: pnpm/action-setup@v3
        with:
          version: 9

      - uses: actions/setup-node@v4
        with:
          node-version: 20
          cache: pnpm

      - run: pnpm install

      - name: Install Playwright browsers
        run: pnpm exec playwright install --with-deps chromium

      - name: Build the app
        run: pnpm build
        env:
          DATABASE_URL: ${{ secrets.DATABASE_URL }}
          NEXTAUTH_SECRET: ${{ secrets.NEXTAUTH_SECRET }}

      - name: Run Playwright tests
        run: pnpm exec playwright test
        env:
          BASE_URL: http://localhost:3000

      - uses: actions/upload-artifact@v4
        if: always()
        with:
          name: playwright-report
          path: playwright-report/
          retention-days: 30
```
