# P9.2 — Real-Time Dashboard

**Phase:** 9 — Advanced Features  
**Topic:** SSE from Route Handler, live data table, Recharts live charts  
**Type:** Project

---

## Goal
Build a real-time analytics dashboard that receives live server-push data via Server-Sent Events (SSE), displays it in an auto-updating data table, and visualizes it with Recharts live line charts.

## What to Build
- SSE Route Handler streaming multiple metric types every second
- Live metrics table (top N rows, sorted by most recent)
- Live line chart (time series — last 60 seconds of data)
- Live bar chart (current values by category)
- Connection status indicator
- Pause/resume button
- Download data as CSV

## Architecture

```
app/
  dashboard/
    page.tsx              ← assembles the dashboard
  api/
    metrics/
      route.ts            ← SSE Route Handler streaming metrics
components/
  LiveTable.tsx           ← auto-updating data table
  LiveLineChart.tsx       ← time series chart (Recharts)
  LiveBarChart.tsx        ← current values bar chart (Recharts)
  ConnectionBadge.tsx
  hooks/
    useMetricsStream.ts   ← EventSource hook with pause/resume
```

```ts
// app/api/metrics/route.ts
import { NextRequest } from "next/server";

export const runtime = "nodejs";

const SERVERS = ["web-01", "web-02", "web-03", "api-01", "api-02"];
const REGIONS = ["us-east-1", "eu-west-1", "ap-southeast-1"];

function sseEvent(type: string, data: unknown) {
  return `event: ${type}\ndata: ${JSON.stringify(data)}\n\n`;
}

function generateMetric() {
  const server = SERVERS[Math.floor(Math.random() * SERVERS.length)];
  const region = REGIONS[Math.floor(Math.random() * REGIONS.length)];
  return {
    id: crypto.randomUUID(),
    ts: Date.now(),
    server,
    region,
    cpu: Math.round(20 + Math.random() * 70),         // 20-90%
    memory: Math.round(30 + Math.random() * 55),       // 30-85%
    requestsPerSec: Math.round(50 + Math.random() * 450),
    latencyMs: Math.round(10 + Math.random() * 190),
    errorRate: parseFloat((Math.random() * 2).toFixed(2)),
  };
}

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    start(controller) {
      let tick = 0;

      // Send initial connection event
      controller.enqueue(
        encoder.encode(sseEvent("connected", { ts: Date.now() }))
      );

      const interval = setInterval(() => {
        tick++;

        // Send 3-5 metric events per tick (simulating multiple servers reporting)
        const count = 3 + Math.floor(Math.random() * 3);
        for (let i = 0; i < count; i++) {
          controller.enqueue(
            encoder.encode(sseEvent("metric", generateMetric()))
          );
        }

        // Every 10 ticks: send a summary event
        if (tick % 10 === 0) {
          controller.enqueue(
            encoder.encode(
              sseEvent("summary", {
                totalRequests: tick * 200,
                avgLatency: Math.round(50 + Math.random() * 30),
                errorCount: Math.floor(tick * 0.01),
                ts: Date.now(),
              })
            )
          );
        }
      }, 1000);

      req.signal.addEventListener("abort", () => {
        clearInterval(interval);
        controller.close();
      });
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      Connection: "keep-alive",
      "X-Accel-Buffering": "no",
    },
  });
}
```

```ts
// hooks/useMetricsStream.ts
"use client";

import { useEffect, useRef, useState, useCallback } from "react";

export interface Metric {
  id: string;
  ts: number;
  server: string;
  region: string;
  cpu: number;
  memory: number;
  requestsPerSec: number;
  latencyMs: number;
  errorRate: number;
}

interface StreamState {
  metrics: Metric[];
  connected: boolean;
  paused: boolean;
  eventCount: number;
}

const MAX_METRICS = 200;

export function useMetricsStream() {
  const [state, setState] = useState<StreamState>({
    metrics: [],
    connected: false,
    paused: false,
    eventCount: 0,
  });

  const esRef = useRef<EventSource | null>(null);
  const pausedRef = useRef(false);
  const bufferRef = useRef<Metric[]>([]);

  function connect() {
    if (esRef.current) return;

    const es = new EventSource("/api/metrics");
    esRef.current = es;

    es.onopen = () => setState((s) => ({ ...s, connected: true }));
    es.onerror = () => setState((s) => ({ ...s, connected: false }));

    es.addEventListener("metric", (e) => {
      const metric: Metric = JSON.parse(e.data);

      if (pausedRef.current) {
        // Buffer incoming events while paused
        bufferRef.current.push(metric);
        return;
      }

      setState((s) => ({
        ...s,
        metrics: [...s.metrics, metric].slice(-MAX_METRICS),
        eventCount: s.eventCount + 1,
      }));
    });
  }

  useEffect(() => {
    connect();
    return () => {
      esRef.current?.close();
      esRef.current = null;
    };
  }, []);

  const pause = useCallback(() => {
    pausedRef.current = true;
    setState((s) => ({ ...s, paused: true }));
  }, []);

  const resume = useCallback(() => {
    pausedRef.current = false;
    const buffered = bufferRef.current;
    bufferRef.current = [];
    setState((s) => ({
      ...s,
      paused: false,
      metrics: [...s.metrics, ...buffered].slice(-MAX_METRICS),
      eventCount: s.eventCount + buffered.length,
    }));
  }, []);

  return { ...state, pause, resume };
}
```

```tsx
// components/LiveLineChart.tsx
"use client";

import {
  LineChart, Line, XAxis, YAxis, CartesianGrid, Tooltip,
  ResponsiveContainer, Legend
} from "recharts";
import { useMemo } from "react";
import type { Metric } from "@/hooks/useMetricsStream";

interface Props {
  metrics: Metric[];
}

export function LiveLineChart({ metrics }: Props) {
  // Aggregate to per-second buckets, keeping the last 60 seconds
  const chartData = useMemo(() => {
    const now = Date.now();
    const buckets: Record<number, { cpu: number[]; latency: number[]; rps: number[] }> = {};

    for (const m of metrics) {
      const sec = Math.floor(m.ts / 1000);
      if (now / 1000 - sec > 60) continue; // older than 60s
      if (!buckets[sec]) buckets[sec] = { cpu: [], latency: [], rps: [] };
      buckets[sec].cpu.push(m.cpu);
      buckets[sec].latency.push(m.latencyMs);
      buckets[sec].rps.push(m.requestsPerSec);
    }

    return Object.entries(buckets)
      .sort(([a], [b]) => Number(a) - Number(b))
      .map(([sec, vals]) => ({
        time: new Date(Number(sec) * 1000).toLocaleTimeString(),
        cpu: Math.round(vals.cpu.reduce((s, v) => s + v, 0) / vals.cpu.length),
        latency: Math.round(vals.latency.reduce((s, v) => s + v, 0) / vals.latency.length),
        rps: Math.round(vals.rps.reduce((s, v) => s + v, 0) / vals.rps.length),
      }));
  }, [metrics]);

  return (
    <div className="bg-white border rounded-2xl p-5">
      <h3 className="font-semibold mb-4">Live Metrics (60s)</h3>
      <ResponsiveContainer width="100%" height={250}>
        <LineChart data={chartData} margin={{ top: 5, right: 20, bottom: 5, left: 0 }}>
          <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
          <XAxis dataKey="time" tick={{ fontSize: 11 }} interval="preserveStartEnd" />
          <YAxis yAxisId="left" tick={{ fontSize: 11 }} unit="%" />
          <YAxis yAxisId="right" orientation="right" tick={{ fontSize: 11 }} unit="ms" />
          <Tooltip />
          <Legend />
          <Line
            yAxisId="left"
            type="monotone"
            dataKey="cpu"
            stroke="#3b82f6"
            dot={false}
            strokeWidth={2}
            isAnimationActive={false}
          />
          <Line
            yAxisId="right"
            type="monotone"
            dataKey="latency"
            stroke="#f59e0b"
            dot={false}
            strokeWidth={2}
            isAnimationActive={false}
          />
        </LineChart>
      </ResponsiveContainer>
    </div>
  );
}
```

```tsx
// components/LiveBarChart.tsx
"use client";

import {
  BarChart, Bar, XAxis, YAxis, CartesianGrid, Tooltip,
  ResponsiveContainer, Cell
} from "recharts";
import { useMemo } from "react";
import type { Metric } from "@/hooks/useMetricsStream";

export function LiveBarChart({ metrics }: Props) {
  const data = useMemo(() => {
    // Latest metric per server
    const latest: Record<string, Metric> = {};
    for (const m of metrics) {
      if (!latest[m.server] || m.ts > latest[m.server].ts) {
        latest[m.server] = m;
      }
    }
    return Object.values(latest).map((m) => ({
      server: m.server,
      cpu: m.cpu,
      memory: m.memory,
    }));
  }, [metrics]);

  return (
    <div className="bg-white border rounded-2xl p-5">
      <h3 className="font-semibold mb-4">Current CPU % by Server</h3>
      <ResponsiveContainer width="100%" height={220}>
        <BarChart data={data} barSize={32}>
          <CartesianGrid strokeDasharray="3 3" stroke="#f0f0f0" />
          <XAxis dataKey="server" tick={{ fontSize: 11 }} />
          <YAxis domain={[0, 100]} tick={{ fontSize: 11 }} unit="%" />
          <Tooltip formatter={(v) => [`${v}%`, "CPU"]} />
          <Bar dataKey="cpu" radius={[4, 4, 0, 0]} isAnimationActive={false}>
            {data.map((entry, i) => (
              <Cell
                key={i}
                fill={entry.cpu > 80 ? "#ef4444" : entry.cpu > 60 ? "#f59e0b" : "#22c55e"}
              />
            ))}
          </Bar>
        </BarChart>
      </ResponsiveContainer>
    </div>
  );
}
```

```tsx
// components/LiveTable.tsx
"use client";

import type { Metric } from "@/hooks/useMetricsStream";

export function LiveTable({ metrics }: { metrics: Metric[] }) {
  // Show most recent 20 events, newest first
  const rows = [...metrics].reverse().slice(0, 20);

  function cpuColor(v: number) {
    if (v > 80) return "text-red-600";
    if (v > 60) return "text-yellow-600";
    return "text-green-600";
  }

  return (
    <div className="bg-white border rounded-2xl overflow-hidden">
      <div className="p-5 border-b">
        <h3 className="font-semibold">Live Event Log</h3>
      </div>
      <div className="overflow-x-auto">
        <table className="w-full text-sm">
          <thead>
            <tr className="border-b bg-gray-50">
              {["Server", "Region", "CPU%", "Memory%", "RPS", "Latency", "Error%"].map((h) => (
                <th key={h} className="text-left px-4 py-2 font-medium text-gray-500 text-xs">{h}</th>
              ))}
            </tr>
          </thead>
          <tbody>
            {rows.map((row) => (
              <tr key={row.id} className="border-b last:border-0 hover:bg-gray-50 transition-colors">
                <td className="px-4 py-2 font-mono text-xs">{row.server}</td>
                <td className="px-4 py-2 text-gray-500 text-xs">{row.region}</td>
                <td className={`px-4 py-2 font-medium ${cpuColor(row.cpu)}`}>{row.cpu}%</td>
                <td className={`px-4 py-2 font-medium ${cpuColor(row.memory)}`}>{row.memory}%</td>
                <td className="px-4 py-2">{row.requestsPerSec}</td>
                <td className={`px-4 py-2 ${row.latencyMs > 100 ? "text-orange-600" : ""}`}>{row.latencyMs}ms</td>
                <td className={`px-4 py-2 ${row.errorRate > 1 ? "text-red-600" : "text-gray-500"}`}>{row.errorRate}%</td>
              </tr>
            ))}
          </tbody>
        </table>
      </div>
    </div>
  );
}
```

```tsx
// app/dashboard/page.tsx
"use client";

import { useMetricsStream } from "@/hooks/useMetricsStream";
import { LiveLineChart } from "@/components/LiveLineChart";
import { LiveBarChart } from "@/components/LiveBarChart";
import { LiveTable } from "@/components/LiveTable";

export default function DashboardPage() {
  const { metrics, connected, paused, eventCount, pause, resume } = useMetricsStream();

  const latest = metrics[metrics.length - 1];

  function downloadCSV() {
    const header = "id,ts,server,region,cpu,memory,rps,latency,errorRate\n";
    const rows = metrics.map((m) =>
      `${m.id},${m.ts},${m.server},${m.region},${m.cpu},${m.memory},${m.requestsPerSec},${m.latencyMs},${m.errorRate}`
    ).join("\n");
    const blob = new Blob([header + rows], { type: "text/csv" });
    const url = URL.createObjectURL(blob);
    const a = document.createElement("a");
    a.href = url;
    a.download = "metrics.csv";
    a.click();
    URL.revokeObjectURL(url);
  }

  return (
    <div className="max-w-7xl mx-auto py-10 px-4 space-y-6">
      {/* Header */}
      <div className="flex items-center justify-between">
        <div>
          <h1 className="text-2xl font-bold">Real-Time Dashboard</h1>
          <p className="text-sm text-gray-500">{eventCount} events received</p>
        </div>
        <div className="flex items-center gap-3">
          <div className="flex items-center gap-1.5 text-sm">
            <span className={`w-2 h-2 rounded-full ${connected ? "bg-green-500 animate-pulse" : "bg-red-500"}`} />
            {connected ? "Live" : "Reconnecting..."}
          </div>
          <button
            onClick={paused ? resume : pause}
            className="px-4 py-2 text-sm border rounded-lg hover:bg-gray-50 transition-colors"
          >
            {paused ? "▶ Resume" : "⏸ Pause"}
          </button>
          <button
            onClick={downloadCSV}
            className="px-4 py-2 text-sm border rounded-lg hover:bg-gray-50 transition-colors"
          >
            ↓ CSV
          </button>
        </div>
      </div>

      {/* KPI tiles */}
      {latest && (
        <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
          {[
            { label: "CPU (latest)", value: `${latest.cpu}%` },
            { label: "Memory (latest)", value: `${latest.memory}%` },
            { label: "Req/s (latest)", value: latest.requestsPerSec },
            { label: "P50 Latency", value: `${latest.latencyMs}ms` },
          ].map((kpi) => (
            <div key={kpi.label} className="bg-white border rounded-2xl p-5">
              <p className="text-xs text-gray-500">{kpi.label}</p>
              <p className="text-2xl font-bold mt-1">{kpi.value}</p>
            </div>
          ))}
        </div>
      )}

      {/* Charts */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
        <LiveLineChart metrics={metrics} />
        <LiveBarChart metrics={metrics} />
      </div>

      {/* Table */}
      <LiveTable metrics={metrics} />
    </div>
  );
}
```

## Deliverables
- [ ] `/api/metrics` streams SSE events continuously using `ReadableStream`
- [ ] Multiple named event types: `connected`, `metric`, `summary`
- [ ] `useMetricsStream` hook manages `EventSource` lifecycle
- [ ] Pause button stops updating the UI (buffers incoming events)
- [ ] Resume button flushes the buffer and resumes live updates
- [ ] Connection status dot shows green (connected) or red (disconnected)
- [ ] Line chart updates in real time with no animation flicker (`isAnimationActive={false}`)
- [ ] Bar chart shows current CPU per server, color-coded by severity
- [ ] Data table shows the 20 most recent events, newest first
- [ ] CSV download produces a valid file with all buffered metrics
- [ ] App handles reconnection automatically (EventSource built-in retry)
