# P6.1 — Performance-Optimized Image Gallery

**Phase:** 6 — Performance & Core Web Vitals  
**Topic:** next/image, blur placeholders, lazy loading, LCP < 2.5s  
**Type:** Project

---

## Goal
Build a masonry/grid photo gallery with optimized images achieving LCP < 2.5 seconds, using `next/image` with proper `sizes`, `priority`, and `blurDataURL` for a great perceived performance experience.

## What to Build
- Gallery grid with 30+ images from Unsplash
- Hero image with `priority` (above the fold, counts for LCP)
- Lazy-loaded grid images with blur-up placeholder effect
- Lightbox on click (full-size image)
- Infinite scroll or "Load more" pagination
- Lighthouse performance score ≥ 90

## Architecture

```
app/
  gallery/
    page.tsx            ← Server Component: fetches image metadata
    loading.tsx         ← skeleton loading state
components/
  GalleryGrid.tsx       ← Client Component with intersection observer
  GalleryImage.tsx      ← single optimized image tile
  Lightbox.tsx          ← modal with full image
lib/
  images.ts             ← image data + plaiceholder blur generation
```

```ts
// lib/images.ts
import { getPlaiceholder } from "plaiceholder";

export interface GalleryImage {
  id: string;
  src: string;
  alt: string;
  width: number;
  height: number;
  blurDataURL: string;
  photographer: string;
}

// In a real app, fetch from Unsplash API
const IMAGE_IDS = [
  "photo-1506905925346-21bda4d32df4",
  "photo-1469474968028-56623f02e42e",
  "photo-1501854140801-50d01698950b",
  "photo-1447752875215-b2761acb3c5d",
  "photo-1433086966358-54859d0ed716",
  "photo-1505144808419-1957a94ca61e",
  // ... more IDs
];

export async function getGalleryImages(): Promise<GalleryImage[]> {
  const images = await Promise.all(
    IMAGE_IDS.map(async (id, i) => {
      const src = `https://images.unsplash.com/${id}?w=800&q=80`;

      // Generate blur placeholder server-side
      const { base64 } = await getPlaiceholder(src, { size: 10 });

      return {
        id,
        src,
        alt: `Landscape photo ${i + 1}`,
        width: 800,
        height: 600,
        blurDataURL: base64,
        photographer: "Unsplash",
      };
    })
  );

  return images;
}
```

```tsx
// app/gallery/page.tsx
import { Suspense } from "react";
import { getGalleryImages } from "@/lib/images";
import { GalleryGrid } from "@/components/GalleryGrid";
import Image from "next/image";
import type { Metadata } from "next";

export const metadata: Metadata = {
  title: "Photo Gallery",
  description: "A performance-optimized photo gallery.",
};

export default async function GalleryPage() {
  const images = await getGalleryImages();
  const [hero, ...rest] = images;

  return (
    <div className="max-w-7xl mx-auto px-4 py-8">
      <h1 className="text-3xl font-bold mb-8">Gallery</h1>

      {/* Hero image — priority: true for LCP */}
      <div className="relative w-full aspect-[21/9] rounded-2xl overflow-hidden mb-8 bg-gray-100">
        <Image
          src={hero.src}
          alt={hero.alt}
          fill
          priority               // fetch eagerly — this is the LCP element
          sizes="100vw"
          className="object-cover"
          placeholder="blur"
          blurDataURL={hero.blurDataURL}
        />
      </div>

      {/* Grid */}
      <Suspense fallback={<GallerySkeleton />}>
        <GalleryGrid images={rest} />
      </Suspense>
    </div>
  );
}

function GallerySkeleton() {
  return (
    <div className="columns-2 md:columns-3 lg:columns-4 gap-4 space-y-4">
      {Array.from({ length: 12 }).map((_, i) => (
        <div
          key={i}
          className="w-full rounded-xl bg-gray-100 animate-pulse"
          style={{ height: `${200 + (i % 3) * 80}px` }}
        />
      ))}
    </div>
  );
}
```

```tsx
// components/GalleryGrid.tsx
"use client";

import { useState } from "react";
import { GalleryImage as GalleryImageComponent } from "./GalleryImage";
import { Lightbox } from "./Lightbox";
import type { GalleryImage } from "@/lib/images";

const PAGE_SIZE = 12;

export function GalleryGrid({ images }: { images: GalleryImage[] }) {
  const [selected, setSelected] = useState<GalleryImage | null>(null);
  const [visible, setVisible] = useState(PAGE_SIZE);

  return (
    <>
      {/* Masonry grid using CSS columns */}
      <div className="columns-2 md:columns-3 lg:columns-4 gap-3 space-y-3">
        {images.slice(0, visible).map((image, i) => (
          <GalleryImageComponent
            key={image.id}
            image={image}
            onClick={() => setSelected(image)}
            // First 4 images load eagerly; rest are lazy
            eager={i < 4}
          />
        ))}
      </div>

      {/* Load more */}
      {visible < images.length && (
        <div className="flex justify-center mt-10">
          <button
            onClick={() => setVisible((v) => v + PAGE_SIZE)}
            className="px-8 py-3 bg-gray-900 text-white rounded-xl text-sm font-medium hover:bg-gray-700 transition-colors"
          >
            Load more photos
          </button>
        </div>
      )}

      {/* Lightbox */}
      {selected && (
        <Lightbox image={selected} onClose={() => setSelected(null)} />
      )}
    </>
  );
}
```

```tsx
"use client";

import Image from "next/image";
import type { GalleryImage } from "@/lib/images";

interface Props {
  image: GalleryImage;
  onClick: () => void;
  eager?: boolean;
}

export function GalleryImage({ image, onClick, eager }: Props) {
  return (
    <button
      onClick={onClick}
      className="group relative block w-full overflow-hidden rounded-xl break-inside-avoid bg-gray-100 cursor-zoom-in"
      aria-label={`View ${image.alt}`}
    >
      <Image
        src={image.src}
        alt={image.alt}
        width={image.width}
        height={image.height}
        // Critical: sizes tells the browser which image to download
        sizes="(max-width: 640px) 50vw, (max-width: 1024px) 33vw, 25vw"
        // loading="eager" for the first few images, "lazy" for the rest
        loading={eager ? "eager" : "lazy"}
        placeholder="blur"
        blurDataURL={image.blurDataURL}
        className="w-full h-auto object-cover transition-transform duration-300 group-hover:scale-105"
      />
      {/* Hover overlay */}
      <div className="absolute inset-0 bg-black/0 group-hover:bg-black/20 transition-colors" />
    </button>
  );
}
```

```tsx
// components/Lightbox.tsx
"use client";

import { useEffect } from "react";
import Image from "next/image";
import type { GalleryImage } from "@/lib/images";

interface Props {
  image: GalleryImage;
  onClose: () => void;
}

export function Lightbox({ image, onClose }: Props) {
  // Close on Escape key
  useEffect(() => {
    const handler = (e: KeyboardEvent) => { if (e.key === "Escape") onClose(); };
    window.addEventListener("keydown", handler);
    return () => window.removeEventListener("keydown", handler);
  }, [onClose]);

  // Prevent body scroll
  useEffect(() => {
    document.body.style.overflow = "hidden";
    return () => { document.body.style.overflow = ""; };
  }, []);

  return (
    <div
      className="fixed inset-0 z-50 flex items-center justify-center bg-black/90"
      onClick={onClose}
    >
      <button
        onClick={onClose}
        className="absolute top-4 right-4 text-white/80 hover:text-white text-4xl leading-none"
        aria-label="Close"
      >
        ×
      </button>

      <div
        className="relative max-w-5xl max-h-[90vh] w-full mx-4"
        onClick={(e) => e.stopPropagation()}
      >
        <Image
          src={image.src.replace("w=800", "w=1600")}
          alt={image.alt}
          width={image.width * 2}
          height={image.height * 2}
          sizes="(max-width: 1024px) 100vw, 80vw"
          priority  // load high-quality version eagerly
          className="max-h-[90vh] w-auto h-auto mx-auto rounded-lg"
          placeholder="blur"
          blurDataURL={image.blurDataURL}
        />
        <p className="text-white/60 text-xs text-center mt-2">{image.photographer}</p>
      </div>
    </div>
  );
}
```

## Performance Checklist

### LCP < 2.5s
- `priority` on the above-the-fold hero image prevents render-blocking resource discovery delay
- `blurDataURL` from plaiceholder shows a color preview instantly while the image loads
- `sizes` attribute ensures the browser downloads the smallest adequate image variant

### Avoiding layout shift (CLS = 0)
- All `<Image>` elements have explicit `width` and `height` — Next.js reserves space automatically
- Using `fill` with a container that has explicit dimensions also works

### Loading strategy
- Above-the-fold: `priority` (preload link in `<head>`, no lazy loading)
- First 4 grid images: `loading="eager"` (no lazy, but no preload)
- Rest: `loading="lazy"` (default — browser only fetches when near viewport)
