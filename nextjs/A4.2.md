## Goal
Manage client-side application state with Zustand — from simple stores through middleware and the slice pattern for large apps.

## Background

Zustand is a minimal, unopinionated state management library for React. Unlike Redux, there is no boilerplate, no reducers, no action creators, and no Provider required. You call `create()` with a function that receives `set` and `get`, and it returns a hook.

The core insight: Zustand stores are just closures. `set` triggers re-renders only in components that subscribe to the specific part of state that changed. This makes Zustand performant by default.

**Selectors** are the key to performance. Instead of `useStore()` (subscribes to everything), use `useStore((s) => s.count)` (subscribes only to `count`). This prevents unnecessary re-renders.

**Middleware** are higher-order functions that wrap `create`. `devtools` integrates with Redux DevTools browser extension. `persist` serialises state to `localStorage` or any storage adapter. `immer` lets you write mutative code that Zustand converts to immutable updates.

For large applications, the **slice pattern** splits state into logical modules ("slices"), each defined in its own file, composed together in a single store.

### 2. Basic store
```ts
// src/stores/counterStore.ts
import { create } from "zustand";
import { devtools } from "zustand/middleware";

type CounterState = {
  count: number;
  increment: () => void;
  decrement: () => void;
  reset: () => void;
  incrementBy: (amount: number) => void;
};

export const useCounterStore = create<CounterState>()(
  devtools(
    (set) => ({
      count: 0,
      increment: () => set((s) => ({ count: s.count + 1 }), false, "increment"),
      decrement: () => set((s) => ({ count: s.count - 1 }), false, "decrement"),
      reset:     () => set({ count: 0 }, false, "reset"),
      incrementBy: (amount) =>
        set((s) => ({ count: s.count + amount }), false, "incrementBy"),
    }),
    { name: "CounterStore" }
  )
);
```
```tsx
// src/app/zustand-demo/page.tsx
"use client";

import { useCounterStore } from "@/stores/counterStore";

// Selector pattern — only re-renders when count changes
function CountDisplay() {
  const count = useCounterStore((s) => s.count);
  return <span className="text-4xl font-mono font-bold">{count}</span>;
}

// This component only re-renders when actions change (they don't)
function CountControls() {
  const { increment, decrement, reset, incrementBy } = useCounterStore();
  return (
    <div className="flex gap-2">
      <button onClick={decrement} className="px-4 py-2 bg-gray-200 rounded">−</button>
      <button onClick={increment} className="px-4 py-2 bg-blue-600 text-white rounded">+</button>
      <button onClick={() => incrementBy(10)} className="px-4 py-2 bg-green-600 text-white rounded">+10</button>
      <button onClick={reset} className="px-4 py-2 bg-red-100 text-red-700 rounded">Reset</button>
    </div>
  );
}

export default function ZustandDemoPage() {
  return (
    <div className="space-y-4 p-8">
      <h1 className="text-2xl font-bold">Zustand Counter</h1>
      <CountDisplay />
      <CountControls />
    </div>
  );
}
```

### 3. Cart store with real-world shape
```ts
// src/stores/cartStore.ts
import { create } from "zustand";
import { devtools, persist } from "zustand/middleware";

export type CartItem = {
  id: number;
  name: string;
  price: number;
  quantity: number;
};

type CartState = {
  items: CartItem[];
  addItem: (item: Omit<CartItem, "quantity">) => void;
  removeItem: (id: number) => void;
  updateQuantity: (id: number, quantity: number) => void;
  clearCart: () => void;
  total: () => number;
  itemCount: () => number;
};

export const useCartStore = create<CartState>()(
  devtools(
    persist(
      (set, get) => ({
        items: [],

        addItem: (item) =>
          set(
            (s) => {
              const existing = s.items.find((i) => i.id === item.id);
              if (existing) {
                return {
                  items: s.items.map((i) =>
                    i.id === item.id ? { ...i, quantity: i.quantity + 1 } : i
                  ),
                };
              }
              return { items: [...s.items, { ...item, quantity: 1 }] };
            },
            false,
            "addItem"
          ),

        removeItem: (id) =>
          set((s) => ({ items: s.items.filter((i) => i.id !== id) }), false, "removeItem"),

        updateQuantity: (id, quantity) =>
          set(
            (s) => ({
              items: quantity <= 0
                ? s.items.filter((i) => i.id !== id)
                : s.items.map((i) => (i.id === id ? { ...i, quantity } : i)),
            }),
            false,
            "updateQuantity"
          ),

        clearCart: () => set({ items: [] }, false, "clearCart"),

        // Derived state as methods (called via get())
        total: () => get().items.reduce((sum, i) => sum + i.price * i.quantity, 0),
        itemCount: () => get().items.reduce((sum, i) => sum + i.quantity, 0),
      }),
      {
        name: "cart-storage", // localStorage key
        // Only persist the items array, not the functions
        partialize: (s) => ({ items: s.items }),
      }
    ),
    { name: "CartStore" }
  )
);
```

### 4. Slice pattern for large stores
```ts
// src/stores/slices/uiSlice.ts
import type { StateCreator } from "zustand";

export type UiSlice = {
  sidebarOpen: boolean;
  toggleSidebar: () => void;
  activeModal: string | null;
  openModal: (name: string) => void;
  closeModal: () => void;
};

export const createUiSlice: StateCreator<UiSlice> = (set) => ({
  sidebarOpen: false,
  toggleSidebar: () => set((s) => ({ sidebarOpen: !s.sidebarOpen })),
  activeModal: null,
  openModal: (name) => set({ activeModal: name }),
  closeModal: () => set({ activeModal: null }),
});
```
```ts
// src/stores/slices/userSlice.ts
import type { StateCreator } from "zustand";

export type UserSlice = {
  user: { id: string; name: string; email: string } | null;
  setUser: (user: UserSlice["user"]) => void;
  clearUser: () => void;
};

export const createUserSlice: StateCreator<UserSlice> = (set) => ({
  user: null,
  setUser: (user) => set({ user }),
  clearUser: () => set({ user: null }),
});
```
```ts
// src/stores/appStore.ts
import { create } from "zustand";
import { devtools } from "zustand/middleware";
import { createUiSlice, type UiSlice } from "./slices/uiSlice";
import { createUserSlice, type UserSlice } from "./slices/userSlice";

type AppStore = UiSlice & UserSlice;

export const useAppStore = create<AppStore>()(
  devtools(
    (...args) => ({
      ...createUiSlice(...args),
      ...createUserSlice(...args),
    }),
    { name: "AppStore" }
  )
);

// Typed selector hooks for better DX
export const useSidebar = () => useAppStore((s) => s.sidebarOpen);
export const useToggleSidebar = () => useAppStore((s) => s.toggleSidebar);
export const useCurrentUser = () => useAppStore((s) => s.user);
```

### 5. Accessing store outside React (e.g., in a utility)
```ts
// src/lib/api.ts (using store state outside a component)
import { useCartStore } from "@/stores/cartStore";

export function getCartHeaders() {
  // Access store state directly without a hook
  const { items } = useCartStore.getState();
  return {
    "X-Cart-Count": String(items.length),
  };
}
```
