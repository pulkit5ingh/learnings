## Goal
Understand Core Web Vitals, measure them on a real deployed Next.js app, and fix the most common issues.

## Background

Core Web Vitals (CWV) are Google's user-experience metrics. They directly impact search rankings:

**LCP (Largest Contentful Paint)** — How fast does the main content appear? Target: < 2.5s. The LCP element is usually a hero image or the largest text block. Common causes of poor LCP: render-blocking resources, large unoptimised images, slow server response, no `priority` prop on hero image.

**INP (Interaction to Next Paint)** — How responsive is the page to interactions? Replaced FID in 2024. Target: < 200ms. Common causes: heavy event handlers, synchronous state updates blocking paint, third-party scripts.

**CLS (Cumulative Layout Shift)** — Does content jump around as it loads? Target: < 0.1. Common causes: images without dimensions, dynamically injected content above existing content, fonts causing FOUT (flash of unstyled text), ads.

Every fix must be verified with a measurement. The cycle is: **Measure → Identify → Fix → Measure again**.

## Tasks

### 1. Add CWV measurement to your app
```tsx
// src/app/layout.tsx (add WebVitals)
import { WebVitals } from "@/components/WebVitals";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        {children}
        <WebVitals />
      </body>
    </html>
  );
}
```
```tsx
// src/components/WebVitals.tsx
"use client";

import { useReportWebVitals } from "next/web-vitals";

export function WebVitals() {
  useReportWebVitals((metric) => {
    // Log to console in dev
    if (process.env.NODE_ENV === "development") {
      console.log(`[CWV] ${metric.name}:`, Math.round(metric.value), "ms");
    }

    // Send to your analytics in production
    if (process.env.NODE_ENV === "production") {
      // Example: send to your own endpoint
      fetch("/api/analytics/vitals", {
        method: "POST",
        body: JSON.stringify(metric),
        headers: { "Content-Type": "application/json" },
      }).catch(() => {}); // fire and forget
    }
  });

  return null;
}
```

### 3. LCP — Fix hero image
```tsx
// BEFORE (bad LCP):
// <img src="/hero.jpg" />  ← no preload, no size hints

// AFTER (good LCP):
import Image from "next/image";

export function HeroImage() {
  return (
    <div className="relative h-[70vh] w-full">
      <Image
        src="/hero.jpg"         // local image
        alt="Hero"
        fill
        priority               // ← Adds <link rel="preload">
        sizes="100vw"
        quality={85}
        className="object-cover"
      />
    </div>
  );
}
```

### 4. CLS — Fix images without dimensions
```tsx
// BEFORE (causes CLS — browser doesn't know the aspect ratio):
<img src="photo.jpg" /> // no width/height

// AFTER option 1 — explicit dimensions:
<Image src="photo.jpg" alt="Photo" width={800} height={600} />

// AFTER option 2 — fill mode with aspect-ratio wrapper:
<div className="relative aspect-video w-full">
  <Image src="photo.jpg" alt="Photo" fill className="object-cover" />
</div>
```

### 5. CLS — Prevent font layout shift
```tsx
// BAD — uses system font until web font loads (FOUT → CLS)
// <link href="https://fonts.googleapis.com/css2?family=Inter" rel="stylesheet" />

// GOOD — next/font eliminates FOUT:
import { Inter } from "next/font/google";

const inter = Inter({
  subsets: ["latin"],
  display: "swap",
  // size-adjust and ascent-override are computed automatically
  // to match the fallback font metrics → zero CLS
});
```

### 6. INP — Defer heavy state updates
```tsx
// BEFORE — heavy computation blocks paint
function SearchInput() {
  const [results, setResults] = useState([]);

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value;
    // This runs 1000 items through filter synchronously BEFORE the input updates
    const filtered = massiveList.filter(item => item.includes(value));
    setResults(filtered);
  }
  // ...
}

// AFTER — use startTransition to mark the expensive update as non-urgent
import { startTransition, useState } from "react";

function SearchInput() {
  const [inputValue, setInputValue] = useState("");
  const [results, setResults] = useState([]);

  function handleChange(e: React.ChangeEvent<HTMLInputElement>) {
    const value = e.target.value;
    setInputValue(value);   // urgent: update input immediately

    startTransition(() => {
      // non-urgent: React can interrupt this if the user types again
      const filtered = massiveList.filter(item => item.includes(value));
      setResults(filtered);
    });
  }
  // ...
}
```

### 7. CLS — Prevent content jump from dynamic data
```tsx
// BAD — skeleton collapses to 0 height, then content expands → CLS
function UserGreeting() {
  const { data: user } = useUser();
  if (!user) return null; // ← zero height then expands → CLS!
  return <p>Hello, {user.name}</p>;
}

// GOOD — skeleton has same dimensions as real content
function UserGreeting() {
  const { data: user } = useUser();
  if (!user) {
    return <div className="h-6 w-32 bg-gray-200 rounded animate-pulse" />;
  }
  return <p className="h-6">Hello, {user.name}</p>;
}
```

### 9. PageSpeed Insights API (automated)
```ts
// src/scripts/check-cwv.ts (run with: ts-node src/scripts/check-cwv.ts)
async function checkPageSpeed(url: string) {
  const apiKey = process.env.PAGESPEED_API_KEY;
  const endpoint = `https://www.googleapis.com/pagespeedonline/v5/runPagespeed?url=${encodeURIComponent(url)}&strategy=mobile&key=${apiKey}`;

  const res = await fetch(endpoint);
  const data = await res.json();

  const cwv = data.loadingExperience?.metrics;
  console.log("LCP:", cwv?.LARGEST_CONTENTFUL_PAINT_MS?.category);
  console.log("CLS:", cwv?.CUMULATIVE_LAYOUT_SHIFT_SCORE?.category);
  console.log("INP:", cwv?.INTERACTION_TO_NEXT_PAINT?.category);
}

checkPageSpeed("https://your-app.vercel.app");
```
