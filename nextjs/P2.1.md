# P2.1 — Blog with ISR

**Phase:** 2 — Data Fetching & Rendering  
**Topic:** MDX posts, generateStaticParams, ISR revalidation, RSS feed via Route Handler  
**Type:** Project

---

## Goal
Build a blog that statically generates pages from MDX files, revalidates stale pages with ISR, and exposes an RSS feed via a Route Handler.

## What to Build
- Blog index with list of posts sorted by date
- Individual post pages generated statically from MDX
- ISR: revalidate pages every 60 seconds (or on-demand)
- RSS 2.0 feed at `/feed.xml` via Route Handler
- Reading time estimate on each post
- Tags/categories with filtering

## Architecture

```
app/
  blog/
    page.tsx              ← posts list
    [slug]/
      page.tsx            ← single post (MDX + ISR)
  api/
    feed/
      route.ts            ← RSS 2.0 XML feed
content/
  posts/
    hello-world.mdx
    nextjs-ssr-vs-ssg.mdx
lib/
  posts.ts                ← getAllPosts, getPost, reading time
components/
  PostCard.tsx
  TagBadge.tsx
```

```ts
// lib/posts.ts
import fs from "fs";
import path from "path";
import matter from "gray-matter";

const POSTS_DIR = path.join(process.cwd(), "content/posts");

export interface PostMeta {
  title: string;
  description: string;
  date: string;   // ISO date string e.g. "2024-03-15"
  tags: string[];
  author: string;
}

export interface Post extends PostMeta {
  slug: string;
  content: string;
  readingTime: number; // in minutes
}

export function getAllPosts(): Post[] {
  const files = fs.readdirSync(POSTS_DIR).filter((f) => f.endsWith(".mdx"));

  return files
    .map((file) => {
      const slug = file.replace(".mdx", "");
      const raw = fs.readFileSync(path.join(POSTS_DIR, file), "utf8");
      const { data, content } = matter(raw);
      const wordCount = content.trim().split(/\s+/).length;
      return {
        slug,
        ...(data as PostMeta),
        content,
        readingTime: Math.ceil(wordCount / 200),
      };
    })
    .sort((a, b) => new Date(b.date).getTime() - new Date(a.date).getTime());
}

export function getPost(slug: string): Post | null {
  const filePath = path.join(POSTS_DIR, `${slug}.mdx`);
  if (!fs.existsSync(filePath)) return null;
  const raw = fs.readFileSync(filePath, "utf8");
  const { data, content } = matter(raw);
  const wordCount = content.trim().split(/\s+/).length;
  return {
    slug,
    ...(data as PostMeta),
    content,
    readingTime: Math.ceil(wordCount / 200),
  };
}
```

```tsx
// app/blog/page.tsx
import type { Metadata } from "next";
import { getAllPosts } from "@/lib/posts";
import { PostCard } from "@/components/PostCard";

export const revalidate = 60; // ISR: revalidate at most every 60s

export const metadata: Metadata = {
  title: "Blog",
  description: "Articles on Next.js, React, and web development.",
};

export default async function BlogPage({
  searchParams,
}: {
  searchParams: { tag?: string };
}) {
  const posts = getAllPosts();
  const tag = searchParams.tag;
  const filtered = tag ? posts.filter((p) => p.tags.includes(tag)) : posts;

  const allTags = Array.from(new Set(posts.flatMap((p) => p.tags)));

  return (
    <section className="py-16 px-4 max-w-3xl mx-auto">
      <h1 className="text-3xl font-bold mb-2">Blog</h1>
      <p className="text-gray-500 mb-8">{posts.length} articles</p>

      {/* Tag filter */}
      <div className="flex flex-wrap gap-2 mb-10">
        <a
          href="/blog"
          className={`px-3 py-1 rounded-full text-sm border ${!tag ? "bg-gray-900 text-white border-gray-900" : "border-gray-200 hover:bg-gray-50"}`}
        >
          All
        </a>
        {allTags.map((t) => (
          <a
            key={t}
            href={`/blog?tag=${t}`}
            className={`px-3 py-1 rounded-full text-sm border ${tag === t ? "bg-gray-900 text-white border-gray-900" : "border-gray-200 hover:bg-gray-50"}`}
          >
            {t}
          </a>
        ))}
      </div>

      <div className="space-y-6">
        {filtered.map((post) => (
          <PostCard key={post.slug} post={post} />
        ))}
      </div>
    </section>
  );
}
```

```tsx
// app/blog/[slug]/page.tsx
import { notFound } from "next/navigation";
import type { Metadata } from "next";
import { MDXRemote } from "next-mdx-remote/rsc";
import { getAllPosts, getPost } from "@/lib/posts";

export const revalidate = 60;

interface Props {
  params: { slug: string };
}

export async function generateStaticParams() {
  return getAllPosts().map((p) => ({ slug: p.slug }));
}

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const post = getPost(params.slug);
  if (!post) return { title: "Not Found" };
  return {
    title: post.title,
    description: post.description,
    openGraph: {
      type: "article",
      publishedTime: post.date,
      authors: [post.author],
      tags: post.tags,
    },
  };
}

export default function PostPage({ params }: Props) {
  const post = getPost(params.slug);
  if (!post) notFound();

  return (
    <article className="py-16 px-4 max-w-3xl mx-auto">
      {/* Header */}
      <header className="mb-10">
        <div className="flex flex-wrap gap-2 mb-4">
          {post.tags.map((t) => (
            <a key={t} href={`/blog?tag=${t}`} className="px-2 py-0.5 bg-blue-100 text-blue-700 rounded text-xs font-medium">
              {t}
            </a>
          ))}
        </div>
        <h1 className="text-4xl font-bold leading-tight mb-4">{post.title}</h1>
        <div className="flex items-center gap-4 text-sm text-gray-500">
          <span>{post.author}</span>
          <span>·</span>
          <time dateTime={post.date}>
            {new Date(post.date).toLocaleDateString("en-US", {
              year: "numeric", month: "long", day: "numeric",
            })}
          </time>
          <span>·</span>
          <span>{post.readingTime} min read</span>
        </div>
      </header>

      {/* Content */}
      <div className="prose prose-gray max-w-none">
        <MDXRemote source={post.content} />
      </div>
    </article>
  );
}
```

```ts
// app/api/feed/route.ts
import { getAllPosts } from "@/lib/posts";
import { NextResponse } from "next/server";

const SITE_URL = process.env.NEXT_PUBLIC_SITE_URL ?? "https://example.com";

export const revalidate = 3600; // revalidate feed hourly

export async function GET() {
  const posts = getAllPosts().slice(0, 20); // latest 20

  const rss = `<?xml version="1.0" encoding="UTF-8"?>
<rss version="2.0" xmlns:atom="http://www.w3.org/2005/Atom">
  <channel>
    <title>My Blog</title>
    <link>${SITE_URL}/blog</link>
    <description>Articles on Next.js, React, and web development.</description>
    <language>en-us</language>
    <lastBuildDate>${new Date().toUTCString()}</lastBuildDate>
    <atom:link href="${SITE_URL}/api/feed" rel="self" type="application/rss+xml"/>
    ${posts
      .map(
        (post) => `
    <item>
      <title><![CDATA[${post.title}]]></title>
      <link>${SITE_URL}/blog/${post.slug}</link>
      <guid isPermaLink="true">${SITE_URL}/blog/${post.slug}</guid>
      <description><![CDATA[${post.description}]]></description>
      <pubDate>${new Date(post.date).toUTCString()}</pubDate>
      ${post.tags.map((t) => `<category>${t}</category>`).join("\n      ")}
    </item>`
      )
      .join("")}
  </channel>
</rss>`;

  return new NextResponse(rss, {
    headers: {
      "Content-Type": "application/rss+xml; charset=utf-8",
      "Cache-Control": "s-maxage=3600, stale-while-revalidate=86400",
    },
  });
}
```

## Deliverables
- [ ] `/blog` lists all posts sorted by date
- [ ] Tag filter in URL (`?tag=nextjs`) filters the list without a page reload losing the filter
- [ ] `/blog/[slug]` renders full MDX post with author, date, and reading time
