# P3.1 — Landing Page

**Phase:** 3 — Styling  
**Topic:** Tailwind CSS, responsive design, dark mode toggle, Framer Motion animations  
**Type:** Project

---

## Goal
Build a polished marketing landing page with a hero, features, pricing, and footer — fully responsive, with a dark mode toggle and smooth Framer Motion animations.

## What to Build
- Hero section with animated headline and CTA buttons
- Features grid (3 or 6 cards) with hover effects
- Pricing section (3 tiers)
- Footer with links
- Dark mode toggle (next-themes)
- Framer Motion: fade-in-up on scroll, staggered children, smooth page transitions

## Architecture

```
app/
  layout.tsx            ← ThemeProvider wrapper
  page.tsx              ← assembles all sections
components/
  sections/
    Hero.tsx
    Features.tsx
    Pricing.tsx
    Footer.tsx
  ui/
    ThemeToggle.tsx
    AnimatedSection.tsx ← reusable scroll-triggered animation wrapper
  Header.tsx
```

```tsx
// components/ui/ThemeToggle.tsx
"use client";

import { useTheme } from "next-themes";
import { useEffect, useState } from "react";

export function ThemeToggle() {
  const { theme, setTheme } = useTheme();
  const [mounted, setMounted] = useState(false);

  useEffect(() => setMounted(true), []);
  if (!mounted) return <div className="w-9 h-9" />; // prevent hydration flash

  return (
    <button
      onClick={() => setTheme(theme === "dark" ? "light" : "dark")}
      className="w-9 h-9 flex items-center justify-center rounded-lg border border-gray-200 dark:border-gray-700 hover:bg-gray-100 dark:hover:bg-gray-800 transition-colors"
      aria-label="Toggle dark mode"
    >
      {theme === "dark" ? (
        // Sun icon
        <svg className="w-4 h-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M12 3v1m0 16v1m9-9h-1M4 12H3m15.364-6.364l-.707.707M6.343 17.657l-.707.707M17.657 17.657l-.707-.707M6.343 6.343l-.707-.707M12 8a4 4 0 100 8 4 4 0 000-8z" />
        </svg>
      ) : (
        // Moon icon
        <svg className="w-4 h-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M20.354 15.354A9 9 0 018.646 3.646 9.003 9.003 0 0012 21a9.003 9.003 0 008.354-5.646z" />
        </svg>
      )}
    </button>
  );
}
```

```tsx
// components/ui/AnimatedSection.tsx
"use client";

import { motion, useInView } from "framer-motion";
import { useRef } from "react";

interface Props {
  children: React.ReactNode;
  className?: string;
  delay?: number;
}

export function AnimatedSection({ children, className, delay = 0 }: Props) {
  const ref = useRef<HTMLDivElement>(null);
  const isInView = useInView(ref, { once: true, margin: "-100px" });

  return (
    <motion.div
      ref={ref}
      initial={{ opacity: 0, y: 32 }}
      animate={isInView ? { opacity: 1, y: 0 } : {}}
      transition={{ duration: 0.6, delay, ease: [0.22, 1, 0.36, 1] }}
      className={className}
    >
      {children}
    </motion.div>
  );
}
```
