# P1.1 — Portfolio Site

**Phase:** 1 — Next.js & App Router Fundamentals  
**Topic:** Static App Router pages, Metadata API, dynamic project pages, custom 404  
**Type:** Project

---

## Goal
Build a personal portfolio site using Next.js 14 App Router with static pages, a dynamic projects section, proper SEO metadata, and a custom 404 page.

## What to Build
- Home page with hero, about, and skills sections
- Projects listing page with dynamic `[slug]` pages per project
- Contact page (static)
- Custom `not-found.tsx` (404) page
- `generateMetadata` for every page (title, description, OG tags)
- `layout.tsx` with shared header and footer

## Architecture

```
app/
  layout.tsx              ← root layout: <html>, <body>, <Header>, <Footer>
  page.tsx                ← home: hero + about + skills
  not-found.tsx           ← custom 404 page
  projects/
    page.tsx              ← projects listing (all cards)
    [slug]/
      page.tsx            ← individual project page
  contact/
    page.tsx              ← contact page (static)
  globals.css
components/
  Header.tsx
  Footer.tsx
  ProjectCard.tsx
lib/
  projects.ts             ← project data + getProject() helper
```

```tsx
// app/projects/[slug]/page.tsx
import { notFound } from "next/navigation";
import type { Metadata } from "next";
import { getProject, projects } from "@/lib/projects";

interface Props {
  params: { slug: string };
}

export async function generateStaticParams() {
  return projects.map((p) => ({ slug: p.slug }));
}

export async function generateMetadata({ params }: Props): Promise<Metadata> {
  const project = getProject(params.slug);
  if (!project) return { title: "Not Found" };
  return {
    title: project.title,
    description: project.description,
    openGraph: {
      title: project.title,
      description: project.description,
      images: [{ url: project.coverImage }],
    },
  };
}

export default function ProjectPage({ params }: Props) {
  const project = getProject(params.slug);
  if (!project) notFound();

  return (
    <article className="py-16 px-4 max-w-3xl mx-auto">
      <a href="/projects" className="text-blue-600 text-sm hover:underline">
        ← All projects
      </a>
      <h1 className="text-4xl font-bold mt-4 mb-2">{project.title}</h1>
      <div className="flex flex-wrap gap-2 mb-6">
        {project.tech.map((t) => (
          <span key={t} className="px-2 py-0.5 bg-blue-100 text-blue-700 rounded text-xs font-medium">
            {t}
          </span>
        ))}
      </div>
      <p className="text-gray-600 text-lg leading-relaxed mb-8">{project.longDescription}</p>
      <div className="flex gap-4">
        <a
          href={project.github}
          target="_blank"
          rel="noopener noreferrer"
          className="px-5 py-2 border rounded-lg text-sm hover:bg-gray-50 transition-colors"
        >
          GitHub →
        </a>
        {project.demo && (
          <a
            href={project.demo}
            target="_blank"
            rel="noopener noreferrer"
            className="px-5 py-2 bg-blue-600 text-white rounded-lg text-sm hover:bg-blue-700 transition-colors"
          >
            Live Demo →
          </a>
        )}
      </div>
    </article>
  );
}
```

## Deliverables
- [ ] Home, Projects, and Contact pages render without errors
- [ ] `/projects/[slug]` renders correct content for each project slug
- [ ] `generateStaticParams` pre-renders all project pages at build time
- [ ] Visiting a non-existent slug shows the custom `not-found.tsx` page
- [ ] Visiting any unknown URL shows the 404 page
- [ ] `generateMetadata` sets unique title and description per page
- [ ] `<title>` in `<head>` uses the template `"Project Name | John Doe"`
- [ ] OG tags present on project pages (check with `curl -s URL | grep og:`)
- [ ] Header navigation links are active and functional
- [ ] App builds without TypeScript errors (`pnpm build`)
