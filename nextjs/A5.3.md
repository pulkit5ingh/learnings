# A5.3 — Session & Tokens

**Phase:** 5 — Authentication  
**Topic:** JWT vs database sessions, accessing session in Server Components / Route Handlers / Client  
**Type:** Assignment

---

## Goal
Understand JWT vs database session strategies and know how to access the session from any context in a Next.js app.

## Background

Auth.js supports two session strategies: **JWT** and **database**.

With **JWT sessions**, the session data is stored in a signed, encrypted cookie. No database round-trip is needed to read the session — the cookie IS the session. This makes it fast and works without a database. The trade-off: you cannot invalidate a session without making it expire or adding a blocklist.

With **database sessions**, only a random session token is stored in the cookie. Each request queries the database to look up the full session. This allows instant invalidation (delete the row) and lets you store arbitrary session data. The trade-off: every authenticated request hits the database.

For most apps, JWT sessions are the right choice. Use database sessions when you need instant revocation (e.g., after a security incident or account deletion).

**Accessing the session** depends on your context:
- **Server Component**: `const session = await auth()`
- **Route Handler**: `const session = await auth()` (same API)
- **Middleware**: `req.auth` (provided by the auth middleware)
- **Client Component**: `useSession()` from `next-auth/react` (requires `SessionProvider`)

## Tasks

### 1. JWT strategy (default) — config reminder
```ts
// src/auth.ts
export const { handlers, auth, signIn, signOut } = NextAuth({
  // ...providers
  session: {
    strategy: "jwt",        // store session in JWT cookie
    maxAge: 30 * 24 * 60 * 60, // 30 days
    updateAge: 24 * 60 * 60,   // refresh token if older than 24h
  },
});
```

### 2. Database strategy with Prisma adapter
```ts
// src/auth.ts (database strategy)
import NextAuth from "next-auth";
import { PrismaAdapter } from "@auth/prisma-adapter";
import { prisma } from "@/lib/prisma";
import Google from "next-auth/providers/google";

export const { handlers, auth, signIn, signOut } = NextAuth({
  adapter: PrismaAdapter(prisma),
  providers: [Google({ /* ... */ })],
  session: {
    strategy: "database",   // session stored in DB
  },
  callbacks: {
    session({ session, user }) {
      // With DB sessions, `user` comes from the DB (not JWT)
      session.user.id = user.id;
      return session;
    },
  },
});
```

```prisma

  id           String   @id @default(cuid())
  sessionToken String   @unique
  userId       String
  expires      DateTime
  user         User     @relation(fields: [userId], references: [id], onDelete: Cascade)

  id            String    @id @default(cuid())

  identifier String
  token      String   @unique
  expires    DateTime
  @@unique([identifier, token])
}
```

### 3. Session in Server Components
```tsx
// src/app/server-session/page.tsx
import { auth } from "@/auth";
import { redirect } from "next/navigation";

export default async function ServerSessionPage() {
  // auth() is the ONLY way to read session in Server Components
  const session = await auth();

  if (!session) redirect("/login");

  return (
    <div className="p-6 space-y-2">
      <h1 className="text-xl font-bold">Session from Server Component</h1>
      <pre className="p-4 bg-gray-50 rounded text-sm overflow-auto">
        {JSON.stringify(session, null, 2)}
      </pre>
    </div>
  );
}
```

### 4. Session in Route Handlers
```ts
// src/app/api/me/route.ts
import { auth } from "@/auth";
import { NextResponse } from "next/server";

export async function GET() {
  const session = await auth();

  if (!session) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  return NextResponse.json({
    user: {
      id: session.user.id,
      name: session.user.name,
      email: session.user.email,
      role: (session.user as any).role,
    },
  });
}
```

### 5. Session in Client Components with SessionProvider
```tsx
// src/providers/SessionProvider.tsx
"use client";

import { SessionProvider as NextAuthSessionProvider } from "next-auth/react";

export function SessionProvider({ children }: { children: React.ReactNode }) {
  return <NextAuthSessionProvider>{children}</NextAuthSessionProvider>;
}
```
```tsx
// src/app/layout.tsx (add SessionProvider)
import { SessionProvider } from "@/providers/SessionProvider";

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en">
      <body>
        <SessionProvider>{children}</SessionProvider>
      </body>
    </html>
  );
}
```
```tsx
// src/components/UserMenu.tsx
"use client";

import { useSession, signOut } from "next-auth/react";

export function UserMenu() {
  const { data: session, status } = useSession();

  if (status === "loading") {
    return <div className="w-8 h-8 rounded-full bg-gray-200 animate-pulse" />;
  }

  if (!session) {
    return (
      <a href="/login" className="text-sm font-medium text-blue-600 hover:underline">
        Sign in
      </a>
    );
  }

  return (
    <div className="flex items-center gap-3">
      <div className="text-right hidden sm:block">
        <p className="text-sm font-medium">{session.user?.name}</p>
        <p className="text-xs text-gray-500">{session.user?.email}</p>
      </div>
      <button
        onClick={() => signOut({ callbackUrl: "/" })}
        className="text-sm text-gray-600 hover:underline"
      >
        Sign out
      </button>
    </div>
  );
}
```

### 6. Session in middleware (via req.auth)
```ts
// src/middleware.ts
import { auth } from "@/auth";

export default auth((req) => {
  const { auth: session, nextUrl } = req;

  // Access session data directly from req.auth
  const userId = session?.user?.id;
  const role = (session?.user as any)?.role;

  // Log for debugging
  console.log(`[middleware] ${nextUrl.pathname} — user: ${userId ?? "unauthenticated"}`);
});
```

### 7. Session refresh and update
```ts
// src/auth.ts — callbacks for session update
callbacks: {
  async jwt({ token, trigger, session }) {
    // trigger === "update" when update() is called from client
    if (trigger === "update" && session?.name) {
      token.name = session.name; // merge updated data
    }
    return token;
  },
}
```
```tsx
// Client: update session after profile edit
import { useSession } from "next-auth/react";
const { update } = useSession();
await update({ name: "New Name" }); // triggers jwt callback with trigger = "update"
```
