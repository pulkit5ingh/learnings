## Goal
Write comprehensive component tests that cover interactions, conditional UI, and API calls mocked with Mock Service Worker.

## Background

Mock Service Worker (MSW) is the gold standard for mocking API calls in tests. Unlike `jest.spyOn(global, "fetch")`, MSW intercepts requests at the network level — your component makes real `fetch()` calls, and MSW intercepts them before they hit the network. This means you test the full component code path, including error handling.

MSW works in both browser (via Service Worker) and Node.js (via `http-interceptor`). In Jest, you use the `node` package. You define handlers that match URL patterns and return mocked responses. Each test can override handlers to test different scenarios.

**Testing patterns:**
- Use `waitFor` when you need to assert that something happens after async work completes
- Use `within` to scope queries to a specific container when there are multiple similar elements
- Mock timers with `jest.useFakeTimers()` for debounce/throttle tests
- Test accessibility with `jest-axe`

### 2. Define MSW handlers
```ts
// src/mocks/handlers.ts
import { http, HttpResponse } from "msw";

export const handlers = [
  // GET /api/users
  http.get("/api/users", () => {
    return HttpResponse.json([
      { id: 1, name: "Alice", email: "alice@example.com" },
      { id: 2, name: "Bob",   email: "bob@example.com"   },
    ]);
  }),

  // GET /api/users/:id
  http.get("/api/users/:id", ({ params }) => {
    const id = Number(params.id);
    if (id === 999) {
      return HttpResponse.json({ error: "Not found" }, { status: 404 });
    }
    return HttpResponse.json({ id, name: "Alice", email: "alice@example.com" });
  }),

  // POST /api/users
  http.post("/api/users", async ({ request }) => {
    const body = await request.json() as { name: string; email: string };
    return HttpResponse.json(
      { id: Date.now(), ...body },
      { status: 201 }
    );
  }),
];
```

// jest.setup.ts (updated)
import "@testing-library/jest-dom";

### 4. Test a component that fetches data
```tsx
// src/components/UserList.tsx
"use client";

import { useState, useEffect } from "react";

type User = { id: number; name: string; email: string };

export function UserList() {
  const [users, setUsers] = useState<User[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetch("/api/users")
      .then((r) => {
        if (!r.ok) throw new Error("Failed to load users");
        return r.json();
      })
      .then(setUsers)
      .catch((e) => setError(e.message))
      .finally(() => setLoading(false));
  }, []);

  if (loading) return <p>Loading users…</p>;
  if (error) return <p role="alert">Error: {error}</p>;
  return (
    <ul>
      {users.map((u) => (
        <li key={u.id}>{u.name} — {u.email}</li>
      ))}
    </ul>
  );
}
```
```tsx
// src/components/__tests__/UserList.test.tsx
import { render, screen } from "@testing-library/react";
import { server } from "@/mocks/server";
import { http, HttpResponse } from "msw";
import { UserList } from "../UserList";

describe("UserList", () => {
  it("shows loading state initially", () => {
    render(<UserList />);
    expect(screen.getByText("Loading users…")).toBeInTheDocument();
  });

  it("renders users after successful fetch", async () => {
    render(<UserList />);

    // findBy waits for the async fetch to complete
    expect(await screen.findByText("Alice — alice@example.com")).toBeInTheDocument();
    expect(screen.getByText("Bob — bob@example.com")).toBeInTheDocument();
  });

  it("shows error message when fetch fails", async () => {
    // Override the handler just for this test
    server.use(
      http.get("/api/users", () => {
        return HttpResponse.json({ error: "Server error" }, { status: 500 });
      })
    );

    render(<UserList />);

    const errorEl = await screen.findByRole("alert");
    expect(errorEl).toHaveTextContent("Error:");
  });
});
```

### 5. Test form submission with API call
```tsx
// src/components/CreateUserForm.tsx
"use client";

import { useState } from "react";

export function CreateUserForm({ onCreated }: { onCreated: () => void }) {
  const [name, setName] = useState("");
  const [email, setEmail] = useState("");
  const [status, setStatus] = useState<"idle" | "loading" | "success" | "error">("idle");

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setStatus("loading");
    try {
      const res = await fetch("/api/users", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ name, email }),
      });
      if (!res.ok) throw new Error("Failed");
      setStatus("success");
      onCreated();
    } catch {
      setStatus("error");
    }
  }

  return (
    <form onSubmit={handleSubmit}>
      <label htmlFor="name">Name</label>
      <input id="name" value={name} onChange={(e) => setName(e.target.value)} required />
      <label htmlFor="email">Email</label>
      <input id="email" type="email" value={email} onChange={(e) => setEmail(e.target.value)} required />
      <button type="submit" disabled={status === "loading"}>
        {status === "loading" ? "Creating…" : "Create"}
      </button>
      {status === "success" && <p role="status">User created!</p>}
      {status === "error" && <p role="alert">Failed to create user</p>}
    </form>
  );
}
```
```tsx
// src/components/__tests__/CreateUserForm.test.tsx
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";
import { CreateUserForm } from "../CreateUserForm";

describe("CreateUserForm", () => {
  it("submits form and shows success message", async () => {
    const user = userEvent.setup();
    const onCreated = jest.fn();
    render(<CreateUserForm onCreated={onCreated} />);

    await user.type(screen.getByLabelText("Name"), "Charlie");
    await user.type(screen.getByLabelText("Email"), "charlie@example.com");
    await user.click(screen.getByRole("button", { name: "Create" }));

    // Wait for async submission
    await screen.findByRole("status");
    expect(screen.getByText("User created!")).toBeInTheDocument();
    expect(onCreated).toHaveBeenCalledTimes(1);
  });

  it("shows error message on API failure", async () => {
    const { server } = await import("@/mocks/server");
    const { http, HttpResponse } = await import("msw");

    server.use(
      http.post("/api/users", () =>
        HttpResponse.json({ error: "Server error" }, { status: 500 })
      )
    );

    const user = userEvent.setup();
    render(<CreateUserForm onCreated={jest.fn()} />);

    await user.type(screen.getByLabelText("Name"), "Eve");
    await user.type(screen.getByLabelText("Email"), "eve@example.com");
    await user.click(screen.getByRole("button", { name: "Create" }));

    await screen.findByRole("alert");
    expect(screen.getByText("Failed to create user")).toBeInTheDocument();
  });

  it("disables button while submitting", async () => {
    const user = userEvent.setup();
    render(<CreateUserForm onCreated={jest.fn()} />);

    await user.type(screen.getByLabelText("Name"), "Alice");
    await user.type(screen.getByLabelText("Email"), "alice@example.com");

    const button = screen.getByRole("button");
    await user.click(button);

    // During submission, button shows loading text
    // (may need to check before async resolves)
    await screen.findByRole("status"); // wait for completion
    expect(screen.getByRole("button", { name: "Create" })).not.toBeDisabled();
  });
});
```

### 6. Conditional rendering tests
```tsx
// src/components/__tests__/conditional.test.tsx
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";
import { useState } from "react";

function Toggle({ label }: { label: string }) {
  const [open, setOpen] = useState(false);
  return (
    <div>
      <button onClick={() => setOpen((o) => !o)}>{open ? "Close" : "Open"}</button>
      {open && <div role="dialog" aria-label={label}><p>Dialog content</p></div>}
    </div>
  );
}

describe("Toggle", () => {
  it("does not render dialog initially", () => {
    render(<Toggle label="Settings" />);
    expect(screen.queryByRole("dialog")).not.toBeInTheDocument();
  });

  it("renders dialog when Open is clicked", async () => {
    const user = userEvent.setup();
    render(<Toggle label="Settings" />);

    await user.click(screen.getByRole("button", { name: "Open" }));

    expect(screen.getByRole("dialog", { name: "Settings" })).toBeInTheDocument();
    expect(screen.getByRole("button", { name: "Close" })).toBeInTheDocument();
  });

  it("hides dialog when Close is clicked", async () => {
    const user = userEvent.setup();
    render(<Toggle label="Settings" />);

    await user.click(screen.getByRole("button", { name: "Open" }));
    await user.click(screen.getByRole("button", { name: "Close" }));

    expect(screen.queryByRole("dialog")).not.toBeInTheDocument();
  });
});
```
