## Goal
Build forms that submit data via Server Actions — no API routes needed — and implement optimistic UI updates.

## Background

**Server Actions** are async functions that run on the server but are called from the client. They eliminate the need for a separate API route for mutations. You define them with the `"use server"` directive either at the top of a file (making all exports Server Actions) or at the top of an async function inside a Server Component.

Server Actions integrate directly with HTML `<form>` elements via the `action` attribute. When the form submits, the browser sends a `POST` request (or the React runtime intercepts it for progressive enhancement), and the Server Action runs on the server with full access to databases, secrets, and file system.

`useFormState` (React 18 + Next.js) wraps a Server Action and gives you back the action's return value as state. `useFormStatus` is a hook that must be used inside the form's child component — it tells you whether the parent form is currently pending.

`useOptimistic` lets you immediately update the UI before the server confirms the action. If the server action fails, the optimistic update is rolled back. This creates a snappy feel even on slow connections.

## Tasks

### 1. Minimal Server Action with a form
```tsx
// src/app/actions/page.tsx
import { revalidatePath } from "next/cache";

// ✅ "use server" at function level (inside a Server Component file)
async function addTodo(formData: FormData) {
  "use server";

  const text = formData.get("text") as string;
  if (!text?.trim()) return;

  // In a real app: await db.todo.create({ data: { text } })
  console.log("Server: creating todo:", text);

  revalidatePath("/actions");
}

export default function ActionsPage() {
  return (
    <div className="max-w-md">
      <h1 className="text-2xl font-bold mb-4">Server Actions</h1>
      <form action={addTodo} className="flex gap-2">
        <input
          name="text"
          placeholder="New todo…"
          className="flex-1 px-3 py-2 border rounded"
          required
        />
        <button
          type="submit"
          className="px-4 py-2 bg-blue-600 text-white rounded"
        >
          Add
        </button>
      </form>
    </div>
  );
}
```

### 2. Server Action in a separate file (for use from Client Components)
```ts
// src/app/actions/todo-actions.ts
"use server";

import { revalidatePath } from "next/cache";

export type ActionState = {
  error?: string;
  success?: string;
};

// Simple in-memory store for demo (use a DB in real projects)
const todos: { id: number; text: string }[] = [];
let nextId = 1;

export async function createTodo(
  prevState: ActionState,
  formData: FormData
): Promise<ActionState> {
  const text = formData.get("text") as string;

  if (!text?.trim()) {
    return { error: "Todo text cannot be empty" };
  }

  if (text.length > 100) {
    return { error: "Todo text must be under 100 characters" };
  }

  // Simulate DB latency
  await new Promise((r) => setTimeout(r, 500));

  todos.push({ id: nextId++, text: text.trim() });
  revalidatePath("/actions-with-state");

  return { success: `Created: "${text}"` };
}

export async function getTodos() {
  return [...todos];
}
```

// src/app/actions-with-state/TodoForm.tsx
"use client";

import { useFormState } from "react-dom";
import { createTodo } from "../actions/todo-actions";
import { SubmitButton } from "./SubmitButton";

const INITIAL_STATE = {};

export function TodoForm() {
  const [state, formAction] = useFormState(createTodo, INITIAL_STATE);

  return (
    <form action={formAction} className="space-y-2">
      <div>
        <input
          name="text"
          placeholder="New todo…"
          className="w-full px-3 py-2 border rounded"
        />
        {state.error && (
          <p className="text-sm text-red-600 mt-1">{state.error}</p>
        )}
        {state.success && (
          <p className="text-sm text-green-600 mt-1">{state.success}</p>
        )}
      </div>
      <SubmitButton />
    </form>
  );
}

### 4. useFormStatus — disable button while pending
```tsx
// src/app/actions-with-state/SubmitButton.tsx
"use client";

import { useFormStatus } from "react-dom";

export function SubmitButton() {
  const { pending } = useFormStatus();

  return (
    <button
      type="submit"
      disabled={pending}
      className="px-4 py-2 bg-blue-600 text-white rounded disabled:opacity-50 disabled:cursor-not-allowed"
    >
      {pending ? "Adding…" : "Add Todo"}
    </button>
  );
}
```

```tsx
// src/app/optimistic/OptimisticList.tsx
"use client";

import { useOptimistic, useState, useTransition } from "react";
import { createTodo } from "../actions/todo-actions";

type Todo = { id: number; text: string };

export function OptimisticList({ initialTodos }: { initialTodos: Todo[] }) {
  const [isPending, startTransition] = useTransition();
  const [optimisticTodos, addOptimisticTodo] = useOptimistic(
    initialTodos,
    (state: Todo[], newText: string) => [
      ...state,
      { id: Date.now(), text: newText + " (saving…)" },
    ]
  );

  async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();
    const form = e.currentTarget;
    const text = (form.elements.namedItem("text") as HTMLInputElement).value;
    form.reset();

    startTransition(async () => {
      addOptimisticTodo(text); // Instantly update UI
      const formData = new FormData();
      formData.set("text", text);
      await createTodo({}, formData); // Server confirms
    });
  }

  return (
    <div className="max-w-md space-y-4">
      <h1 className="text-2xl font-bold">Optimistic Updates</h1>
      <form onSubmit={handleSubmit} className="flex gap-2">
        <input name="text" className="flex-1 px-3 py-2 border rounded" placeholder="New todo…" />
        <button type="submit" disabled={isPending}
          className="px-4 py-2 bg-green-600 text-white rounded disabled:opacity-50">
          Add
        </button>
      </form>
      <ul className="space-y-2">
        {optimisticTodos.map((t) => (
          <li key={t.id} className={`p-2 border rounded ${t.text.includes("saving") ? "opacity-60" : ""}`}>
            {t.text}
          </li>
        ))}
      </ul>
    </div>
  );
}
```
