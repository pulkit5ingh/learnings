## Goal
Add English and Spanish internationalization to a Next.js app using next-intl, with locale-based routing and translated metadata.

## Background

Internationalization (i18n) in Next.js involves two concerns: **routing** (serving the right URL for the right locale) and **translation** (rendering content in the right language).

**next-intl** is the most popular i18n library for the App Router. It supports:
- Locale routing via middleware (e.g., `/en/about`, `/es/about`)
- Type-safe message dictionaries
- Pluralization, date formatting, number formatting
- Server Components AND Client Components
- Translated metadata via `generateMetadata`

The routing strategy: all content routes are nested under a `[locale]` dynamic segment (`app/[locale]/page.tsx`). Middleware intercepts requests, detects the preferred locale (from URL, cookie, or Accept-Language header), and redirects/rewrites accordingly.

**RTL languages** (Arabic, Hebrew, Persian) require `dir="rtl"` on the `<html>` element. next-intl makes this easy to implement by checking the locale.

### 3. Configure next-intl
```ts
// src/i18n.ts
import { getRequestConfig } from "next-intl/server";

export const LOCALES = ["en", "es"] as const;
export type Locale = (typeof LOCALES)[number];
export const DEFAULT_LOCALE: Locale = "en";

export default getRequestConfig(async ({ locale }) => ({
  messages: (await import(`../messages/${locale}.json`)).default,
}));
```

### 4. Middleware for locale routing
```ts
// src/middleware.ts
import createMiddleware from "next-intl/middleware";
import { LOCALES, DEFAULT_LOCALE } from "./i18n";

export default createMiddleware({
  locales: LOCALES,
  defaultLocale: DEFAULT_LOCALE,
  localePrefix: "always", // /en/about, /es/about (always show locale in URL)
  // Alternative: "as-needed" — /about (default), /es/about (non-default)
});

export const config = {
  matcher: ["/((?!_next|_vercel|.*\\..*).*)"],
};
```

### 6. Root [locale] layout
```tsx
// src/app/[locale]/layout.tsx
import { NextIntlClientProvider } from "next-intl";
import { getMessages } from "next-intl/server";
import { LOCALES, type Locale } from "@/i18n";
import { notFound } from "next/navigation";
import { Inter } from "next/font/google";

const inter = Inter({ subsets: ["latin"], variable: "--font-inter" });

export function generateStaticParams() {
  return LOCALES.map((locale) => ({ locale }));
}

const RTL_LOCALES = ["ar", "he", "fa"];

export default async function LocaleLayout({
  children,
  params: { locale },
}: {
  children: React.ReactNode;
  params: { locale: string };
}) {
  if (!LOCALES.includes(locale as Locale)) notFound();

  const messages = await getMessages();
  const direction = RTL_LOCALES.includes(locale) ? "rtl" : "ltr";

  return (
    <html lang={locale} dir={direction} className={inter.variable}>
      <body>
        <NextIntlClientProvider messages={messages}>
          {children}
        </NextIntlClientProvider>
      </body>
    </html>
  );
}
```

### 7. Use translations in Server Components
```tsx
// src/app/[locale]/page.tsx
import { useTranslations } from "next-intl";
import { getTranslations } from "next-intl/server";
import type { Metadata } from "next";

export async function generateMetadata({
  params: { locale },
}: {
  params: { locale: string };
}): Promise<Metadata> {
  const t = await getTranslations({ locale, namespace: "home" });
  const tCommon = await getTranslations({ locale, namespace: "common" });

  return {
    title: t("title", { appName: tCommon("appName") }),
  };
}

export default function HomePage() {
  const t = useTranslations("home");
  const tCommon = useTranslations("common");

  return (
    <div className="text-center py-20">
      <h1 className="text-4xl font-bold">
        {t("title", { appName: tCommon("appName") })}
      </h1>
      <p className="mt-4 text-xl text-gray-600">{t("subtitle")}</p>
      <a href="#" className="mt-8 inline-block px-6 py-3 bg-blue-600 text-white rounded-lg">
        {t("cta")}
      </a>
    </div>
  );
}
```

### 8. Language switcher
```tsx
// src/components/LanguageSwitcher.tsx
"use client";

import { useLocale } from "next-intl";
import { usePathname, useRouter } from "next/navigation";
import { LOCALES } from "@/i18n";

const LOCALE_LABELS: Record<string, string> = {
  en: "English",
  es: "Español",
};

export function LanguageSwitcher() {
  const locale = useLocale();
  const pathname = usePathname();
  const router = useRouter();

  function switchLocale(newLocale: string) {
    // Replace the locale segment in the current URL
    const currentLocale = locale;
    const newPath = pathname.replace(`/${currentLocale}`, `/${newLocale}`);
    router.push(newPath);
  }

  return (
    <div className="flex gap-1">
      {LOCALES.map((loc) => (
        <button
          key={loc}
          onClick={() => switchLocale(loc)}
          className={`px-2 py-1 text-sm rounded ${
            loc === locale ? "bg-blue-600 text-white" : "border hover:bg-gray-50"
          }`}
        >
          {LOCALE_LABELS[loc]}
        </button>
      ))}
    </div>
  );
}
```

### 9. useTranslations in Client Components
```tsx
// src/components/DeleteConfirmDialog.tsx
"use client";

import { useTranslations } from "next-intl";

export function DeleteConfirmDialog({ onConfirm, onCancel }: {
  onConfirm: () => void;
  onCancel: () => void;
}) {
  const t = useTranslations("common");

  return (
    <div className="fixed inset-0 bg-black/50 flex items-center justify-center">
      <div className="bg-white rounded-xl p-6 max-w-sm w-full">
        <p className="font-medium">Are you sure you want to delete this?</p>
        <div className="flex gap-2 mt-4">
          <button onClick={onCancel} className="flex-1 py-2 border rounded">
            {t("cancel")}
          </button>
          <button onClick={onConfirm} className="flex-1 py-2 bg-red-600 text-white rounded">
            {t("delete")}
          </button>
        </div>
      </div>
    </div>
  );
}
```
