# P4.2 — Multi-Step Form

**Phase:** 4 — State Management & Client Data Fetching  
**Topic:** React Hook Form + Zod, step navigation, form persistence, mutation  
**Type:** Project

---

## Goal
Build a multi-step onboarding form (4 steps) using React Hook Form and Zod for per-step validation, with form state persisted to `localStorage` so users can resume after a page reload, and final submission via React Query mutation.

## What to Build
- Step 1: Personal info (name, email, phone)
- Step 2: Account setup (username, password, confirm password)
- Step 3: Preferences (role, team size, newsletter toggle)
- Step 4: Review screen (show all entries before submit)
- Progress indicator showing current step
- Back/Next navigation with per-step Zod validation
- `localStorage` persistence: form data survives reloads
- Final submission via `useMutation` → POST to `/api/onboard`

## Architecture

```
app/
  onboard/
    page.tsx            ← multi-step form shell
  api/
    onboard/
      route.ts          ← POST endpoint
components/
  steps/
    PersonalInfoStep.tsx
    AccountStep.tsx
    PreferencesStep.tsx
    ReviewStep.tsx
  StepIndicator.tsx
lib/
  onboardSchema.ts      ← Zod schemas per step + full schema
```

```ts
// lib/onboardSchema.ts
import { z } from "zod";

export const personalInfoSchema = z.object({
  firstName: z.string().min(1, "First name is required").max(50),
  lastName: z.string().min(1, "Last name is required").max(50),
  email: z.string().email("Invalid email address"),
  phone: z.string().regex(/^\+?[\d\s\-()]{7,15}$/, "Invalid phone number").optional().or(z.literal("")),
});

export const accountSchema = z
  .object({
    username: z
      .string()
      .min(3, "Username must be at least 3 characters")
      .max(30)
      .regex(/^[a-z0-9_-]+$/, "Only lowercase letters, numbers, underscores, and hyphens"),
    password: z
      .string()
      .min(8, "Password must be at least 8 characters")
      .regex(/[A-Z]/, "Must contain an uppercase letter")
      .regex(/[0-9]/, "Must contain a number"),
    confirmPassword: z.string(),
  })
  .refine((data) => data.password === data.confirmPassword, {
    message: "Passwords do not match",
    path: ["confirmPassword"],
  });

export const preferencesSchema = z.object({
  role: z.enum(["developer", "designer", "manager", "other"], {
    required_error: "Please select a role",
  }),
  teamSize: z.enum(["solo", "2-10", "11-50", "50+"], {
    required_error: "Please select a team size",
  }),
  newsletter: z.boolean().default(false),
});

export const fullOnboardSchema = personalInfoSchema
  .merge(accountSchema.omit({ confirmPassword: true }))
  .merge(preferencesSchema);

export type PersonalInfo = z.infer<typeof personalInfoSchema>;
export type Account = z.infer<typeof accountSchema>;
export type Preferences = z.infer<typeof preferencesSchema>;
export type OnboardData = z.infer<typeof fullOnboardSchema>;
```

```tsx
// app/onboard/page.tsx
"use client";

import { useState, useEffect } from "react";
import { useForm } from "react-hook-form";
import { zodResolver } from "@hookform/resolvers/zod";
import { useMutation } from "@tanstack/react-query";
import {
  personalInfoSchema, accountSchema, preferencesSchema,
  type PersonalInfo, type Account, type Preferences
} from "@/lib/onboardSchema";
import { PersonalInfoStep } from "@/components/steps/PersonalInfoStep";
import { AccountStep } from "@/components/steps/AccountStep";
import { PreferencesStep } from "@/components/steps/PreferencesStep";
import { ReviewStep } from "@/components/steps/ReviewStep";
import { StepIndicator } from "@/components/StepIndicator";

// pnpm add @hookform/resolvers

type FormData = PersonalInfo & Account & Preferences;

const STORAGE_KEY = "onboard-form-data";
const STEPS = ["Personal Info", "Account", "Preferences", "Review"];

const STEP_SCHEMAS = [personalInfoSchema, accountSchema, preferencesSchema];

export default function OnboardPage() {
  const [step, setStep] = useState(0);
  const [completed, setCompleted] = useState(false);

  const form = useForm<FormData>({
    // Per-step resolver — only validates fields in the current step's schema
    resolver: zodResolver(STEP_SCHEMAS[step] ?? personalInfoSchema),
    defaultValues: {},
    mode: "onTouched",
  });

  // Load persisted data on mount
  useEffect(() => {
    const saved = localStorage.getItem(STORAGE_KEY);
    if (saved) {
      try {
        const parsed = JSON.parse(saved);
        form.reset(parsed);
        if (parsed._step) setStep(parsed._step);
      } catch {}
    }
  }, []);

  // Persist form data on every change
  useEffect(() => {
    const sub = form.watch((values) => {
      localStorage.setItem(STORAGE_KEY, JSON.stringify({ ...values, _step: step }));
    });
    return () => sub.unsubscribe();
  }, [form.watch, step]);

  const mutation = useMutation({
    mutationFn: async (data: FormData) => {
      const res = await fetch("/api/onboard", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(data),
      });
      if (!res.ok) throw new Error("Submission failed");
      return res.json();
    },
    onSuccess: () => {
      localStorage.removeItem(STORAGE_KEY);
      setCompleted(true);
    },
  });

  async function handleNext() {
    const valid = await form.trigger();
    if (!valid) return;
    setStep((s) => s + 1);
  }

  async function handleSubmit(data: FormData) {
    mutation.mutate(data);
  }

  if (completed) {
    return (
      <div className="min-h-screen flex items-center justify-center">
        <div className="text-center space-y-4">
          <div className="text-5xl">🎉</div>
          <h1 className="text-2xl font-bold">You're all set!</h1>
          <p className="text-gray-500">Your account has been created.</p>
        </div>
      </div>
    );
  }

  return (
    <div className="min-h-screen flex items-center justify-center px-4 py-16">
      <div className="w-full max-w-lg">
        <StepIndicator steps={STEPS} currentStep={step} />

        <div className="mt-8 bg-white border rounded-2xl p-8 shadow-sm">
          <h2 className="text-xl font-bold mb-6">{STEPS[step]}</h2>

          <form onSubmit={form.handleSubmit(handleSubmit)}>
            {step === 0 && <PersonalInfoStep form={form} />}
            {step === 1 && <AccountStep form={form} />}
            {step === 2 && <PreferencesStep form={form} />}
            {step === 3 && <ReviewStep values={form.getValues()} />}

            <div className="flex justify-between mt-8 pt-6 border-t">
              <button
                type="button"
                onClick={() => setStep((s) => s - 1)}
                disabled={step === 0}
                className="px-5 py-2.5 border rounded-xl text-sm font-medium disabled:opacity-40 hover:bg-gray-50 transition-colors"
              >
                Back
              </button>

              {step < STEPS.length - 1 ? (
                <button
                  type="button"
                  onClick={handleNext}
                  className="px-5 py-2.5 bg-blue-600 text-white rounded-xl text-sm font-medium hover:bg-blue-700 transition-colors"
                >
                  Continue
                </button>
              ) : (
                <button
                  type="submit"
                  disabled={mutation.isPending}
                  className="px-5 py-2.5 bg-blue-600 text-white rounded-xl text-sm font-medium hover:bg-blue-700 disabled:opacity-50 transition-colors"
                >
                  {mutation.isPending ? "Submitting..." : "Create account"}
                </button>
              )}
            </div>
          </form>

          {mutation.isError && (
            <p className="mt-3 text-sm text-red-600 text-center">
              {(mutation.error as Error).message}
            </p>
          )}
        </div>

        {/* Resume indicator */}
        <p className="text-xs text-center text-gray-400 mt-4">
          Your progress is saved automatically
        </p>
      </div>
    </div>
  );
}
```

```tsx
// components/StepIndicator.tsx
interface Props {
  steps: string[];
  currentStep: number;
}

export function StepIndicator({ steps, currentStep }: Props) {
  return (
    <div className="flex items-center gap-0">
      {steps.map((label, i) => (
        <div key={label} className="flex items-center flex-1 last:flex-none">
          <div className="flex flex-col items-center gap-1.5">
            <div
              className={`w-8 h-8 rounded-full flex items-center justify-center text-sm font-semibold transition-colors ${
                i < currentStep
                  ? "bg-blue-600 text-white"
                  : i === currentStep
                  ? "bg-blue-600 text-white ring-4 ring-blue-100"
                  : "bg-gray-100 text-gray-400"
              }`}
            >
              {i < currentStep ? "✓" : i + 1}
            </div>
            <span className={`text-xs whitespace-nowrap ${i === currentStep ? "text-blue-600 font-medium" : "text-gray-400"}`}>
              {label}
            </span>
          </div>
          {i < steps.length - 1 && (
            <div className={`flex-1 h-0.5 mx-1 mb-4 transition-colors ${i < currentStep ? "bg-blue-600" : "bg-gray-200"}`} />
          )}
        </div>
      ))}
    </div>
  );
}
```

```ts
import { NextRequest, NextResponse } from "next/server";

export async function POST(req: NextRequest) {
  const body = await req.json();
  const parsed = fullOnboardSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.flatten() }, { status: 400 });
```

## Deliverables
- [ ] 4-step form with distinct sections (Personal, Account, Preferences, Review)
- [ ] Clicking "Continue" on step 1 validates only step 1 fields (not later fields)
- [ ] Clicking "Back" preserves already-entered data
- [ ] Refreshing the page restores the current step and entered values from `localStorage`
- [ ] Password mismatch on step 2 shows an error on `confirmPassword`
- [ ] Step indicator shows completed (✓), active (highlighted), and upcoming steps
