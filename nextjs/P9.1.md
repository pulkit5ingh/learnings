# P9.1 — i18n SaaS App

**Phase:** 9 — Advanced Features  
**Topic:** next-intl, 2 languages, locale switching, translated metadata  
**Type:** Project

---

## Goal
Build an internationalized SaaS landing page and app supporting English and Spanish, with locale-aware routing, a persistent language switcher, translated metadata, and locale-aware date/number formatting.

## What to Build
- Full site available at `/en/...` and `/es/...`
- Language switcher in the header (persists via cookie)
- Translated page titles, descriptions (via `generateMetadata`)
- Locale-aware date formatting (e.g. "March 15, 2024" vs "15 de marzo de 2024")
- Locale-aware number formatting (currency, large numbers)
- Pluralization for item counts
- Marketing landing page + authenticated dashboard, both translated

## Architecture

```
messages/
  en.json               ← all English strings
  es.json               ← all Spanish strings
src/
  i18n.ts               ← next-intl config
  middleware.ts         ← locale detection + routing
  app/
    [locale]/
      layout.tsx        ← NextIntlClientProvider + html lang + dir
      page.tsx          ← landing page
      dashboard/
        page.tsx        ← dashboard (requires auth)
      pricing/
        page.tsx
components/
  LanguageSwitcher.tsx
  LocaleLink.tsx        ← Link that prepends the current locale
```

```ts
// src/i18n.ts
import { getRequestConfig } from "next-intl/server";
import { notFound } from "next/navigation";

export const LOCALES = ["en", "es"] as const;
export type Locale = (typeof LOCALES)[number];
export const DEFAULT_LOCALE: Locale = "en";

export default getRequestConfig(async ({ locale }) => {
  if (!LOCALES.includes(locale as Locale)) notFound();

  return {
    messages: (await import(`../messages/${locale}.json`)).default,
    timeZone: "UTC",
    now: new Date(),
  };
});
```

```tsx
// src/app/[locale]/layout.tsx
import { NextIntlClientProvider } from "next-intl";
import { getMessages, getTranslations } from "next-intl/server";
import { Inter } from "next/font/google";
import { LOCALES, type Locale } from "@/i18n";
import { notFound } from "next/navigation";
import type { Metadata } from "next";
import { Nav } from "@/components/Nav";

const inter = Inter({ subsets: ["latin", "latin-ext"], variable: "--font-inter" });

export async function generateStaticParams() {
  return LOCALES.map((locale) => ({ locale }));
}

export async function generateMetadata({
  params: { locale },
}: {
  params: { locale: string };
}): Promise<Metadata> {
  const t = await getTranslations({ locale, namespace: "common" });
  return {
    title: {
      default: t("appName"),
      template: `%s — ${t("appName")}`,
    },
  };
}

export default async function LocaleLayout({
  children,
  params: { locale },
}: {
  children: React.ReactNode;
  params: { locale: string };
}) {
  if (!LOCALES.includes(locale as Locale)) notFound();

  const messages = await getMessages();

  return (
    <html lang={locale} className={inter.variable}>
      <body className="min-h-screen bg-white text-gray-900 antialiased">
        <NextIntlClientProvider messages={messages}>
          <Nav locale={locale} />
          <main>{children}</main>
        </NextIntlClientProvider>
      </body>
    </html>
  );
}
```

```tsx
// components/LanguageSwitcher.tsx
"use client";

import { useLocale } from "next-intl";
import { usePathname, useRouter } from "next/navigation";
import { LOCALES } from "@/i18n";

const LABELS: Record<string, string> = { en: "🇬🇧 EN", es: "🇪🇸 ES" };

export function LanguageSwitcher() {
  const locale = useLocale();
  const pathname = usePathname();
  const router = useRouter();

  function switchTo(newLocale: string) {
    // Replace /en/ with /es/ (or vice versa) in the path
    const newPath = pathname.replace(`/${locale}`, `/${newLocale}`);
    router.push(newPath);
    // Persist locale preference in a cookie
    document.cookie = `NEXT_LOCALE=${newLocale}; path=/; max-age=31536000; SameSite=lax`;
  }

  return (
    <div className="flex gap-1 rounded-lg border p-1">
      {LOCALES.map((loc) => (
        <button
          key={loc}
          onClick={() => switchTo(loc)}
          className={`px-2.5 py-1 rounded-md text-sm font-medium transition-colors ${
            loc === locale
              ? "bg-gray-900 text-white"
              : "text-gray-500 hover:text-gray-900"
          }`}
        >
          {LABELS[loc]}
        </button>
      ))}
    </div>
  );
}
```

- [ ] `/en` serves English content and `/es` serves Spanish content
- [ ] Middleware redirects `/` to `/en` (default locale)
- [ ] `<html lang="en">` and `<html lang="es">` set correctly per locale
- [ ] Language switcher changes locale and keeps the user on the same page
- [ ] Locale preference persists via cookie across page reloads
- [ ] Dashboard shows dates in locale-appropriate format (e.g. "March 15" vs "15 de marzo")
- [ ] Dashboard shows currency in locale-appropriate format ($12,500 vs same with . separator in es)
- [ ] `generateMetadata` returns translated title and description per locale
- [ ] Page titles in `<head>` use the translated template
- [ ] Missing translation keys are caught at TypeScript compile time (next-intl type safety)
- [ ] `pnpm build` completes without errors
