## Goal
Configure a Next.js project properly and implement complete SEO metadata including OpenGraph, robots, and sitemap.

## Background

`next.config.js` (or `next.config.mjs`) is the central configuration file for a Next.js project. It controls image domains, redirects, rewrites, environment variables exposure, experimental features, and much more. You will edit it frequently in real projects, so understanding its shape is essential.

The **Metadata API** is the App Router's replacement for the old `<Head>` component. You export a `metadata` object from any `layout.tsx` or `page.tsx`, and Next.js automatically injects the correct `<meta>` tags into the HTML. For dynamic metadata (e.g., a blog post title based on its slug), you export an async `generateMetadata` function instead, which receives the same `params` as the page.

OpenGraph (OG) metadata controls how your page appears when shared on social platforms. The most important tags are `og:title`, `og:description`, `og:image`, and `og:url`. Next.js supports OG image generation at `/opengraph-image.tsx` — you can render React components as images using the `ImageResponse` API.

`robots.txt` and `sitemap.xml` are discoverable by search engines. In the App Router you generate them programmatically from files named `robots.ts` and `sitemap.ts` inside the `app/` directory — they export functions that return typed objects Next.js serialises automatically.

## Tasks

### 1. Configure next.config.js
```js
// next.config.js
/** @type {import('next').NextConfig} */
const nextConfig = {
  // Allow images from these external domains
  images: {
    remotePatterns: [
      {
        protocol: "https",
        hostname: "images.unsplash.com",
        pathname: "/**",
      },
      {
        protocol: "https",
        hostname: "avatars.githubusercontent.com",
      },
    ],
  },

  // Permanent redirect example
  async redirects() {
    return [
      {
        source: "/old-blog/:slug",
        destination: "/blog/:slug",
        permanent: true, // 308
      },
    ];
  },

  // Rewrite example (proxy /api/v1/* to an external service)
  async rewrites() {
    return [
      {
        source: "/api/v1/:path*",
        destination: "https://api.example.com/:path*",
      },
    ];
  },

  // Enable standalone output for Docker deployments
  output: "standalone",

  // Experimental features (opt-in)
  experimental: {
    typedRoutes: true, // type-checks <Link href> values
  },
};

module.exports = nextConfig;
```

```tsx
// src/app/about/page.tsx
import type { Metadata } from "next";

export const metadata: Metadata = {
  title: "About Us",
  description: "Learn about our team and mission.",
  openGraph: {

    description: "Learn about our team and mission.",
    url: "https://myapp.com/about",
    siteName: "My App",
    images: [
      {
        url: "https://myapp.com/og-about.png",

      },

    card: "summary_large_image",

  },
};

export default function AboutPage() {
  return <h1 className="text-2xl font-bold">About Us</h1>;
}
```

### 3. Dynamic metadata with generateMetadata
```tsx
// src/app/blog/[slug]/page.tsx
import type { Metadata } from "next";
import { notFound } from "next/navigation";

type Post = { slug: string; title: string; excerpt: string };

async function getPost(slug: string): Promise<Post | null> {
  const posts: Post[] = [
    { slug: "hello-world", title: "Hello World", excerpt: "My first post." },
    { slug: "nextjs-tips", title: "Next.js Tips", excerpt: "Tips for Next.js." },
  ];
  return posts.find((p) => p.slug === slug) ?? null;
}

// Called BEFORE the page renders — used to build <head>
export async function generateMetadata({
  params,
}: {
  params: { slug: string };
}): Promise<Metadata> {
  const post = await getPost(params.slug);

  if (!post) {
    return { title: "Post Not Found" };
  }

  return {
    title: post.title,
    description: post.excerpt,
    openGraph: {
      title: post.title,
      description: post.excerpt,
      type: "article",
      url: `https://myapp.com/blog/${post.slug}`,
      images: [
        {
          url: `https://myapp.com/api/og?title=${encodeURIComponent(post.title)}`,
          width: 1200,
          height: 630,
        },
      ],
    },
  };
}

export default async function BlogPostPage({
  params,
}: {
  params: { slug: string };
}) {
  const post = await getPost(params.slug);
  if (!post) notFound();

  return (
    <article>
      <h1 className="text-3xl font-bold">{post.title}</h1>
      <p className="mt-4 text-gray-600">{post.excerpt}</p>
    </article>
  );
}
```

### 4. OG Image generation
```tsx
// src/app/api/og/route.tsx
import { ImageResponse } from "next/og";
import { NextRequest } from "next/server";

export const runtime = "edge";

export function GET(request: NextRequest) {
  const { searchParams } = new URL(request.url);
  const title = searchParams.get("title") ?? "My App";

  return new ImageResponse(
    (
      <div
        style={{
          display: "flex",
          flexDirection: "column",
          alignItems: "center",
          justifyContent: "center",
          width: "100%",
          height: "100%",
          backgroundColor: "#0f172a",
          color: "#f1f5f9",
          fontFamily: "sans-serif",
          padding: "60px",
        }}
      >
        <p style={{ fontSize: 24, color: "#94a3b8", marginBottom: 16 }}>
          My App
        </p>
        <h1 style={{ fontSize: 64, fontWeight: 700, textAlign: "center" }}>
          {title}
        </h1>
      </div>
    ),
    { width: 1200, height: 630 }
  );
}
```

### 5. Robots.txt
```ts
// src/app/robots.ts
import { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        allow: "/",
        disallow: ["/dashboard/", "/api/"],
      },
    ],
    sitemap: "https://myapp.com/sitemap.xml",
  };
}
```
Visit `/robots.txt` to see the generated output.

### 6. Sitemap
```ts
// src/app/sitemap.ts
import { MetadataRoute } from "next";

export default function sitemap(): MetadataRoute.Sitemap {
  const BASE = "https://myapp.com";

  const staticRoutes = ["/", "/about", "/blog", "/contact"].map((route) => ({
    url: `${BASE}${route}`,
    lastModified: new Date(),
    changeFrequency: "monthly" as const,
    priority: route === "/" ? 1 : 0.8,
  }));

  // In a real app, fetch posts from DB here
  const posts = [
    { slug: "hello-world", updatedAt: new Date("2024-01-01") },
    { slug: "nextjs-tips", updatedAt: new Date("2024-02-01") },
  ];

  const postRoutes = posts.map((post) => ({
    url: `${BASE}/blog/${post.slug}`,
    lastModified: post.updatedAt,
    changeFrequency: "weekly" as const,
    priority: 0.6,
  }));

  return [...staticRoutes, ...postRoutes];
}
```
Visit `/sitemap.xml` to see the generated XML.
