# A7.4 — Testing Next.js Specifics

**Phase:** 7 — Testing  
**Topic:** Mock next/navigation, test Server Actions, test Route Handlers, Playwright for page flows  
**Type:** Assignment

---

## Goal
Test Next.js-specific patterns: mock the App Router navigation hooks, test Server Actions and Route Handlers in isolation.

## Background

Testing Next.js components that use `useRouter`, `usePathname`, or `useSearchParams` requires mocking the `next/navigation` module. Without mocking, these hooks throw errors in the Jest/jsdom environment because they depend on the Next.js router context.

Server Actions and Route Handlers are just async functions — they can be imported and called directly in tests. For Server Actions, you call the function with arguments as your component would. For Route Handlers, you construct a `Request` object and call the exported function directly.

**Testing Server Actions:**
- Call the action function directly with test data
- Mock database calls with `jest.mock()`
- Verify `revalidatePath` was called
- Verify return values for error/success states

**Testing Route Handlers:**
- Create a `new Request("http://localhost/api/route", { method: "POST", body: ... })`
- Call the exported `GET`/`POST` function with the request
- Assert on the `Response` object

## Tasks

### 1. Mock next/navigation
```ts
// src/__mocks__/next/navigation.ts — Jest auto-mocking
jest.mock("next/navigation", () => ({
  useRouter: () => ({
    push: jest.fn(),
    replace: jest.fn(),
    back: jest.fn(),
    forward: jest.fn(),
    prefetch: jest.fn(),
    refresh: jest.fn(),
  }),
  usePathname: () => "/current-path",
  useSearchParams: () => new URLSearchParams("page=1&filter=all"),
  redirect: jest.fn(),
  notFound: jest.fn(),
}));
```

Or per-test override:
```ts
// In a test file
const mockPush = jest.fn();
jest.mock("next/navigation", () => ({
  useRouter: () => ({ push: mockPush, replace: jest.fn() }),
  usePathname: () => "/",
  useSearchParams: () => new URLSearchParams(),
}));
```

### 2. Test a component that uses useRouter
```tsx
// src/components/BackButton.tsx
"use client";

import { useRouter } from "next/navigation";

export function BackButton({ label = "Go back" }: { label?: string }) {
  const router = useRouter();
  return (
    <button onClick={() => router.back()}>{label}</button>
  );
}
```
```tsx
// src/components/__tests__/BackButton.test.tsx
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";

const mockBack = jest.fn();

jest.mock("next/navigation", () => ({
  useRouter: () => ({ back: mockBack, push: jest.fn(), replace: jest.fn() }),
  usePathname: () => "/",
  useSearchParams: () => new URLSearchParams(),
}));

import { BackButton } from "../BackButton";

describe("BackButton", () => {
  beforeEach(() => mockBack.mockClear());

  it("calls router.back when clicked", async () => {
    const user = userEvent.setup();
    render(<BackButton />);

    await user.click(screen.getByRole("button", { name: "Go back" }));
    expect(mockBack).toHaveBeenCalledTimes(1);
  });

  it("renders custom label", () => {
    render(<BackButton label="Return to list" />);
    expect(screen.getByRole("button", { name: "Return to list" })).toBeInTheDocument();
  });
});
```

### 3. Test a component that uses useSearchParams
```tsx
// src/components/SearchFilters.tsx
"use client";

import { useSearchParams, useRouter, usePathname } from "next/navigation";

export function SearchFilters() {
  const searchParams = useSearchParams();
  const router = useRouter();
  const pathname = usePathname();
  const active = searchParams.get("filter") ?? "all";

  function setFilter(filter: string) {
    const params = new URLSearchParams(searchParams.toString());
    params.set("filter", filter);
    router.push(`${pathname}?${params.toString()}`);
  }

  return (
    <div>
      {["all", "active", "done"].map((f) => (
        <button
          key={f}
          onClick={() => setFilter(f)}
          aria-pressed={active === f}
          className={active === f ? "font-bold" : ""}
        >
          {f}
        </button>
      ))}
    </div>
  );
}
```
```tsx
// src/components/__tests__/SearchFilters.test.tsx
import { render, screen } from "@testing-library/react";
import userEvent from "@testing-library/user-event";

const mockPush = jest.fn();

jest.mock("next/navigation", () => ({
  useSearchParams: () => new URLSearchParams("filter=active"),
  useRouter: () => ({ push: mockPush }),
  usePathname: () => "/todos",
}));

import { SearchFilters } from "../SearchFilters";

describe("SearchFilters", () => {
  beforeEach(() => mockPush.mockClear());

  it("marks the current filter as pressed", () => {
    render(<SearchFilters />);
    expect(screen.getByRole("button", { name: "active" })).toHaveAttribute("aria-pressed", "true");
    expect(screen.getByRole("button", { name: "all" })).toHaveAttribute("aria-pressed", "false");
  });

  it("calls router.push with updated filter", async () => {
    const user = userEvent.setup();
    render(<SearchFilters />);

    await user.click(screen.getByRole("button", { name: "done" }));
    expect(mockPush).toHaveBeenCalledWith("/todos?filter=done");
  });
});
```

### 4. Test a Server Action
```ts
// src/app/todos/actions.ts
"use server";

import { revalidatePath } from "next/cache";

type State = { error?: string; success?: boolean };

// Simple in-memory store for testing
const todos: { id: number; text: string }[] = [];

export async function createTodo(prevState: State, formData: FormData): Promise<State> {
  const text = formData.get("text") as string;
  if (!text?.trim()) return { error: "Text is required" };
  if (text.length > 100) return { error: "Too long" };
  todos.push({ id: Date.now(), text });
  revalidatePath("/todos");
  return { success: true };
}
```
```ts
// src/app/todos/__tests__/actions.test.ts
// Mock next/cache since it requires Next.js runtime
jest.mock("next/cache", () => ({
  revalidatePath: jest.fn(),
}));

import { createTodo } from "../actions";
import { revalidatePath } from "next/cache";

describe("createTodo", () => {
  it("returns error for empty text", async () => {
    const formData = new FormData();
    formData.set("text", "");
    const result = await createTodo({}, formData);
    expect(result.error).toBe("Text is required");
  });

  it("returns error for text over 100 chars", async () => {
    const formData = new FormData();
    formData.set("text", "a".repeat(101));
    const result = await createTodo({}, formData);
    expect(result.error).toBe("Too long");
  });

  it("creates todo and revalidates path on success", async () => {
    const formData = new FormData();
    formData.set("text", "Buy groceries");
    const result = await createTodo({}, formData);
    expect(result.success).toBe(true);
    expect(revalidatePath).toHaveBeenCalledWith("/todos");
  });
});
```

### 5. Test a Route Handler
```ts
// src/app/api/todos/route.ts
import { NextRequest, NextResponse } from "next/server";

const store: { id: number; text: string }[] = [];

export async function GET() {
  return NextResponse.json(store);
}

export async function POST(request: NextRequest) {
  const body = await request.json();
  if (!body.text?.trim()) {
    return NextResponse.json({ error: "Text required" }, { status: 400 });
  }
  const todo = { id: Date.now(), text: body.text };
  store.push(todo);
  return NextResponse.json(todo, { status: 201 });
}
```
```ts
// src/app/api/todos/__tests__/route.test.ts
import { GET, POST } from "../route";
import { NextRequest } from "next/server";

describe("GET /api/todos", () => {
  it("returns empty array initially", async () => {
    const response = await GET();
    const data = await response.json();
    expect(Array.isArray(data)).toBe(true);
  });
});

describe("POST /api/todos", () => {
  it("creates a todo and returns 201", async () => {
    const request = new NextRequest("http://localhost/api/todos", {
      method: "POST",
      body: JSON.stringify({ text: "Buy milk" }),
      headers: { "Content-Type": "application/json" },
    });

    const response = await POST(request);
    expect(response.status).toBe(201);

    const data = await response.json();
    expect(data.text).toBe("Buy milk");
    expect(data.id).toBeDefined();
  });

  it("returns 400 for missing text", async () => {
    const request = new NextRequest("http://localhost/api/todos", {
      method: "POST",
      body: JSON.stringify({ text: "" }),
      headers: { "Content-Type": "application/json" },
    });

    const response = await POST(request);
    expect(response.status).toBe(400);

    const data = await response.json();
    expect(data.error).toBe("Text required");
  });
});
```
