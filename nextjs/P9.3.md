# P9.3 — Full-Stack SaaS Template

**Phase:** 9 — Advanced Features

## Goal
Build a complete production-ready SaaS starter template that combines all skills from the curriculum: Auth.js v5 authentication, Stripe subscription billing, a multi-section dashboard, user settings with dark mode, and full internationalization with next-intl.

## What to Build
- Authentication (Auth.js v5 — Google + Credentials)
- Stripe subscriptions: Free, Pro, Enterprise plans
- Stripe webhook handling (subscription created/updated/deleted)
- Protected dashboard shell with sidebar navigation
- User settings page (profile, billing, preferences)
- Dark mode toggle (next-themes)
- i18n with English + Spanish (next-intl)
- Middleware that guards auth + admin routes with role checks

```ts
export const PLANS = {

      annual: process.env.STRIPE_PRO_ANNUAL_PRICE_ID!,
    },
    features: ["Unlimited projects", "100 GB storage", "Priority support", "Analytics"],
    limits: { projects: Infinity, storage: 100 },
  },
  enterprise: {
    id: "enterprise",
    name: "Enterprise",
    description: "For large organizations.",
    price: { monthly: 99, annual: 79 },
    priceId: {
      monthly: process.env.STRIPE_ENTERPRISE_MONTHLY_PRICE_ID!,
      annual: process.env.STRIPE_ENTERPRISE_ANNUAL_PRICE_ID!,
    },
    features: ["Everything in Pro", "1 TB storage", "Dedicated support", "SSO / SAML", "SLA"],
    limits: { projects: Infinity, storage: 1000 },
  },
} as const;

export type PlanId = keyof typeof PLANS;

export function getUserPlan(stripePriceId?: string | null): PlanId {
  if (!stripePriceId) return "free";
  for (const [planId, plan] of Object.entries(PLANS)) {
    if (
      plan.priceId &&
      (plan.priceId.monthly === stripePriceId || plan.priceId.annual === stripePriceId)
    ) {
      return planId as PlanId;
    }
  }
  return "free";
}

export function isPro(stripePriceId?: string | null, periodEnd?: Date | null): boolean {
  const plan = getUserPlan(stripePriceId);
  const active = !periodEnd || periodEnd > new Date();
  return (plan === "pro" || plan === "enterprise") && active;
}
```

### Stripe Billing

```ts
// lib/stripe.ts
import Stripe from "stripe";

export const stripe = new Stripe(process.env.STRIPE_SECRET_KEY!, {
  apiVersion: "2024-04-10",
  typescript: true,
});

export async function getOrCreateStripeCustomer(userId: string, email: string): Promise<string> {
  const { prisma } = await import("./db");
  const user = await prisma.user.findUnique({ where: { id: userId } });

  if (user?.stripeCustomerId) return user.stripeCustomerId;

  const customer = await stripe.customers.create({
    email,
    metadata: { userId },
  });

  await prisma.user.update({
    where: { id: userId },
    data: { stripeCustomerId: customer.id },
  });

  return customer.id;
}
```

```ts
// app/api/create-checkout/route.ts
import { NextRequest, NextResponse } from "next/server";
import { auth } from "@/auth";
import { stripe, getOrCreateStripeCustomer } from "@/lib/stripe";

export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session?.user?.id) {
    return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
  }

  const { priceId, locale } = await req.json();

  const customerId = await getOrCreateStripeCustomer(
    session.user.id,
    session.user.email!
  );

  const siteUrl = process.env.NEXT_PUBLIC_SITE_URL!;

  const checkout = await stripe.checkout.sessions.create({
    customer: customerId,
    mode: "subscription",
    payment_method_types: ["card"],
    line_items: [{ price: priceId, quantity: 1 }],
    success_url: `${siteUrl}/${locale}/settings/billing?success=1`,
    cancel_url: `${siteUrl}/${locale}/pricing`,
    metadata: { userId: session.user.id },
    allow_promotion_codes: true,
    subscription_data: {
      metadata: { userId: session.user.id },
    },
  });

  return NextResponse.json({ url: checkout.url });
}
```

```ts
    where: { id: session.user.id },

    customer: user.stripeCustomerId,
    return_url: `${process.env.NEXT_PUBLIC_SITE_URL}/en/settings/billing`,
  });

  return NextResponse.json({ url: portal.url });
}
```

```ts
// app/api/webhooks/stripe/route.ts
import { NextRequest, NextResponse } from "next/server";
import { stripe } from "@/lib/stripe";
import { prisma } from "@/lib/db";
import Stripe from "stripe";

export async function POST(req: NextRequest) {
  const body = await req.text();
  const signature = req.headers.get("stripe-signature")!;

  let event: Stripe.Event;

  try {
    event = stripe.webhooks.constructEvent(
      body,
      signature,
      process.env.STRIPE_WEBHOOK_SECRET!
    );
  } catch (err) {
    return NextResponse.json({ error: "Invalid signature" }, { status: 400 });
  }

  const sub = event.data.object as Stripe.Subscription;

  switch (event.type) {
    case "customer.subscription.created":
    case "customer.subscription.updated": {
      const userId = sub.metadata?.userId;
      if (!userId) break;

      await prisma.user.update({
        where: { id: userId },
        data: {
          stripeSubscriptionId: sub.id,
          stripePriceId: sub.items.data[0].price.id,
          stripeCurrentPeriodEnd: new Date(sub.current_period_end * 1000),
        },
      });
      break;
    }

    case "customer.subscription.deleted": {
      const userId = sub.metadata?.userId;
      if (!userId) break;

      await prisma.user.update({
        where: { id: userId },
        data: {
          stripeSubscriptionId: null,
          stripePriceId: null,
          stripeCurrentPeriodEnd: null,
        },
      });
      break;
    }
  }

  return NextResponse.json({ received: true });
}
```

```tsx
// components/BillingStatus.tsx
"use client";

import { useState } from "react";
import { PLANS, type PlanId } from "@/lib/plans";

interface Props {
  plan: PlanId;
  active: boolean;
  periodEnd: Date | null | undefined;
  hasCustomer: boolean;
  locale: string;
}

export function BillingStatus({ plan, active, periodEnd, hasCustomer, locale }: Props) {
  const [loading, setLoading] = useState(false);

  async function openPortal() {
    setLoading(true);
    const res = await fetch("/api/create-portal", { method: "POST" });
    const data = await res.json();
    if (data.url) window.location.href = data.url;
    setLoading(false);
  }

  async function subscribe(priceId: string) {
    setLoading(true);
    const res = await fetch("/api/create-checkout", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ priceId, locale }),
    });
    const data = await res.json();
    if (data.url) window.location.href = data.url;
    setLoading(false);
  }

  const currentPlan = PLANS[plan];

  return (
    <div className="border rounded-2xl p-6 bg-white dark:bg-gray-900 space-y-6">
      {/* Current plan */}
      <div>
        <div className="flex items-center justify-between">
          <div>
            <h2 className="font-semibold text-lg">{currentPlan.name}</h2>
            <p className="text-sm text-gray-500">{currentPlan.description}</p>
          </div>
          <span className={`px-3 py-1 rounded-full text-xs font-bold ${active ? "bg-green-100 text-green-700" : "bg-gray-100 text-gray-600"}`}>
            {active ? "Active" : "Free"}
          </span>
        </div>
        {active && periodEnd && (
          <p className="text-xs text-gray-400 mt-2">
            Renews on {new Date(periodEnd).toLocaleDateString("en-US", { month: "long", day: "numeric", year: "numeric" })}
          </p>
        )}
      </div>

      {/* Actions */}
      <div className="flex gap-3">
        {hasCustomer && (
          <button
            onClick={openPortal}
            disabled={loading}
            className="px-4 py-2 border rounded-xl text-sm font-medium hover:bg-gray-50 disabled:opacity-50 transition-colors"
          >
            {loading ? "Loading..." : "Manage billing"}
          </button>
        )}
        {plan === "free" && (
          <button
            onClick={() => subscribe(PLANS.pro.priceId.monthly)}
            disabled={loading}
            className="px-4 py-2 bg-blue-600 text-white rounded-xl text-sm font-medium hover:bg-blue-700 disabled:opacity-50 transition-colors"
          >
            Upgrade to Pro
          </button>
        )}
      </div>

      {/* Plan features */}
      <ul className="space-y-1.5 pt-2 border-t">
        {currentPlan.features.map((f) => (
          <li key={f} className="flex items-center gap-2 text-sm text-gray-600 dark:text-gray-400">
            <svg className="w-4 h-4 text-green-500 shrink-0" fill="none" viewBox="0 0 24 24" stroke="currentColor">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2.5} d="M5 13l4 4L19 7" />
            </svg>
            {f}
          </li>
        ))}
      </ul>
    </div>
  );
}
```

### Middleware — Auth + Roles + Locale

```ts
// src/middleware.ts
import { auth } from "@/auth";
import createIntlMiddleware from "next-intl/middleware";
import { NextResponse } from "next/server";
import { LOCALES, DEFAULT_LOCALE } from "./i18n";

const intlMiddleware = createIntlMiddleware({
  locales: LOCALES,
  defaultLocale: DEFAULT_LOCALE,
  localePrefix: "always",
});

const APP_PATHS = ["/dashboard", "/settings", "/admin"];

export default auth((req) => {
  const { pathname } = req.nextUrl;
  const session = req.auth;
  const locale = LOCALES.find((l) => pathname.startsWith(`/${l}/`) || pathname === `/${l}`) ?? DEFAULT_LOCALE;

  // Check if it's a protected app path
  const isProtected = APP_PATHS.some((p) => pathname.startsWith(`/${locale}${p}`));

  if (isProtected && !session) {
    return NextResponse.redirect(new URL(`/${locale}/sign-in?callbackUrl=${pathname}`, req.url));
  }

  if (pathname.startsWith(`/${locale}/admin`) && session?.user?.role !== "admin") {
    return NextResponse.redirect(new URL(`/${locale}/access-denied`, req.url));
  }

  return intlMiddleware(req);
});

export const config = {
  matcher: ["/((?!api|_next|_vercel|.*\\..*).*)", "/"],
};
```
