# P2.2 — Dashboard with Server Actions

**Phase:** 2 — Data Fetching & Rendering  
**Topic:** Form → Server Action → DB write, optimistic UI, revalidatePath  
**Type:** Project

---

## Goal
Build a task management dashboard where creating, updating, and deleting tasks go through Server Actions, with optimistic UI using `useOptimistic`, and proper cache invalidation via `revalidatePath`.

## What to Build
- Task list fetched server-side (Server Component)
- "Add task" form wired to a Server Action
- Optimistic task addition (task appears instantly)
- Mark task as done (toggle, Server Action)
- Delete task (Server Action + confirm)
- `revalidatePath` to keep the cache fresh after mutations

## Architecture

```
app/
  dashboard/
    page.tsx          ← Server Component: fetches tasks, renders TaskList
    actions.ts        ← Server Actions: createTask, toggleTask, deleteTask
  layout.tsx
components/
  TaskList.tsx        ← Client Component with useOptimistic
  AddTaskForm.tsx     ← Client Component wrapping the form
lib/
  db.ts               ← simulated DB (Map) or Prisma client
  types.ts
```

```ts
// lib/db.ts
// Simulated in-memory DB — swap for Prisma in a real project
import { Task } from "./types";

const tasks = new Map<string, Task>([
  ["1", { id: "1", title: "Set up Next.js project", completed: true, createdAt: new Date("2024-01-01") }],
  ["2", { id: "2", title: "Build the dashboard", completed: false, createdAt: new Date("2024-01-02") }],
]);

export async function getTasks(): Promise<Task[]> {
  return Array.from(tasks.values()).sort(
    (a, b) => b.createdAt.getTime() - a.createdAt.getTime()
  );
}

export async function createTaskInDB(title: string): Promise<Task> {
  const task: Task = {
    id: crypto.randomUUID(),
    title,
    completed: false,
    createdAt: new Date(),
  };
  tasks.set(task.id, task);
  return task;
}

export async function toggleTaskInDB(id: string): Promise<Task | null> {
  const task = tasks.get(id);
  if (!task) return null;
  task.completed = !task.completed;
  return task;
}

export async function deleteTaskInDB(id: string): Promise<void> {
  tasks.delete(id);
}
```

```ts
// app/dashboard/actions.ts
"use server";

import { revalidatePath } from "next/cache";
import { createTaskInDB, toggleTaskInDB, deleteTaskInDB } from "@/lib/db";
import { z } from "zod";

const CreateTaskSchema = z.object({
  title: z.string().min(1, "Title is required").max(200),
});

export async function createTask(
  prevState: { error?: string },
  formData: FormData
) {
  const parsed = CreateTaskSchema.safeParse({ title: formData.get("title") });

  if (!parsed.success) {
    return { error: parsed.error.issues[0].message };
  }

  await createTaskInDB(parsed.data.title);
  revalidatePath("/dashboard");
  return {};
}

export async function toggleTask(id: string) {
  await toggleTaskInDB(id);
  revalidatePath("/dashboard");
}

export async function deleteTask(id: string) {
  await deleteTaskInDB(id);
  revalidatePath("/dashboard");
}
```

```tsx
import { useFormState, useFormStatus } from "react-dom";
```

```tsx
// components/TaskList.tsx
"use client";

import { useOptimistic, useTransition } from "react";
import { toggleTask, deleteTask } from "@/app/dashboard/actions";
import type { Task } from "@/lib/types";

export function TaskList({ tasks }: { tasks: Task[] }) {
  const [optimisticTasks, updateOptimistic] = useOptimistic(
    tasks,
    (state, action: { type: "toggle"; id: string } | { type: "delete"; id: string }) => {
      if (action.type === "toggle") {
        return state.map((t) =>
          t.id === action.id ? { ...t, completed: !t.completed } : t
        );
      }
      if (action.type === "delete") {
        return state.filter((t) => t.id !== action.id);
      }
      return state;
    }
  );

  const [, startTransition] = useTransition();

  function handleToggle(id: string) {
    startTransition(async () => {
      updateOptimistic({ type: "toggle", id });
      await toggleTask(id);
    });
  }

  function handleDelete(id: string) {
    startTransition(async () => {
      updateOptimistic({ type: "delete", id });
      await deleteTask(id);
    });
  }

  if (!optimisticTasks.length) {
    return (
      <p className="text-center py-12 text-gray-400">
        No tasks yet. Add one above!
      </p>
    );
  }

  return (
    <ul className="space-y-2">
      {optimisticTasks.map((task) => (
        <li
          key={task.id}
          className="flex items-center gap-3 p-4 bg-white border rounded-xl group hover:border-gray-300 transition-colors"
        >
          {/* Checkbox */}
          <button
            onClick={() => handleToggle(task.id)}
            className={`w-5 h-5 rounded border-2 flex items-center justify-center shrink-0 transition-colors ${
              task.completed
                ? "bg-blue-600 border-blue-600"
                : "border-gray-300 hover:border-blue-400"
            }`}
            aria-label={task.completed ? "Mark incomplete" : "Mark complete"}
          >
            {task.completed && (
              <svg className="w-3 h-3 text-white" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={3} d="M5 13l4 4L19 7" />
              </svg>
            )}
          </button>

          {/* Title */}
          <span className={`flex-1 text-sm ${task.completed ? "line-through text-gray-400" : "text-gray-900"}`}>
            {task.title}
          </span>

          {/* Date */}
          <span className="text-xs text-gray-300 hidden group-hover:block">
            {new Date(task.createdAt).toLocaleDateString()}
          </span>

          {/* Delete */}
          <button
            onClick={() => handleDelete(task.id)}
            className="opacity-0 group-hover:opacity-100 p-1 text-gray-400 hover:text-red-500 transition-all"
            aria-label="Delete task"
          >
            <svg className="w-4 h-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
              <path strokeLinecap="round" strokeLinejoin="round" strokeWidth={2} d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
            </svg>
          </button>
        </li>
      ))}
    </ul>
  );
}
```

## Deliverables
- [ ] Dashboard page renders tasks server-side (no loading spinner on initial load)
- [ ] Adding a task via the form calls the Server Action and shows it immediately (optimistic)
- [ ] If the Server Action fails, the optimistic update rolls back
- [ ] Form shows Zod validation error when title is empty
- [ ] `useFormStatus` disables the submit button while the action is in flight
- [ ] Toggling a task immediately updates the checkbox (optimistic) and syncs with the server
- [ ] Deleting a task immediately removes it from the list (optimistic)
- [ ] `revalidatePath("/dashboard")` keeps the Next.js cache in sync
- [ ] No `useState` for the task list — all state comes from the server + `useOptimistic`
