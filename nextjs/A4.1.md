## Goal
Manage server state on the client with TanStack Query — fetching, caching, mutations, and optimistic updates.

## Background

TanStack Query (formerly React Query) solves the "server state" problem on the client. Server state is data that lives on the server and must be fetched, cached, and kept in sync with the backend. It's different from UI state (which lives locally) and requires different tools.

Without TanStack Query, you'd manage loading/error/data states manually with `useEffect` and `useState`, deal with race conditions, implement your own caching, and build refetch-on-focus logic from scratch. TanStack Query handles all of this automatically.

The **queryKey** is the identity of a query. When the key changes, React Query re-fetches. When you call `invalidateQueries`, all queries matching the key are marked stale and refetched. This is how you keep data fresh after mutations.

**useMutation** handles data-modifying operations. It provides `mutate`, `isPending`, `isSuccess`, `isError`, and callbacks like `onSuccess`, `onError`, and `onMutate` (for optimistic updates).

### 1. Install and configure
```bash
pnpm add @tanstack/react-query @tanstack/react-query-devtools
```
```tsx
// src/providers/QueryProvider.tsx
"use client";

import { QueryClient, QueryClientProvider } from "@tanstack/react-query";
import { ReactQueryDevtools } from "@tanstack/react-query-devtools";
import { useState } from "react";

export function QueryProvider({ children }: { children: React.ReactNode }) {
  // useState ensures each browser tab gets its own QueryClient
  const [queryClient] = useState(
    () =>
      new QueryClient({
        defaultOptions: {
          queries: {
            staleTime: 60 * 1000, // data stays fresh for 1 minute
            gcTime: 5 * 60 * 1000, // garbage collect after 5 minutes
            retry: 1,
            refetchOnWindowFocus: true,
          },
        },
      })
  );

  return (
    <QueryClientProvider client={queryClient}>
      {children}
      <ReactQueryDevtools initialIsOpen={false} />
    </QueryClientProvider>
  );
}
```

### 2. useQuery — fetch a list
```tsx
// src/hooks/usePosts.ts
import { useQuery } from "@tanstack/react-query";

export type Post = { id: number; title: string; body: string; userId: number };

async function fetchPosts(): Promise<Post[]> {
  const res = await fetch("https://jsonplaceholder.typicode.com/posts?_limit=10");
  if (!res.ok) throw new Error("Failed to fetch posts");
  return res.json();
}

export function usePosts() {
  return useQuery({
    queryKey: ["posts"],
    queryFn: fetchPosts,
  });
}
```
```tsx
// src/app/rq-demo/page.tsx
"use client";

import { usePosts } from "@/hooks/usePosts";
```

### 3. useQuery with dynamic key (dependent query)
```ts
// src/hooks/usePost.ts
import { useQuery } from "@tanstack/react-query";
import type { Post } from "./usePosts";

async function fetchPost(id: number): Promise<Post> {
  const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`);
  if (!res.ok) throw new Error("Post not found");
  return res.json();
}

export function usePost(id: number) {
  return useQuery({
    queryKey: ["posts", id],    // key changes when id changes → refetch
    queryFn: () => fetchPost(id),
    enabled: id > 0,            // don't fetch if id is invalid
  });
}

// src/hooks/useComments.ts
export function useComments(postId: number) {
  return useQuery({
    queryKey: ["posts", postId, "comments"],
    queryFn: async () => {
      const res = await fetch(
        `https://jsonplaceholder.typicode.com/posts/${postId}/comments`
      );
      return res.json();
    },
    enabled: postId > 0,
  });
}
```

### 4. useMutation — create and delete
```ts
// src/hooks/usePostMutations.ts
import { useMutation, useQueryClient } from "@tanstack/react-query";
import type { Post } from "./usePosts";

export function useCreatePost() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (newPost: Omit<Post, "id">) => {
      const res = await fetch("https://jsonplaceholder.typicode.com/posts", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify(newPost),
      });
      if (!res.ok) throw new Error("Failed to create post");
      return res.json() as Promise<Post>;
    },
    onSuccess: () => {
      // Invalidate the posts list so it refetches
      queryClient.invalidateQueries({ queryKey: ["posts"] });
    },
  });
}

export function useDeletePost() {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (id: number) => {
      const res = await fetch(`https://jsonplaceholder.typicode.com/posts/${id}`, {
        method: "DELETE",
      });
      if (!res.ok) throw new Error("Failed to delete");
    },
    onSuccess: (_, id) => {
      // Remove this specific post from the cache immediately
      queryClient.removeQueries({ queryKey: ["posts", id] });
      queryClient.invalidateQueries({ queryKey: ["posts"] });
    },
  });
}
```

### 5. Optimistic updates with useMutation
```ts
// src/hooks/useToggleLike.ts
import { useMutation, useQueryClient } from "@tanstack/react-query";

type Post = { id: number; title: string; liked: boolean };

export function useToggleLike(postId: number) {
  const queryClient = useQueryClient();

  return useMutation({
    mutationFn: async (liked: boolean) => {
      // Simulated API call
      await new Promise((r) => setTimeout(r, 800));
      return { liked };
    },

    // Called BEFORE the API call — update cache immediately
    onMutate: async (newLiked: boolean) => {
      await queryClient.cancelQueries({ queryKey: ["posts", postId] });

      const previous = queryClient.getQueryData<Post>(["posts", postId]);

      // Optimistically update
      queryClient.setQueryData<Post>(["posts", postId], (old) =>
        old ? { ...old, liked: newLiked } : old
      );

      return { previous }; // context passed to onError
    },

    // Roll back on error
    onError: (_err, _vars, context) => {
      if (context?.previous) {
        queryClient.setQueryData(["posts", postId], context.previous);
      }
    },

    // Always sync with server after success or error
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ["posts", postId] });
    },
  });
}
```

### 6. Full CRUD component
```tsx
// src/app/rq-crud/page.tsx
"use client";

import { useState } from "react";
import { usePosts } from "@/hooks/usePosts";
import { useCreatePost, useDeletePost } from "@/hooks/usePostMutations";

export default function RqCrudPage() {
  const { data: posts, isLoading } = usePosts();
  const createPost = useCreatePost();
  const deletePost = useDeletePost();
  const [title, setTitle] = useState("");

  function handleCreate() {
    if (!title.trim()) return;
    createPost.mutate(
      { title, body: "New post body", userId: 1 },
      { onSuccess: () => setTitle("") }
    );
  }

  return (
    <div className="max-w-2xl space-y-4">
      <div className="flex gap-2">
        <input
          value={title}
          onChange={(e) => setTitle(e.target.value)}
          placeholder="New post title"
          className="flex-1 px-3 py-2 border rounded"
        />
        <button
          onClick={handleCreate}
          disabled={createPost.isPending}
          className="px-4 py-2 bg-blue-600 text-white rounded disabled:opacity-50"
        >
          {createPost.isPending ? "Creating…" : "Create"}
        </button>
      </div>

      {isLoading ? (
        <p>Loading…</p>
      ) : (
        <ul className="space-y-2">
          {posts?.map((post) => (
            <li key={post.id} className="flex items-center justify-between p-3 border rounded">
              <span className="text-sm">{post.title.slice(0, 50)}</span>
              <button
                onClick={() => deletePost.mutate(post.id)}
                disabled={deletePost.isPending}
                className="text-red-500 text-sm hover:underline disabled:opacity-50"
              >
                Delete
              </button>
            </li>
          ))}
        </ul>
      )}
    </div>
  );
}
```
