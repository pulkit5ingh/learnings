**Phase:** 5 — Authentication

**Type:** Assignment

## Goal
Implement a complete authentication flow: signup, email verification placeholder, login, logout, and account linking.

## Background

A production auth flow involves more than just NextAuth setup. Users need to sign up, optionally verify their email, log in, and manage their accounts. OAuth complicates this: a user might first sign up with Google, then try to sign in with their email/password and hit an "account already exists" error.

Auth.js v5 has a built-in error page system. When authentication fails, it redirects to your custom error page with an error code as a query param. The most common error codes: `OAuthAccountNotLinked` (trying to sign in with a different provider for an existing email), `CredentialsSignin` (wrong email/password), and `Callback` (OAuth callback error).

**Account linking** allows a single user to connect multiple OAuth providers to one account. Auth.js handles this automatically when using a database adapter — it matches accounts by email. Without a database adapter, you must handle linking manually.

### 1. Signup page with Credentials provider
```tsx
// src/app/register/page.tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import Link from "next/link";

export default function RegisterPage() {
  const router = useRouter();
  const [error, setError] = useState("");
  const [loading, setLoading] = useState(false);

  async function handleSubmit(e: React.FormEvent<HTMLFormElement>) {
    e.preventDefault();
    setError("");
    setLoading(true);

    const formData = new FormData(e.currentTarget);
    const name = formData.get("name") as string;
    const email = formData.get("email") as string;
    const password = formData.get("password") as string;

    try {
      const res = await fetch("/api/auth/register", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ name, email, password }),
      });

      if (!res.ok) {
        const data = await res.json();
        setError(data.error ?? "Registration failed");
        return;
      }

      // Auto sign-in after registration
      router.push("/login?message=registered");
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="min-h-screen flex items-center justify-center">
      <div className="w-full max-w-sm space-y-4 p-6 border rounded-xl">
        <h1 className="text-2xl font-bold text-center">Create account</h1>

        {error && (
          <p className="p-3 bg-red-50 text-red-700 rounded text-sm">{error}</p>
        )}

        <form onSubmit={handleSubmit} className="space-y-3">
          <input name="name" placeholder="Full name" required
            className="w-full px-3 py-2 border rounded" />
          <input name="email" type="email" placeholder="Email" required
            className="w-full px-3 py-2 border rounded" />
          <input name="password" type="password" placeholder="Password (8+ chars)" required minLength={8}
            className="w-full px-3 py-2 border rounded" />
          <button type="submit" disabled={loading}
            className="w-full py-2.5 bg-blue-600 text-white rounded disabled:opacity-50">
            {loading ? "Creating account…" : "Create account"}
          </button>
        </form>

        <p className="text-sm text-center text-gray-500">
          Already have an account?{" "}
          <Link href="/login" className="text-blue-600 hover:underline">Sign in</Link>
        </p>
      </div>
    </div>
  );
}
```

### 2. Registration API Route Handler
```ts
// src/app/api/auth/register/route.ts
import { NextRequest, NextResponse } from "next/server";
import { z } from "zod";
import bcrypt from "bcryptjs";

// pnpm add bcryptjs @types/bcryptjs

const registerSchema = z.object({
  name: z.string().min(2),
  email: z.string().email(),
  password: z.string().min(8),
});

export async function POST(request: NextRequest) {
  const body = await request.json();
  const parsed = registerSchema.safeParse(body);

  if (!parsed.success) {
    return NextResponse.json({ error: parsed.error.issues[0].message }, { status: 400 });
  }

  const { name, email, password } = parsed.data;

  // Check if user exists (replace with real DB check)
  const existingUsers = ["alice@example.com", "bob@example.com"];
  if (existingUsers.includes(email)) {
    return NextResponse.json({ error: "Email already registered" }, { status: 409 });
  }

  const hashedPassword = await bcrypt.hash(password, 12);

  // In real app: await prisma.user.create({ data: { name, email, password: hashedPassword } })
  console.log("Would create user:", { name, email, hashedPassword });

  return NextResponse.json({ success: true }, { status: 201 });
}
```

### 3. Custom error page
```tsx
// src/app/auth/error/page.tsx
import Link from "next/link";

const ERROR_MESSAGES: Record<string, string> = {
  OAuthAccountNotLinked:
    "An account with this email already exists using a different sign-in method. Please sign in with the original method.",
  CredentialsSignin:
    "Invalid email or password. Please try again.",
  OAuthSignin:
    "Could not sign in with this OAuth provider. Please try again.",
  OAuthCallback:
    "OAuth callback error. Please try again.",
  EmailSignin:
    "Could not send verification email. Please try again.",
  SessionRequired:
    "You must be signed in to access this page.",
  Default:
    "An unexpected error occurred. Please try again.",
};

export default function AuthErrorPage({
  searchParams,
}: {
  searchParams: { error?: string };
}) {
  const error = searchParams.error ?? "Default";
  const message = ERROR_MESSAGES[error] ?? ERROR_MESSAGES.Default;

  return (
    <div className="min-h-screen flex items-center justify-center">
      <div className="max-w-sm w-full p-6 text-center space-y-4">
        <div className="w-16 h-16 bg-red-100 rounded-full flex items-center justify-center mx-auto">
          <span className="text-red-600 text-2xl">!</span>
        </div>
        <h1 className="text-xl font-bold">Authentication Error</h1>
        <p className="text-gray-600 text-sm">{message}</p>
        <p className="text-xs text-gray-400">Error code: {error}</p>
        <div className="flex gap-2 justify-center">
          <Link href="/login"
            className="px-4 py-2 bg-blue-600 text-white rounded text-sm hover:bg-blue-700">
            Try again
          </Link>
          <Link href="/"
            className="px-4 py-2 border rounded text-sm hover:bg-gray-50">
            Go home
          </Link>
        </div>
      </div>
    </div>
  );
}
```

### 6. Password hashing utility
```ts
// src/lib/password.ts
import bcrypt from "bcryptjs";

export async function hashPassword(password: string): Promise<string> {
  return bcrypt.hash(password, 12);
}

export async function verifyPassword(
  password: string,
  hashedPassword: string
): Promise<boolean> {
  return bcrypt.compare(password, hashedPassword);
}
```
