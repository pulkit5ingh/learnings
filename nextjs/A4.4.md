# A4.4 — URL State

**Phase:** 4 — State Management & Client Data Fetching  
**Topic:** useSearchParams, useRouter, nuqs for type-safe URL state, filter/sort/pagination in URL  
**Type:** Assignment

## Goal
Store filter, sort, and pagination state in the URL so that pages are shareable and browser-navigable.

## Background

URL state is underused in most React apps. When a user filters a product list by "category: electronics, price: 0-500" and sorts by "rating desc" at page 3, that state should live in the URL: `/products?category=electronics&maxPrice=500&sort=rating&order=desc&page=3`. This makes the result shareable, bookmarkable, and correctly handled by the browser's back/forward buttons.

In Next.js App Router, `useSearchParams()` reads the current URL's search params in a Client Component. `useRouter()` lets you push/replace navigation. However, manually parsing and serialising query strings is verbose and error-prone.

**nuqs** (Next.js URL Query State) is a library that provides typed hooks for URL state. `useQueryState("page", parseAsInteger.withDefault(1))` gives you a `page: number` state that reads from and writes to the URL automatically, with type safety and default values.

### 3. Basic useSearchParams (without nuqs)
```tsx
// src/app/search/page.tsx
"use client";

import { useSearchParams, useRouter, usePathname } from "next/navigation";
import { useState, useTransition } from "react";

export default function SearchPage() {
  const searchParams = useSearchParams();
  const router = useRouter();
  const pathname = usePathname();
  const [, startTransition] = useTransition();

  const query = searchParams.get("q") ?? "";

  function updateSearch(value: string) {
    const params = new URLSearchParams(searchParams.toString());
    if (value) {
      params.set("q", value);
    } else {
      params.delete("q");
    }
    startTransition(() => {
      router.replace(`${pathname}?${params.toString()}`);
    });
  }

  return (
    <div className="p-6">
      <input
        defaultValue={query}
        onChange={(e) => updateSearch(e.target.value)}
        placeholder="Search…"
        className="px-3 py-2 border rounded w-64"
      />
      <p className="mt-4 text-sm text-gray-500">
        Current query: <code>{query || "(empty)"}</code>
      </p>
    </div>
  );
}
```

### 4. nuqs — type-safe URL state
```tsx
// src/app/nuqs-demo/page.tsx
"use client";

import { useQueryState, parseAsInteger, parseAsString } from "nuqs";

export default function NuqsDemoPage() {
  const [page, setPage] = useQueryState("page", parseAsInteger.withDefault(1));
  const [search, setSearch] = useQueryState("q", parseAsString.withDefault(""));
  const [tab, setTab] = useQueryState("tab", parseAsString.withDefault("all"));

  return (
    <div className="p-6 space-y-4">
      <h1 className="text-xl font-bold">nuqs Demo</h1>
      <p className="text-sm text-gray-500">Watch the URL update as you change these controls.</p>

      <div className="flex gap-2">
        {["all", "active", "completed"].map((t) => (
          <button key={t} onClick={() => setTab(t)}
            className={`px-3 py-1.5 rounded ${tab === t ? "bg-blue-600 text-white" : "border"}`}>
            {t}
          </button>
        ))}
      </div>

      <input
        value={search}
        onChange={(e) => setSearch(e.target.value || null)}
        placeholder="Search…"
        className="px-3 py-2 border rounded"
      />

      <div className="flex items-center gap-3">
        <button onClick={() => setPage((p) => Math.max(1, p - 1))}
          disabled={page <= 1} className="px-3 py-1.5 border rounded disabled:opacity-50">
          ← Prev
        </button>
        <span>Page {page}</span>
        <button onClick={() => setPage((p) => p + 1)}
          className="px-3 py-1.5 border rounded">
          Next →
        </button>
      </div>
    </div>
  );
}
```

### 5. Full filter/sort/pagination for a product list
```tsx
// src/app/products-filtered/useProductFilters.ts
"use client";

import {
  useQueryState,
  parseAsInteger,
  parseAsString,
  parseAsStringEnum,
  createSerializer,
} from "nuqs";

export const CATEGORIES = ["all", "electronics", "clothing", "books"] as const;
export const SORT_OPTIONS = ["price-asc", "price-desc", "name-asc", "rating"] as const;

export function useProductFilters() {
  const [category, setCategory] = useQueryState(
    "category",
    parseAsStringEnum([...CATEGORIES]).withDefault("all")
  );
  const [sort, setSort] = useQueryState(
    "sort",
    parseAsStringEnum([...SORT_OPTIONS]).withDefault("name-asc")
  );
  const [page, setPage] = useQueryState("page", parseAsInteger.withDefault(1));
  const [search, setSearch] = useQueryState("q", parseAsString.withDefault(""));
  const [minPrice, setMinPrice] = useQueryState("minPrice", parseAsInteger.withDefault(0));
  const [maxPrice, setMaxPrice] = useQueryState("maxPrice", parseAsInteger.withDefault(1000));

  function resetFilters() {
    setCategory("all");
    setSort("name-asc");
    setPage(1);
    setSearch("");
    setMinPrice(0);
    setMaxPrice(1000);
  }

  return {
    category, setCategory,
    sort, setSort,
    page, setPage,
    search, setSearch,
    minPrice, setMinPrice,
    maxPrice, setMaxPrice,
    resetFilters,
  };
}
```
```tsx
// src/app/products-filtered/page.tsx
"use client";

import { useProductFilters, CATEGORIES, SORT_OPTIONS } from "./useProductFilters";

type Product = { id: number; name: string; category: string; price: number; rating: number };

const MOCK_PRODUCTS: Product[] = Array.from({ length: 20 }, (_, i) => ({
  id: i + 1,
  name: `Product ${i + 1}`,
  category: CATEGORIES[1 + (i % 3)],
  price: 10 + i * 15,
  rating: 3 + (i % 3) * 0.5,
}));

const PAGE_SIZE = 6;

export default function ProductsFilteredPage() {
  const { category, setCategory, sort, setSort, page, setPage,
          search, setSearch, minPrice, maxPrice, resetFilters } = useProductFilters();

  const filtered = MOCK_PRODUCTS
    .filter((p) => category === "all" || p.category === category)
    .filter((p) => p.name.toLowerCase().includes(search.toLowerCase()))
    .filter((p) => p.price >= minPrice && p.price <= maxPrice)
    .sort((a, b) => {
      if (sort === "price-asc") return a.price - b.price;
      if (sort === "price-desc") return b.price - a.price;
      if (sort === "rating") return b.rating - a.rating;
      return a.name.localeCompare(b.name);
    });

  const totalPages = Math.ceil(filtered.length / PAGE_SIZE);
  const paginated = filtered.slice((page - 1) * PAGE_SIZE, page * PAGE_SIZE);

  return (
    <div className="p-6 max-w-4xl space-y-4">
      <div className="flex flex-wrap gap-3 items-center">
        <input value={search} onChange={(e) => { setSearch(e.target.value); setPage(1); }}
          placeholder="Search products…" className="px-3 py-2 border rounded" />

        <select value={category} onChange={(e) => { setCategory(e.target.value as any); setPage(1); }}
          className="px-3 py-2 border rounded">
          {CATEGORIES.map((c) => <option key={c} value={c}>{c}</option>)}
        </select>

        <select value={sort} onChange={(e) => setSort(e.target.value as any)}
          className="px-3 py-2 border rounded">
          {SORT_OPTIONS.map((s) => <option key={s} value={s}>{s}</option>)}
        </select>

        <button onClick={resetFilters} className="px-3 py-2 text-sm text-gray-600 hover:underline">
          Reset
        </button>
      </div>

      <p className="text-sm text-gray-500">{filtered.length} products found</p>

      <div className="grid grid-cols-2 sm:grid-cols-3 gap-4">
        {paginated.map((p) => (
          <div key={p.id} className="p-4 border rounded">
            <p className="font-semibold">{p.name}</p>
            <p className="text-sm text-gray-500 capitalize">{p.category}</p>
            <p className="text-green-600 mt-1">${p.price}</p>
            <p className="text-yellow-500 text-sm">★ {p.rating}</p>
          </div>
        ))}
      </div>

      <div className="flex gap-2">
        <button onClick={() => setPage((p) => p - 1)} disabled={page <= 1}
          className="px-3 py-1.5 border rounded disabled:opacity-50">← Prev</button>
        <span className="px-3 py-1.5">Page {page} / {totalPages}</span>
        <button onClick={() => setPage((p) => p + 1)} disabled={page >= totalPages}
          className="px-3 py-1.5 border rounded disabled:opacity-50">Next →</button>
      </div>
    </div>
  );
}
```

- [ ] Search input updates `?q=` in the URL as you type
- [ ] Category and sort dropdowns update URL params
- [ ] Pagination updates `?page=` and resets to 1 when filters change
- [ ] Refreshing the page restores all filter/sort/page state from the URL
- [ ] Sharing the URL with another browser tab shows the same results
- [ ] `resetFilters` clears all params and shows all products at page 1
