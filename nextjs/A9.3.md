## Goal
Add real-time data updates to a Next.js app using both Server-Sent Events (SSE) from a Route Handler and Socket.io from a separate Node server.

## Background

Next.js App Router supports two real-time patterns:

**Server-Sent Events (SSE)** via Route Handlers — unidirectional server→client push over HTTP. Simple, firewall-friendly, built into the browser (`EventSource` API). Great for live feeds, notifications, dashboards.

**WebSockets via Socket.io** — bidirectional. Socket.io runs as a separate Node server (or Express adapter) since Next.js Route Handlers do not support WebSocket upgrades. The Next.js client connects to it.

### When to use which
| | SSE | WebSockets |
|---|---|---|
| Direction | Server → Client | Bidirectional |
| Protocol | HTTP | ws:// |
| Browser API | `EventSource` | `WebSocket` |
| Next.js support | Native (Route Handler) | Separate server |
| Good for | Feeds, dashboards, notifications | Chat, games, collaboration |

## Tasks

### 1. SSE via Route Handler

```ts
// app/api/events/route.ts
import { NextRequest } from "next/server";

export const runtime = "nodejs"; // required — Edge doesn't support long-lived connections well

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    start(controller) {
      // Send an initial event immediately
      controller.enqueue(
        encoder.encode(`data: ${JSON.stringify({ type: "connected", ts: Date.now() })}\n\n`)
      );

      // Simulate live data — send a random number every second
      const interval = setInterval(() => {
        const payload = JSON.stringify({
          type: "metric",
          value: Math.round(Math.random() * 100),
          ts: Date.now(),
        });
        controller.enqueue(encoder.encode(`data: ${payload}\n\n`));
      }, 1000);

      // Cleanup when the client disconnects
      req.signal.addEventListener("abort", () => {
        clearInterval(interval);
        controller.close();
      });
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache, no-transform",
      Connection: "keep-alive",
      "X-Accel-Buffering": "no", // disable nginx buffering
    },
  });
}
```

### 2. EventSource in a Client Component

```tsx
// components/LiveMetric.tsx
"use client";

import { useEffect, useRef, useState } from "react";

interface Metric {
  type: string;
  value: number;
  ts: number;
}

export function LiveMetric() {
  const [metrics, setMetrics] = useState<Metric[]>([]);
  const [connected, setConnected] = useState(false);
  const esRef = useRef<EventSource | null>(null);

  useEffect(() => {
    const es = new EventSource("/api/events");
    esRef.current = es;

    es.onopen = () => setConnected(true);

    es.onmessage = (event) => {
      const data: Metric = JSON.parse(event.data);
      if (data.type === "metric") {
        setMetrics((prev) => [...prev.slice(-19), data]); // keep last 20
      }
    };

    es.onerror = () => {
      setConnected(false);
      // EventSource auto-reconnects — no manual retry needed
    };

    return () => {
      es.close();
      setConnected(false);
    };
  }, []);

  return (
    <div className="p-4 border rounded-xl space-y-2">
      <div className="flex items-center gap-2">
        <span
          className={`w-2 h-2 rounded-full ${connected ? "bg-green-500" : "bg-red-500"}`}
        />
        <span className="text-sm text-gray-500">
          {connected ? "Connected" : "Reconnecting..."}
        </span>
      </div>

      <div className="flex gap-1 items-end h-20">
        {metrics.map((m, i) => (
          <div
            key={m.ts}
            title={`${m.value}`}
            style={{ height: `${m.value}%` }}
            className="flex-1 bg-blue-500 rounded-t transition-all duration-300"
          />
        ))}
      </div>

      <p className="text-xs text-gray-400">
        Last value: {metrics[metrics.length - 1]?.value ?? "—"}
      </p>
    </div>
  );
}
```

### 3. Named SSE events

The SSE spec supports named events via `event:` lines. Use these to multiplex different event types on a single connection.

```ts
// app/api/feed/route.ts — multiple named event types
import { NextRequest } from "next/server";

export const runtime = "nodejs";

function sseEvent(event: string, data: unknown): string {
  return `event: ${event}\ndata: ${JSON.stringify(data)}\n\n`;
}

export async function GET(req: NextRequest) {
  const encoder = new TextEncoder();

  const stream = new ReadableStream({
    start(controller) {
      let tick = 0;

      const interval = setInterval(() => {
        tick++;

        // Every tick: send a metric
        controller.enqueue(
          encoder.encode(sseEvent("metric", { value: Math.random() * 100, tick }))
        );

        // Every 5 ticks: send a notification
        if (tick % 5 === 0) {
          controller.enqueue(
            encoder.encode(sseEvent("notification", { message: `Checkpoint ${tick}` }))
          );
        }
      }, 1000);

      req.signal.addEventListener("abort", () => {
        clearInterval(interval);
        controller.close();
      });
    },
  });

  return new Response(stream, {
    headers: {
      "Content-Type": "text/event-stream",
      "Cache-Control": "no-cache",
      Connection: "keep-alive",
    },
  });
}
```

### 4. Socket.io — separate server setup

```bash
pnpm add socket.io socket.io-client
```

```ts
// server/socket.ts — run with: npx ts-node server/socket.ts
import { createServer } from "http";
import { Server } from "socket.io";

const httpServer = createServer();

const io = new Server(httpServer, {
  cors: {
    origin: "http://localhost:3000",
    methods: ["GET", "POST"],
  },
});

io.on("connection", (socket) => {
  console.log("Client connected:", socket.id);

  // Client joins a room by name
  socket.on("join-room", (room: string) => {
    socket.join(room);
    socket.emit("joined", { room });
  });

  // Broadcast a chat message to the room
  socket.on("message", ({ room, text }: { room: string; text: string }) => {
    io.to(room).emit("message", {
      id: crypto.randomUUID(),
      text,
      sender: socket.id,
      ts: Date.now(),
    });
  });

  socket.on("disconnect", () => {
    console.log("Client disconnected:", socket.id);
  });
});

// Emit server-driven live metrics to all clients
setInterval(() => {
  io.emit("metric", { value: Math.random() * 100, ts: Date.now() });
}, 1000);

httpServer.listen(3001, () => {
  console.log("Socket.io server running on :3001");
});
```

### 5. Socket.io client hook

```ts
// hooks/useSocket.ts
"use client"; // this file is imported only by Client Components

import { useEffect, useRef, useState } from "react";
import { io, Socket } from "socket.io-client";

const SOCKET_URL = process.env.NEXT_PUBLIC_SOCKET_URL ?? "http://localhost:3001";

export function useSocket() {
  const socketRef = useRef<Socket | null>(null);
  const [connected, setConnected] = useState(false);

  useEffect(() => {
    const socket = io(SOCKET_URL, {
      transports: ["websocket"], // skip long-polling
    });
    socketRef.current = socket;

    socket.on("connect", () => setConnected(true));
    socket.on("disconnect", () => setConnected(false));

    return () => {
      socket.disconnect();
    };
  }, []);

  return { socket: socketRef.current, connected };
}
```

### 6. Chat Client Component using Socket.io

```tsx
// components/ChatRoom.tsx
"use client";

import { useEffect, useRef, useState } from "react";
import { useSocket } from "@/hooks/useSocket";

interface Message {
  id: string;
  text: string;
  sender: string;
  ts: number;
}

export function ChatRoom({ room }: { room: string }) {
  const { socket, connected } = useSocket();
  const [messages, setMessages] = useState<Message[]>([]);
  const [input, setInput] = useState("");
  const bottomRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    if (!socket) return;

    socket.emit("join-room", room);

    socket.on("message", (msg: Message) => {
      setMessages((prev) => [...prev, msg]);
    });

    return () => {
      socket.off("message");
    };
  }, [socket, room]);

  // Auto-scroll to bottom
  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages]);

  function sendMessage(e: React.FormEvent) {
    e.preventDefault();
    if (!socket || !input.trim()) return;
    socket.emit("message", { room, text: input.trim() });
    setInput("");
  }

  return (
    <div className="flex flex-col h-[500px] border rounded-xl overflow-hidden">
      {/* Header */}
      <div className="px-4 py-2 border-b flex items-center gap-2 bg-gray-50">
        <span className={`w-2 h-2 rounded-full ${connected ? "bg-green-500" : "bg-gray-400"}`} />
        <span className="font-medium">#{room}</span>
      </div>

      {/* Messages */}
      <div className="flex-1 overflow-y-auto p-4 space-y-2">
        {messages.map((msg) => (
          <div key={msg.id} className="flex flex-col">
            <span className="text-xs text-gray-400">
              {msg.sender.slice(0, 6)} · {new Date(msg.ts).toLocaleTimeString()}
            </span>
            <p className="text-sm bg-white border rounded-lg px-3 py-2 inline-block max-w-xs">
              {msg.text}
            </p>
          </div>
        ))}
        <div ref={bottomRef} />
      </div>

      {/* Input */}
      <form onSubmit={sendMessage} className="p-2 border-t flex gap-2">
        <input
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Type a message..."
          className="flex-1 border rounded px-3 py-1.5 text-sm"
          disabled={!connected}
        />
        <button
          type="submit"
          disabled={!connected || !input.trim()}
          className="px-4 py-1.5 bg-blue-600 text-white text-sm rounded disabled:opacity-50"
        >
          Send
        </button>
      </form>
    </div>
  );
}
```

### 7. Streaming responses (LLM-style)

Route Handlers can also stream text incrementally using `ReadableStream` — useful for AI chat interfaces.

```ts
// app/api/stream/route.ts
export async function POST(req: Request) {
  const { prompt } = await req.json();
  const encoder = new TextEncoder();

  // Simulate streaming by chunking a response word by word
  const words = `Streaming response for: ${prompt}`.split(" ");

  const stream = new ReadableStream({
    async start(controller) {
      for (const word of words) {
        controller.enqueue(encoder.encode(word + " "));
        await new Promise((r) => setTimeout(r, 100)); // simulate latency
      }
      controller.close();
    },
  });

  return new Response(stream, {
    headers: { "Content-Type": "text/plain; charset=utf-8" },
  });
}
```

```tsx
// components/StreamingText.tsx
"use client";

import { useState } from "react";

export function StreamingText() {
  const [output, setOutput] = useState("");
  const [streaming, setStreaming] = useState(false);

  async function startStream() {
    setOutput("");
    setStreaming(true);

    const res = await fetch("/api/stream", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ prompt: "Tell me something interesting" }),
    });

    const reader = res.body!.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      setOutput((prev) => prev + decoder.decode(value));
    }

    setStreaming(false);
  }

  return (
    <div className="space-y-4">
      <button
        onClick={startStream}
        disabled={streaming}
        className="px-4 py-2 bg-blue-600 text-white rounded disabled:opacity-50"
      >
        {streaming ? "Streaming..." : "Start Stream"}
      </button>
      <pre className="p-4 bg-gray-100 rounded min-h-[100px] whitespace-pre-wrap text-sm">
        {output || "Output will appear here..."}
      </pre>
    </div>
  );
}
```
