## Goal
Add meaningful animations to a Next.js app using both Tailwind's built-in animation utilities and Framer Motion for complex choreography.

## Background

Animations serve two purposes in UI: they communicate state changes to the user, and they provide delight. The key is restraint — animations should be fast (150–400ms), serve a purpose, and respect `prefers-reduced-motion`.

**Tailwind animate** utilities cover common cases: `animate-spin`, `animate-ping`, `animate-bounce`, `animate-pulse`, and any custom keyframes you define in `tailwind.config.ts`.

**Framer Motion** is the standard animation library for React. Its core primitive is `motion.div` (and `motion.span`, `motion.button`, etc.) which accept `initial`, `animate`, and `exit` props. `variants` let you define animation states as named objects and coordinate parent-child animations. `AnimatePresence` is required to animate elements that unmount (exit). `layout` prop enables automatic layout animations when a component's size or position changes.

For simple hover/focus effects, CSS `transition` utilities in Tailwind are the right tool — they are zero-runtime-cost. Use Framer Motion only when you need physics-based springs, exit animations, or complex orchestration.

### 4. Basic Framer Motion
```tsx
// src/app/framer-demo/page.tsx
"use client";

import { motion } from "framer-motion";
import { useState } from "react";

export default function FramerDemoPage() {
  const [isVisible, setIsVisible] = useState(true);

  return (
    <div className="p-8 space-y-8">
      <h1 className="text-2xl font-bold">Framer Motion</h1>

      {/* Basic animate */}
      <motion.div
        initial={{ opacity: 0, y: 20 }}
        animate={{ opacity: 1, y: 0 }}
        transition={{ duration: 0.4, ease: "easeOut" }}
        className="p-4 bg-blue-100 rounded-lg"
      >
        Fades + slides in on mount
      </motion.div>

      {/* Hover + tap */}
      <motion.button
        whileHover={{ scale: 1.05 }}
        whileTap={{ scale: 0.95 }}
        className="px-6 py-3 bg-green-600 text-white rounded-lg"
      >
        Spring button
      </motion.button>

      {/* Toggle */}
      <div>
        <button onClick={() => setIsVisible((v) => !v)}
          className="px-4 py-2 border rounded mb-4">
          Toggle box
        </button>
        {isVisible && (
          <motion.div
            initial={{ opacity: 0, scale: 0.8 }}
            animate={{ opacity: 1, scale: 1 }}
            exit={{ opacity: 0, scale: 0.8 }}
            transition={{ type: "spring", stiffness: 300, damping: 25 }}
            className="p-6 bg-purple-100 rounded-lg"
          >
            Spring-animated box
          </motion.div>
        )}
      </div>
    </div>
  );
}
```

### 5. AnimatePresence for exit animations
```tsx
// src/app/animate-presence/page.tsx
"use client";

import { AnimatePresence, motion } from "framer-motion";
import { useState } from "react";

type Notification = { id: number; text: string };

let nextId = 1;

export default function AnimatePresencePage() {
  const [notifications, setNotifications] = useState<Notification[]>([]);

  function add() {
    setNotifications((n) => [
      { id: nextId++, text: `Notification ${nextId - 1}` },
      ...n,
    ]);
  }

  function remove(id: number) {
    setNotifications((n) => n.filter((item) => item.id !== id));
  }

  return (
    <div className="p-8 max-w-sm">
      <button onClick={add} className="px-4 py-2 bg-blue-600 text-white rounded mb-4">
        Add Notification
      </button>
      <ul className="space-y-2">
        <AnimatePresence initial={false}>
          {notifications.map((n) => (
            <motion.li
              key={n.id}
              layout
              initial={{ opacity: 0, x: 50, height: 0 }}
              animate={{ opacity: 1, x: 0, height: "auto" }}
              exit={{ opacity: 0, x: -50, height: 0 }}
              transition={{ duration: 0.25 }}
              className="flex items-center justify-between p-3 bg-white border rounded-lg shadow-sm"
            >
              <span className="text-sm">{n.text}</span>
              <button
                onClick={() => remove(n.id)}
                className="text-gray-400 hover:text-red-500 text-lg leading-none"
              >
                ×
              </button>
            </motion.li>
          ))}
        </AnimatePresence>
      </ul>
    </div>
  );
}
```

### 6. Variants for orchestration
```tsx
// src/app/variants-demo/page.tsx
"use client";

import { motion } from "framer-motion";

const container = {
  hidden: { opacity: 0 },
  show: {
    opacity: 1,
    transition: {
      staggerChildren: 0.08, // each child delays by 80ms
    },
  },
};

const item = {
  hidden: { opacity: 0, y: 20 },
  show: { opacity: 1, y: 0, transition: { type: "spring", stiffness: 300, damping: 24 } },
};

const CARDS = ["React", "Next.js", "TypeScript", "Tailwind CSS", "Framer Motion", "Zustand"];

export default function VariantsDemoPage() {
  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold mb-6">Staggered List</h1>
      <motion.ul
        variants={container}
        initial="hidden"
        animate="show"
        className="grid grid-cols-2 sm:grid-cols-3 gap-4"
      >
        {CARDS.map((card) => (
          <motion.li
            key={card}
            variants={item}
            whileHover={{ y: -4 }}
            className="p-4 bg-white border rounded-lg shadow-sm cursor-default"
          >
            {card}
          </motion.li>
        ))}
      </motion.ul>
    </div>
  );
}
```
