# P6.2 — Optimized E-Commerce Page

**Phase:** 6 — Performance & Core Web Vitals  
**Topic:** Bundle analysis, code splitting with dynamic(), CWV improvements  
**Type:** Project

---

## Goal
Build an e-commerce product detail page (PDP) optimized for Core Web Vitals — analyze the bundle, code-split heavy components, and measure before/after CWV scores.

## What to Build
- Product detail page with images, description, reviews
- Review chart component (heavy library — Recharts) loaded with `dynamic()`
- Related products carousel (heavy — `embla-carousel`) loaded lazily
- Bundle analysis with `@next/bundle-analyzer`
- Before/after comparison: identify the heaviest imports and fix them
- `<Suspense>` wrappers with skeleton fallbacks

## Architecture

```
app/
  products/
    [id]/
      page.tsx            ← product detail page
      loading.tsx         ← full page skeleton
components/
  ProductImages.tsx       ← image gallery (static import — above fold)
  ProductInfo.tsx         ← add to cart, price (static — above fold)
  ReviewSection.tsx       ← lazy: loads ReviewChart
  RelatedProducts.tsx     ← lazy: loads carousel
  skeletons/
    ReviewSkeleton.tsx
    CarouselSkeleton.tsx
lib/
  products.ts
```

```tsx
// app/products/[id]/page.tsx
import { Suspense } from "react";
import dynamic from "next/dynamic";
import { notFound } from "next/navigation";
import { getProduct } from "@/lib/products";
import { ProductImages } from "@/components/ProductImages";
import { ProductInfo } from "@/components/ProductInfo";
import { ReviewSkeleton } from "@/components/skeletons/ReviewSkeleton";
import { CarouselSkeleton } from "@/components/skeletons/CarouselSkeleton";

// Heavy components loaded lazily — NOT in the initial bundle
const ReviewSection = dynamic(
  () => import("@/components/ReviewSection").then((m) => m.ReviewSection),
  { loading: () => <ReviewSkeleton />, ssr: false }
);

const RelatedProducts = dynamic(
  () => import("@/components/RelatedProducts").then((m) => m.RelatedProducts),
  { loading: () => <CarouselSkeleton /> }
);

interface Props {
  params: { id: string };
}

export default async function ProductPage({ params }: Props) {
  const product = await getProduct(params.id);
  if (!product) notFound();

  return (
    <div className="max-w-6xl mx-auto px-4 py-12">
      {/* Above-the-fold: statically imported — part of the initial JS */}
      <div className="grid grid-cols-1 lg:grid-cols-2 gap-12 mb-16">
        <ProductImages images={product.images} />
        <ProductInfo product={product} />
      </div>

      {/* Below-the-fold: code-split via dynamic() */}
      <Suspense fallback={<ReviewSkeleton />}>
        <ReviewSection productId={product.id} reviews={product.reviews} />
      </Suspense>

      <Suspense fallback={<CarouselSkeleton />}>
        <RelatedProducts category={product.category} currentId={product.id} />
      </Suspense>
    </div>
  );
}
```

```tsx
// components/ReviewSection.tsx
// This file imports Recharts — it will be in a separate chunk
import {
  BarChart, Bar, XAxis, YAxis, Tooltip, ResponsiveContainer, Cell
} from "recharts";

interface Review {
  rating: number;
  author: string;
  content: string;
  date: string;
}

interface Props {
  productId: string;
  reviews: Review[];
}

function buildRatingDistribution(reviews: Review[]) {
  const counts = [0, 0, 0, 0, 0];
  for (const r of reviews) counts[r.rating - 1]++;
  return [5, 4, 3, 2, 1].map((star) => ({
    star: `${star}★`,
    count: counts[star - 1],
  }));
}

export function ReviewSection({ reviews }: Props) {
  const distribution = buildRatingDistribution(reviews);
  const avg = reviews.reduce((s, r) => s + r.rating, 0) / (reviews.length || 1);

  return (
    <section className="py-12 border-t">
      <h2 className="text-2xl font-bold mb-8">Customer Reviews</h2>

      <div className="grid grid-cols-1 md:grid-cols-3 gap-8 mb-10">
        {/* Average score */}
        <div className="text-center">
          <p className="text-6xl font-bold">{avg.toFixed(1)}</p>
          <p className="text-yellow-500 text-xl mt-1">{"★".repeat(Math.round(avg))}</p>
          <p className="text-sm text-gray-500 mt-1">{reviews.length} reviews</p>
        </div>

        {/* Rating distribution chart */}
        <div className="col-span-2">
          <ResponsiveContainer width="100%" height={160}>
            <BarChart data={distribution} layout="vertical" barSize={16}>
              <XAxis type="number" hide />
              <YAxis type="category" dataKey="star" width={32} tick={{ fontSize: 12 }} />
              <Tooltip formatter={(v) => [`${v} reviews`, ""]} />
              <Bar dataKey="count" radius={[0, 4, 4, 0]}>
                {distribution.map((_, i) => (
                  <Cell key={i} fill={i === 0 ? "#22c55e" : i < 3 ? "#eab308" : "#ef4444"} />
                ))}
              </Bar>
            </BarChart>
          </ResponsiveContainer>
        </div>
      </div>

      {/* Individual reviews */}
      <div className="space-y-6">
        {reviews.map((review, i) => (
          <div key={i} className="pb-6 border-b last:border-0">
            <div className="flex items-center gap-3 mb-2">
              <div className="w-8 h-8 rounded-full bg-blue-100 text-blue-600 text-sm font-semibold flex items-center justify-center">
                {review.author[0]}
              </div>
              <div>
                <p className="font-medium text-sm">{review.author}</p>
                <p className="text-yellow-400 text-xs">{"★".repeat(review.rating)}{"☆".repeat(5 - review.rating)}</p>
              </div>
              <time className="ml-auto text-xs text-gray-400">{new Date(review.date).toLocaleDateString()}</time>
            </div>
            <p className="text-sm text-gray-600 leading-relaxed">{review.content}</p>
          </div>
        ))}
      </div>
    </section>
  );
}
```

```tsx
// components/RelatedProducts.tsx
// This file imports embla-carousel — separate chunk
"use client";

import useEmblaCarousel from "embla-carousel-react";
import Image from "next/image";

interface Props {
  category: string;
  currentId: string;
}

const RELATED_MOCK = Array.from({ length: 8 }, (_, i) => ({
  id: `r-${i}`,
  name: `Related Product ${i + 1}`,
  price: 29.99 + i * 10,
  image: `https://picsum.photos/seed/r${i}/300/300`,
}));

export function RelatedProducts({ category, currentId }: Props) {
  const [emblaRef] = useEmblaCarousel({ loop: false, align: "start", slidesToScroll: 2 });

  const related = RELATED_MOCK.filter((p) => p.id !== currentId);

  return (
    <section className="py-12 border-t">
      <h2 className="text-2xl font-bold mb-6">You might also like</h2>
      <div ref={emblaRef} className="overflow-hidden">
        <div className="flex gap-4">
          {related.map((product) => (
            <div key={product.id} className="shrink-0 w-52">
              <div className="relative aspect-square rounded-xl overflow-hidden bg-gray-100 mb-3">
                <Image src={product.image} alt={product.name} fill className="object-cover" sizes="208px" />
              </div>
              <p className="text-sm font-medium truncate">{product.name}</p>
              <p className="text-sm text-gray-500">${product.price.toFixed(2)}</p>
            </div>
          ))}
        </div>
      </div>
    </section>
  );
}
```

## Bundle Optimization Notes

### Before optimization (hypothetical)
- Initial JS: 450 kB
- Recharts + embla in the main bundle
- LCP: ~3.2s (blocking on chart rendering)

### After optimization
- Initial JS: 180 kB (60% reduction)
- Recharts in a separate chunk (~120 kB) loaded after hydration
- embla in a separate chunk (~30 kB) loaded lazily
- LCP: ~1.8s

### Identifying what to split
```bash
ANALYZE=true pnpm build
# Opens two treemap visualizations: client and server bundles
# Look for: recharts, @mui, lodash, moment — large libraries not needed above the fold
```

### `dynamic()` cheat sheet
```ts
// Basic dynamic import
const Heavy = dynamic(() => import("./Heavy"));

// With custom loading fallback
const Heavy = dynamic(() => import("./Heavy"), {
  loading: () => <Skeleton />,
});

// Disable SSR (for browser-only libraries like window-dependent charting)
const Chart = dynamic(() => import("./Chart"), { ssr: false });

// Named export
const { MyComponent } = dynamic(
  () => import("./lib").then((m) => ({ default: m.MyComponent }))
);
```
